24/**Copies global values from a PETSc vector to a STL vector.*/
25void chi_math::PETScUtils::CopyGlobalVecToSTLvector(
26 Vec x,
27 const std::vector<int64_t>& global_indices,
28 std::vector<double> &data)
29{
30 //=================================== Populating local indices
31 size_t N = global_indices.size();
32 std::vector<int64_t> local_indices(N,0);
33 size_t counter=0;
34 for (auto val : global_indices)
35 {
36 local_indices[counter] = counter;
37 ++counter;
38 }
39
40 //=================================== Creating PETSc vector
41 Vec local_vec;
42 VecCreateSeq(PETSC_COMM_SELF,global_indices.size()+1,&local_vec);
43 VecSet(local_vec,0.0);
44
45 //=================================== Create and transfer index sets
46 IS global_set;
47 IS local_set;
48 ISCreateGeneral(PETSC_COMM_SELF, N, global_indices.data(),
49 PETSC_COPY_VALUES,&global_set);
50 ISCreateGeneral(PETSC_COMM_SELF, N, local_indices.data(),
51 PETSC_COPY_VALUES,&local_set);
52 VecScatter scat;
53 VecScatterCreate(x,global_set,local_vec,local_set,&scat);
54 VecScatterBegin(scat,x,local_vec,INSERT_VALUES,SCATTER_FORWARD);
55 VecScatterEnd(scat,x,local_vec,INSERT_VALUES,SCATTER_FORWARD);
56
57 //=================================== Copy to STL
58 data.clear();
59 data.resize(N,0.0);
60 const double* x_ref;
61 VecGetArrayRead(local_vec,&x_ref);
62
63 for (size_t i=0; i<N; ++i)
64 data[i] = x_ref[i];
65
66 VecRestoreArrayRead(x,&x_ref);
67
68 //=================================== Cleanup
69 ISDestroy(&global_set);
70 ISDestroy(&local_set);
71
72 VecDestroy(&local_vec);
73}
