228 chi_math::CylindricalAngularQuadrature::InitializeParameters()
229 {
230  const auto pi_sum_q_weights = static_cast<double>(1);
231 
232  fac_diamond_difference.resize(weights.size(), 1);
233  fac_streaming_operator.resize(weights.size(), 0);
234  for (size_t p = 0; p < map_directions.size(); ++p)
235  {
236  // interface quantities initialised to starting direction values
237  double alpha_interface = 0;
238  double phi_interface = abscissae[map_directions[p].front()].phi;
239  std::vector<double> mu_interface(2, std::cos(phi_interface));
240 
241  // initialisation permits to forego start direction and final direction
242  for (size_t q = 1; q < map_directions[p].size()-1; ++q)
243  {
244  const auto k = map_directions[p][q];
245  const auto w_pq = weights[k];
246  const auto mu_pq = omegas[k].x;
247  const auto phi_pq = abscissae[k].phi;
248 
249  alpha_interface -= w_pq * mu_pq;
250 
251  phi_interface -= w_pq * pi_sum_q_weights;
252  mu_interface[0] = mu_interface[1];
253  mu_interface[1] = std::cos(phi_interface);
254 
255  const auto mu = std::cos(phi_pq);
256  const auto tau = (mu - mu_interface[0]) / (mu_interface[1] - mu_interface[0]);
257 
258  fac_diamond_difference[k] = tau;
259  fac_streaming_operator[k] = alpha_interface / (w_pq * tau) + mu_pq;
260  }
261  }
