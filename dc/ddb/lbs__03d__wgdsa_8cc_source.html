8/**Initializes the Within-Group DSA solver. */
9void lbs::SteadySolver::InitWGDSA(LBSGroupset& groupset)
10{
11 if (groupset.apply_wgdsa)
12 {
13 //=========================================== Make UnknownManager
14 const size_t gs_G = groupset.groups.size();
15 chi_math::UnknownManager uk_man;
16 uk_man.AddUnknown(chi_math::UnknownType::VECTOR_N, gs_G);
17
18 //=========================================== Make boundary conditions
19 typedef chi_mesh::sweep_management::BoundaryType SwpBndryType;
20 typedef lbs::acceleration::BoundaryCondition BC;
21 typedef lbs::acceleration::BCType BCType;
22
23 std::vector<BC> bcs;
24 size_t bid = 0;
25 for (auto& lbs_bndry : sweep_boundaries)
26 {
27 if (lbs_bndry->Type() == SwpBndryType::REFLECTING)
28 bcs.push_back({BCType::ROBIN,{0.0,1.0,0.0}});
29 else//dirichlet
30 bcs.push_back({BCType::DIRICHLET,{0.0,0.0,0.0}});
31 ++bid;
32 }
33
34 //=========================================== Make xs map
35 typedef lbs::acceleration::Multigroup_D_and_sigR MGXs;
36 typedef std::map<int, lbs::acceleration::Multigroup_D_and_sigR> MapMatID2XS;
37 MapMatID2XS map_mat_id_2_mgxs;
38 for (const auto& mat_id_xs_pair : matid_to_xs_map)
39 {
40 const auto& mat_id = mat_id_xs_pair.first;
41 const auto& xs = mat_id_xs_pair.second;
42
43 std::vector<double> Dg (gs_G, 0.0);
44 std::vector<double> sigR(gs_G, 0.0);
45
46 size_t g = 0;
47 for (size_t gprime=groupset.groups.front().id;
48 gprime<=groupset.groups.back().id; ++gprime)
49 {
50 Dg[g] = xs->diffusion_coeff[gprime];
51 sigR[g] = xs->sigma_removal[gprime];
52 ++g;
53 }//for g
54
55 map_mat_id_2_mgxs.insert(std::make_pair(mat_id,MGXs{Dg,sigR}));
56 }
57
58 //=========================================== Create solver
59 const auto& sdm = *discretization;
60
61 auto solver =
62 std::make_shared<acceleration::DiffusionMIPSolver>(
63 std::string(TextName()+"_WGDSA"),
64 *grid,sdm,
65 uk_man,
66 bcs,
67 map_mat_id_2_mgxs,
68 unit_cell_matrices,
69 true); //verbosity
70
71 solver->options.residual_tolerance = groupset.wgdsa_tol;
72 solver->options.max_iters = groupset.wgdsa_max_iters;
73 solver->options.verbose = groupset.wgdsa_verbose;
74 solver->options.additional_options_string = groupset.wgdsa_string;
75
76 solver->Initialize();
77
78 delta_phi_local.assign(sdm.GetNumLocalDOFs(uk_man),0.0);
79
80 solver->AssembleAand_b(delta_phi_local);
81
82 delta_phi_local.resize(0);
83 delta_phi_local.shrink_to_fit();
84
85 groupset.wgdsa_solver = solver;
86 }
87}

98void lbs::SteadySolver::
99 AssembleWGDSADeltaPhiVector(LBSGroupset& groupset,
100 const std::vector<double>&ref_phi_old,
101 const std::vector<double>&ref_phi_new)
102{
103 const auto& sdm = *discretization;
104 const auto& dphi_uk_man = groupset.wgdsa_solver->UnknownStructure();
105 const auto& phi_uk_man = flux_moments_uk_man;
106
107 const int gsi = groupset.groups.front().id;
108 const size_t gss = groupset.groups.size();
109
110 delta_phi_local.clear();
111 delta_phi_local.assign(sdm.GetNumLocalDOFs(dphi_uk_man), 0.0);
112
113 for (const auto& cell : grid->local_cells)
114 {
115 const auto& cell_mapping = sdm.GetCellMapping(cell);
116 const size_t num_nodes = cell_mapping.NumNodes();
117 const auto& sigma_s = matid_to_xs_map[cell.material_id]->sigma_s_gtog;
118
119 for (size_t i=0; i < num_nodes; i++)
120 {
121 const int64_t dphi_map = sdm.MapDOFLocal(cell, i, dphi_uk_man, 0, 0);
122 const int64_t phi_map = sdm.MapDOFLocal(cell, i, phi_uk_man, 0, gsi);
123
124 double* delta_phi_mapped = &delta_phi_local[dphi_map];
125 const double* phi_old_mapped = &ref_phi_old[phi_map];
126 const double* phi_new_mapped = &ref_phi_new[phi_map];
127
128 for (size_t g=0; g<gss; g++)
129 {
130 delta_phi_mapped[g] =
131 sigma_s[gsi+g]*(phi_new_mapped[g] - phi_old_mapped[g]);
132 }//for g
133 }//for node
134 }//for cell
