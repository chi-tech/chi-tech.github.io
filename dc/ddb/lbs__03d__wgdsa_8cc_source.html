8/**Initializes the Within-Group DSA solver. */
9void lbs::LBSSolver::InitWGDSA(LBSGroupset& groupset,
10 bool vaccum_bcs_are_dirichlet/*=true*/)
11{
12 if (groupset.apply_wgdsa_)
13 {
14 //=========================================== Make UnknownManager
15 const size_t num_gs_groups = groupset.groups_.size();
16 chi_math::UnknownManager uk_man;
17 uk_man.AddUnknown(chi_math::UnknownType::VECTOR_N, num_gs_groups);
18
19 //=========================================== Make boundary conditions
20 typedef chi_mesh::sweep_management::BoundaryType SwpBndryType;
21 typedef lbs::acceleration::BoundaryCondition BC;
22 typedef lbs::acceleration::BCType BCType;
23
24 std::map<uint64_t, BC> bcs;
25 for (auto& [bid,lbs_bndry] : sweep_boundaries_)
26 {
27 if (lbs_bndry->Type() == SwpBndryType::REFLECTING)
28 bcs[bid] = {BCType::ROBIN,{0.0,1.0,0.0}};
29 else if (lbs_bndry->Type() == SwpBndryType::INCIDENT_VACCUUM)
30 if (vaccum_bcs_are_dirichlet)
31 bcs[bid] = {BCType::DIRICHLET,{0.0,0.0,0.0}};
32 else
33 bcs[bid] = {BCType::ROBIN,{0.25,0.5}};
34 else//dirichlet
35 bcs[bid] = {BCType::DIRICHLET,{0.0,0.0,0.0}};
36 }
37
38 //=========================================== Make xs map
39 typedef lbs::acceleration::Multigroup_D_and_sigR MGXS;
40 typedef std::map<int, lbs::acceleration::Multigroup_D_and_sigR> MatID2XSMap;
41 MatID2XSMap matid_2_mgxs_map;
42 for (const auto& matid_xs_pair : matid_to_xs_map_)
43 {
44 const auto& mat_id = matid_xs_pair.first;
45 const auto& xs = matid_xs_pair.second;
46
47 std::vector<double> D (num_gs_groups, 0.0);
48 std::vector<double> sigma_r(num_gs_groups, 0.0);
49
50 size_t g = 0;
51 const auto& diffusion_coeff = xs->DiffusionCoefficient();
52 const auto& sigma_removal = xs->SigmaRemoval();
53 for (size_t gprime = groupset.groups_.front().id_;
54 gprime <= groupset.groups_.back().id_; ++gprime)
55 {
56 D[g] = diffusion_coeff[gprime];
57 sigma_r[g] = sigma_removal[gprime];
58 ++g;
59 }//for g
60
61 matid_2_mgxs_map.insert(std::make_pair(mat_id, MGXS{D, sigma_r}));
62 }
63
64 //=========================================== Create solver
65 const auto& sdm = *discretization_;
66
67 auto solver =
68 std::make_shared<acceleration::DiffusionMIPSolver>(
69 std::string(TextName()+"_WGDSA"),
70 *grid_ptr_, sdm,
71 uk_man,
72 bcs,
73 matid_2_mgxs_map,
74 unit_cell_matrices_,
75 true); //verbosity
76
77 solver->options.residual_tolerance = groupset.wgdsa_tol_;
78 solver->options.max_iters = groupset.wgdsa_max_iters_;
79 solver->options.verbose = groupset.wgdsa_verbose_;
80 solver->options.additional_options_string = groupset.wgdsa_string_;
81
82 solver->Initialize();
83
84 std::vector<double> dummy_rhs(sdm.GetNumLocalDOFs(uk_man),0.0);
85
86 solver->AssembleAand_b(dummy_rhs);
87
88 groupset.wgdsa_solver_ = solver;
89 }
90}

109

112

120 for (size_t i=0; i < num_nodes; i++)
121 {
122 const int64_t dphi_map = sdm.MapDOFLocal(cell, i, dphi_uk_man, 0, 0);

124
125 double* output_mapped = &output_phi_local[dphi_map];
126 const double* phi_in_mapped = &phi_in[phi_map];
127
128 for (size_t g=0; g<gss; g++)
129 {
130 output_mapped[g] = phi_in_mapped[g];
131 }//for g
132 }//for node
