38  << "Number of azimuthal angles : "

44  // sweep chunks

46  MainSweepScheduler sweepScheduler(DEPTH_OF_GRAPH,
47  groupset->angle_agg);

50  // available inside
51  // Action
52  KSP_DATA_CONTEXT data_context;
53  data_context.solver = this;

59

63  int globl_size = glob_dof_count*num_moments*groupset_numgrps;
64  MatCreateShell(PETSC_COMM_WORLD,local_size,
65  local_size,
66  globl_size,
67  globl_size,
68  &data_context,&A);
69 
70  //================================================== Set the action-operator
71  MatShellSetOperation(A, MATOP_MULT,(void (*)(void)) NPTMatrixAction_Ax);
72 
73  //================================================== Initial vector assembly
74  phi_new_local.assign(phi_new_local.size(),0.0);
75 
76  VecCreate(PETSC_COMM_WORLD,&phi_new);
77  VecCreate(PETSC_COMM_WORLD,&phi_old);
78  VecCreate(PETSC_COMM_WORLD,&q_fixed);
79 
80  VecSetSizes(phi_new,
81  local_size, //Local size
82  globl_size); //Global size
83  VecSetType(phi_new,VECMPI);
84  VecSet(phi_new,0.0);
85  VecDuplicate(phi_new,&phi_old);

90  KSP ksp;

98  PCSetType(pc,PCNONE);
99 
100  KSPSetTolerances(ksp,1.e-50,
101  groupset->residual_tolerance,1.0e50,
102  groupset->max_iterations);
103  KSPGMRESSetRestart(ksp,groupset->gmres_restart_intvl);
104  KSPSetApplicationContext(ksp,&data_context);
105  KSPSetConvergenceTest(ksp,&KSPConvergenceTestNPT,NULL,NULL);
106  KSPSetInitialGuessNonzero(ksp,PETSC_TRUE);
107  KSPSetUp(ksp);
108 
109  //================================================== Compute b
110  chi_log.Log(LOG_0) << "Computing b";
111  SetSource(group_set_num,USE_MATERIAL_SOURCE,SUPPRESS_PHI_OLD);
112  sweep_chunk->SetDestinationPhi(&phi_new_local);
113 
114  sweepScheduler.Sweep(sweep_chunk);
115 
116  //=================================================== Apply WGDSA
117  if (groupset->apply_wgdsa)
118  {
119  AssembleWGDSADeltaPhiVector(groupset, phi_old_local.data(), phi_new_local.data());
120  ((chi_diffusion::Solver*)groupset->wgdsa_solver)->ExecuteS(true,false);
121  DisAssembleWGDSADeltaPhiVector(groupset, phi_new_local.data());
122  }
123  if (groupset->apply_tgdsa)
124  {
125  AssembleTGDSADeltaPhiVector(groupset, phi_old_local.data(), phi_new_local.data());
126  ((chi_diffusion::Solver*)groupset->tgdsa_solver)->ExecuteS(true,false);
127  DisAssembleTGDSADeltaPhiVector(groupset, phi_new_local.data());
128  }
129 
130  //=================================================== Assemble vectors
131  AssembleVector(groupset,q_fixed,phi_new_local.data());
132  AssembleVector(groupset,phi_old,phi_old_local.data());
133 
134  //=================================================== Retool for GMRES
135  sweep_chunk->SetDestinationPhi(&phi_new_local);
136  sweep_chunk->suppress_surface_src = true; //Action of Ax specific
137 
138  VecCopy(phi_old,phi_new);
139 
140  //**************** CALL GMRES SOLVE ******************
141  chi_log.Log(LOG_0) << "Starting iterations";
142  KSPSolve(ksp,q_fixed,phi_new);
143  //****************************************************
144 
145  KSPConvergedReason reason;
146  KSPGetConvergedReason(ksp,&reason);
147  if (reason<0)
148  chi_log.Log(LOG_0WARNING) << "GMRES solver failed.";
149 
150 
151  DisAssembleVector(groupset, phi_new, phi_new_local.data());
152  DisAssembleVector(groupset, phi_new, phi_old_local.data());
153 
154  //==================================================== Clean up
155  KSPDestroy(&ksp);
156  VecDestroy(&phi_new);
157  VecDestroy(&phi_old);
158  VecDestroy(&q_fixed);
159

162  double sweep_time = chi_global_timings[16]/chi_global_timings[17];
163  double source_time= chi_global_timings[18]/chi_global_timings[19];
164  size_t num_angles = groupset->quadrature->abscissae.size();
165  long int num_unknowns = (long int)glob_dof_count*
166  (long int)num_angles*
167  (long int)groupset->groups.size();
168  chi_log.Log(LOG_0)
169  << "\n\n";
170  chi_log.Log(LOG_0)
171  << " Set Src Time/sweep (s): "
172  << source_time*1.0e-3;
173  chi_log.Log(LOG_0)
174  << " Sweep Time/Unknown (ns): "
175  << sweep_time*1.0e9*chi_mpi.process_count/num_unknowns;
176  chi_log.Log(LOG_0)
177  << " Number of unknowns per sweep: " << num_unknowns;
178  chi_log.Log(LOG_0)
179  << "\n\n";
