459  int dep_mapped_index = glob_order_mapping[dep_loc];
460 
461  if (glob_sweep_order_rank[dep_mapped_index] > max_rank)

463  max_rank = glob_sweep_order_rank[dep_mapped_index];
464  }
465  }
