132/**Input parameters based constructor.*/
133lbs::LBSGroupset::LBSGroupset(const chi::InputParameters& params,
134 const int id,
135 const LBSSolver& lbs_solver)
136 : ChiObject(params), id_(id)
137{
138 const std::string fname = __FUNCTION__;
139
140 // ============================================ Add groups
141 const auto groups_from_to =
142 params.GetParamVectorValue<size_t>("groups_from_to");
143 ChiInvalidArgumentIf(groups_from_to.size() != 2,
144 "Parameter \"groups_from_to\" can only have 2 entries");
145
146 const size_t from = groups_from_to[0];
147 const size_t to = groups_from_to[1];
148 ChiInvalidArgumentIf(to < from,
149 "\"to\" field is less than the \"from\" field.");
150
151 try
152 {
153 for (size_t g = from; g <= to; ++g)
154 {
155 groups_.push_back(lbs_solver.Groups().at(g));
156 } // for g
157 }
158 catch (const std::exception& exc)
159 {
160 throw std::invalid_argument("Attempting to group to groupset that is not"
161 "part of the solver.");
162 }
163
164 master_num_grp_subsets_ = params.GetParamValue<int>("groupset_num_subsets");
165
166 // ============================================ Add quadrature
167 const size_t quad_handle =
168 params.GetParamValue<size_t>("angular_quadrature_handle");
169 quadrature_ = Chi::GetStackItemPtr<chi_math::AngularQuadrature>(
170 Chi::angular_quadrature_stack, quad_handle, fname);
171
172 // ============================================ Angle aggregation
173 const auto angle_agg_typestr =
174 params.GetParamValue<std::string>("angle_aggregation_type");
175 if (angle_agg_typestr == "polar")
176 angleagg_method_ = AngleAggregationType::POLAR;
177 else if (angle_agg_typestr == "single")
178 angleagg_method_ = AngleAggregationType::SINGLE;
179 else if (angle_agg_typestr == "azimuthal")
180 angleagg_method_ = AngleAggregationType::AZIMUTHAL;
181
182 master_num_ang_subsets_ =
183 params.GetParamValue<int>("angle_aggregation_num_subsets");
184
185 // ============================================ Inner solver
186 const auto inner_linear_method =
187 params.GetParamValue<std::string>("inner_linear_method");
188 if (inner_linear_method == "richardson")
189 iterative_method_ = IterativeMethod::KRYLOV_RICHARDSON;
190 else if (inner_linear_method == "gmres")
191 iterative_method_ = IterativeMethod::KRYLOV_GMRES;
192 else if (inner_linear_method == "bicgstab")
193 iterative_method_ = IterativeMethod::KRYLOV_BICGSTAB;
194
195 allow_cycles_ = params.GetParamValue<bool>("allow_cycles");
196 residual_tolerance_ = params.GetParamValue<double>("l_abs_tol");
197 max_iterations_ = params.GetParamValue<int>("l_max_its");
198
199 // ============================================ Misc.
200 log_sweep_events_ = params.GetParamValue<bool>("log_sweep_events");
201
202 // ============================================ DSA
203 apply_wgdsa_ = params.GetParamValue<bool>("apply_wgdsa");
204 apply_tgdsa_ = params.GetParamValue<bool>("apply_tgdsa");
205
206 wgdsa_tol_ = params.GetParamValue<double>("wgdsa_l_abs_tol");
207 tgdsa_tol_ = params.GetParamValue<double>("tgdsa_l_abs_tol");
208
209 wgdsa_max_iters_ = params.GetParamValue<int>("wgdsa_l_max_its");
210 tgdsa_max_iters_ = params.GetParamValue<int>("tgdsa_l_max_its");
211
212 wgdsa_verbose_ = params.GetParamValue<bool>("wgdsa_verbose");
213 tgdsa_verbose_ = params.GetParamValue<bool>("tgdsa_verbose");
214
215 wgdsa_string_ = params.GetParamValue<std::string>("wgdsa_petsc_options");

221void lbs::LBSGroupset::BuildDiscMomOperator(unsigned int scattering_order,
222 lbs::GeometryType geometry_type)
223{
224 if (geometry_type == lbs::GeometryType::ONED_SLAB ||
225 geometry_type == lbs::GeometryType::ONED_CYLINDRICAL ||
226 geometry_type == lbs::GeometryType::ONED_SPHERICAL)
227 {
228 quadrature_->BuildDiscreteToMomentOperator(scattering_order, 1);
229 }
230 else if (geometry_type == lbs::GeometryType::TWOD_CARTESIAN ||
231 geometry_type == lbs::GeometryType::TWOD_CYLINDRICAL)
232 {
233 quadrature_->BuildDiscreteToMomentOperator(scattering_order, 2);
234 }
235 else if (geometry_type == lbs::GeometryType::THREED_CARTESIAN)
236 {
237 quadrature_->BuildDiscreteToMomentOperator(scattering_order, 3);
238 }

280/**Constructs the groupset subsets.*/
281void lbs::LBSGroupset::PrintSweepInfoFile(size_t ev_tag,
282 const std::string& file_name)
283{
284 if (not log_sweep_events_) return;
285
286 std::ofstream ofile;
287 ofile.open(file_name, std::ofstream::out);
288
289 ofile << "Groupset Sweep information "
290 << "location " << Chi::mpi.location_id << "\n";
291
292 //======================================== Print all anglesets
293 for (int q = 0; q < angle_agg_->angle_set_groups.size(); ++q)
294 {
295 ofile << "Angle-set group " << q << ":\n";
296 auto& ang_set_grp = angle_agg_->angle_set_groups[q];
297 int num_ang_sets_per_grp = (int)ang_set_grp.angle_sets.size();
298 for (int as = 0; as < num_ang_sets_per_grp; ++as)
299 {
300 auto ang_set = ang_set_grp.angle_sets[as];
301
302 int ang_set_num = as + q * num_ang_sets_per_grp;
303
304 ofile << " Angle-set " << ang_set_num << " angles [# varphi theta]:\n";
305
306 for (auto& ang_num : ang_set->angles)
307 {
308 const auto& angle = quadrature_->abscissae_[ang_num];
309
310 ofile << " " << ang_num << " " << angle.phi << " " << angle.theta
311 << "\n";
312 }
313 }
314 }
315
316 //======================================== Print event history
317 ofile << Chi::log.PrintEventHistory(ev_tag);
318
319 ofile.close();
