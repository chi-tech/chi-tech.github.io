94void lbs::LBSGroupset::BuildSubsets()
95{
96 grp_subsets.clear();
97 grp_subset_sizes.clear();
98
99 //=================================== Groupset subsets
100 int num_gs_subsets = 1;
101 if (master_num_grp_subsets <= groups.size())
102 num_gs_subsets = master_num_grp_subsets;
103
104 int gs_subset_size = floor(groups.size()/num_gs_subsets);
105
106 for (int ss=0; ss<num_gs_subsets; ss++)
107 {
108 int subset_ranki = ss*gs_subset_size;
109 int subset_size = gs_subset_size;
110
111 if (ss == (num_gs_subsets-1))
112 subset_size = (int)groups.size() - ss*gs_subset_size;
113
114// grp_subsets.push_back(GsSubSet(subset_ranki,subset_ranki+subset_size-1));
115 grp_subsets.emplace_back(subset_ranki,subset_ranki+subset_size-1);
116 grp_subset_sizes.push_back(subset_size);
117
118 chi::log.Log()
119 << "Groupset subset " << ss << " "
120 << subset_ranki << "->" << subset_ranki+subset_size-1;
121 }//for ss
122
123 //=================================== Angle subsets
124 ang_subsets_top.clear();
125 ang_subsets_bot.clear();
126 ang_subset_sizes_top.clear();
127 ang_subset_sizes_bot.clear();
128 if (quadrature->type == chi_math::AngularQuadratureType::ProductQuadrature)
129 {
130 auto prodquadrature =
131 std::static_pointer_cast<chi_math::ProductQuadrature>(quadrature);
132 int num_pol_angls_hemi = (int)prodquadrature->polar_ang.size()/2;
133 int num_an_subsets = 1;
134 if (master_num_ang_subsets <= num_pol_angls_hemi)
135 num_an_subsets = master_num_ang_subsets;
136
137 int an_subset_size = floor(num_pol_angls_hemi/num_an_subsets);
138
139 //==================== Top hemishpere
140 for (int ss=0; ss<num_an_subsets; ss++)
141 {
142 int subset_ranki = ss*an_subset_size;
143 int subset_size = an_subset_size;
144
145 if (ss == (num_an_subsets-1))
146 subset_size = num_pol_angls_hemi - ss*an_subset_size;
147
148// ang_subsets_top.push_back(AngSubSet(subset_ranki,subset_ranki+subset_size-1));
149 ang_subsets_top.emplace_back(subset_ranki,subset_ranki+subset_size-1);
150 ang_subset_sizes_top.push_back(subset_size);
151
152 if (angleagg_method != lbs::AngleAggregationType::SINGLE)
153 chi::log.Log()
154 << "Top-hemi Angle subset " << ss << " "
155 << subset_ranki << "->" << subset_ranki+subset_size-1;
156 }//for ss
157
158 //==================== Bottom hemisphere
159 for (int ss=0; ss<num_an_subsets; ss++)
160 {
161 int subset_ranki = ss*an_subset_size + num_pol_angls_hemi;
162 int subset_size = an_subset_size;
163
164 if (ss == (num_an_subsets-1))
165 subset_size = num_pol_angls_hemi - ss*an_subset_size;
166
167// ang_subsets_bot.push_back(AngSubSet(subset_ranki,subset_ranki+subset_size-1));
168 ang_subsets_bot.emplace_back(subset_ranki,subset_ranki+subset_size-1);
169 ang_subset_sizes_bot.push_back(subset_size);
170
171 if (angleagg_method != lbs::AngleAggregationType::SINGLE)
172 chi::log.Log()
173 << "Bot-hemi Angle subset " << ss << " "
174 << subset_ranki << "->" << subset_ranki+subset_size-1;
175 }//for ss
176 }
