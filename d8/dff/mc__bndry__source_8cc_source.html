31 /**Initializes references to necessary mesh elements.
32  *
33  * Since this is a boundary source the major difficulty here is
34  * to get the sampling direction right. Sampled directions need
35  * to be rotation to the opposite of the outward pointing boundary
36  * normals. For this we assemble a rotation matrix for each face
37  * that needs to be sampled.*/
38 void chi_montecarlon::BoundarySource::
39  Initialize(chi_mesh::MeshContinuum* ref_grid,
40  SpatialDiscretization_FV* ref_fv_sdm,
41  chi_montecarlon::Solver* ref_solver)
42 {
43  chi_log.Log(LOG_0) << "Initializing Boundary Source";
44  grid = ref_grid;
45  fv_sdm = ref_fv_sdm;
46 
47  const int ALL_BOUNDRIES = -1;
48 
49  //============================================= Build surface src patchess
50  // The surface points
51  double total_patch_area = 0.0;
52  for (auto& cell : grid->local_cells)
53  {
54  auto fv_view = fv_sdm->MapFeView(cell.local_id);
55 
56 
57  int f=0;
58  for (auto& face : cell.faces)
59  {
60  if (not grid->IsCellBndry(face.neighbor)) {++f; continue;}
61 
62  //Determine if face will be sampled
63  bool sample_face = false;
64  if (ref_bndry == ALL_BOUNDRIES)
65  sample_face = true;
66  else if ((ref_bndry != ALL_BOUNDRIES) and
67  (ref_bndry == abs(face.neighbor)))
68  sample_face = true;
69 
70  if (sample_face)
71  {
72  double face_area = fv_view->face_area[f];
73  chi_mesh::Matrix3x3 R;
74 
75  chi_mesh::Vector3 n = face.normal*-1.0;
76  chi_mesh::Vector3 khat(0.0,0.0,1.0);
77 
78  if (n.Dot(khat) > 0.9999999)
79  R.SetDiagonalVec(1.0,1.0,1.0);
80  else if (n.Dot(khat) < -0.9999999)
81  R.SetDiagonalVec(1.0,1.0,-1.0);
82  else
83  {
84  chi_mesh::Vector3 binorm = khat.Cross(n);
85  binorm = binorm/binorm.Norm();
86 
87  chi_mesh::Vector3 tangent = binorm.Cross(n);
88  tangent = tangent/tangent.Norm();
89 
90  R.SetColJVec(0,tangent);
91  R.SetColJVec(1,binorm);
92  R.SetColJVec(2,n);
93  }
94 
95  total_patch_area += face_area;
96  source_patches.emplace_back(cell.local_id,f,R,face_area);
97  }
98  ++f;
99  }//for face
100  }//for cell
101 
102  //============================================= Build source cdf
103  source_patch_cdf.resize(source_patches.size(),0.0);
104  double cumulative_value = 0.0;
105  int p = 0;
106  for (auto& source_patch : source_patches)
107  {
108  cumulative_value += std::get<3>(source_patch);
109  source_patch_cdf[p] = cumulative_value/total_patch_area;
110  ++p;
111  }
112 
113  int local_num_patches = source_patches.size();
114  int global_num_patches = 0;
115 
116  MPI_Allreduce(&local_num_patches,
117  &global_num_patches,
118  1,
119  MPI_INT,
120  MPI_SUM,
121  MPI_COMM_WORLD);
122 
123  if (global_num_patches == 0)
124  {
125  chi_log.Log(LOG_ALLERROR)
126  << "chi_montecarlon::BoundarySource::Initialize"
127  " No source patches.";
128  exit(EXIT_FAILURE);
129  }
130 }

133 /**Source routine for boundary source.*/
134 chi_montecarlon::Particle chi_montecarlon::BoundarySource::
135  CreateParticle(chi_montecarlon::RandomNumberGenerator *rng)
136 {
137  chi_montecarlon::Particle new_particle;
138 
139  if (source_patch_cdf.empty())
140  {
141  new_particle.alive = false;
142  return new_particle;
143  }
144 
145  //======================================== Sample source patch
146  int source_patch_sample = std::lower_bound(
147  source_patch_cdf.begin(),
148  source_patch_cdf.end(),
149  rng->Rand()) - source_patch_cdf.begin();
150 
151  auto& source_patch = source_patches[source_patch_sample];
152 
153  //======================================== Get references
154  int cell_local_id = std::get<0>(source_patch);
155  int f = std::get<1>(source_patch);
156  auto& RotationMatrix = std::get<2>(source_patch);
157  auto& cell = grid->local_cells[cell_local_id];
158  auto& face = cell.faces[f];
159  auto fv_view = fv_sdm->MapFeView(cell_local_id);
160 
161  //======================================== Sample position
162  if (cell.Type() == chi_mesh::CellType::SLAB)
163  new_particle.pos = *grid->vertices[face.vertex_ids[0]];
164  else if (cell.Type() == chi_mesh::CellType::POLYGON)
165  {
166  chi_mesh::Vertex& v0 = *grid->vertices[face.vertex_ids[0]];
167  chi_mesh::Vertex& v1 = *grid->vertices[face.vertex_ids[1]];
168  double w = rng->Rand();
169  new_particle.pos = v0*w + v1*(1.0-w);
170  }
171  else if (cell.Type() == chi_mesh::CellType::POLYHEDRON)
172  {
173  auto polyh_cell = (chi_mesh::CellPolyhedron*)(&cell);
174  auto polyh_fv_view = (PolyhedronFVView*)fv_view;
175 
176  auto edges = polyh_cell->GetFaceEdges(f);
177 
178  //===================== Sample side
179  double rn = rng->Rand();
180  int s = -1;
181  double cumulated_area = 0.0;
182  for (auto face_side_area : polyh_fv_view->face_side_area[f])
183  {
184  s++;
185  if (rn < ((face_side_area+cumulated_area)/polyh_fv_view->face_area[f]))
186  break;
187  cumulated_area+=face_side_area;
188  }
189 
190  double u = rng->Rand();
191  double v = rng->Rand();
192  while ((u+v)>1.0)
193  {u = rng->Rand(); v = rng->Rand();}
194 
195  chi_mesh::Vector3& v0 = *grid->vertices[edges[s][0]];
196  new_particle.pos = v0 + polyh_fv_view->face_side_vectors[f][s][0]*u +
197  polyh_fv_view->face_side_vectors[f][s][1]*v;
198  }
199 
200  //======================================== Sample direction
201  double costheta = rng->Rand(); //Sample half-range only
202  double theta = acos(sqrt(costheta));
203  double varphi = rng->Rand()*2.0*M_PI;
204 
205  chi_mesh::Vector3 ref_dir;
206  ref_dir.x = sin(theta)*cos(varphi);
207  ref_dir.y = sin(theta)*sin(varphi);
208  ref_dir.z = cos(theta);
209 
210  new_particle.dir = RotationMatrix*ref_dir;
211 
212  //======================================== Sample energy
213  new_particle.egrp = 0;
214 
215  //======================================== Determine weight
216  new_particle.w = 1.0;
217 
218  new_particle.cur_cell_global_id = cell.global_id;
219  new_particle.cur_cell_local_id = cell.local_id;
220 
221  return new_particle;
222 }
