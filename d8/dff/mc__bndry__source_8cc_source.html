31 /**Initializes references to necessary mesh elements.
32  *
33  * Since this is a boundary source the major difficulty here is
34  * to get the sampling direction right. Sampled directions need
35  * to be rotation to the opposite of the outward pointing boundary
36  * normals. For this we assemble a rotation matrix for each face
37  * that needs to be sampled.*/
38 void chi_montecarlon::BoundarySource::
39  Initialize(chi_mesh::MeshContinuum* ref_grid,
40  SpatialDiscretization_FV* ref_fv_sdm,
41  chi_montecarlon::Solver* ref_solver)
42 {
43  chi_log.Log(LOG_0) << "Initializing Boundary Source";
44  grid = ref_grid;
45  fv_sdm = ref_fv_sdm;
46 
47  const int ALL_BOUNDRIES = -1;
48 
49  //============================================= Build surface src patchess
50  // The surface points
51  double total_patch_area = 0.0;
52  for (auto& cell_glob_index : grid->local_cell_glob_indices)
53  {
54  auto cell = grid->cells[cell_glob_index];
55  auto fv_view = fv_sdm->MapFeView(cell_glob_index);
56 
57 
58  int f=0;
59  for (auto& face : cell->faces)
60  {
61  if (not grid->IsCellBndry(face.neighbor)) {++f; continue;}
62 
63  //Determine if face will be sampled
64  bool sample_face = false;
65  if (ref_bndry == ALL_BOUNDRIES)
66  sample_face = true;
67  else if ((ref_bndry != ALL_BOUNDRIES) and
68  (ref_bndry == abs(face.neighbor)))
69  sample_face = true;
70 
71  if (sample_face)
72  {
73  double face_area = fv_view->face_area[f];
74  chi_mesh::Matrix3x3 R;
75 
76  chi_mesh::Vector n = cell->faces[f].normal*-1.0;
77  chi_mesh::Vector khat(0.0,0.0,1.0);
78 
79  if (n.Dot(khat) > 0.9999999)
80  R.SetDiagonalVec(1.0,1.0,1.0);
81  else if (n.Dot(khat) < -0.9999999)
82  R.SetDiagonalVec(1.0,1.0,-1.0);
83  else
84  {
85  chi_mesh::Vector binorm = khat.Cross(n);
86  binorm = binorm/binorm.Norm();
87 
88  chi_mesh::Vector tangent = binorm.Cross(n);
89  tangent = tangent/tangent.Norm();
90 
91  R.SetColJVec(0,tangent);
92  R.SetColJVec(1,binorm);
93  R.SetColJVec(2,n);
94  }
95 
96  total_patch_area += face_area;
97  source_patches.emplace_back(cell_glob_index,f,R,face_area);
98  }
99  ++f;
100  }//for face
101  }//for cell
102 
103  //============================================= Build source cdf
104  source_patch_cdf.resize(source_patches.size(),0.0);
105  double cumulative_value = 0.0;
106  int p = 0;
107  for (auto& source_patch : source_patches)
108  {
109  cumulative_value += std::get<3>(source_patch);
110  source_patch_cdf[p] = cumulative_value/total_patch_area;
111  ++p;
112  }
113 }

116 /**Source routine for boundary source.*/
117 chi_montecarlon::Particle chi_montecarlon::BoundarySource::
118  CreateParticle(chi_montecarlon::RandomNumberGenerator *rng)
119 {
120  chi_montecarlon::Particle new_particle;
121 
122  if (source_patch_cdf.empty())
123  {
124  new_particle.alive = false;
125  return new_particle;
126  }
127 
128  //======================================== Sample source patch
129  int source_patch_sample = std::lower_bound(
130  source_patch_cdf.begin(),
131  source_patch_cdf.end(),
132  rng->Rand()) - source_patch_cdf.begin();
133 
134  auto& source_patch = source_patches[source_patch_sample];
135 
136  //======================================== Get references
137  int cell_glob_index = std::get<0>(source_patch);
138  int f = std::get<1>(source_patch);
139  auto& RotationMatrix = std::get<2>(source_patch);
140  auto cell = grid->cells[cell_glob_index];
141  auto face = cell->faces[f];
142  auto fv_view = fv_sdm->MapFeView(cell_glob_index);
143 
144  //======================================== Sample position
145  if (cell->Type() == chi_mesh::CellType::SLAB)
146  new_particle.pos = *grid->nodes[face.vertex_ids[0]];
147  else if (cell->Type() == chi_mesh::CellType::POLYGON)
148  {
149  chi_mesh::Vertex& v0 = *grid->nodes[face.vertex_ids[0]];
150  chi_mesh::Vertex& v1 = *grid->nodes[face.vertex_ids[1]];
151  double w = rng->Rand();
152  new_particle.pos = v0*w + v1*(1.0-w);
153  }
154  else if (cell->Type() == chi_mesh::CellType::POLYHEDRON)
155  {
156  auto polyh_cell = (chi_mesh::CellPolyhedron*)cell;
157  auto polyh_fv_view = (PolyhedronFVView*)fv_view;
158 
159  auto edges = polyh_cell->GetFaceEdges(f);
160 
161  //===================== Sample side
162  double rn = rng->Rand();
163  int s = -1;
164  double cumulated_area = 0.0;
165  for (auto face_side_area : polyh_fv_view->face_side_area[f])
166  {
167  s++;
168  if (rn < ((face_side_area+cumulated_area)/polyh_fv_view->face_area[f]))
169  break;
170  cumulated_area+=face_side_area;
171  }
172 
173  double u = rng->Rand();
174  double v = rng->Rand();
175  while ((u+v)>1.0)
176  {u = rng->Rand(); v = rng->Rand();}
177 
178  chi_mesh::Vector& v0 = *grid->nodes[edges[s][0]];
179  new_particle.pos = v0 + polyh_fv_view->face_side_vectors[f][s][0]*u +
180  polyh_fv_view->face_side_vectors[f][s][1]*v;
181  }
182 
183  //======================================== Sample direction
184  double costheta = rng->Rand(); //Sample half-range only
185  double theta = acos(sqrt(costheta));
186  double varphi = rng->Rand()*2.0*M_PI;
187 
188  chi_mesh::Vector ref_dir;
189  ref_dir.x = sin(theta)*cos(varphi);
190  ref_dir.y = sin(theta)*sin(varphi);
191  ref_dir.z = cos(theta);
192 
193  new_particle.dir = RotationMatrix*ref_dir;
194 
195  //======================================== Sample energy
196  new_particle.egrp = 0;
197 
198  //======================================== Determine weight
199  new_particle.w = 1.0;
200 
201  new_particle.cur_cell_ind = cell_glob_index;
202 
203 // chi_log.Log(LOG_ALL) << new_particle.pos.PrintS();
204 // usleep(100000);
205 
206  return new_particle;
207 }
