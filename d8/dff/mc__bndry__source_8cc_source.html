38 void chi_montecarlon::BoundarySource::
39  Initialize(chi_mesh::MeshContinuum* ref_grid,
40  CHI_DISCRETIZATION_FV* ref_fv_sdm)
41 {
42  grid = ref_grid;
43  fv_sdm = ref_fv_sdm;
44 
45  int num_local_cells = grid->local_cell_glob_indices.size();
46  for (int lc=0; lc<num_local_cells; lc++)
47  {
48  int cell_glob_index = grid->local_cell_glob_indices[lc];
49  auto cell = grid->cells[cell_glob_index];
50 
51  //$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$ SLAB
52  if (typeid(*cell) == typeid(chi_mesh::CellSlab))
53  {
54  auto slab_cell = (chi_mesh::CellSlab*)cell;
55 
56  int num_faces = 2;
57  for (int f=0; f<num_faces; f++)
58  {
59  int neighbor = slab_cell->edges[f];
60 
61  //================================== If all boundaries
62  if ((neighbor < 0) and (ref_bndry == MC_ALL_BOUNDARIES))
63  {
64  FACE_REF* new_face_ref = new FACE_REF(cell_glob_index,f);
65  chi_mesh::Matrix3x3& R = new_face_ref->RotMatrix;
66 
67  R.SetRowIVec(0,chi_mesh::Vector(1.0,0.0,0.0));
68  R.SetRowIVec(1,chi_mesh::Vector(0.0,1.0,0.0));
69  R.SetRowIVec(2,slab_cell->face_normals[f]*-1.0);
70 
71  ref_cell_faces.push_back(new_face_ref);
72  }
73  //================================== If specific bndries
74  else if ((neighbor < 0) and (ref_bndry == abs(neighbor)))
75  {
76  FACE_REF* new_face_ref = new FACE_REF(cell_glob_index,f);
77  chi_mesh::Matrix3x3& R = new_face_ref->RotMatrix;
78 
79  R.SetRowIVec(0,chi_mesh::Vector(1.0,0.0,0.0));
80  R.SetRowIVec(1,chi_mesh::Vector(0.0,1.0,0.0));
81  R.SetRowIVec(2,slab_cell->face_normals[f]*-1.0);
82 
83  ref_cell_faces.push_back(new_face_ref);
84  }
85 
86  }//for faces
87  }//if slab
88  // $$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$ POLYGON
89  else if (typeid(*cell) == typeid(chi_mesh::CellPolygon))
90  {
91  auto poly_cell = (chi_mesh::CellPolygon*)cell;
92  auto cell_fv_view = (PolygonFVView*)fv_sdm->MapFeView(cell_glob_index);
93 
94  int num_faces = poly_cell->edges.size();
95  for (int f=0; f<num_faces; f++)
96  {
97  int neighbor = poly_cell->edges[f][EDGE_NEIGHBOR];
98 
99  //================================== If all boundaries
100  if ((neighbor < 0) and (ref_bndry == MC_ALL_BOUNDARIES))
101  {
102  FACE_REF* new_face_ref = new FACE_REF(cell_glob_index,f);
103  chi_mesh::Matrix3x3& R = new_face_ref->RotMatrix;
104 
105  chi_mesh::Vector n = poly_cell->edgenormals[f]*-1.0;
106  chi_mesh::Vector khat(0.0,0.0,1.0);
107 
108  if (n.Dot(khat) > 0.9999)
109  R.SetDiagonalVec(1.0,1.0,1.0);
110  else
111  {
112  chi_mesh::Vector binorm = khat.Cross(n);
113  binorm = binorm/binorm.Norm();
114 
115  chi_mesh::Vector tangent = binorm.Cross(n);
116  tangent = tangent/tangent.Norm();
117 
118  R.SetRowIVec(0,tangent);
119  R.SetRowIVec(1,binorm);
120  R.SetRowIVec(2,n);
121  }
122 
123  new_face_ref->area = cell_fv_view->face_area[f];
124 
125  ref_cell_faces.push_back(new_face_ref);
126  }
127  //================================== If specific bndries
128  else if ((neighbor < 0) and (ref_bndry == abs(neighbor)))
129  {
130  FACE_REF* new_face_ref = new FACE_REF(cell_glob_index,f);
131  chi_mesh::Matrix3x3& R = new_face_ref->RotMatrix;
132 
133  chi_mesh::Vector n = poly_cell->edgenormals[f]*-1.0;
134  chi_mesh::Vector khat(0.0,0.0,1.0);
135 
136  if (n.Dot(khat) > 0.9999)
137  R.SetDiagonalVec(1.0,1.0,1.0);
138  else
139  {
140  chi_mesh::Vector binorm = khat.Cross(n);
141  binorm = binorm/binorm.Norm();
142 
143  chi_mesh::Vector tangent = binorm.Cross(n);
144  tangent = tangent/tangent.Norm();
145 
146  R.SetRowIVec(0,tangent);
147  R.SetRowIVec(1,binorm);
148  R.SetRowIVec(2,n);
149  }
150 
151  new_face_ref->area = cell_fv_view->face_area[f];
152 
153  ref_cell_faces.push_back(new_face_ref);
154  }
155  }
156  }//for polygon
157  else
158  {
159  chi_log.Log(LOG_ALLERROR)
160  << "chi_montecarlon: Unsupported cell type encountered in "
161  << "call to BoundarySource::Initialize.";
162  exit(EXIT_FAILURE);
163  }
164  }//for local cell
165 
166  //============================================= Build cdf for surface
167  // sampling
168  double total_area = 0.0;
169  int num_ref_faces = ref_cell_faces.size();
170  for (int rf=0; rf<num_ref_faces; rf++)
171  total_area += ref_cell_faces[rf]->area;
172 
173  double intgl = 0.0;
174  face_cdf.resize(num_ref_faces);
175  for (int rf=0; rf<num_ref_faces; rf++)
176  {
177  intgl += ref_cell_faces[rf]->area;
178  face_cdf[rf] = intgl/total_area;
179  }
180 }

200  FACE_REF* face_ref = ref_cell_faces[f];

202  chi_mesh::CellSlab* slab_cell =

204  int ref_vert_index = slab_cell->v_indices[face_ref->face_num];
205  chi_mesh::Vertex* ref_vert = grid->nodes[ref_vert_index];
206 
207  //====================================== Sample direction
208  double costheta = rng->Rand(); //Sample half-range only
209  double theta = acos(costheta);
210  double varphi = rng->Rand()*2.0*M_PI;
211 
212  chi_mesh::Vector ref_dir;
213  ref_dir.x = sin(theta)*cos(varphi);
214  ref_dir.y = sin(theta)*sin(varphi);
215  ref_dir.z = cos(theta);
216 
217  //====================================== Set quantities

219  new_particle.dir = face_ref->RotMatrix*ref_dir;
220

222  new_particle.w = costheta;
223  new_particle.cur_cell_ind = face_ref->cell_glob_index;
224 
225  return new_particle;
226  }//Slab cells
227  else

230  }
