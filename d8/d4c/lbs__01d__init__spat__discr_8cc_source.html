24void lbs::LBSSolver::ComputeUnitIntegrals()
25{
26 Chi::log.Log() << "Computing unit integrals.\n";
27 const auto& sdm = *discretization_;
28
29 //======================================== Define spatial weighting functions
30 struct SpatialWeightFunction //SWF
31 {
32 virtual double operator()(const chi_mesh::Vector3& pt) const
33 { return 1.0; }
34 virtual ~SpatialWeightFunction() = default;
35 };
36
37 struct SphericalSWF : public SpatialWeightFunction
38 {
39 double operator()(const chi_mesh::Vector3& pt) const override
40 { return pt[2]*pt[2]; }
41 };
42
43 struct CylindricalSWF : public SpatialWeightFunction
44 {
45 double operator()(const chi_mesh::Vector3& pt) const override
46 { return pt[0]; }
47 };
48
49 auto swf_ptr = std::make_shared<SpatialWeightFunction>();
50 if (options_.geometry_type == lbs::GeometryType::ONED_SPHERICAL)
51 swf_ptr = std::make_shared<SphericalSWF>();
52 if (options_.geometry_type == lbs::GeometryType::TWOD_CYLINDRICAL)
53 swf_ptr = std::make_shared<CylindricalSWF>();
54
55 auto ComputeCellUnitIntegrals = [&sdm](const chi_mesh::Cell& cell,
56 const SpatialWeightFunction& swf)
57 {
58 const auto& cell_mapping = sdm.GetCellMapping(cell);
59 const size_t cell_num_faces = cell.faces_.size();
60 const size_t cell_num_nodes = cell_mapping.NumNodes();
61 const auto vol_qp_data = cell_mapping.MakeVolumeQuadraturePointData();
62
63 MatDbl IntV_gradshapeI_gradshapeJ(cell_num_nodes, VecDbl(cell_num_nodes));
64 MatVec3 IntV_shapeI_gradshapeJ(cell_num_nodes, VecVec3(cell_num_nodes));
65 MatDbl IntV_shapeI_shapeJ(cell_num_nodes, VecDbl(cell_num_nodes));
66 VecDbl IntV_shapeI(cell_num_nodes);
67
68 std::vector<MatDbl> IntS_shapeI_shapeJ(cell_num_faces);
69 std::vector<MatVec3> IntS_shapeI_gradshapeJ(cell_num_faces);
70 std::vector<VecDbl> IntS_shapeI(cell_num_faces);
71
72 //Volume integrals
73 for (unsigned int i = 0; i < cell_num_nodes; ++i)
74 {
75 for (unsigned int j = 0; j < cell_num_nodes; ++j)
76 {
77 for (const auto& qp : vol_qp_data.QuadraturePointIndices())
78 {
79 IntV_gradshapeI_gradshapeJ[i][j]
80 += swf(vol_qp_data.QPointXYZ(qp)) *
81 vol_qp_data.ShapeGrad(i, qp).Dot(vol_qp_data.ShapeGrad(j, qp)) *
82 vol_qp_data.JxW(qp); //K-matrix
83
84 IntV_shapeI_gradshapeJ[i][j]
85 += swf(vol_qp_data.QPointXYZ(qp)) *
86 vol_qp_data.ShapeValue(i, qp) *
87 vol_qp_data.ShapeGrad(j, qp) *
88 vol_qp_data.JxW(qp); //G-matrix
89
90 IntV_shapeI_shapeJ[i][j]
91 += swf(vol_qp_data.QPointXYZ(qp)) *
92 vol_qp_data.ShapeValue(i, qp) *
93 vol_qp_data.ShapeValue(j, qp) *
94 vol_qp_data.JxW(qp); //M-matrix
95 }// for qp
96 }// for j
97
98 for (const auto& qp : vol_qp_data.QuadraturePointIndices())
99 {
100 IntV_shapeI[i]
101 += swf(vol_qp_data.QPointXYZ(qp)) *
102 vol_qp_data.ShapeValue(i, qp) * vol_qp_data.JxW(qp);
103 }// for qp
104 }//for i
105
106 // surface integrals
107 for (size_t f = 0; f < cell_num_faces; ++f)
108 {
109 const auto faces_qp_data = cell_mapping.MakeFaceQuadraturePointData(f);
110 IntS_shapeI_shapeJ[f].resize(cell_num_nodes, VecDbl(cell_num_nodes));
111 IntS_shapeI[f].resize(cell_num_nodes);
112 IntS_shapeI_gradshapeJ[f].resize(cell_num_nodes, VecVec3(cell_num_nodes));
113
114 for (unsigned int i = 0; i < cell_num_nodes; ++i)
115 {
116 for (unsigned int j = 0; j < cell_num_nodes; ++j)
117 {
118 for (const auto& qp : faces_qp_data.QuadraturePointIndices())
119 {
120 IntS_shapeI_shapeJ[f][i][j]
121 += swf(faces_qp_data.QPointXYZ(qp)) *
122 faces_qp_data.ShapeValue(i, qp) *
123 faces_qp_data.ShapeValue(j, qp) *
124 faces_qp_data.JxW(qp);
125 IntS_shapeI_gradshapeJ[f][i][j]
126 += swf(faces_qp_data.QPointXYZ(qp)) *
127 faces_qp_data.ShapeValue(i, qp) *
128 faces_qp_data.ShapeGrad(j, qp) *
129 faces_qp_data.JxW(qp);
130 }// for qp
131 }//for j
132
133 for (const auto& qp : faces_qp_data.QuadraturePointIndices())
134 {
135 IntS_shapeI[f][i]
136 += swf(faces_qp_data.QPointXYZ(qp)) *
137 faces_qp_data.ShapeValue(i, qp) * faces_qp_data.JxW(qp);
138 }// for qp
139 }//for i
140 }//for f
141
142 return
143 UnitCellMatrices{IntV_gradshapeI_gradshapeJ, //K-matrix
144 IntV_shapeI_gradshapeJ, //G-matrix
145 IntV_shapeI_shapeJ, //M-matrix
146 IntV_shapeI, //Vi-vectors
147
148 IntS_shapeI_shapeJ, //face M-matrices
149 IntS_shapeI_gradshapeJ, //face G-matrices
150 IntS_shapeI}; //face Si-vectors
151 };
152
153 const size_t num_local_cells = grid_ptr_->local_cells.size();
154 unit_cell_matrices_.resize(num_local_cells);
155
156 for (const auto& cell : grid_ptr_->local_cells)
157 unit_cell_matrices_[cell.local_id_] = ComputeCellUnitIntegrals(cell, *swf_ptr);
158
159 const auto ghost_ids = grid_ptr_->cells.GetGhostGlobalIDs();
160 for (uint64_t ghost_id : ghost_ids)
161 unit_ghost_cell_matrices_[ghost_id] =
162 ComputeCellUnitIntegrals(grid_ptr_->cells[ghost_id],*swf_ptr);
163
164 //============================================= Assessing global unit cell
165 // matrix storage
166 std::array<size_t,2> num_local_ucms = {unit_cell_matrices_.size(),
167 unit_ghost_cell_matrices_.size()};
168 std::array<size_t,2> num_globl_ucms = {0,0};
169
170 MPI_Allreduce(num_local_ucms.data(), //sendbuf
171 num_globl_ucms.data(), //recvbuf
172 2, MPIU_SIZE_T, //count+datatype
173 MPI_SUM, //operation
174 Chi::mpi.comm); //comm
175
176
177
178 Chi::mpi.Barrier();
179 Chi::log.Log()
180 << "Ghost cell unit cell-matrix ratio: "
181 << (double)num_globl_ucms[1]*100/(double)num_globl_ucms[0]
