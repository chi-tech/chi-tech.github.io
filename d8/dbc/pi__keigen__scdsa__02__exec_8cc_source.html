21
22 /**Lambda for the creation of scattering sources but the
23 * input vector is only the zeroth moment*/
24 auto SetLBSScatterSourcePhi0 =
25 [this, &phi_temp](const VecDbl& input,
26 const bool additive,
27 const bool suppress_wg_scat = false)

29 ProjectBackPhi0(front_gs_, input, phi_temp);

33 using namespace chi_math;
34
35 k_eff_ = 1.0;
36 double k_eff_prev = 1.0;
37 double k_eff_change = 1.0;
38
39 //================================================== Start power iterations
40 int nit = 0;
41 bool converged = false;
42 while (nit < max_iters_)
43 {
44 //================================= Set the fission source
45 SetLBSFissionSource(phi_old_local_, /*additive=*/false);
46 Scale(q_moments_local_, 1.0 / k_eff_);
47
48 auto Sf_ell = q_moments_local_;
49 auto Sf0_ell = CopyOnlyPhi0(front_gs_, q_moments_local_);
50
51 //================================= This solves the inners for transport
52 primary_ags_solver_->Setup();
53 primary_ags_solver_->Solve();
54
55 // lph_i = l + 1/2,i
56 auto phi0_lph_i = CopyOnlyPhi0(front_gs_, phi_new_local_);
57
58 // Now we produce lph_ip1 = l + 1/2, i+1
59 q_moments_local_ = Sf_ell; // Restore 1/k F phi_l
60 SetLBSScatterSource(phi_new_local_, /*additive=*/true);
61
62 front_wgs_context_->ApplyInverseTransportOperator(NO_FLAGS_SET); // Sweep
63
64 auto phi0_lph_ip1 = CopyOnlyPhi0(front_gs_, phi_new_local_);
65
66 //====================================== Power Iteration Acceleration
67 SetLBSScatterSourcePhi0(phi0_lph_ip1 - phi0_lph_i, /*additive=*/false);
68 auto Ss_res = CopyOnlyPhi0(front_gs_, q_moments_local_);
69
70 double production_k = lbs_solver_.ComputeFissionProduction(phi_new_local_);
71
72 VecDbl epsilon_k(phi0_lph_ip1.size(), 0.0);
73 auto epsilon_kp1 = epsilon_k;
74
75 double lambda_k = k_eff_;
76 double lambda_kp1 = lambda_k;
77
78 for (size_t k = 0; k < accel_pi_max_its_; ++k)
79 {
80 ProjectBackPhi0(front_gs_,
81 /*in*/ epsilon_k + phi0_lph_ip1,
82 /*out*/ phi_temp);
83
84 //double production_k = lbs_solver_.ComputeFissionProduction(phi_temp);
85
86 SetLBSFissionSource(phi_temp, /*additive=*/false);
87 Scale(q_moments_local_, 1.0 / lambda_k);
88
89 auto Sfaux = CopyOnlyPhi0(front_gs_, q_moments_local_);
90
91 // Inner iterations seems extremely wasteful therefore I
92 // am leaving this at 1 iteration here for further investigation.
93 for (int i = 0; i < 1; ++i)
94 {
95 SetLBSScatterSourcePhi0(epsilon_k,
96 /*additive=*/false,
97 /*suppress_wg_scat=*/true);
98
99 auto Ss = CopyOnlyPhi0(front_gs_, q_moments_local_);
100
101 // Solve the diffusion system
102 diffusion_solver_->Assemble_b(Ss + Sfaux + Ss_res - Sf0_ell);
103 diffusion_solver_->Solve(epsilon_kp1, /*use_initial_guess=*/true);
104
105 epsilon_k = epsilon_kp1;
106 }
107
108 ProjectBackPhi0(front_gs_,
109 /*in*/ epsilon_kp1 + phi0_lph_ip1,
110 /*out*/ phi_old_local_);
111
112 double production_kp1 =
113 lbs_solver_.ComputeFissionProduction(phi_old_local_);
114
115 lambda_kp1 = production_kp1 / (production_k / lambda_k);
116
117 const double lambda_change = std::fabs(1.0 - lambda_kp1 / lambda_k);
118 if (accel_pi_verbose_ >= 1)
119 Chi::log.Log() << "PISCDSA iteration " << k << " lambda " << lambda_kp1
120 << " lambda change " << lambda_change;
121
122 if (lambda_change < accel_pi_k_tol_) break;
123
124 lambda_k = lambda_kp1;
125 epsilon_k = epsilon_kp1;
126 production_k = production_kp1;
127 } // acceleration
128
129 ProjectBackPhi0(front_gs_,
130 /*in*/ epsilon_kp1 + phi0_lph_ip1,
131 /*out*/ phi_new_local_);
132 lbs_solver_.GSScopedCopyPrimarySTLvectors(front_gs_,
133 /*in*/ phi_new_local_,
134 /*out*/ phi_old_local_);
135
136 const double production =
137 lbs_solver_.ComputeFissionProduction(phi_old_local_);
138 lbs_solver_.ScalePhiVector(PhiSTLOption::PHI_OLD, lambda_kp1 / production);
139
140 //================================= Recompute k-eigenvalue
141 k_eff_ = lambda_kp1;
142 double reactivity = (k_eff_ - 1.0) / k_eff_;
143
144 //================================= Check convergence, bookkeeping
145 k_eff_change = fabs(k_eff_ - k_eff_prev) / k_eff_;
146 k_eff_prev = k_eff_;
147 nit += 1;
148
149 if (k_eff_change < std::max(k_tolerance_, 1.0e-12)) converged = true;
150
151 //================================= Print iteration summary
152 if (lbs_solver_.Options().verbose_outer_iterations)
153 {
154 std::stringstream k_iter_info;
155 k_iter_info << Chi::program_timer.GetTimeString() << " "
156 << " Iteration " << std::setw(5) << nit << " k_eff "
157 << std::setw(11) << std::setprecision(7) << k_eff_
158 << " k_eff change " << std::setw(12) << k_eff_change
159 << " reactivity " << std::setw(10) << reactivity * 1e5;
160 if (converged) k_iter_info << " CONVERGED\n";
161
162 Chi::log.Log() << k_iter_info.str();
163 }
164
165 if (converged) break;
166 } // for k iterations
167
168 //================================================== Print summary
169 Chi::log.Log() << "\n";
170 Chi::log.Log() << " Final k-eigenvalue : "
171 << std::setprecision(7) << k_eff_;
172 Chi::log.Log() << " Final change : "
173 << std::setprecision(6) << k_eff_change << " (num_TrOps:"
174 << front_wgs_context_->counter_applications_of_inv_op_ << ")"
175 << "\n";
176 Chi::log.Log() << "\n";
177
178 if (lbs_solver_.Options().use_precursors)
179 {
180 lbs_solver_.ComputePrecursors();
181 chi_math::Scale(lbs_solver_.PrecursorsNewLocal(), 1.0 / k_eff_);
182 }
183
184 lbs_solver_.UpdateFieldFunctions();
185
186 Chi::log.Log()
187 << "LinearBoltzmann::KEigenvalueSolver execution completed\n\n";
188}
