28  int cell_local_ind = cell->cell_local_id;
29 
30  int ir = cell_local_ind*num_grps + prtcl.egrp;
31 
32  double tracklength = (pf - prtcl.pos).Norm();
33 
34  double tally_contrib = tracklength*prtcl.w;
35 
36  phi_tally[ir] += tally_contrib;
37  phi_tally_sqr[ir] += tally_contrib*tally_contrib;
38

41  chi_log.Log(LOG_ALLERROR)
42  << "Tracklength corruption."
43  << " pos " << prtcl.pos.PrintS()

45  exit(EXIT_FAILURE);
46  }
47 
48  if (make_pwld)
49  {
50  segment_lengths.clear();
51  segment_lengths.push_back(tracklength);
52  chi_mesh::PopulateRaySegmentLengths(grid, cell,
53  segment_lengths,
54  prtcl.pos, pf,prtcl.dir);
55 
56 
57 
58  auto cell_pwl_view = pwl_discretization->MapFeView(prtcl.cur_cell_ind);
59  int map = local_cell_pwl_dof_array_address[cell_local_ind];
60 
61  double last_segment_length = 0.0;
62 
63  for (auto segment_length : segment_lengths)

76  phi_pwl_tally[ir] += pwl_tally_contrib;
77  phi_pwl_tally_sqr[ir] += pwl_tally_contrib*pwl_tally_contrib;

85 /**Merges tallies from multiple locations.*/
86 void chi_montecarlon::Solver::RendesvouzTallies()
87 {
88  TULL temp_nps_global = 0;
89  MPI_Allreduce(&nps,&temp_nps_global,1,MPI_UNSIGNED_LONG_LONG,MPI_SUM,MPI_COMM_WORLD);
90 
91  nps_global += temp_nps_global;
92 
93  //============================================= If mesh global
94  if (mesh_is_global)
95  {
96  //============================ Merge phi_local
97  int num_values = phi_tally.size();
98  std::vector<double> temp_phi_global(num_values,0.0);
99 
100  MPI_Allreduce(phi_tally.inner(),
101  temp_phi_global.inner(),
102  num_values,MPI_DOUBLE,
103  MPI_SUM,MPI_COMM_WORLD);
104 
105  for (int i=0; i<num_values; i++)
106  phi_global[i] += temp_phi_global[i];
107 
108  //============================ Merge the square of phi_local
109  phi_global_tally_sqr.assign(num_values,0.0);
110 
111  MPI_Allreduce(phi_tally_sqr.inner(),
112  phi_global_tally_sqr.inner(),
113  num_values,MPI_DOUBLE,
114  MPI_SUM,MPI_COMM_WORLD);
115 
116  //============================ Reset tallies
117  phi_tally.assign(phi_tally.size(),0.0);
118  phi_tally_sqr.assign(phi_tally_sqr.size(),0.0);
119  }
120  //============================================= Mesh partitioned
121  else
122  {
123  int num_values = phi_tally.size();
124 
125  phi_global_tally_sqr.assign(num_values,0.0);
126 
127  for (int i=0; i<num_values; i++)
128  {
129  phi_global[i] += phi_tally[i];
130  phi_global_tally_sqr[i] = phi_tally_sqr[i];
131  }
132 
133  //============================ Reset tallies
134  phi_tally.assign(phi_tally.size(),0.0);
135  phi_tally_sqr.assign(phi_tally_sqr.size(),0.0);
136  }

194 void chi_montecarlon::Solver::ComputeRelativeStdDev()
195 {
196  max_relative_error = 0.0;
197 
198  int num_cells = grid->local_cell_glob_indices.size();
199  for (int lc=0; lc<num_cells; lc++)
200  {
201  int hi = 0;
202  int lo = num_grps-1;
203  if (group_hi_bound >= 0)
204  hi = group_hi_bound;
205  if (group_lo_bound >=0 && group_lo_bound >= group_hi_bound)
206  lo = group_lo_bound;
207 
208  for (int g=0; g<num_grps; g++)
209  {
210  int ir = lc*num_grps + g;
211 
212  TULL n = batch_sizes[current_batch];
213 
214  double x_avg = phi_global[ir]/nps_global;
215  double x2_avg = (phi_global_tally_sqr[ir]/n);
216 
217  double stddev = sqrt((x2_avg - x_avg*x_avg)/nps_global);
218 
219  if (!std::isinf(stddev/x_avg))
220  {
221  phi_local_relsigma[ir] = stddev/x_avg;
222 
223  if (g==0 && lc == 0)
224  {
225  max_relative_error = phi_local_relsigma[ir];
226  max_relative_error2 = x2_avg;
227  max_relative_error3 = x_avg*x_avg;
228  }
229  }

231  {
232  phi_local_relsigma[ir] = 0.0;
233  }
234 
235 
236  if (std::isinf(phi_local_relsigma[ir]))

245  }
246  }//for g
