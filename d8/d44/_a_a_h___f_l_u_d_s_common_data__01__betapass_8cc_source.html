22 // first because the delayed information does not follow the
23 // Task Dependency Graph and hence when a location receives its delayed
24 // information, the information might not have been sent yet.
25
26 //=============================================== Send delayed successor
27 //information

33 std::vector<int>());

35 {
36 int locJ = location_successors[deplocI];
37
38 std::vector<int>::const_iterator delayed_successor =
39 std::find(delayed_location_successors.begin(),
40 delayed_location_successors.end(),
41 locJ);
42 if ((delayed_successor == delayed_location_successors.end())) continue;
43

50 static_cast<int>(multi_face_indices[deplocI].size()),
51 MPI_INT,

53 101 + tag_index,

55 &send_requests[deplocI]);
