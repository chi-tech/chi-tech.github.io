43/**Input parameters based construction.*/
44LBSSolver::LBSSolver(const chi_objects::InputParameters& params)
45 : chi_physics::Solver(params)
46{
47 //=================================== Make groups
48 const size_t num_groups = params.GetParamValue<size_t>("num_groups");
49 for (size_t g = 0; g < num_groups; ++g)
50 groups_.push_back(LBSGroup(static_cast<int>(g)));
51
52 //=================================== Make groupsets
53 const auto& groupsets_array = params.GetParam("groupsets");
54
55 const size_t num_gs = groupsets_array.NumParameters();
56 for (size_t gs = 0; gs < num_gs; ++gs)
57 {
58 const auto& groupset_params = groupsets_array.GetParam(gs);
59
60 chi_objects::InputParameters gs_input_params =
61 LBSGroupset::GetInputParameters();
62 gs_input_params.SetObjectType("LBSSolver:LBSGroupset");
63 gs_input_params.AssignParameters(groupset_params);
64
65 groupsets_.emplace_back(gs_input_params, gs, *this);
66 } // for gs
67
68 //=================================== Options
69 if (params.ParametersAtAssignment().Has("options"))
70 {
71 auto options_params = LBSSolver::OptionsBlock();
72 options_params.AssignParameters(params.GetParam("options"));
73
74 this->SetOptions(options_params);
75 }
76}
