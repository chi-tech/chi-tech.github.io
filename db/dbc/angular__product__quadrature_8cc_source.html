107/**Optimizes the angular quadrature for polar symmetry by removing
108 * all the direction with downward pointing polar angles.
109 *
110 * \param normalization float. (Optional) The default is a negative number
111 * which does not apply any normalization. If a
112 * positive number is provided, the weights will be
113 * normalized to sum to this number.*/
114void chi_math::ProductQuadrature::
115 OptimizeForPolarSymmetry(const double normalization)
116{
117 std::vector<chi_math::QuadraturePointPhiTheta> new_abscissae;
118 std::vector<double> new_weights;
119 std::vector<chi_mesh::Vector3> new_omegas;
120 std::vector<double> new_polar_ang;
121 std::vector<double> new_azimu_ang;
122
123 const size_t num_pol = polar_ang.size();
124 const size_t num_azi = azimu_ang.size();
125
126 std::vector<unsigned int> new_polar_map;
127 for (size_t p=0; p<num_pol; ++p)
128 if (polar_ang[p] < M_PI_2)
129 {
130 new_polar_ang.push_back(polar_ang[p]);
131 new_polar_map.push_back(p);
132 }
133 new_azimu_ang = azimu_ang;
134
135 const size_t new_num_pol = new_polar_ang.size();
136 double weight_sum = 0.0;
137 for (size_t a=0; a<num_azi; ++a)
138 for (size_t p=0; p<new_num_pol; ++p)
139 {
140 const auto pmap = new_polar_map[p];
141 const auto dmap = GetAngleNum(pmap,a);
142 new_weights.push_back(weights[dmap]);
143 weight_sum += weights[dmap];
144 }
145
146 if (normalization > 0.0)
147 for (double& w : new_weights)
148 w *= normalization/weight_sum;
149
150
151 AssembleCosines(new_azimu_ang, new_polar_ang,new_weights,false);
152 polar_ang = new_polar_ang;
153 azimu_ang = new_azimu_ang;
154}

183/**Constructor for Angular Gauss-Legendre-Legendre.*/
184chi_math::AngularQuadratureProdGLL::
185 AngularQuadratureProdGLL(int Na, int Np, bool verbose)
186{
187 chi_math::QuadratureGaussLegendre gl_polar(Np*2);
188 chi_math::QuadratureGaussLegendre gl_azimu(Na*4);
189
190 //================================================= Create azimuthal angles
191 azimu_ang.clear();
192 for (unsigned int i = 0; i < (Na*4); ++i)
193 azimu_ang.emplace_back(M_PI*gl_azimu.qpoints[i][0] + M_PI);
194
195 //================================================== Create polar angles
196 polar_ang.clear();
197 for (unsigned int j = 0; j < (Np*2); ++j)
198 polar_ang.emplace_back(M_PI-acos(gl_polar.qpoints[j][0]));
199
200 //================================================== Create combined weights
201 std::vector<double> weights;
202 for (unsigned int i = 0; i < azimu_ang.size(); ++i)
203 for (unsigned int j = 0; j < polar_ang.size(); ++j)
204 weights.emplace_back(M_PI*gl_azimu.weights[i]*gl_polar.weights[j]);
205
206 //================================================== Initialize
207 AssembleCosines(azimu_ang, polar_ang, weights, verbose);
208}
