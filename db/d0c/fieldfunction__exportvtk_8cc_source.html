145/**Uploads just the geometry portion of a cell to VTK.*/
146void chi_physics::FieldFunction::
147 UploadCellGeometry(const chi_mesh::Cell &cell,
148 int64_t& node_counter,
149 vtkNew<vtkPoints>& points,
150 vtkNew<vtkUnstructuredGrid> &ugrid)
151{
152 size_t num_verts = cell.vertex_ids.size();
153
154 std::vector<vtkIdType> cell_vids(num_verts);
155 for (size_t v=0; v<num_verts; v++)
156 {
157 uint64_t vgi = cell.vertex_ids[v];
158 std::vector<double> d_node(3);
159 d_node[0] = grid->vertices[vgi].x;
160 d_node[1] = grid->vertices[vgi].y;
161 d_node[2] = grid->vertices[vgi].z;
162
163 points->InsertPoint(node_counter,d_node.data());
164 cell_vids[v] = node_counter++;
165 }
166
167 if (cell.Type() == chi_mesh::CellType::SLAB)
168 {
169 ugrid->InsertNextCell(VTK_LINE,
170 static_cast<vtkIdType>(num_verts),
171 cell_vids.data());
172 }
173 if (cell.Type() == chi_mesh::CellType::POLYGON)
174 {
175 ugrid->InsertNextCell(VTK_POLYGON,
176 static_cast<vtkIdType>(num_verts),
177 cell_vids.data());
178 }
179 if (cell.Type() == chi_mesh::CellType::POLYHEDRON)
180 {
181 // Build polyhedron faces
182 std::vector<vtkIdType> faces_vids;
183
184 size_t num_faces = cell.faces.size();
185 for (auto& face : cell.faces)
186 {
187 size_t num_fverts = face.vertex_ids.size();
188 std::vector<vtkIdType> face_info(num_fverts);
189 for (size_t fv=0; fv<num_fverts; fv++)
190 {
191 size_t v = 0;
192 for (size_t cv=0; cv<num_verts; ++cv)
193 if (cell.vertex_ids[cv] == face.vertex_ids[fv])
194 { v = cv; break; }
195
196 face_info[fv] = cell_vids[v];
197 }
198
199 faces_vids.push_back(static_cast<vtkIdType>(num_fverts));
200 for (auto vid : face_info)
201 faces_vids.push_back(vid);
202 }//for f
203
204 ugrid->InsertNextCell(VTK_POLYHEDRON,
205 static_cast<vtkIdType>(num_verts),
206 cell_vids.data(),
207 static_cast<vtkIdType>(num_faces),
208 faces_vids.data());
209 }//polyhedron
210}
