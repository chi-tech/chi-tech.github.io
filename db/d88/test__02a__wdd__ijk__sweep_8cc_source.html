42
43 int n = 0;
44 for (const auto& omega_n : quad.omegas_)
45 {
46 if (Chi::mpi.location_id == 0 and verbose)
47 std::cout << "Sweep angle " << n << " " << omega_n.PrintStr() << std::endl;
48
49 //====================================== Determine sweep ordering
50 if (omega_n.x > 0.0) iorder = chi_math::Range<int>(0, Nx);
51 else iorder = chi_math::Range<int>(Nx-1,-1,-1);
52
53 if (omega_n.y > 0.0) jorder = chi_math::Range<int>(0, Ny);
54 else jorder = chi_math::Range<int>(Ny-1,-1,-1);
55
56 if (omega_n.z > 0.0) korder = chi_math::Range<int>(0, Nz);
57 else korder = chi_math::Range<int>(Nz-1,-1,-1);
58
59 //====================================== Sweep cells
60 IJKArrayDbl psi_ds_x(mesh_divs);
61 IJKArrayDbl psi_ds_y(mesh_divs);
62 IJKArrayDbl psi_ds_z(mesh_divs);
63 for (auto k : korder)
64 for (auto j : jorder)
65 for (auto i : iorder)
66 {
67 double psi_us_x = (omega_n.x > 0.0)? bcs[0] : bcs[1];
68 double psi_us_y = (omega_n.y > 0.0)? bcs[2] : bcs[3];
69 double psi_us_z = (omega_n.z > 0.0)? bcs[4] : bcs[5];
70
71 if (omega_n.x > 0.0 and i > 0 ) psi_us_x = psi_ds_x(i-1,j ,k );
72 if (omega_n.x < 0.0 and i < (Nx-1)) psi_us_x = psi_ds_x(i+1,j ,k );
73
74 if (omega_n.y > 0.0 and j > 0 ) psi_us_y = psi_ds_y(i ,j-1,k );
75 if (omega_n.y < 0.0 and j < (Ny-1)) psi_us_y = psi_ds_y(i ,j+1,k );
76
77 if (omega_n.z > 0.0 and k > 0 ) psi_us_z = psi_ds_z(i ,j ,k-1);
78 if (omega_n.z < 0.0 and k < (Nz-1)) psi_us_z = psi_ds_z(i ,j ,k+1);
79
80 double rhs = q(i,j,k)/(4.0*M_PI);
81 if (Nx > 1) rhs += 2.0*std::fabs(omega_n.x)*psi_us_x/dx;
82 if (Ny > 1) rhs += 2.0*std::fabs(omega_n.y)*psi_us_y/dy;
83 if (Nz > 1) rhs += 2.0*std::fabs(omega_n.z)*psi_us_z/dz;
84
85 double lhs = sigma_t(i,j,k);
86 if (Nx > 1) lhs += 2.0*std::fabs(omega_n.x)/dx;
87 if (Ny > 1) lhs += 2.0*std::fabs(omega_n.y)/dy;
88 if (Nz > 1) lhs += 2.0*std::fabs(omega_n.z)/dz;
89
90 double psi_ijk = rhs/lhs;
91
92 phi_0(i,j,k) += D2M[0][n]*psi_ijk;
93
94 psi_ds_x(i,j,k) = 2.0*psi_ijk - psi_us_x;
95 psi_ds_y(i,j,k) = 2.0*psi_ijk - psi_us_y;
96 psi_ds_z(i,j,k) = 2.0*psi_ijk - psi_us_z;
97 }
98 ++n;
99 }//for omega
100
101 return phi_0;
102}
