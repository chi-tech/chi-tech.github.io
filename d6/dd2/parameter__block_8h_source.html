28/**A ParameterBlock is a conceptually simple data structure that supports
29 * a hierarchy of primitive parameters. There really are just 4 member variables
30 * on a ParameterBlock object, they are 1) the type (as an enum), 2) the
31 * name of the block, 3) a pointer to a value (which can only be a primitive
32 * type), and 4) a vector of child parameters.
33 *
34 * If a ParameterBlock has a primitive type, i.e., BOOLEAN, FLOAT, STRING, or
35 * INTEGER, then the value_ptr will contain a pointer to the value of a
36 * primitive type. Otherwise, for types ARRAY and BLOCK, the ParameterBlock
37 * will not have a value_ptr and instead the vector member will contain
38 * sub-parameters.*/

232 /**Converts the parameters of an array-type parameter block to a vector of
233 * primitive types and returns it.*/
234 template <typename T>
235 std::vector<T> GetVectorValue() const
236 {
237 if (Type() != ParameterBlockType::ARRAY)
238 throw std::logic_error(error_origin_scope_ +
239 std::string(__PRETTY_FUNCTION__) +
240 ": Invalid type requested for parameter of type " +
241 ParameterBlockTypeName(Type()));
242
243 std::vector<T> vec;
244 if (parameters_.empty()) return vec;
245
246 // Check the first sub-param is of the right type
247 const auto& front_param = parameters_.front();
248
249 // Check that all other parameters are of the required type
250 for (const auto& param : parameters_)
251 if (param.Type() != front_param.Type())
252 throw std::logic_error(
253 error_origin_scope_ + " " + std::string(__PRETTY_FUNCTION__) +
254 ": Parameter \"" + name_ +
255 "\", cannot construct vector from block because "
256 "the sub_parameters do not all have the correct type. param->" +
257 ParameterBlockTypeName(param.Type()) + " vs param0->" +
258 ParameterBlockTypeName(front_param.Type()));
259
260 const size_t num_params = parameters_.size();
261 for (size_t k = 0; k < num_params; ++k)
262 {
263 const auto& param = GetParam(k);
264 vec.push_back(param.GetValue<T>());
265 }
266
267 return vec;
268 }
