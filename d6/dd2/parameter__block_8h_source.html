27/**A ParameterBlock is a conceptually simple data structure that supports
28 * a hierarchy of primitive parameters. There really are just 4 member variables
29 * on a ParameterBlock object, they are 1) the type (as an enum), 2) the
30 * name of the block, 3) a pointer to a value (which can only be a primitive
31 * type), and 4) a vector of child parameters.
32 *
33 * If a ParameterBlock has a primitive type, i.e., BOOLEAN, FLOAT, STRING, or
34 * INTEGER, then the value_ptr will contain a pointer to the value of a
35 * primitive type. Otherwise, for types ARRAY and BLOCK, the ParameterBlock
36 * will not have a value_ptr and instead the vector member will contain
37 * sub-parameters.*/

219 /**Converts the parameters of an array-type parameter block to a vector of
220 * primitive types and returns it.*/
221 template <typename T>
222 std::vector<T> GetVectorValue() const
223 {
224 if (Type() != ParameterBlockType::ARRAY)
225 throw std::logic_error(error_origin_scope_ +
226 std::string(__PRETTY_FUNCTION__) +
227 ": Invalid type requested for parameter of type " +
228 ParameterBlockTypeName(Type()));
229
230 std::vector<T> vec;
231 if (parameters_.empty()) return vec;
232
233 // Check the first sub-param is of the right type
234 const auto& front_param = parameters_.front();
235
236 // Check that all other parameters are of the required type
237 for (const auto& param : parameters_)
238 if (param.Type() != front_param.Type())
239 throw std::logic_error(
240 error_origin_scope_ + std::string(__PRETTY_FUNCTION__) +
241 ": Cannot construct vector from block because "
242 "the sub_parameters do not all have the correct type. param->" +
243 ParameterBlockTypeName(param.Type()) + " vs param0->" +
244 ParameterBlockTypeName(front_param.Type()));
245
246 const size_t num_params = parameters_.size();
247 for (size_t k = 0; k < num_params; ++k)
248 {
249 const auto& param = GetParam(k);
250 vec.push_back(param.GetValue<T>());
251 }
252
253 return vec;
254 }
