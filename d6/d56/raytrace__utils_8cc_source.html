4 /**Computes the intersection of a line with a plane.
5  *
6  * The first step of this algorithm is to compute v0 and v1. These
7  * are vectors from the plane's reference point to each of the line-points,
8  * respectively.
9  * We then take the dot-products of these
10  * vectors with the plane normal.
11  * We then say that the vectors have a positive sense if the dot-product
12  * is positive and a negative sense if the dot-product is negative.
13  * If the senses are not equal then the line intersects the plane.
14  *
15  * Since the face normal is a normalized vector the dot-product of v0 or v1
16  * will give the projection of the relevant vector along the normal to the
17  * plane. We can use this projection to compute a weight associated with
18  * each vector. This also then allows us to compute the intersection point.
19 
20  \param plane_normal The normal associated with the plane
21  \param plane_point The reference point for the plane
22  \param line_point_0 The line's initial point
23  \param line_point_1 The line's destination point
24  \param intersection_point The point to be populated with the intersection
25  point
26  \param weights The weights associated with this intersection
27 
28  \return Returns true if the line intersects the plane and false otherwise.
29 
30  \author Jan*/

39  chi_mesh::Vector v0 = line_point_0 - plane_point;
40  chi_mesh::Vector v1 = line_point_1 - plane_point;
41 
42  double dotp_0 = plane_normal.Dot(v0);
43  double dotp_1 = plane_normal.Dot(v1);
44 
45  bool sense_0 = (dotp_0 >= 0.0);
46  bool sense_1 = (dotp_1 >= 0.0);
47 
48  if (sense_0 != sense_1)

50  double dotp_total = std::fabs(dotp_0) + std::fabs(dotp_1);
51  weights.first = (std::fabs(dotp_0)/dotp_total);
52  weights.second = 1.0 - weights.first;
53  intersection_point =
54  line_point_0*weights.second +
55  line_point_1*weights.first;
