10 /**Computes the intersection of a line with a plane.
11  *
12  * The first step of this algorithm is to compute v0 and v1. These
13  * are vectors from the plane's reference point to each of the line-points,
14  * respectively.
15  * We then take the dot-products of these
16  * vectors with the plane normal.
17  * We then say that the vectors have a positive sense if the dot-product
18  * is positive and a negative sense if the dot-product is negative.
19  * If the senses are not equal then the line intersects the plane.
20  *
21  * Since the face normal is a normalized vector the dot-product of v0 or v1
22  * will give the projection of the relevant vector along the normal to the
23  * plane. We can use this projection to compute a weight associated with
24  * each vector. This also then allows us to compute the intersection point.
25 
26  \param plane_normal The normal associated with the plane
27  \param plane_point The reference point for the plane
28  \param line_point_0 The line's initial point
29  \param line_point_1 The line's destination point
30  \param intersection_point The point to be populated with the intersection
31  point
32  \param weights The weights associated with this intersection
33 
34  \return Returns true if the line intersects the plane and false otherwise.
35 
36  \author Jan*/
37 bool chi_mesh::
38 CheckPlaneLineIntersect(const chi_mesh::Normal& plane_normal,
39  const chi_mesh::Vector3& plane_point,
40  const chi_mesh::Vector3& line_point_0,
41  const chi_mesh::Vector3& line_point_1,
42  chi_mesh::Vector3& intersection_point,
43  std::pair<double,double>& weights)
44 {
45  chi_mesh::Vector3 v0 = line_point_0 - plane_point;
46  chi_mesh::Vector3 v1 = line_point_1 - plane_point;
47 
48  double dotp_0 = plane_normal.Dot(v0);
49  double dotp_1 = plane_normal.Dot(v1);
50 
51  bool sense_0 = (dotp_0 >= 0.0);
52  bool sense_1 = (dotp_1 >= 0.0);
53 
54  if (sense_0 != sense_1)
55  {
56  double dotp_total = std::fabs(dotp_0) + std::fabs(dotp_1);
57  weights.first = (std::fabs(dotp_0)/dotp_total);
58  weights.second = 1.0 - weights.first;
59  intersection_point =
60  line_point_0*weights.second +
61  line_point_1*weights.first;
62 
63  return true;
64  }
65 
66  return false;
67 }

120 /**Given a triangle defined by three points, computes whether a line
121  * intersects this triangle.
122  *
123  * */
124 bool
125 chi_mesh::CheckLineIntersectTriangle2(
126  const chi_mesh::Vector3& tri_point0,
127  const chi_mesh::Vector3& tri_point1,
128  const chi_mesh::Vector3& tri_point2,
129  const chi_mesh::Vector3& ray_posi,
130  const chi_mesh::Vector3& ray_dir,
131  chi_mesh::Vector3& intersection_point,
132  double* distance_to_intersection)
133 {
134  double epsilon = 1.0e-12;
135  chi_mesh::Vector3 edge1 = tri_point1 - tri_point0;
136  chi_mesh::Vector3 edge2 = tri_point2 - tri_point0;
137 
138  // Compute characteristic vector for incident angle
139  // This vector becomes perpendicular to the plane
140  // when the ray is parallel to triangle
141  chi_mesh::Vector3 h = ray_dir.Cross(edge2);
142 
143  // If h is indeed perpendicular to the plane,
144  // the dot product of the other leg with this h will be close
145  // to zero.
146  double a = edge1.Dot(h);
147  if (std::fabs(a) < epsilon)
148  return false;
149 
150  chi_mesh::Vector3 s = ray_posi - tri_point0;
151 
152  double f = 1.0/a;
153 
154  // If, s projected onto h, is greater than,
155  // v01 projected onto h, or negative, there is now way
156  // the ray can intersect
157  double u = f*(s.Dot(h));
158  if (u<0.0 or u>1.0)
159  return false;
160 
161  chi_mesh::Vector3 q = s.Cross(edge1);
162 
163  // If, q projected onto omega, is greater than,
164  // v01 projected onto h, or negative, there is now way
165  // the ray can intersect
166  double v = f*ray_dir.Dot(q);
167  if (v<0.0 or (u+v)>1.0)
168  return false;
169 
170  double t = f*edge2.Dot(q);
171 
172  if (distance_to_intersection != nullptr)
173  *distance_to_intersection = t;
174 
175  if (t > epsilon and t<(1.0/epsilon))
176  {
177  intersection_point = ray_posi + ray_dir*t;
178  return true;
179  } else
180  {
181  return false;
182  }
183 }

219 /** This functions checks the intersection of a plane with a tetrahedron.
220  * The equation of a plane is
221  * nx(x-x0) + ny(y-y0) + nz(z-z0) = 0
222  * Where the plane normal is (nx,ny,nz) and the plane point is (x0,y0,z0).
223  * If we form a dot product between the normal and a vector
224  * (x-x0,y-y0,z-z0) then sign of the result gives the sense to the surface.
225  * Therefore, if we encounter differing senses then the plane is indeed
226  * intersecting.*/
227 bool chi_mesh::
228 CheckPlaneTetIntersect(const chi_mesh::Normal& plane_normal,
229  const chi_mesh::Vector3& plane_point,
230  const std::vector<chi_mesh::Vector3>& tet_points)
231 {
232  bool current_sense = false;
233 
234  size_t num_points = tet_points.size();
235  for (size_t i=0; i<num_points; i++)
236  {
237  chi_mesh::Vector3 v = tet_points[i] - plane_point;
238  double dotp = plane_normal.Dot(v);
239 
240  bool new_sense = (dotp >= 0.0);
241 
242  if (i==0)
243  current_sense = new_sense;
244  else if (new_sense != current_sense)
