13 /** Creates actual z-levels for the input layer specification.*/
14 void chi_mesh::VolumeMesherExtruder::SetupLayers(int default_layer_count)
15 {
16  //================================================== Create default layers if no

20  chi_log.Log(LOG_0WARNING)

41  last_z += dz;

43  }
44  }
45  }
46 
47  chi_log.Log(LOG_0)
48  << "VolumeMesherExtruder: Total number of cell layers is "

90 bool chi_mesh::VolumeMesherExtruder::

101  int pid = GetCellKBAPartitionIDFromCentroid(projected_centroid);
102  if (pid == chi_mpi.location_id)
103  return true;
104  }
105 
106  const size_t last_z_level = vertex_layers.size() - 1;
107  const size_t z_level_below = z_level - 1;
108  const size_t z_level_above = z_level + 1;
109 
110  //======================================== Build z-levels to search
111  std::vector<size_t> z_levels_to_search;
112  z_levels_to_search.reserve(3);
113 
114  z_levels_to_search.push_back(z_level);
115  if (z_level != 0) z_levels_to_search.push_back(z_level_below);
116  if (z_level != (last_z_level-1)) z_levels_to_search.push_back(z_level_above);
117 
118  //======================================== Search template cell's
119  // lateral neighbors
120  const auto& vertex_subs =
121  template_unpartitioned_mesh->vertex_cell_subscriptions;
122  for (uint64_t vid : template_cell.vertex_ids)
123  for (uint64_t cid : vertex_subs[vid])
124  {
125  if (cid == template_cell.local_id) continue;
126 
127  auto& candidate_cell =
128  template_continuum.local_cells[cid];
129  auto& cc_centroid = candidate_cell.centroid;
130 
131  for (size_t z : z_levels_to_search)
132  {
133  auto projected_centroid = ProjectCentroidToLevel(cc_centroid, z);
134  int pid = GetCellKBAPartitionIDFromCentroid(projected_centroid);
135  if (pid == chi_mpi.location_id)
136  return true;
137  }
138  }//for cid

141  // longitudinal neighbors
142  for (size_t z : z_levels_to_search)
143  {
144  if (z == z_level) continue;
145 
146  auto projected_centroid = ProjectCentroidToLevel(template_cell.centroid, z);
147  int pid = GetCellKBAPartitionIDFromCentroid(projected_centroid);
