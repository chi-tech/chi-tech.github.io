13/** Creates actual z-levels for the input layer specification.*/
14void chi_mesh::VolumeMesherExtruder::SetupLayers(int default_layer_count)
15{
16 //================================================== Create default layers if no
17 // input layers are provided
18 if (input_layers.empty())
19 {
20 chi::log.Log0Warning()
21 << "VolumeMesherExtruder: No extrusion layers have been specified. "
22 << "A default single layer will be used with height 1.0 and a single "
23 << "subdivision.";
24 double dz = 1.0/default_layer_count;
25 for (int k=0; k<=default_layer_count; k++)
26 {
27 vertex_layers.push_back(k*dz);
28 }
29 }
30 else
31 {
32 double last_z=0.0;
33 vertex_layers.push_back(last_z);
34
35 for (const auto& input_layer : input_layers)
36 {
37 double dz = input_layer.height/input_layer.sub_divisions;
38
39 for (int k=0; k<input_layer.sub_divisions; k++)
40 {
41 last_z += dz;
42 vertex_layers.push_back(last_z);
43 }
44 }
45 }
46
47 chi::log.Log()
48 << "VolumeMesherExtruder: Total number of cell layers is "
49 << vertex_layers.size()-1;
50}

53/**Projects a centroid to an extruded equivalent layer.*/
54chi_mesh::Vector3 chi_mesh::VolumeMesherExtruder::
55 ProjectCentroidToLevel(const chi_mesh::Vector3 &centroid,
56 const size_t level)
57{
58 double z_location = 0.5 * (vertex_layers[level] + vertex_layers[level+1]);
59
60 auto centroid_projected = centroid;
61 centroid_projected.z = z_location;
62
63 return centroid_projected;
64}

67/**Computes a cell's partition id based on a centroid.*/
68int chi_mesh::VolumeMesherExtruder::
69 GetCellKBAPartitionIDFromCentroid(chi_mesh::Vector3& centroid)
70{
71 int px = options.partition_x;
72 int py = options.partition_y;
73
74 chi_mesh::Cell n_gcell(CellType::GHOST, CellType::GHOST);
75 n_gcell.centroid = centroid;
76
77 auto xyz_partition_indices = GetCellXYZPartitionID(&n_gcell);
78
79 int nxi = std::get<0>(xyz_partition_indices);
80 int nyi = std::get<1>(xyz_partition_indices);
81 int nzi = std::get<2>(xyz_partition_indices);
82
83 return nzi*px*py + nyi*px + nxi;
84}

88/**Determines if a template cell is in the current partition or a
89 * direct neighbor to the current partition.*/
90bool chi_mesh::VolumeMesherExtruder::
91 HasLocalScope(
92 const chi_mesh::Cell& template_cell,
93 const chi_mesh::MeshContinuum& template_continuum,
94 size_t z_level)
95{
96 //======================================== Check if the template cell
97 // is in the current partition
98 {
99 auto& centroid = template_cell.centroid;
100 auto projected_centroid = ProjectCentroidToLevel(centroid, z_level);
101 int pid = GetCellKBAPartitionIDFromCentroid(projected_centroid);
102 if (pid == chi::mpi.location_id)
103 return true;
104 }
105
106 const size_t last_z_level = vertex_layers.size() - 1;
107 const size_t z_level_below = z_level - 1;
108 const size_t z_level_above = z_level + 1;
109
110 //======================================== Build z-levels to search
111 std::vector<size_t> z_levels_to_search;
112 z_levels_to_search.reserve(3);
113
114 z_levels_to_search.push_back(z_level);
115 if (z_level != 0) z_levels_to_search.push_back(z_level_below);
116 if (z_level != (last_z_level-1)) z_levels_to_search.push_back(z_level_above);
117
118 //======================================== Search template cell's
119 // lateral neighbors
120 const auto& vertex_subs =
121 template_unpartitioned_mesh->vertex_cell_subscriptions;
122 for (uint64_t vid : template_cell.vertex_ids)
123 for (uint64_t cid : vertex_subs[vid])
124 {
125 if (cid == template_cell.local_id) continue;
126
127 auto& candidate_cell =
128 template_continuum.local_cells[cid];
129 auto& cc_centroid = candidate_cell.centroid;
130
131 for (size_t z : z_levels_to_search)
132 {
133 auto projected_centroid = ProjectCentroidToLevel(cc_centroid, z);
134 int pid = GetCellKBAPartitionIDFromCentroid(projected_centroid);
135 if (pid == chi::mpi.location_id)
136 return true;
137 }
138 }//for cid
139
140 //======================================== Search template cell's
141 // longitudinal neighbors
142 for (size_t z : z_levels_to_search)
143 {
144 if (z == z_level) continue;
145
146 auto projected_centroid = ProjectCentroidToLevel(template_cell.centroid, z);
147 int pid = GetCellKBAPartitionIDFromCentroid(projected_centroid);
148 if (pid == chi::mpi.location_id)
149 return true;
150 }//for z
151
152 return false;
153}

156/**Makes an extruded cell from a template cell.*/
157std::unique_ptr<chi_mesh::Cell> chi_mesh::VolumeMesherExtruder::
158 MakeExtrudedCell(const chi_mesh::Cell &template_cell,
159 const chi_mesh::MeshContinuum &grid,
160 size_t z_level,
161 uint64_t cell_global_id,
162 int partition_id,
163 size_t num_template_cells)
164{
165 const size_t tc_num_verts = template_cell.vertex_ids.size();
166
167 //========================================= Create polyhedron
168 auto cell = std::make_unique<chi_mesh::Cell>(CellType::POLYHEDRON,
169 CellType::POLYHEDRON);
170 cell->global_id = cell_global_id;
171 //cell->local_id set when added to mesh
172 cell->partition_id = partition_id;
173 cell->centroid = ProjectCentroidToLevel(template_cell.centroid, z_level);
174
175 //========================================= Populate cell v-indices
176 cell->vertex_ids.reserve(2*tc_num_verts);
177 for (auto tc_vid : template_cell.vertex_ids)
178 cell->vertex_ids.push_back(tc_vid + z_level*node_z_index_incr);
179
180 for (auto tc_vid : template_cell.vertex_ids)
181 cell->vertex_ids.push_back(tc_vid + (z_level+1)*node_z_index_incr);
182
183
184 //========================================= Create side faces
185 for (auto& face : template_cell.faces)
186 {
187 chi_mesh::CellFace newFace;
188
189 newFace.vertex_ids.resize(4,-1);
190 newFace.vertex_ids[0] = face.vertex_ids[0] + z_level*node_z_index_incr;
191 newFace.vertex_ids[1] = face.vertex_ids[1] + z_level*node_z_index_incr;
192 newFace.vertex_ids[2] = face.vertex_ids[1] + (z_level+1)*node_z_index_incr;
193 newFace.vertex_ids[3] = face.vertex_ids[0] + (z_level+1)*node_z_index_incr;
194
195 //Compute centroid
196 const auto& v0 = grid.vertices[newFace.vertex_ids[0]];
197 const auto& v1 = grid.vertices[newFace.vertex_ids[1]];
198 const auto& v2 = grid.vertices[newFace.vertex_ids[2]];
199 const auto& v3 = grid.vertices[newFace.vertex_ids[3]];
200
201 chi_mesh::Vertex vfc = (v0+v1+v2+v3)/4.0;
202 newFace.centroid = vfc;
203
204 //Compute normal
205 chi_mesh::Vector3 va = v0 - vfc;
206 chi_mesh::Vector3 vb = v1 - vfc;
207
208 chi_mesh::Vector3 vn = va.Cross(vb);
209
210 newFace.normal = (vn/vn.Norm());
211
212 //Set neighbor
213 //The side connections have the same connections as the
214 //template cell + the z_level specifiers of the layer.
215 if (face.has_neighbor)
216 {
217 newFace.neighbor_id = face.neighbor_id +
218 z_level*num_template_cells;
219 newFace.has_neighbor = true;
220 }
221 else
222 newFace.neighbor_id = face.neighbor_id;
223
224 cell->faces.push_back(newFace);
225 } //for side faces
226
227 //========================================= Create top and bottom faces
228 //=============================== Bottom face
229 {
230 chi_mesh::CellFace newFace;
231 //Vertices
232 auto vfc = chi_mesh::Vertex(0.0,0.0,0.0);
233 newFace.vertex_ids.reserve(tc_num_verts);
234 for (int tv=(static_cast<int>(tc_num_verts)-1); tv>=0; tv--)
235 {
236 newFace.vertex_ids.push_back(template_cell.vertex_ids[tv]
237 + z_level*node_z_index_incr);
238 const auto& v = grid.vertices[newFace.vertex_ids.back()];
239 vfc = vfc + v;
240 }
241
242 //Compute centroid
243 vfc = vfc/static_cast<double>(tc_num_verts);
244 newFace.centroid = vfc;
245
246 //Compute normal
247 auto va = grid.vertices[newFace.vertex_ids[0]] - vfc;
248 auto vb = grid.vertices[newFace.vertex_ids[1]] - vfc;
249
250 auto vn = va.Cross(vb);
251 newFace.normal = vn/vn.Norm();
252
253 //Set neighbor
254 if (z_level==0)
255 {
256 newFace.neighbor_id = 0;
257 newFace.has_neighbor = false;
258 }
259 else
260 {
261 newFace.neighbor_id = template_cell.local_id +
262 (z_level-1)*num_template_cells;
263 newFace.has_neighbor = true;
264 }
265
266 cell->faces.push_back(newFace);
267 }
268
269 //=============================== Top face
270 {
271 chi_mesh::CellFace newFace;
272 //Vertices
273 auto vfc = chi_mesh::Vertex(0.0,0.0,0.0);
274 newFace.vertex_ids.reserve(tc_num_verts);
275 for (auto tc_vid : template_cell.vertex_ids)
276 {
277 newFace.vertex_ids.push_back(tc_vid + (z_level+1)*node_z_index_incr);
278 const auto& v = grid.vertices[newFace.vertex_ids.back()];
279 vfc = vfc + v;
280 }
281
282 //Compute centroid
283 vfc = vfc/static_cast<double>(tc_num_verts);
284 newFace.centroid = vfc;
285
286 //Compute normal
287 auto va = grid.vertices[newFace.vertex_ids[0]] - vfc;
288 auto vb = grid.vertices[newFace.vertex_ids[1]] - vfc;
289
290 auto vn = va.Cross(vb);
291 newFace.normal = vn/vn.Norm();
292
293 //Set neighbor
294 if (z_level==(vertex_layers.size()-2))
295 {
296 newFace.neighbor_id = 0;
297 newFace.has_neighbor = false;
298 }
299 else
300 {
301 newFace.neighbor_id = template_cell.local_id +
302 (z_level+1)*num_template_cells;
303 newFace.has_neighbor = true;
304 }
305
306 cell->faces.push_back(newFace);
307 }
308
309 return cell;
