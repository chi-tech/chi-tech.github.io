11/**Makes a source-moments vector from scattering and fission based
12 * on the latest phi-solution.*/
13std::vector<double> lbs::LBSSolver::
14 MakeSourceMomentsFromPhi()
15{
16 size_t num_local_dofs = discretization_->GetNumLocalDOFs(flux_moments_uk_man_);
17
18 std::vector<double> source_moments(num_local_dofs,0.0);
19 for (auto& groupset : groupsets_)
20 {
21 active_set_source_function_(
22 groupset,
23 source_moments,
24 PhiOldLocal(),
25 APPLY_AGS_SCATTER_SOURCES | APPLY_WGS_SCATTER_SOURCES |
26 APPLY_AGS_FISSION_SOURCES | APPLY_WGS_FISSION_SOURCES);
27 }
28
29 return source_moments;
30}

38{

42 //============================================= Open file
43 chi::log.Log() << "Writing flux-moments to files with base-name " << file_base
44 << " and extension .data";

46 std::ofstream::binary | //binary file
47 std::ofstream::out | //no accidental reading
48 std::ofstream::trunc); //clear file contents when opened
49
50 //============================================= Check file is open
51 if (not file.is_open())
52 {
53 chi::log.LogAllWarning()

58 //============================================= Write header
59 std::string header_info =
60 "Chi-Tech LinearBoltzmann: Flux moments file\n"
61 "Header size: 500 bytes\n"
62 "Structure(type-info):\n"
63 "uint64_t num_local_nodes\n"
64 "uint64_t num_moments\n"
65 "uint64_t num_groups\n"
66 "uint64_t num_records\n"
67 "uint64_t num_cells\n"
68 "Each cell:\n"
69 " uint64_t cell_global_id\n"
70 " uint64_t num_nodes\n"
71 " Each node:\n"
72 " double x_position\n"
73 " double y_position\n"
74 " double z_position\n"
75 "Each record:\n"
76 " uint64_t cell_global_id\n"
77 " unsigned int node_number\n"
78 " unsigned int moment_num\n"
79 " unsigned int group_num\n"
80 " double flux_moment_value\n";
81
82 int header_size = (int)header_info.length();
83
84 char header_bytes[500];
85 memset(header_bytes, '-', 500);
86 strncpy(header_bytes, header_info.c_str(),std::min(header_size,499));
87 header_bytes[499]='\0';
88
89 file << header_bytes;
90
91 //============================================= Get relevant items
92 auto NODES_ONLY = chi_math::UnknownManager::GetUnitaryUnknownManager();
93 auto& sdm = discretization_;
94 uint64_t num_local_nodes = discretization_->GetNumLocalDOFs(NODES_ONLY);
95 uint64_t num_moments_t = static_cast<uint64_t>(num_moments_);
96 uint64_t num_groups_t = static_cast<uint64_t>(num_groups_);
97 uint64_t num_local_dofs = discretization_->GetNumLocalDOFs(flux_moments_uk_man_);
98 uint64_t num_local_cells = grid_ptr_->local_cells.size();
99
100 //============================================= Write num_ quantities
101 file.write((char*)&num_local_nodes,sizeof(uint64_t));
102 file.write((char*)&num_moments_t ,sizeof(uint64_t));
103 file.write((char*)&num_groups_t ,sizeof(uint64_t));
104 file.write((char*)&num_local_dofs ,sizeof(uint64_t));
105 file.write((char*)&num_local_cells,sizeof(uint64_t));
106
107 //============================================= Write nodal positions for
108 // each cell
109 for (const auto& cell : grid_ptr_->local_cells)
110 {
111 uint64_t cell_global_id = static_cast<uint64_t>(cell.global_id_);
112 file.write((char *) &cell_global_id, sizeof(uint64_t));
113
114 uint64_t num_nodes = discretization_->GetCellNumNodes(cell);
115 file.write((char *) &num_nodes, sizeof(uint64_t));
116
117 auto node_locations = discretization_->GetCellNodeLocations(cell);
118 for (const auto& node : node_locations)
119 {
120 file.write((char *) &node.x, sizeof(double));
121 file.write((char *) &node.y, sizeof(double));
122 file.write((char *) &node.z, sizeof(double));
123 }//for node
124 }//for cell
125
126 //============================================= Write per dof data
127 for (const auto& cell : grid_ptr_->local_cells)
128 for (unsigned int i=0; i<sdm->GetCellNumNodes(cell); ++i)
129 for (unsigned int m=0; m<num_moments_t; ++m)
130 for (unsigned int g=0; g < num_groups_; ++g)
131 {
132 uint64_t cell_global_id = cell.global_id_;
133 uint64_t dof_map = sdm->MapDOFLocal(cell, i, flux_moments_uk_man_, m, g);
134
135 assert(dof_map < flux_moments.size());
136 double value = flux_moments[dof_map];
137
138 file.write((char*)&cell_global_id,sizeof(uint64_t));
139 file.write((char*)&i ,sizeof(unsigned int));
140 file.write((char*)&m ,sizeof(unsigned int));
141 file.write((char*)&g ,sizeof(unsigned int));
142 file.write((char*)&value ,sizeof(double));
143 }
144
145 //============================================= Clean-up
146 file.close();
147}

154 std::vector<double>& flux_moments,

163 chi::log.Log() << "Reading flux-moments file " << file_name;

165 std::ofstream::binary | //binary file
166 std::ofstream::in); //no accidental writing
167
168 //============================================= Check file is open
169 if (not file.is_open())
170 {
171 chi::log.LogAllWarning()
172 << __FUNCTION__ << "Failed to open " << file_name;
173 return;
174 }
175
176 //============================================= Get relevant items
177 auto NODES_ONLY = chi_math::UnknownManager::GetUnitaryUnknownManager();
178 auto& sdm = discretization_;
179 uint64_t num_local_nodes = discretization_->GetNumLocalDOFs(NODES_ONLY);
180 uint64_t num_moments_t = static_cast<uint64_t>(num_moments_);
181 uint64_t num_groups_t = static_cast<uint64_t>(num_groups_);
182 uint64_t num_local_dofs = discretization_->GetNumLocalDOFs(flux_moments_uk_man_);
183 uint64_t num_local_cells = grid_ptr_->local_cells.size();
184
185 uint64_t file_num_local_nodes;
186 uint64_t file_num_moments ;
187 uint64_t file_num_groups ;
188 uint64_t file_num_local_dofs ;
189 uint64_t file_num_local_cells;

191 flux_moments.assign(num_local_dofs,0.0);
192
193 //============================================= Read header
194 char header_bytes[500]; header_bytes[499] = '\0';
195 file.read(header_bytes,499);
196
197 file.read((char*)&file_num_local_nodes, sizeof(uint64_t));
198 file.read((char*)&file_num_moments , sizeof(uint64_t));
199 file.read((char*)&file_num_groups , sizeof(uint64_t));
200 file.read((char*)&file_num_local_dofs , sizeof(uint64_t));
201 file.read((char*)&file_num_local_cells, sizeof(uint64_t));
202
203 //============================================= Check compatibility
204 if (not single_file)
205 if (file_num_local_nodes != num_local_nodes or
206 file_num_moments != num_moments_t or
207 file_num_groups != num_groups_t or
208 file_num_local_dofs != num_local_dofs or
209 file_num_local_cells != num_local_cells)
210 {
211 std::stringstream outstr;
212 outstr << "num_local_nodes: " << file_num_local_nodes << " vs "
213 << num_local_nodes << "\n";
214 outstr << "num_moments_ : " << file_num_moments << " vs "
215 << num_moments_t << "\n";
216 outstr << "num_groups : " << file_num_groups << " vs "
217 << num_groups_ << "\n";
218 outstr << "num_local_dofs : " << file_num_local_dofs << " vs "
219 << num_local_dofs << "\n";
220 outstr << "num_local_cells: " << file_num_local_cells << " vs "
221 << num_local_cells << "\n";
222 chi::log.LogAll()
223 << "Incompatible DOF data found in file " << file_name << "\n"
224 << "File data vs system:\n" << outstr.str();
225 file.close();
226 return;
227 }
228
229 //============================================= Read cell nodal locations
230 std::map<uint64_t, std::map<uint64_t,uint64_t>> file_cell_nodal_mapping;
231 for (uint64_t c=0; c < file_num_local_cells; ++c)
232 {
233 //============================ Read cell-id and num_nodes
234 uint64_t cell_global_id;
235 uint64_t num_nodes;
236
237 file.read((char*)&cell_global_id, sizeof(uint64_t));
238 file.read((char*)&num_nodes , sizeof(uint64_t));
239
240 //============================ Read node locations
241 std::vector<chi_mesh::Vector3> file_node_locations;
242 file_node_locations.reserve(num_nodes);
243 for (uint64_t n=0; n < num_nodes; ++n)
244 {
245 double x,y,z;
246 file.read((char*)&x, sizeof(double));
247 file.read((char*)&y, sizeof(double));
248 file.read((char*)&z, sizeof(double));
249
250 file_node_locations.emplace_back(x,y,z);
251 }//for file node n
252
253 if (not grid_ptr_->IsCellLocal(cell_global_id)) continue;
254
255 const auto& cell = grid_ptr_->cells[cell_global_id];
256
257 //================ Now map file nodes to system nodes
258 auto system_node_locations = discretization_->GetCellNodeLocations(cell);
259 std::map<uint64_t,uint64_t> mapping;
260
261 //Check num_nodes equal
262 if (system_node_locations.size() != num_nodes)
263 throw std::logic_error(std::string(__FUNCTION__) +
264 ": Incompatible number of nodes for a cell was encountered. Mapping "
265 "could not be performed.");
266
267 bool mapping_successful = true; //Assume true, now try to disprove
268
269 const auto &sys_nodes = system_node_locations;
270 const auto &file_nodes = file_node_locations;
271 size_t num_system_nodes = system_node_locations.size();
272
273 for (uint64_t n = 0; n < num_nodes; ++n)
274 {
275 bool mapping_found = false;
276 for (uint64_t m = 0; m < num_system_nodes; ++m)
277 if ((sys_nodes[m] - file_nodes[n]).NormSquare() < 1.0e-12) {
278 mapping[n] = m; mapping_found = true; }
279
280 if (not mapping_found) {
281 mapping_successful = false; break; }
282 }//for n
283
284 if (not mapping_successful)
285 throw std::logic_error(std::string(__FUNCTION__) +
286 ": Incompatible node locations for a cell was encountered. Mapping "
287 "unsuccessful.");
288
289 file_cell_nodal_mapping[cell_global_id] = std::move(mapping);
290 }//for c (cell in file)
291
292 //============================================= Commit to reading the file
293 for (size_t dof=0; dof < file_num_local_dofs; ++dof)
294 {
295 uint64_t cell_global_id;
296 unsigned int node;
297 unsigned int moment;
298 unsigned int group;
299 double flux_value;
300
301 file.read((char*)&cell_global_id,sizeof(uint64_t));
302 file.read((char*)&node ,sizeof(unsigned int));
303 file.read((char*)&moment ,sizeof(unsigned int));
304 file.read((char*)&group ,sizeof(unsigned int));
305 file.read((char*)&flux_value ,sizeof(double));
306
307 if (grid_ptr_->IsCellLocal(cell_global_id))
308 {
309 const auto& cell = grid_ptr_->cells[cell_global_id];
310 const auto& node_mapping = file_cell_nodal_mapping.at(cell_global_id);
311
312 size_t node_mapped = node_mapping.at(node);
313
314 size_t dof_map = sdm->MapDOFLocal(cell,
315 node_mapped,
316 flux_moments_uk_man_,
317 moment,
318 group);
319
320 assert(dof_map < flux_moments.size());
321 flux_moments[dof_map] = flux_value;
322 }//if cell is local
323 }//for dof
324
325 //============================================= Clean-up
326 file.close();
327}
