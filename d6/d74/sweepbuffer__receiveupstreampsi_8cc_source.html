45 {

48 {
49 int msg_avail = 1;

51 MPI_Iprobe(comm_set->MapIonJ(locJ,chi::mpi.location_id),

54 &msg_avail,MPI_STATUS_IGNORE);

57 {
58 ready_to_execute = false;
59 break;

61
62

70 message_size,
71 MPI_DOUBLE,

73 max_num_mess*angle_set_num + m, //tag
74 comm_set->communicators[chi::mpi.location_id],
75 MPI_STATUS_IGNORE);
76
77 if (error_code != MPI_SUCCESS)

79 std::stringstream err_stream;

91 err_stream << error_string << "\n";
92 MPI_Error_string(error_code, error_string, &length_of_error_string);
93 err_stream << error_string << "\n";
94 chi::log.LogAllWarning() << err_stream.str();
95 }
96 }//if not message already received
97 }//for message
98
99 if (!ready_to_execute) break;
100 }//for predecessor
101
102 if (!ready_to_execute)
103 return AngleSetStatus::RECEIVING;
104 else
105 return AngleSetStatus::READY_TO_EXECUTE;
