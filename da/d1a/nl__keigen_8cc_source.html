84void XXNonLinearKEigen::Execute()
85{
86 if (reinit_phi_1_)
87 lbs_solver_.SetPhiVectorScalarValues(lbs_solver_.PhiOldLocal(), 1.0);
88
89 if (num_free_power_its_ > 0)
90 {
91 double k_eff = 1.0;
92 PowerIterationKEigen(lbs_solver_,
93 nl_solver_.ToleranceOptions().nl_absolute_tol,
94 num_free_power_its_,
95 k_eff);
96 }
97
98 nl_solver_.Setup();
99 nl_solver_.Solve();
100
101 if (lbs_solver_.Options().use_precursors)
102 {
103 lbs_solver_.ComputePrecursors();
104 chi_math::Scale(lbs_solver_.PrecursorsNewLocal(),
105 1.0 / nl_context_->kresid_func_context_.k_eff);
106 }
107
108 lbs_solver_.UpdateFieldFunctions();
109
110 Chi::log.Log()
111 << "LinearBoltzmann::KEigenvalueSolver execution completed\n\n";
112}
