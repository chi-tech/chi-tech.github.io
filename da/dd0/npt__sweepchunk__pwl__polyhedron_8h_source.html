109  void Sweep(chi_mesh::SweepManagement::AngleSet* angle_set)
110  {
111  int outface_master_counter=0;
112 
113  if (!a_and_b_initialized)
114  {
115  Amat.resize(max_cell_dofs,std::vector<double>(max_cell_dofs));
116  Atemp.resize(max_cell_dofs,std::vector<double>(max_cell_dofs));
117  b.resize(G,std::vector<double>(max_cell_dofs,0.0));
118  source.resize(max_cell_dofs,0.0);
119 
120  a_and_b_initialized = true;
121  }
122 
123  chi_mesh::SweepManagement::SPDS* spds = angle_set->GetSPDS();
124  chi_mesh::SweepManagement::FLUDS* fluds = angle_set->fluds;
125 
126  GsSubSet& subset = groupset->grp_subsets[angle_set->ref_subset];
127  int gs_ss_size = groupset->grp_subset_sizes[angle_set->ref_subset];
128 
129  int gs_ss_begin = subset.first;
130  int gs_ss_end = subset.second;
131 
132  //Groupset subset first group number
133  int gs_gi = groupset->groups[gs_ss_begin]->id;
134 
135 
136  int deploc_face_counter = -1;
137  int preloc_face_counter = -1;
138  int bndry_face_counter = -1;
139 
140  double* phi = x->data();
141  double* psi = zero_mg_src.data();
142  double* q_mom = q_moments->data();
143 
144 
145  //========================================================== Loop over each cell
146  size_t num_loc_cells = spds->spls->item_id.size();
147  for (int cr_i=0; cr_i<num_loc_cells; cr_i++)
148  {
149  int cell_g_index = spds->spls->item_id[cr_i];
150  auto cell = grid_view->cells[cell_g_index];
151 
152 
153  chi_mesh::CellPolyhedron* polyh_cell =
154  (chi_mesh::CellPolyhedron*)cell;
155  PolyhedronFEView* cell_fe_view =
156  (PolyhedronFEView*)grid_fe_view->MapFeView(cell_g_index);
157  NPT_CELLVIEW_FULL* transport_view =
158  (NPT_CELLVIEW_FULL*)(*grid_transport_view)[polyh_cell->cell_local_id];
159 
160  int cell_dofs = cell_fe_view->dofs;
161  int xs_id = transport_view->xs_id;
162  double* sigma_tg = (*xsections)[xs_id]->sigma_tg.data();
163 
164 
165  //=================================================== Get Cell matrices
166  const std::vector<std::vector<chi_mesh::Vector>>& L =
167  cell_fe_view->IntV_shapeI_gradshapeJ;
168 
169  const std::vector<std::vector<double>>& M =
170  cell_fe_view->IntV_shapeI_shapeJ;
171 
172  const std::vector<std::vector<std::vector<double>>>& N =
173  cell_fe_view->IntS_shapeI_shapeJ;
174 
175  //=================================================== Loop over angles in set
176  int ni_deploc_face_counter = deploc_face_counter;
177  int ni_preloc_face_counter = preloc_face_counter;
178  int ni_bndry_face_counter = bndry_face_counter;
179  for (int n=0; n<angle_set->angles.size(); n++)
180  {
181  deploc_face_counter = ni_deploc_face_counter;
182  preloc_face_counter = ni_preloc_face_counter;
183  bndry_face_counter = ni_bndry_face_counter;
184 
185  angle_num = angle_set->angles[n];
186  omega = *groupset->quadrature->omegas[angle_num];
187  wn = groupset->quadrature->weights[angle_num];
188 
189  //============================================ Gradient matrix
190  for (int i=0; i<cell_dofs; i++)
191  {
192  for (int j=0; j<cell_dofs; j++)
193  {
194  Amat[i][j] = omega.Dot(L[i][j]);
195  }//for j
196  }//for i
197 
198  for (int gsg=0; gsg<gs_ss_size; gsg++)
199  b[gsg].assign(cell_dofs,0.0);
200 
201 
202  //============================================ Surface integrals
203  int num_faces = polyh_cell->faces.size();
204  int in_face_counter=-1;
205  int internal_face_bndry_counter = -1;
206  for (int f=0; f<num_faces; f++)
207  {
208 
209  double mu = omega.Dot(polyh_cell->faces[f]->geometric_normal);
210  int face_neighbor = transport_view->face_f_adj_part_id[f];
211 
212  //============================= Set flags
213  if (mu>=0.0)
214  transport_view->face_f_upwind_flag[f] = false;
215  else
216  transport_view->face_f_upwind_flag[f] = true;
217 
218  //This counter update-logic is for mapping an incident boundary
219  //condition. Because it is cheap, the cell faces was mapped to a
220  //corresponding boundary during initialization and is
221  //independent of angle. Accessing things like reflective boundary
222  //angular fluxes (and complex boundary conditions), requires the
223  //more general bndry_face_counter.
224  int bndry_map = -1;
225  if (face_neighbor<0)
226  {
227  internal_face_bndry_counter++;
228  bndry_map = transport_view->
229  face_boundary_id[internal_face_bndry_counter];}
230 
231  if (mu < 0.0) //UPWIND
232  {
233  //============================== Increment face counters
234 
235  if (face_neighbor == LOCAL)
236  {in_face_counter++;}
237 
238  if ((face_neighbor != LOCAL) && (face_neighbor>=0))
239  {preloc_face_counter++;}
240 
241  if (face_neighbor<0)
242  {bndry_face_counter++;}
243 
244 
245  //============================== Loop over face vertices
246  int num_face_indices = polyh_cell->faces[f]->v_indices.size();
247  for (int fi=0; fi<num_face_indices; fi++)
248  {
249  int i = cell_fe_view->face_dof_mappings[f]->cell_dof[fi];
250 
251  //=========== Loop over face unknowns
252  for (int fj=0; fj<num_face_indices; fj++)
253  {
254  int j = cell_fe_view->face_dof_mappings[f]->cell_dof[fj];
255 
256  // %%%%% LOCAL CELL DEPENDENCY %%%%%
257  if (face_neighbor == LOCAL)
258  {psi = fluds->UpwindPsi(cr_i,in_face_counter,fj,0,n);}
259  // %%%%% NON-LOCAL CELL DEPENDENCY %%%%%
260  else if (face_neighbor >= 0)
261  {psi = fluds->NLUpwindPsi(preloc_face_counter,fj,0,n);}
262  // %%%%% BOUNDARY CELL DEPENDENCY %%%%%
263  else if (!suppress_surface_src)
264  {psi = angle_set->PsiBndry(bndry_face_counter,
265  bndry_map,fj,gs_gi,n);
266  //if (gs_gi>9) psi = zero_mg_src.data();
267  }
268  else
269  {psi = zero_mg_src.data();}
270 
271  double mu_Nij = -mu*N[f][i][j];
272 
273  Amat[i][j] += mu_Nij;
274 
275  for (int gsg=0; gsg<gs_ss_size; gsg++)
276  b[gsg][i] += psi[gsg]*mu_Nij;
277  }
278  };
279 
280  }//if mu<0.0
281 
282  }//for f
283 
284  //========================================== Looping over groups
285  double sigma_tgr = 0.0;
286  double temp_src = 0.0;
287  int gi_deploc_face_counter = deploc_face_counter;
288  int gi_preloc_face_counter = preloc_face_counter;
289  for (int gsg=0; gsg<gs_ss_size; gsg++)
290  {
291  deploc_face_counter = gi_deploc_face_counter;
292  preloc_face_counter = gi_preloc_face_counter;
293 
294  g = gs_gi+gsg;
295 
296  //============================= Contribute source moments
297  double m2d = 0.0;
298  for (int i=0; i<cell_fe_view->dofs; i++)
299  {
300  temp_src = 0.0;
301  for (int m=0; m<num_moms; m++)
302  {
303  m2d = groupset->m2d_op[m][angle_num];
304  int ir = transport_view->MapDOF(i,m,g);
305  temp_src += m2d*q_mom[ir];
306  }
307  source[i] = temp_src;
308  }
309 
310  //============================= Mass Matrix and Source
311  sigma_tgr = sigma_tg[g];
312  for (int i=0; i<cell_fe_view->dofs; i++)
313  {
314  double temp = 0.0;
315  for (int j=0; j<cell_fe_view->dofs; j++)
316  {
317  double Mij = M[i][j];
318  Atemp[i][j] = Amat[i][j] + Mij*sigma_tgr;
319  temp += Mij*source[j];
320  }//for j
321  b[gsg][i] += temp;
322  }//for i
323 
324 
325  //============================= Solve system
326  chi_math_handler.GaussElimination(Atemp,b[gsg],cell_fe_view->dofs);
327 
328 
329  }//for g
330 
331 
332 
333 
334  //============================= Accumulate flux
335  double wn_d2m = 0.0;
336  for (int m=0; m<num_moms; m++)
337  {
338  wn_d2m = groupset->d2m_op[m][angle_num];
339  for (int i=0; i<cell_fe_view->dofs; i++)
340  {
341  int ir = transport_view->MapDOF(i,m,gs_gi);
342  for (int gsg=0; gsg<gs_ss_size; gsg++)
343  {
344  phi[ir+gsg] += wn_d2m*b[gsg][i];
345  }
346  }
347  }
348 
349  //============================================= Outgoing fluxes
350  int out_face_counter=-1;
351  for (int f=0; f<polyh_cell->faces.size(); f++)
352  {
353  int face_neighbor = transport_view->face_f_adj_part_id[f];
354  bool face_incident = transport_view->face_f_upwind_flag[f];
355  double* psi = zero_mg_src.data();
356 
357  //============================= Set flags and counters
358  if (face_incident) continue;
359  else out_face_counter++;
360 
361  //============================= Store outgoing Psi Locally
362  if ((face_neighbor == LOCAL) && (!face_incident))
363  {
364  for (int fi=0; fi<polyh_cell->faces[f]->v_indices.size(); fi++)
365  {
366  int i = cell_fe_view->face_dof_mappings[f]->cell_dof[fi];
367  psi = fluds->OutgoingPsi(cr_i,out_face_counter,fi,n);
368 
369  for (int gsg=0; gsg<gs_ss_size; gsg++)
370  psi[gsg] = b[gsg][i];
371  }
372  }//
373  //============================= Store outgoing Psi Non-Locally
374  else if ((face_neighbor != LOCAL) &&
375  (face_neighbor >= 0) &&
376  (!face_incident))
377  {
378  deploc_face_counter++;
379  for (int fi=0; fi<polyh_cell->faces[f]->v_indices.size(); fi++)
380  {
381  int i = cell_fe_view->face_dof_mappings[f]->cell_dof[fi];
382  psi = fluds->NLOutgoingPsi(deploc_face_counter,fi,n);
383 
384  for (int gsg=0; gsg<gs_ss_size; gsg++)
385  psi[gsg] = b[gsg][i];
386  }//for fdof
387  }//if non-local
388  }//for f
389 
390 
391  }//for n
392 
393  }// for cell
394 
395  }//Sweep function
