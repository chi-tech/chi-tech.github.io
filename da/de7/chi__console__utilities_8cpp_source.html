15/** Executes the given file in the Lua engine.
16\author Jan*/
17int chi_objects::ChiConsole::ExecuteFile(const std::string& fileName,
18 int argc,
19 char** argv) const
20{
21 lua_State* L = this->console_state_;
22 if (not fileName.empty())
23 {
24 if (argc > 0)
25 {
26 lua_newtable(L);
27 for (int i = 1; i <= argc; i++)
28 {
29 lua_pushnumber(L, i);
30 lua_pushstring(L, argv[i - 1]);
31 lua_settable(L, -3);
32 }
33 lua_setglobal(L, "chiArgs");
34 }
35 int error = luaL_dofile(this->console_state_, fileName.c_str());
36
37 if (error > 0)
38 {
39 chi::log.LogAllError()
40 << "LuaError: " << lua_tostring(this->console_state_, -1);
41 return EXIT_FAILURE;
42 }
43 }
44 return EXIT_SUCCESS;
45}

62/**Basic addition to registry. Used by the other public methods
63 * to registry a text-key to a lua function.*/
64void chi_objects::ChiConsole::AddFunctionToRegistry(
65 const std::string& name_in_lua, lua_CFunction function_ptr)
66{
67 auto& console = GetInstance();
68
69 // Check if the function name is already there
70 if (console.lua_function_registry_.count(name_in_lua) > 0)
71 {
72 const auto& current_entry = console.lua_function_registry_.at(name_in_lua);
73
74 throw std::logic_error(std::string(__PRETTY_FUNCTION__) +
75 ": Attempted "
76 "to register lua function \"" +
77 name_in_lua +
78 "\" but the function "
79 "is already taken by " +
80 current_entry.function_raw_name);
81 }
82
83 console.lua_function_registry_.insert(std::make_pair(
84 name_in_lua, LuaFunctionRegistryEntry{function_ptr, name_in_lua}));
85}

130/**Sets/Forms a lua function in the state using a namespace structure.*/
131void chi_objects::ChiConsole::SetLuaFuncNamespaceTableStructure(
132 const std::string& full_lua_name, lua_CFunction function_ptr)
133{
134 auto L = GetInstance().console_state_;
135 const auto lua_name_split = chi_misc_utils::StringSplit(full_lua_name, "::");
136
137 if (lua_name_split.size() == 1)
138 {
139 lua_pushcfunction(L, function_ptr);
140 lua_setglobal(L, lua_name_split.back().c_str());
141 return;
142 }
143
144 const std::vector<std::string> table_names(lua_name_split.begin(),
145 lua_name_split.end() - 1);
146
147 FleshOutLuaTableStructure(table_names);
148
149 lua_pushstring(L, lua_name_split.back().c_str());
150 lua_pushcfunction(L, function_ptr);
151 lua_settable(L, -3);
152
153 lua_pop(L, lua_gettop(L));
154}

191// ##################################################################
192/**Fleshes out a path in a table tree. For example, given
193 * "fee::foo::fah::koo, this routine will make sure that
194 * fee.foo.fah.koo is defined as a table tree structure. The routine will
195 * create a table structure where one is needed and leave existing ones alone.
196 *
197 * At the end of the routine the last table in the structure will be on top
198 * of the stack.*/
199void chi_objects::ChiConsole::FleshOutLuaTableStructure(
200 const std::vector<std::string>& table_names)
201{
202 auto L = GetInstance().console_state_;
203
204 for (const auto& table_name : table_names)
205 {
206 // The first entry needs to be in lua's global scope,
207 // so it looks a little different
208 if (table_name == table_names.front())
209 {
210 lua_getglobal(L, table_name.c_str());
211 if (not lua_istable(L, -1))
212 {
213 std::cout << "Creating base table " << table_name << "\n";
214 lua_pop(L, 1);
215 lua_newtable(L);
216 lua_setglobal(L, table_name.c_str());
217 lua_getglobal(L, table_name.c_str());
218 }
219 }
220 else
221 {
222 lua_getfield(L, -1, table_name.c_str());
223 if (not lua_istable(L, -1))
224 {
225 std::cout << "Creating sub-table " << table_name << "\n";
226 lua_pop(L, 1);
227 lua_pushstring(L, table_name.c_str());
228 lua_newtable(L);
229 lua_settable(L, -3);
230 lua_getfield(L, -1, table_name.c_str());
231 }
232 }
233 } // for table_key in table_keys
234}

237/**Assumes a table is on top of the stack, then loads the table
238 * with chunks that call registered methods of the class with the first
239 * argument being the object's handle.*/
240void chi_objects::ChiConsole::SetObjectMethodsToTable(
241 const std::string& class_name, size_t handle)
242{
243 auto& console = GetInstance();
244 auto L = console.console_state_;
245
246 const auto& class_method_registry = console.class_method_registry_;
247 if (class_method_registry.count(class_name) == 0) return;
248
249 const auto& method_list = class_method_registry.at(class_name);
250 for (const auto& method_name : method_list)
251 {
252 std::string function_name = class_name;
253 function_name += "::";
254 function_name += method_name;
255
256 const auto path_names = chi_misc_utils::StringSplit(function_name, "::");
257 std::string path_dot_name;
258 for (const auto& name : path_names)
259 {
260 path_dot_name += name;
261 if (name != path_names.back()) path_dot_name += ".";
262 }
263
264 std::string chunk_code = "local params = ...; ";
265 chunk_code += "return " + path_dot_name + "(" + std::to_string(handle) +
266 ", ...)";
267
268 lua_pushstring(L, method_name.c_str());
269 luaL_loadstring(L, chunk_code.c_str());
270 lua_settable(L, -3);
271 }
