289/**Print the block tree structure into a designated string.*/
290void ParameterBlock::RecursiveDumpToString(std::string& outstr,
291 const std::string& offset) const
292{
293 outstr += offset + this->Name() + " = \n";
294 outstr += offset + "{\n";
295
296 for (const auto& param : parameters_)
297 {
298
299 switch (param.Type())
300 {
301 case ParameterBlockType::BOOLEAN:
302 {
303 outstr += offset + " " + param.Name() + " = ";
304 const bool value = param.Value().BoolValue();
305 outstr += std::string(value ? "true" : "false") + ",\n";
306 break;
307 }
308 case ParameterBlockType::FLOAT:
309 {
310 outstr += offset + " " + param.Name() + " = ";
311 const double value = param.Value().FloatValue();
312 outstr += std::to_string(value) + ",\n";
313 break;
314 }
315 case ParameterBlockType::STRING:
316 {
317 outstr += offset + " " + param.Name() + " = ";
318 const auto& value = param.Value().StringValue();
319 outstr += "\"" + value + "\",\n";
320 break;
321 }
322 case ParameterBlockType::INTEGER:
323 {
324 outstr += offset + " " + param.Name() + " = ";
325 const int64_t value = param.Value().IntegerValue();
326 outstr += std::to_string(value) + ",\n";
327 break;
328 }
329 case ParameterBlockType::ARRAY:
330 case ParameterBlockType::BLOCK:
331 {
332 param.RecursiveDumpToString(outstr, offset + " ");
333 break;
334 }
335 default:
336 break;
337 }
338 } // for parameter
339
340 outstr += offset + "}\n";
341}
