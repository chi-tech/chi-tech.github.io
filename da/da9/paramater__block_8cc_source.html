130/**Changes the block type to array, making it accessible via integer
131 * keys.*/
132void ParameterBlock::ChangeToArray()
133{
134 const std::string fname = __PRETTY_FUNCTION__;
135 if (parameters_.empty())
136 {
137 type_ = ParameterBlockType::ARRAY;
138 return;
139 }
140
141 const auto& first_param = parameters_.front();
142 for (const auto& param : parameters_)
143 if (param.Type() != first_param.Type())
144 throw std::logic_error(
145 error_origin_scope_ + fname +
146 ": Cannot change ParameterBlock to "
147 "array. It has existing parameters and they are not of the same"
148 "type.");
149
150 type_ = ParameterBlockType::ARRAY;
151}

206void ParameterBlock::SortParameters()

291/**Print the block tree structure into a designated string.*/
292void ParameterBlock::RecursiveDumpToString(std::string& outstr,
293 const std::string& offset) const
294{
295 outstr += offset + this->Name() + " = \n";
296 outstr += offset + "{\n";
297
298 for (const auto& param : parameters_)
299 {
300
301 switch (param.Type())
302 {
303 case ParameterBlockType::BOOLEAN:
304 {
305 outstr += offset + " " + param.Name() + " = ";
306 const bool value = param.Value().BoolValue();
307 outstr += std::string(value ? "true" : "false") + ",\n";
308 break;
309 }
310 case ParameterBlockType::FLOAT:
311 {
312 outstr += offset + " " + param.Name() + " = ";
313 const double value = param.Value().FloatValue();
314 outstr += std::to_string(value) + ",\n";
315 break;
316 }
317 case ParameterBlockType::STRING:
318 {
319 outstr += offset + " " + param.Name() + " = ";
320 const auto& value = param.Value().StringValue();
321 outstr += "\"" + value + "\",\n";
322 break;
323 }
324 case ParameterBlockType::INTEGER:
325 {
326 outstr += offset + " " + param.Name() + " = ";
327 const int64_t value = param.Value().IntegerValue();
328 outstr += std::to_string(value) + ",\n";
329 break;
330 }
331 case ParameterBlockType::ARRAY:
332 case ParameterBlockType::BLOCK:
333 {
334 param.RecursiveDumpToString(outstr, offset + " ");
335 break;
336 }
337 default:
338 break;
339 }
340 } // for parameter
341
342 outstr += offset + "}\n";
343}
