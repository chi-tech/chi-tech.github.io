154 /**Checks if data has been initialized and initializes it if not.*/
155 void chi_mesh::SweepManagement::SweepBuffer::
156  InitializeBuffers()
157 {
158  if (!data_initialized)
159  {
160  chi_mesh::SweepManagement::SPDS* spds = angleset->GetSPDS();
161  chi_mesh::SweepManagement::FLUDS* fluds= angleset->fluds;
162 
163  int num_grps = angleset->GetNumGrps();
164  int num_angles = angleset->angles.size();
165 
166  //============================ Resize FLUDS local outgoing Data
167  angleset->local_psi.resize(fluds->local_psi_stride*
168  fluds->local_psi_max_elements*
169  num_grps*num_angles,0.0);
170 
171  //============================ Resize FLUDS non-local outgoing Data
172  angleset->deplocI_outgoing_psi.resize(
173  spds->location_successors.size(),std::vector<double>());
174  for (int deplocI=0; deplocI<spds->location_successors.size(); deplocI++)
175  {
176  angleset->deplocI_outgoing_psi[deplocI].resize(
177  fluds->deplocI_face_dof_count[deplocI]*num_grps*num_angles,0.0);
178  }
179 
180  //============================ Set incident boundaries size
181  std::vector<std::pair<int,int>>& bndry_types = *angleset->ref_boundary_types;
182  angleset->boundryI_incoming_psi.resize(6,std::vector<double>(1,0.0));
183  for (int refb=0; refb<fluds->boundary_dependencies.size(); refb++)
184  {
185  int b = fluds->boundary_dependencies[refb];
186  if (bndry_types[b].first == INCIDENT_ISOTROPIC)
187  {
188  int index_incid_iso = bndry_types[b].second;
189 
190  angleset->boundryI_incoming_psi[b].clear();
191 
192  std::copy(
193  (*angleset->ref_incident_P0_mg_boundaries)[index_incid_iso].begin(),
194  (*angleset->ref_incident_P0_mg_boundaries)[index_incid_iso].end(),
195  std::back_inserter(angleset->boundryI_incoming_psi[b]));
196  }
197  else
198  {
199  angleset->boundryI_incoming_psi[b].clear();
200 
201  std::copy(
202  (*angleset->ref_incident_P0_mg_boundaries).back().begin(),
203  (*angleset->ref_incident_P0_mg_boundaries).back().end(),
204  std::back_inserter(angleset->boundryI_incoming_psi[b]));
205  }
206  }
207 
208  //================================================ Make a memory query
209  double memory_mb = chi_console.GetMemoryUsageInMB();
210 
211  if (memory_mb>chi_global_timings[9])
212  chi_global_timings[9] = memory_mb;
213 
214  chi_global_timings[10] += memory_mb;
215  chi_global_timings[11] += 1.0;

218

350  MPI_STATUS_IGNORE);
351  }
352  }//if not message already received
353  }//for message
354 
355  if (!ready_to_execute) break;
356  }//for predecessor
