49 //----------------------------------------------------------------------------
50 int vtkCleanUnstructuredGrid::RequestData(vtkInformation* vtkNotUsed(request),
51  vtkInformationVector** inputVector, vtkInformationVector* outputVector)
52 {
53  vtkInformation* inInfo = inputVector[0]->GetInformationObject(0);
54  vtkInformation* outInfo = outputVector->GetInformationObject(0);
55 
56  vtkDataSet* input = vtkDataSet::SafeDownCast(inInfo->Get(vtkDataObject::DATA_OBJECT()));
57  vtkUnstructuredGrid* output =
58  vtkUnstructuredGrid::SafeDownCast(outInfo->Get(vtkDataObject::DATA_OBJECT()));
59 
60  if (input->GetNumberOfCells() == 0)
61  {
62  // set up a ugrid with same data arrays as input, but
63  // no points, cells or data.
64  output->Allocate(1);
65  output->GetPointData()->CopyAllocate(input->GetPointData(), VTK_CELL_SIZE);
66  output->GetCellData()->CopyAllocate(input->GetCellData(), 1);
67  vtkPoints* pts = vtkPoints::New();
68  output->SetPoints(pts);
69  pts->Delete();
70  return 1;
71  }
72 
73  output->GetPointData()->CopyAllocate(input->GetPointData());
74  output->GetCellData()->PassData(input->GetCellData());
75 
76  // First, create a new points array that eliminate duplicate points.
77  // Also create a mapping from the old point id to the new.
78  vtkPoints* newPts = vtkPoints::New();
79 
80  // Set the desired precision for the points in the output.
81  if (this->OutputPointsPrecision == vtkAlgorithm::DEFAULT_PRECISION)
82  {
83  // The logical behaviour would be to use the data type from the input.
84  // However, input is a vtkDataSet, which has no point data type; only the
85  // derived class vtkPointSet has a vtkPoints attribute, so only for that
86  // the logical practice can be applied, while for others (currently
87  // vtkImageData and vtkRectilinearGrid) the data type is the default
88  // for vtkPoints - which is VTK_FLOAT.
89  vtkPointSet* ps = vtkPointSet::SafeDownCast(input);
90  if (ps)
91  {
92  newPts->SetDataType(ps->GetPoints()->GetDataType());
93  }
94  }
95  else if (this->OutputPointsPrecision == vtkAlgorithm::SINGLE_PRECISION)
96  {
97  newPts->SetDataType(VTK_FLOAT);
98  }
99  else if (this->OutputPointsPrecision == vtkAlgorithm::DOUBLE_PRECISION)
100  {
101  newPts->SetDataType(VTK_DOUBLE);
102  }
103 
104  vtkIdType num = input->GetNumberOfPoints();
105  vtkIdType id;
106  vtkIdType newId;
107  vtkIdType* ptMap = new vtkIdType[num];
108  double pt[3];
109 
110  this->CreateDefaultLocator(input);
111  if (this->ToleranceIsAbsolute)
112  {
113  this->Locator->SetTolerance(this->AbsoluteTolerance);
114  }
115  else
116  {
117  this->Locator->SetTolerance(this->Tolerance * input->GetLength());
118  }
119  double bounds[6];
120  input->GetBounds(bounds);
121  this->Locator->InitPointInsertion(newPts, bounds);
122 
123  vtkIdType progressStep = num / 100;
124  if (progressStep == 0)
125  {
126  progressStep = 1;
127  }
128  for (id = 0; id < num; ++id)
129  {
130  if (id % progressStep == 0)
131  {
132  this->UpdateProgress(0.8 * ((float)id / num));
133  }
134  input->GetPoint(id, pt);
135  if (this->Locator->InsertUniquePoint(pt, newId))
136  {
137  output->GetPointData()->CopyData(input->GetPointData(), id, newId);
138  }
139  ptMap[id] = newId;
140  }
141  output->SetPoints(newPts);
142  newPts->Delete();
143 
144  // Now copy the cells.
145  vtkIdList* cellPoints = vtkIdList::New();
146  num = input->GetNumberOfCells();
147  output->Allocate(num);
148  for (id = 0; id < num; ++id)
149  {
150  if (id % progressStep == 0)
151  {
152  this->UpdateProgress(0.8 + 0.2 * ((float)id / num));
153  }
154  // special handling for polyhedron cells
155  if (vtkUnstructuredGrid::SafeDownCast(input) && input->GetCellType(id) == VTK_POLYHEDRON)
156  {
157  vtkUnstructuredGrid::SafeDownCast(input)->GetFaceStream(id, cellPoints);
158  vtkUnstructuredGrid::ConvertFaceStreamPointIds(cellPoints, ptMap);
159  }
160  else
161  {
162  input->GetCellPoints(id, cellPoints);
163  for (int i = 0; i < cellPoints->GetNumberOfIds(); i++)
164  {
165  int cellPtId = cellPoints->GetId(i);
166  newId = ptMap[cellPtId];
167  cellPoints->SetId(i, newId);
168  }
169  }
170  output->InsertNextCell(input->GetCellType(id), cellPoints);
171  }
172 
173  delete[] ptMap;
174  cellPoints->Delete();
175  output->Squeeze();
176 
177  return 1;
178 }

187 //----------------------------------------------------------------------------
188 void vtkCleanUnstructuredGrid::CreateDefaultLocator(vtkDataSet* input)
189 {
190  double tol;
191  if (this->ToleranceIsAbsolute)
192  {
193  tol = this->AbsoluteTolerance;
194  }
195  else
196  {
197  if (input)
198  {
199  tol = this->Tolerance * input->GetLength();
200  }
201  else
202  {
203  tol = this->Tolerance;
204  }
205  }
206 
207  if (this->Locator == nullptr)
208  {
209  if (tol == 0.0)
210  {
211  this->Locator = vtkMergePoints::New();
212  this->Locator->Register(this);
213  this->Locator->Delete();
214  }
215  else
216  {
217  this->Locator = vtkPointLocator::New();
218  this->Locator->Register(this);
219  this->Locator->Delete();
220  }
221  }
222  else
223  {
224  // check that the tolerance wasn't changed from zero to non-zero
225  if ((tol > 0.0) && (this->GetLocator()->GetTolerance() == 0.0))
226  {
227  this->SetLocator(nullptr);
228  this->Locator = vtkPointLocator::New();
229  this->Locator->Register(this);
230  this->Locator->Delete();
231  }
232  }
