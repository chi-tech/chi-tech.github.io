255 void chi_mesh::UnpartitionedMesh::BuildMeshConnectivity()
256 {
257  //======================================== Populate vertex
258  // subscriptionns
259  std::vector<std::set<size_t>> vertex_subs(vertices.size());
260  uint64_t cur_cell_id=0;
261  for (auto& cell : raw_cells)
262  {
263  for (auto vid : cell->vertex_ids)
264  vertex_subs[vid].insert(cur_cell_id);
265  ++cur_cell_id;
266  }
267 
268  int num_bndry_faces = 0;
269  for (auto cell : raw_cells)
270  for (auto& face : cell->faces)
271  {
272  if (face.neighbor < 0) ++num_bndry_faces;
273  face.neighbor = -1;
274  }
275 
276  chi_log.Log(LOG_0VERBOSE_1) << chi_program_timer.GetTimeString()
277  << " Number of boundary faces "
278  "before connectivity: " << num_bndry_faces;
279 
280  //======================================== Establish connectivity
281  chi_log.Log() << "Establishing cell connectivity.";
282  std::set<size_t> cells_to_search;
283  cur_cell_id=0;
284  for (auto& cell : raw_cells)
285  {
286  cells_to_search.clear();
287  for (uint64_t vid : cell->vertex_ids)
288  for (uint64_t cell_id : vertex_subs[vid])
289  if (cell_id != cur_cell_id)
290  cells_to_search.insert(cell_id);
291 
292  for (auto& cur_cell_face : cell->faces)
293  {
294  if (cur_cell_face.neighbor >= 0 ) continue;
295 
296  std::set<uint64_t> cfvids(cur_cell_face.vertex_ids.begin(),
297  cur_cell_face.vertex_ids.end());
298 
299  for (uint64_t adj_cell_id : cells_to_search)
300  {
301  auto adj_cell = raw_cells[adj_cell_id];
302 
303  for (auto& adj_cell_face : adj_cell->faces)
304  {
305  if (adj_cell_face.neighbor >= 0) continue;
306  std::set<uint64_t> afvids(adj_cell_face.vertex_ids.begin(),
307  adj_cell_face.vertex_ids.end());
308 
309  if (cfvids == afvids)
310  {
311  cur_cell_face.neighbor = adj_cell_id;
312  adj_cell_face.neighbor = cur_cell_id;
313  goto face_neighbor_found;
314  }
315  }//for adjacent cell face
316  }
317  face_neighbor_found:;
318  }//for face
319 
320  ++cur_cell_id;
321  }//for cell
322 
323  chi_log.Log() << "Done establishing cell connectivity.";
324 
325  num_bndry_faces = 0;
326  for (auto cell : raw_cells)
327  for (auto& face : cell->faces)
328  if (face.neighbor < 0) ++num_bndry_faces;
329 
330  chi_log.Log(LOG_0VERBOSE_1) << chi_program_timer.GetTimeString()
331  << " Number of boundary faces "
