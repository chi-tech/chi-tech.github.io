15/**Populates a face histogram.
16 *
17 * \param master_tolerance Multiple histograms will only be attempted
18 * if the ratio of the maximum dofs-per-face to the average dofs-per-face
19 * is greater than this value. Default 1.2.
20 *
21 * \param slave_tolerance While traversing a sorted list of dofs-per-face,
22 * a new bin will only be generated when the ratio of the listed dofs-per-face
23 * to a running bin average exceeds this value. Defualt 1.1.
24 *
25 * The function populates face_categories which is a structure containing
26 * pairs. Pair.first is the max dofs-per-face for the category and Pair.second
27 * is the number of faces in this category.
28 *
29 * */
30void chi_mesh::MeshContinuum::
31 BuildFaceHistogramInfo(double master_tolerance, double slave_tolerance)
32{
33 if (face_histogram_available) return;
34
35 //================================================== Fill histogram
36 std::vector<size_t> face_size_histogram;
37 for (const auto& cell : local_cells)
38 for (const auto& face : cell.faces)
39 face_size_histogram.push_back(face.vertex_ids.size());
40
41 std::stable_sort(face_size_histogram.begin(), face_size_histogram.end());
42
43 //================================================== Determine total face dofs
44 size_t total_face_dofs_count = 0;
45 for (auto face_size : face_size_histogram)
46 total_face_dofs_count += face_size;
47
48 //================================================== Compute average and ratio
49 size_t smallest_face = face_size_histogram.front();
50 size_t largest_face = face_size_histogram.back();
51 size_t total_num_faces = face_size_histogram.size();
52 double average_dofs_per_face = (double)total_face_dofs_count/total_num_faces;
53
54 std::stringstream outstr;
55 outstr << "\nSmallest face = " << smallest_face;
56 outstr << "\nLargest face = " << largest_face;
57 outstr << "\nTotal face dofs = " << total_face_dofs_count;
58 outstr << "\nTotal faces = " << face_size_histogram.size();
59 outstr << "\nAverage dofs/face = " << average_dofs_per_face;
60 outstr << "\nMax to avg ratio = " << largest_face/average_dofs_per_face;
61 chi::log.LogAllVerbose1() << outstr.str();
62
63 //================================================== Determine number of bins
64 size_t last_bin_num_faces = total_num_faces;
65 if ((largest_face/average_dofs_per_face) > master_tolerance)
66 {
67 chi::log.LogAllVerbose1()
68 << "The ratio of max face dofs to average face dofs "
69 << "is larger than " << master_tolerance
70 << ", therefore a binned histogram "
71 << "will be constructed.";
72
73 //====================================== Build categories
74 size_t running_total_face_dofs = 0;
75 size_t running_face_count = 0;
76 size_t running_face_size = face_size_histogram[0];
77
78 double running_average = face_size_histogram[0];
79
80 for (size_t f=0; f<total_num_faces; ++f)
81 {
82 if ((face_size_histogram[f]/running_average) > slave_tolerance)
83 {
84 face_categories.emplace_back(running_face_size,running_face_count);
85 running_total_face_dofs = 0;
86 running_face_count = 0;
87 }
88
89 running_face_size = face_size_histogram[f];
90 running_total_face_dofs += face_size_histogram[f];
91 running_face_count++;
92 running_average = (double)running_total_face_dofs/double(running_face_count);
93 last_bin_num_faces = running_face_count;
94 }
95 }
96 face_categories.emplace_back(largest_face,last_bin_num_faces);
97
98 //================================================== Verbose print bins
99 outstr.str(std::string());
100 outstr
101 << "A total of " << face_categories.size()
102 << " bins were created:\n";
103
104 size_t bin_counter = -1;
105 for (auto bins : face_categories)
106 {
107 outstr
108 << "Bin " << ++bin_counter << ": "
109 << bins.second << " faces with max face dofs " << bins.first << "\n";
110 }
111
112 chi::log.LogAllVerbose1() << outstr.str();
113
114 face_histogram_available = true;
115}

199void chi_mesh::MeshContinuum::
200FindAssociatedVertices(chi_mesh::CellFace& cur_face,
201 std::vector<short>& dof_mapping) const
202{
203 int associated_face = cur_face.GetNeighborAssociatedFace(*this);
204 //======================================== Check index validity
205 if ((not cur_face.has_neighbor) || (not cur_face.IsNeighborLocal(*this)))
206 {
207 chi::log.LogAllError()
208 << "Invalid cell index encountered in call to "
209 << "MeshContinuum::FindAssociatedVertices. Index points to either a boundary"
210 << "or a non-local cell.";
211 chi::Exit(EXIT_FAILURE);
212 }
213
214 auto& adj_cell = local_cells[cur_face.GetNeighborLocalID(*this)];
215
216 dof_mapping.reserve(cur_face.vertex_ids.size());
217
218 const auto& adj_face = adj_cell.faces[associated_face];
219
220 for (auto cfvid : cur_face.vertex_ids)
221 {
222 bool found = false;
223 short afv = 0;
224 for (auto afvid : adj_face.vertex_ids)
225 {
226 if (cfvid == afvid)
227 {
228 dof_mapping.push_back((short)afv);
229 found = true;
230 break;
231 }
232 afv++;
233 }
234
235 if (!found)
236 {
237 chi::log.LogAllError()
238 << "Face DOF mapping failed in call to "
239 << "MeshContinuum::FindAssociatedVertices. Could not find a matching"
240 "node."
241 << cur_face.neighbor_id << " " << cur_face.centroid.PrintS();
242 chi::Exit(EXIT_FAILURE);
243 }
244 }
