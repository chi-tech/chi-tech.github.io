16/**Populates a face histogram.
17 *
18 * \param master_tolerance Multiple histograms will only be attempted
19 * if the ratio of the maximum dofs-per-face to the average dofs-per-face
20 * is greater than this value. Default 1.2.
21 *
22 * \param slave_tolerance While traversing a sorted list of dofs-per-face,
23 * a new bin will only be generated when the ratio of the listed dofs-per-face
24 * to a running bin average exceeds this value. Defualt 1.1.
25 *
26 * The function populates face_categories which is a structure containing
27 * pairs. Pair.first is the max dofs-per-face for the category and Pair.second
28 * is the number of faces in this category.
29 *
30 * */
31void chi_mesh::MeshContinuum::
32 BuildFaceHistogramInfo(double master_tolerance, double slave_tolerance)
33{
34 if (face_histogram_available) return;
35
36 //================================================== Fill histogram
37 std::vector<size_t> face_size_histogram;
38 for (const auto& cell : local_cells)
39 for (const auto& face : cell.faces)
40 face_size_histogram.push_back(face.vertex_ids.size());
41
42 std::stable_sort(face_size_histogram.begin(), face_size_histogram.end());
43
44 //================================================== Determine total face dofs
45 size_t total_face_dofs_count = 0;
46 for (auto face_size : face_size_histogram)
47 total_face_dofs_count += face_size;
48
49 //================================================== Compute average and ratio
50 size_t smallest_face = face_size_histogram.front();
51 size_t largest_face = face_size_histogram.back();
52 size_t total_num_faces = face_size_histogram.size();
53 double average_dofs_per_face = (double)total_face_dofs_count/total_num_faces;
54
55 std::stringstream outstr;
56 outstr << "\nSmallest face = " << smallest_face;
57 outstr << "\nLargest face = " << largest_face;
58 outstr << "\nTotal face dofs = " << total_face_dofs_count;
59 outstr << "\nTotal faces = " << face_size_histogram.size();
60 outstr << "\nAverage dofs/face = " << average_dofs_per_face;
61 outstr << "\nMax to avg ratio = " << largest_face/average_dofs_per_face;
62 chi::log.LogAllVerbose1() << outstr.str();
63
64 //================================================== Determine number of bins
65 size_t last_bin_num_faces = total_num_faces;
66 if ((largest_face/average_dofs_per_face) > master_tolerance)
67 {
68 chi::log.LogAllVerbose1()
69 << "The ratio of max face dofs to average face dofs "
70 << "is larger than " << master_tolerance
71 << ", therefore a binned histogram "
72 << "will be constructed.";
73
74 //====================================== Build categories
75 size_t running_total_face_dofs = 0;
76 size_t running_face_count = 0;
77 size_t running_face_size = face_size_histogram[0];
78
79 double running_average = face_size_histogram[0];
80
81 for (size_t f=0; f<total_num_faces; ++f)
82 {
83 if ((face_size_histogram[f]/running_average) > slave_tolerance)
84 {
85 face_categories.emplace_back(running_face_size,running_face_count);
86 running_total_face_dofs = 0;
87 running_face_count = 0;
88 }
89
90 running_face_size = face_size_histogram[f];
91 running_total_face_dofs += face_size_histogram[f];
92 running_face_count++;
93 running_average = (double)running_total_face_dofs/double(running_face_count);
94 last_bin_num_faces = running_face_count;
95 }
96 }
97 face_categories.emplace_back(largest_face,last_bin_num_faces);
98
99 //================================================== Verbose print bins
100 outstr.str(std::string());
101 outstr
102 << "A total of " << face_categories.size()
103 << " bins were created:\n";
104
105 size_t bin_counter = -1;
106 for (auto bins : face_categories)
107 {
108 outstr
109 << "Bin " << ++bin_counter << ": "
110 << bins.second << " faces with max face dofs " << bins.first << "\n";
111 }
112
113 chi::log.LogAllVerbose1() << outstr.str();
114
115 face_histogram_available = true;
116}

200void chi_mesh::MeshContinuum::
201FindAssociatedVertices(const chi_mesh::CellFace& cur_face,
202 std::vector<short>& dof_mapping) const
203{
204 int associated_face = cur_face.GetNeighborAssociatedFace(*this);
205 //======================================== Check index validity
206 if ((not cur_face.has_neighbor) || (not cur_face.IsNeighborLocal(*this)))
207 {
208 chi::log.LogAllError()
209 << "Invalid cell index encountered in call to "
210 << "MeshContinuum::FindAssociatedVertices. Index points to either a boundary"
211 << "or a non-local cell.";
212 chi::Exit(EXIT_FAILURE);
213 }
214
215 auto& adj_cell = local_cells[cur_face.GetNeighborLocalID(*this)];
216
217 dof_mapping.reserve(cur_face.vertex_ids.size());
218
219 const auto& adj_face = adj_cell.faces[associated_face];
220
221 for (auto cfvid : cur_face.vertex_ids)
222 {
223 bool found = false;
224 short afv = 0;
225 for (auto afvid : adj_face.vertex_ids)
226 {
227 if (cfvid == afvid)
228 {
229 dof_mapping.push_back((short)afv);
230 found = true;
231 break;
232 }
233 afv++;
234 }
235

250/**Given the current cell, cell A, and its adjacent cell, cell B, with
251 * cell B adjacent to A at the `f`-th face of cell A. Will determine the
252 * `af`-th index of the face on cell B that interface with the `f`-th face
253 * of cell A.*/
254size_t chi_mesh::MeshContinuum::
255 MapCellFace(const chi_mesh::Cell &cur_cell,
256 const chi_mesh::Cell &adj_cell,
257 unsigned int f)
258{
259 const auto& ccface = cur_cell.faces[f]; //current cell face
260 std::set<uint64_t> ccface_vids;
261 for (auto vid : ccface.vertex_ids) ccface_vids.insert(vid);
262
263 size_t fmap;
264 bool map_found = false;
265 for (size_t af=0; af < adj_cell.faces.size(); af++)
266 {
267 const auto& acface = adj_cell.faces[af]; //adjacent cell face
268
269 std::set<uint64_t> acface_vids;
270 for (auto vid : acface.vertex_ids) acface_vids.insert(vid);
271
272 if (acface_vids == ccface_vids)
273 {
274 fmap = af;
275 map_found = true;
276 break;
277 }
278 }//for adj faces
279
280 if (not map_found)
281 throw std::logic_error(
282 "chi_mesh::MeshContinuum::MapCellFace: Mapping failure.");
283
284 return fmap;
285}

343chi_data_types::NDArray<uint64_t> chi_mesh::MeshContinuum::
344 MakeIJKToGlobalIDMapping() const
345{
346 const std::string fname = "MakeIJKToGlobalIDMapping";

348 throw std::logic_error(fname + " can only be run on orthogonal meshes.");
349
350 const auto ijk_info = this->GetIJKInfo();

353 const auto Nz = static_cast<int64_t>(ijk_info[2]);
354

358 for (int k=0; k<Nz; ++k)
359 m_ijk_to_i(i,j,k) = static_cast<uint64_t>(m_ijk_to_i.MapNDtoLin(i,j,k));
360
361 return m_ijk_to_i;
362}

365/**Determines the bounding box size of each cell and returns it as
366 * a list of 3-component vectors, one Vec3 for each cell.*/
367std::vector<chi_mesh::Vector3> chi_mesh::MeshContinuum::
368 MakeCellOrthoSizes() const
369{
370 std::vector<chi_mesh::Vector3> cell_ortho_sizes(local_cells.size());
371 for (const auto& cell : local_cells)
372 {
373 chi_mesh::Vector3 vmin = vertices[cell.vertex_ids.front()];
374 chi_mesh::Vector3 vmax = vmin;
375
376 for (const auto vid : cell.vertex_ids)
377 {
378 const auto& vertex = vertices[vid];
379 vmin.x = std::min(vertex.x,vmin.x);
380 vmin.y = std::min(vertex.y,vmin.y);
381 vmin.z = std::min(vertex.z,vmin.z);
382
383 vmax.x = std::max(vertex.x,vmax.x);
384 vmax.y = std::max(vertex.y,vmax.y);
385 vmax.z = std::max(vertex.z,vmax.z);
386 }
387
388 cell_ortho_sizes[cell.local_id] = vmax-vmin;
389 }//for cell
390
391 return cell_ortho_sizes;
