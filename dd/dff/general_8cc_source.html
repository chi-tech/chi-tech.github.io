440 /**Attempts to apply Dirichlet boundary conditions on the ith row
441  * of the matrix. If it did apply the dirichlet boundary condition
442  * then it will return true which should logically be used to prevent
443  * other matrix columns to be populated.
444  *
445  * \param ir The row which is to receive the possible application of
446  * the dirichlet boundary condition.
447  * \param ir_boundary_type Pointer to a index that will receive the
448  * boundary type.
449  * \param iref The reference vertex index to be used for boundary type
450  * information.
451  *
452  * Note on ir. In a multigroup simulation ir would be the dof number plus
453  * offset calculations (i.e. idof*G+g).
454  *
455  *
456  * */
457 bool chi_diffusion::Solver::ApplyDirichletI(int ir,
458  int *ir_boundary_type,
459  int iref,
460  bool suppress_assembly)
461 {
462  int irefr = iref;
463  if (irefr<0)
464  irefr = ir;
465  //================= Check if i is on boundary
466  *ir_boundary_type = NO_BOUNDARY; //NO_BOUNDARY
467  int ir_boundary_index= 0;
468  if (nodal_boundary_numbers[irefr]<0)
469  {
470  ir_boundary_index = abs(nodal_boundary_numbers[irefr])-1;
471  *ir_boundary_type = boundaries[ir_boundary_index]->type;

473 
474  if (*ir_boundary_type == DIFFUSION_DIRICHLET)

485
