174 /**Relative Residual Convergence test. The test uses
175  * the L2-norm of the residual (\f$ ||b-Ax||_2 \f$), divided by
176  * by the L2-norm of the right hand side (\f$ ||b||_2 \f$) compared
177  * to a tolerance, \f$ \epsilon \f$.
178 
179 \f[
180  \frac{||b-Ax||_2}{||b||_2} < \epsilon
181 \f]
182  *
183  * */
184 PetscErrorCode
185  chi_math::PETScUtils::RelativeResidualConvergenceTest(
186  KSP ksp, PetscInt n,
187  PetscReal rnorm,
188  KSPConvergedReason* convergedReason,
189  void *monitordestroy)
190 {
191  //============================== Compute rhs norm
192  Vec Rhs;
193  KSPGetRhs(ksp,&Rhs);
194  double rhs_norm;
195  VecNorm(Rhs,NORM_2,&rhs_norm);
196  if (rhs_norm < 1.0e-25)
197  rhs_norm = 1.0;
198 
199  //============================== Compute test criterion
200  double tol;
201  int maxIts;
202  KSPGetTolerances(ksp,NULL,&tol,NULL,&maxIts);
203 
204  double relative_residual = rnorm/rhs_norm;
205 
206  if (relative_residual < tol)
207  *convergedReason = KSP_CONVERGED_RTOL;
208 
209  return KSP_CONVERGED_ITERATING;
210 }
