19/**Power iterative scheme for k-eigenvalue calculations.
20\author Zachary Hardy
21*/
22void KEigenvalueSolver::PowerIteration()
23{
24 chi::log.Log()
25 << "\n********** Solving k-eigenvalue problem with "
26 << "the Power Method.\n";
27
28 phi_old_local.assign(phi_old_local.size(), 1.0);
29
30 double F_prev = 1.0;
31 k_eff = 1.0;
32 double k_eff_prev = 1.0;
33 double k_eff_change = 1.0;
34
35 //================================================== Initialize groupsets
36 for (auto& groupset : groupsets)
37 {
38 ComputeSweepOrderings(groupset);
39 InitFluxDataStructures(groupset);
40
41 InitWGDSA(groupset);
42 InitTGDSA(groupset);
43 }
44
45 //================================================== Start power iterations
46 int nit = 0;
47 bool converged = false;
48 while (nit < max_iterations)
49 {
50 MPI_Barrier(MPI_COMM_WORLD);
51 // Divide phi_old by k_eff (phi_old gives better init-quess for GMRES)
52 for (auto& phi : phi_old_local) phi /= k_eff;
53
54 //============================================= Loop over groupsets
55 for (auto& groupset : groupsets)
56 {
57 //======================================== Setup sweep chunk
58 auto sweep_chunk_ptr = SetSweepChunk(groupset);
59 MainSweepScheduler sweep_scheduler(SchedulingAlgorithm::DEPTH_OF_GRAPH,
60 groupset.angle_agg,
61 *sweep_chunk_ptr);
62
63 //======================================== Precompute the fission source
64 q_moments_local.assign(q_moments_local.size(), 0.0);
65 SetSource(groupset, q_moments_local,
66 APPLY_AGS_FISSION_SOURCES |
67 APPLY_WGS_FISSION_SOURCES);
68
69 //======================================== Converge the scattering source
70 // with a fixed fission source
71 using namespace std::placeholders;
72 if (groupset.iterative_method == IterativeMethod::CLASSICRICHARDSON)
73 {
74 ClassicRichardson(groupset, sweep_scheduler,
75 APPLY_WGS_SCATTER_SOURCES |
76 APPLY_AGS_SCATTER_SOURCES,
77 active_set_source_function,
78 options.verbose_inner_iterations);
79 }
80 else if (groupset.iterative_method == IterativeMethod::GMRES)
81 {
82 GMRES(groupset, sweep_scheduler,
83 APPLY_WGS_SCATTER_SOURCES,
84 APPLY_AGS_SCATTER_SOURCES,
85 active_set_source_function,
86 options.verbose_inner_iterations);
87 }
88 else if (groupset.iterative_method == IterativeMethod::KRYLOV_RICHARDSON or
89 groupset.iterative_method == IterativeMethod::KRYLOV_GMRES or
90 groupset.iterative_method == IterativeMethod::KRYLOV_BICGSTAB)
91 {
92 Krylov(groupset, sweep_scheduler,
93 APPLY_WGS_SCATTER_SOURCES,
94 APPLY_AGS_SCATTER_SOURCES,
95 active_set_source_function,
96 options.verbose_inner_iterations);
97 }
98
99 MPI_Barrier(MPI_COMM_WORLD);
100 }//for groupset
101
102 //======================================== Recompute k-eigenvalue
103 double F_new = ComputeFissionProduction(phi_new_local);
104 k_eff = F_new / F_prev * k_eff;
105 double reactivity = (k_eff - 1.0) / k_eff;
106
107 //======================================== Check convergence, book-keeping
108 k_eff_change = fabs(k_eff - k_eff_prev) / k_eff;
109 k_eff_prev = k_eff;
110 F_prev = F_new;
111 nit += 1;
112
113 if (k_eff_change < std::max(tolerance, 1.0e-12))
114 converged = true;
115
116 //======================================== Print iteration summary
117 if (options.verbose_outer_iterations)
118 {
119 std::stringstream k_iter_info;
120 k_iter_info
121 << chi::program_timer.GetTimeString() << " "
122 << " Iteration " << std::setw(5) << nit
123 << " k_eff " << std::setw(10) << k_eff
124 << " k_eff change " << std::setw(10) << k_eff_change
125 << " reactivity " << std::setw(10) << reactivity * 1e5;
126 if (converged) k_iter_info << " CONVERGED\n";
127
128 chi::log.Log() << k_iter_info.str();
129 }
130
131 if (converged) break;
132 }//for k iterations
133
134 //================================================== Cleanup groupsets
135 for (auto& groupset : groupsets)
136 {
137 CleanUpWGDSA(groupset);
138 CleanUpTGDSA(groupset);
139
140 ResetSweepOrderings(groupset);
141 }
142
143 //================================================== Print summary
144 chi::log.Log() << "\n";
145 chi::log.Log()
146 << " Final k-eigenvalue : "
147 << std::setprecision(6) << k_eff;
148 chi::log.Log()
149 << " Final change : "
150 << std::setprecision(6) << k_eff_change;
151 chi::log.Log() << "\n";
152}
