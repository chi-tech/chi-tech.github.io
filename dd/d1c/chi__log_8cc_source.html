203/**Returns a string representation of the event history associated with
204 * the tag. Each event entry will be prepended by the location id and
205 * the program timestamp in seconds. This method uses the
206 * ChiLog::EventInfo::GetString method to append information. This allows
207 * derived classes to implement more sophisticated outputs.*/
208std::string ChiLog::PrintEventHistory(size_t ev_tag)
209{
210 std::stringstream outstr;
211 if (ev_tag >= repeating_events.size())
212 return outstr.str();
213
214 RepeatingEvent& ref_rep_event = repeating_events[ev_tag];
215
216 for (auto& event : ref_rep_event.events)

218 outstr << "[" << chi_mpi.location_id << "] ";

220 char buf[100];

223
224 switch (event.ev_type)

249/**Processes an event given an event operation. See ChiLog for further
250 * reference.*/
251double ChiLog::ProcessEvent(size_t ev_tag,
252 ChiLog::EventOperation ev_operation)
253{
254 if (ev_tag >= repeating_events.size())
255 return 0.0;
256
257 RepeatingEvent& ref_rep_event = repeating_events[ev_tag];
258
259 double ret_val = 0.0;
260 switch (ev_operation)
261 {
262 case EventOperation::NUMBER_OF_OCCURRENCES:
263 {
264 for (auto& event : ref_rep_event.events)
265 {
266 if ( (event.ev_type == EventType::EVENT_CREATED) or
267 (event.ev_type == EventType::SINGLE_OCCURRENCE) or
268 (event.ev_type == EventType::EVENT_BEGIN))
269 ret_val += 1.0;
270 }//for events
271 break;
272 }
273 case EventOperation::TOTAL_DURATION:
274 {
275 double start_time = 0.0;
276 for (auto& event : ref_rep_event.events)
277 {
278 if (event.ev_type == EventType::EVENT_BEGIN)
279 start_time = event.ev_time;
280 if (event.ev_type == EventType::EVENT_END)
281 ret_val += event.ev_time - start_time;
282 }//for events
283 ret_val *= 1000.0;
284 break;
285 }
286 case EventOperation::AVERAGE_DURATION:
287 {
288 double start_time = 0.0;
289 int counter = 0;
290 for (auto& event : ref_rep_event.events)
291 {
292 if (event.ev_type == EventType::EVENT_BEGIN)
293 start_time = event.ev_time;
294
295 if (event.ev_type == EventType::EVENT_END)
296 {
297 ret_val += event.ev_time - start_time;
298 counter++;
299 }
300 }//for events
301 ret_val /= (1000.0*counter);
302 break;
303 }
304 case EventOperation::MAX_VALUE:
305 {
306 ret_val = 0.0;
307 for (auto& event : ref_rep_event.events)
308 {
309 if ((event.ev_type == EventType::SINGLE_OCCURRENCE) or
310 (event.ev_type == EventType::EVENT_BEGIN) or
311 (event.ev_type == EventType::EVENT_END))
312 {
313 if (event.ev_info != nullptr)
314 ret_val = std::max(event.ev_info->arb_value,ret_val);
315 }
316 }//for events
317 break;
318 }
319 case EventOperation::AVERAGE_VALUE:
320 {
321 ret_val = 0.0;
322 int count = 0;
323 for (auto& event : ref_rep_event.events)
324 {
325 if ((event.ev_type == EventType::SINGLE_OCCURRENCE) or
326 (event.ev_type == EventType::EVENT_BEGIN) or
327 (event.ev_type == EventType::EVENT_END))
328 {
329 if (event.ev_info != nullptr)
330 {
331 ret_val += event.ev_info->arb_value;
332 ++count;
333 }
334 }
335 }//for events
336 if (count == 0) count = 1;
337 ret_val /= count;
338 break;
339 }
340 }//switch
341
342 return ret_val;
