227/**Returns a string representation of the event history associated with
228 * the tag. Each event entry will be prepended by the location id and
229 * the program timestamp in seconds. This method uses the
230 * ChiLog::EventInfo::GetString method to append information. This allows
231 * derived classes to implement more sophisticated outputs.*/
232std::string chi::ChiLog::PrintEventHistory(size_t ev_tag)
233{
234 std::stringstream outstr;
235 if (ev_tag >= repeating_events.size())
236 return outstr.str();
237
238 RepeatingEvent& ref_rep_event = repeating_events[ev_tag];
239
240 for (auto& event : ref_rep_event.events)
241 {
242 outstr << "[" << Chi::mpi.location_id << "] ";
243
244 char buf[100];
245 snprintf(buf,100,"%16.9f",event.ev_time/1000.0);
246 outstr << buf << " ";
247
248 switch (event.ev_type)
249 {
250 case EventType::EVENT_CREATED:
251 outstr << "EVENT_CREATED ";
252 break;
253 case EventType::SINGLE_OCCURRENCE:
254 outstr << "SINGLE_OCCURRENCE ";

256 case EventType::EVENT_BEGIN:

262 }
263

273/**Processes an event given an event operation. See ChiLog for further
274 * reference.*/
275double chi::ChiLog::ProcessEvent(size_t ev_tag,
276 chi::ChiLog::EventOperation ev_operation)
277{
278 if (ev_tag >= repeating_events.size())
279 return 0.0;
280
281 RepeatingEvent& ref_rep_event = repeating_events[ev_tag];
282
283 double ret_val = 0.0;
284 switch (ev_operation)
285 {
286 case EventOperation::NUMBER_OF_OCCURRENCES:
287 {
288 for (auto& event : ref_rep_event.events)
289 {
290 if ( (event.ev_type == EventType::EVENT_CREATED) or
291 (event.ev_type == EventType::SINGLE_OCCURRENCE) or
292 (event.ev_type == EventType::EVENT_BEGIN))
293 ret_val += 1.0;
294 }//for events
295 break;
296 }
297 case EventOperation::TOTAL_DURATION:
298 {
299 double start_time = 0.0;
300 for (auto& event : ref_rep_event.events)
301 {
302 if (event.ev_type == EventType::EVENT_BEGIN)
303 start_time = event.ev_time;
304 if (event.ev_type == EventType::EVENT_END)
305 ret_val += event.ev_time - start_time;
306 }//for events
307 ret_val *= 1000.0;
308 break;
309 }
310 case EventOperation::AVERAGE_DURATION:
311 {
312 double start_time = 0.0;
313 int counter = 0;
314 for (auto& event : ref_rep_event.events)
315 {
316 if (event.ev_type == EventType::EVENT_BEGIN)
317 start_time = event.ev_time;
318
319 if (event.ev_type == EventType::EVENT_END)
320 {
321 ret_val += event.ev_time - start_time;
322 counter++;
323 }
324 }//for events
325 ret_val /= (1000.0*counter);
326 break;
327 }
328 case EventOperation::MAX_VALUE:
329 {
330 ret_val = 0.0;
331 for (auto& event : ref_rep_event.events)
332 {
333 if ((event.ev_type == EventType::SINGLE_OCCURRENCE) or
334 (event.ev_type == EventType::EVENT_BEGIN) or
335 (event.ev_type == EventType::EVENT_END))
336 {
337 if (event.ev_info != nullptr)
338 ret_val = std::max(event.ev_info->arb_value,ret_val);
339 }
340 }//for events
341 break;
342 }
343 case EventOperation::AVERAGE_VALUE:
344 {
345 ret_val = 0.0;
346 int count = 0;
347 for (auto& event : ref_rep_event.events)
348 {
349 if ((event.ev_type == EventType::SINGLE_OCCURRENCE) or
350 (event.ev_type == EventType::EVENT_BEGIN) or
351 (event.ev_type == EventType::EVENT_END))
352 {
353 if (event.ev_info != nullptr)
354 {
355 ret_val += event.ev_info->arb_value;
356 ++count;
357 }
358 }
359 }//for events
360 if (count == 0) count = 1;
361 ret_val /= count;
362 break;
363 }
364 }//switch
365
366 return ret_val;
