84

86  int total_reflect_faces = 0;
87  int total_reflect_size = 0;
88 
89  bool reflecting_bcs_initialized=false;
90 
91  int bndry_id=0;
92  for (auto& bndry : sim_boundaries)
93  {
94  if (bndry->IsReflecting())
95  {
96  size_t tot_num_angles = quadrature->abscissae.size();
97  size_t num_local_cells = grid->local_cell_glob_indices.size();
98  auto& rbndry = (BoundaryReflecting&)(*bndry);
99

101  rbndry.angle_readyflags.resize(tot_num_angles,
102  std::vector<bool>(number_of_group_subsets,false));
103 
104  //========================================= Determine reflected angle
105  for (int n=0; n<tot_num_angles; ++n)
106  {
107  auto omega_reflected = (quadrature->omegas[n]) -
108  rbndry.normal*
109  quadrature->omegas[n].Dot(rbndry.normal)*2.0;
110  for (int nstar=0; nstar<tot_num_angles; ++nstar)
111  if (omega_reflected.Dot(quadrature->omegas[nstar])> (1.0-epsilon))
112  {rbndry.reflected_anglenum[n] = nstar;break;}
113 
114  if (rbndry.reflected_anglenum[n]<0)
115  {
116  chi_log.Log(LOG_ALLERROR)
117  << "Reflected angle not found for angle " << n
118  << " with direction " << quadrature->omegas[n].PrintS()
119  << ". This can happen for two reasons: i) A quadrature is used"
120  " that is not symmetric about the axis associated with the "
121  "reflected boundary, or ii) the reflecting boundary is not "
122  "aligned with any reflecting axis of the quadrature.";
123  exit(EXIT_FAILURE);
124  }
125  }
126 
127  //========================================= For angles
128  rbndry.hetero_boundary_flux.clear();
129  rbndry.hetero_boundary_flux_old.clear();
130  rbndry.hetero_boundary_flux.resize(tot_num_angles);
131  for (int n=0; n<tot_num_angles; ++n)
132  {
133  //Only continue if omega is outgoing
134  if ( quadrature->omegas[n].Dot(rbndry.normal)< 0.0 )
135  continue;
136 
137  //================================== For cells
138  auto& cell_vec = rbndry.hetero_boundary_flux[n];
139  cell_vec.resize(num_local_cells);
140  for (const auto& cell : grid->local_cells)
141  {
142  int c = cell.local_id;
143 
144  //=========================== Check cell on ref bndry
145  bool on_ref_bndry = false;
146  for (const auto& face : cell.faces){
147  if ( (not face.has_neighbor) and
148  (face.normal.Dot(rbndry.normal) > 0.999999) )
149  {
150  on_ref_bndry = true;
151  break;
152  }
153  }
154  if (not on_ref_bndry) continue;
155  total_reflect_cells += 1;
156 
157  //=========================== If cell on ref bndry
158  cell_vec[c].resize(cell.faces.size());
159  int f=0;
160  for (const auto& face : cell.faces)
161  {
162  if ( (not face.has_neighbor) and
163  (face.normal.Dot(rbndry.normal) > 0.999999) )
164  {
165  cell_vec[c][f].clear();
166  cell_vec[c][f].resize(face.vertex_ids.size(),
167  std::vector<double>(number_of_groups,0.0));
168  total_reflect_faces += 1;
169  total_reflect_size += face.vertex_ids.size()*number_of_groups;
170  }
171  ++f;
172  }
173  }//for cells
174  }//for angles
175 
176  //========================================= Determine if boundary is
177  // opposing reflecting
178  if ((bndry_id == 1) and (sim_boundaries[0]->IsReflecting()))
179  rbndry.opposing_reflected = true;
180  if ((bndry_id == 3) and (sim_boundaries[2]->IsReflecting()))
181  rbndry.opposing_reflected = true;
182  if ((bndry_id == 5) and (sim_boundaries[4]->IsReflecting()))
183  rbndry.opposing_reflected = true;
184 
185  if (rbndry.opposing_reflected)
186  rbndry.hetero_boundary_flux_old = rbndry.hetero_boundary_flux;
187 
188  reflecting_bcs_initialized = true;
189  }//if reflecting
190 
191  ++bndry_id;
192  }//for bndry
193

195  chi_log.Log(LOG_0) << "Reflecting boundary conditions initialized.";
196

202 std::pair<size_t,size_t> chi_mesh::sweep_management::AngleAggregation::
203  GetNumDelayedAngularDOFs()
204 {

207  return number_angular_unknowns;
208 
209  //======================================== If not developed
210  size_t local_ang_unknowns = 0;
211 
212  //======================================== Opposing reflecting bndries
213  for (auto& bndry : sim_boundaries)
214  {
215  if (bndry->IsReflecting())
216  {

218

221  for (auto& cellvec : angle)

225 
226  }//if reflecting
227  }//for bndry
228 
229  //======================================== Intra-cell cycles
230  for (auto& as_group : angle_set_groups)

232  local_ang_unknowns += angle_set->delayed_local_psi.size();
233 
234  //======================================== Inter location cycles

236  for (auto& angle_set : as_group.angle_sets)
237  for (auto& loc_vector : angle_set->delayed_prelocI_outgoing_psi)
238  local_ang_unknowns += loc_vector.size();
239 
240 
241 
242  size_t global_ang_unknowns = 0;
243  MPI_Allreduce(&local_ang_unknowns,
244  &global_ang_unknowns,
245  1,
246  MPI_UNSIGNED_LONG_LONG,
247  MPI_SUM,
248  MPI_COMM_WORLD);
249 
250  number_angular_unknowns = {local_ang_unknowns,global_ang_unknowns};
