86/** Finds the roots of the Legendre polynomial.
87 *
88 * The algorithm is that depicted in:
89 *
90 * [1] Barrera-Figueroa, et al., "Multiple root finder algorithm for Legendre
91 * and Chebyshev polynomials via Newton's method", Annales Mathematicae et
92 * Informaticae, 33 (2006) pp. 3-13.
93 *
94 * \param N Is the order of the polynomial.
95 * \param roots Is a reference to the roots.
96 * \param max_iters Maximum newton iterations to perform for each root.
97 * Default: 1000.
98 * \param tol Tolerance at which the newton iteration will be terminated.
99 * Default: 1.0e-12.
100 *
101 * \author Jan*/
102std::vector<double> chi_math::QuadratureGaussLegendre::FindRoots(
103 unsigned int N, unsigned int max_iters, double tol)
104{
105 //======================================== Populate init guess
106 //This initial guess proved to be quite important
107 //at higher N since the roots start to get
108 //squeezed to -1 and 1.
109 int num_search_intvls = 1000;
110 if (N>64)
111 num_search_intvls *= 10;
112 if (N>256)
113 num_search_intvls *= 10;
114 if (N>768)
115 num_search_intvls *= 10;
116
117 if (N>2056)
118 {
119 num_search_intvls *= 10;
120 chi_log.Log(LOG_0WARNING)
121 << "chi_math::QuadratureGaussLegendre::FindRoots: "
122 << "The order of the polynomial for which to find the roots is "
123 << "greater than 2056. Accuracy of the root finder will be diminished "
124 << "along with a reduction in stability.";
125 }
126
127 // For this code we simply check to see where the
128 // polynomial changes sign.
129 double delta = 2.0/num_search_intvls;
130 std::vector<double> xk(N, 0.0);
131 int counter = -1;
132 for(size_t i=0; i<num_search_intvls; i++)
133 {
134 double x_i = -1.0 + i*delta;
135 double x_ip1 = x_i + delta;
136
137 if (Legendre(N,x_i)*Legendre(N,x_ip1) < 0.0)
138 xk[++counter] = (x_ip1 + x_i) / 2.0;
139 }
140
141 //======================================== Apply algorithm
142 // Refer to equation 4.3 in [1]. Sum 1 (S1) is used in the
143 // computation of B at x_k. Sum 2 (S2) is used in equation 4.3.
144 // Equation 4.3 is broken up into pieces as follows:
145 // - a = block bracket containing the second derivative
146 // - b = denominator
147 // - c = everything but xold
148 for (int k=0; k<N; k++)
149 {
150 for (size_t iteration=0; iteration<max_iters; iteration++)
151 {
152 double xold = xk[k];
153 double f = Legendre(N,xold); //Function evaluation
154 double fp = dLegendredx(N,xold); //First derivative
155 double fpp = d2Legendredx2(N,xold); //Second derivative
156
157 //===================== Compute sum 1
158 double S1 = 0.0;
159 for (int i=0; i<=(k-1); i++)
160 S1 += 1.0/(xk[k] - xk[i]);
161
162 //===================== Compute B at x_k
163 double B_xk = fp - f*S1;
164
165 //===================== Compute sum 2
166 double S2 = 0.0;
167 for (int i=0; i<=(k-1); i++)
168 S2 += 1.0 / (xk[k] - xk[i]) / (xk[k] - xk[i]);
169
170 //===================== Compute final formula
171 double a = fpp + f*S2;
172 double b = B_xk*B_xk + fp*fp - f*a;
173 double c = 2.0*f*B_xk/b;
174
175 xk[k] = xold - c;
176
177 if (std::fabs(xk[k] - xold) < tol)
178 break;
179 }//for iteration
180 }//for k
181
182 std::stable_sort(xk.begin(), xk.end());
183
184 return xk;
185}
