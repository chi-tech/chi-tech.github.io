85/** Finds the roots of the Legendre polynomial.
86 *
87 * The algorithm is that depicted in:
88 *
89 * [1] Barrera-Figueroa, et al., "Multiple root finder algorithm for Legendre
90 * and Chebyshev polynomials via Newton's method", Annales Mathematicae et
91 * Informaticae, 33 (2006) pp. 3-13.
92 *
93 * \param N Is the order of the polynomial.
94 * \param roots Is a reference to the roots.
95 * \param max_iters Maximum newton iterations to perform for each root.
96 * Default: 1000.
97 * \param tol Tolerance at which the newton iteration will be terminated.
98 * Default: 1.0e-12.
99 *
100 * \author Jan*/
101std::vector<double> chi_math::QuadratureGaussLegendre::FindRoots(
102 unsigned int N, unsigned int max_iters, double tol)
103{
104 //======================================== Populate init guess
105 //This initial guess proved to be quite important
106 //at higher N since the roots start to get
107 //squeezed to -1 and 1.
108 int num_search_intvls = 1000;
109 if (N>64)
110 num_search_intvls *= 10;
111 if (N>256)
112 num_search_intvls *= 10;
113 if (N>768)
114 num_search_intvls *= 10;
115
116 if (N>2056)
117 {
118 num_search_intvls *= 10;
119 Chi::log.Log0Warning()
120 << "chi_math::QuadratureGaussLegendre::FindRoots: "
121 << "The order of the polynomial for which to find the roots is "
122 << "greater than 2056. Accuracy of the root finder will be diminished "
123 << "along with a reduction in stability.";
124 }
125
126 // For this code we simply check to see where the
127 // polynomial changes sign.
128 double delta = 2.0/num_search_intvls;
129 std::vector<double> xk(N, 0.0);
130 int counter = -1;
131 for(size_t i=0; i<num_search_intvls; i++)
132 {
133 double x_i = -1.0 + i*delta;
134 double x_ip1 = x_i + delta;
135
136 if (Legendre(N,x_i)*Legendre(N,x_ip1) < 0.0)
137 xk[++counter] = (x_ip1 + x_i) / 2.0;
138 }
139
140 //======================================== Apply algorithm
141 // Refer to equation 4.3 in [1]. Sum 1 (S1) is used in the
142 // computation of B at x_k. Sum 2 (S2) is used in equation 4.3.
143 // Equation 4.3 is broken up into pieces as follows:
144 // - a = block bracket containing the second derivative
145 // - b = denominator
146 // - c = everything but xold
147 for (int k=0; k<N; k++)
148 {
149 for (size_t iteration=0; iteration<max_iters; iteration++)
150 {
151 double xold = xk[k];
152 double f = Legendre(N,xold); //Function evaluation
153 double fp = dLegendredx(N,xold); //First derivative
154 double fpp = d2Legendredx2(N,xold); //Second derivative
155
156 //===================== Compute sum 1
157 double S1 = 0.0;
158 for (int i=0; i<=(k-1); i++)
159 S1 += 1.0/(xk[k] - xk[i]);
160
161 //===================== Compute B at x_k
162 double B_xk = fp - f*S1;
163
164 //===================== Compute sum 2
165 double S2 = 0.0;
166 for (int i=0; i<=(k-1); i++)
167 S2 += 1.0 / (xk[k] - xk[i]) / (xk[k] - xk[i]);
168
169 //===================== Compute final formula
170 double a = fpp + f*S2;
171 double b = B_xk*B_xk + fp*fp - f*a;
172 double c = 2.0*f*B_xk/b;
173
174 xk[k] = xold - c;
175
176 if (std::fabs(xk[k] - xold) < tol)
177 break;
178 }//for iteration
179 }//for k
180
181 std::stable_sort(xk.begin(), xk.end());
182
183 return xk;
184}
