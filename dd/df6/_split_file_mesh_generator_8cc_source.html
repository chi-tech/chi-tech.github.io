75void SplitFileMeshGenerator::Execute()
76{
77 const int num_mpi = Chi::mpi.process_count;
78 const int num_parts = num_mpi == 1 ? num_parts_ : num_mpi;
79
80 if (Chi::mpi.location_id == 0 and (not read_only_))
81 {
82 //======================================== Execute all input generators
83 // Note these could be empty
84 std::unique_ptr<UnpartitionedMesh> current_umesh = nullptr;
85 for (auto mesh_generator_ptr : inputs_)
86 {
87 auto new_umesh =
88 mesh_generator_ptr->GenerateUnpartitionedMesh(std::move(current_umesh));
89 current_umesh = std::move(new_umesh);
90 }
91
92 //======================================== Generate final umesh
93 current_umesh = GenerateUnpartitionedMesh(std::move(current_umesh));
94
95 Chi::log.Log() << "Writing split-mesh with " << num_parts << " parts";
96 const auto cell_pids = PartitionMesh(*current_umesh, num_parts);
97 WriteSplitMesh(cell_pids, *current_umesh, num_parts);
98 Chi::log.Log() << "Split-mesh with " << num_parts
99 << " parts successfully created";
100 } // if home location
101
102 // Other locations wait here for files to be written
103 Chi::mpi.Barrier();
104
105 if (Chi::mpi.process_count == num_parts)
106 {
107 Chi::log.Log() << "Reading split-mesh";
108 auto mesh_info = ReadSplitMesh();
109
110 auto grid_ptr = SetupLocalMesh(mesh_info);
111
112 auto new_mesher =
113 std::make_shared<chi_mesh::VolumeMesher>(VolumeMesherType::UNPARTITIONED);
114 new_mesher->SetContinuum(grid_ptr);
115
116 if (Chi::current_mesh_handler < 0) chi_mesh::PushNewHandlerAndGetIndex();
117
118 auto& cur_hndlr = chi_mesh::GetCurrentHandler();
119 cur_hndlr.SetVolumeMesher(new_mesher);
120 Chi::log.Log() << "Done reading split-mesh files";
121 }
122 else
123 {
124 Chi::log.Log0Warning()
125 << "After creating a split-mesh with mpi-processes < "
126 "num_parts the program will now auto terminate. This is not an error "
127 "and is the default behavior for the SplitFileMeshGenerator.\n"
128 << Chi::log.GetTimingBlock("ChiTech").MakeGraphString();
129 Chi::Exit(EXIT_SUCCESS);
130 }
131
132 Chi::mpi.Barrier();
133}

136void SplitFileMeshGenerator::WriteSplitMesh(
137 const std::vector<int64_t>& cell_pids,
138 const UnpartitionedMesh& umesh,
139 int num_parts)
140{
141 const std::filesystem::path dir_path =
142 std::filesystem::absolute(split_mesh_dir_path_);
143
144 const auto parent_path = dir_path.parent_path();
145 ChiInvalidArgumentIf(not std::filesystem::exists(parent_path),
146 "Parent path " + parent_path.string() +
147 " does not exist");
148
149 bool root_dir_created = true;
150 if (not std::filesystem::exists(dir_path))
151 root_dir_created = std::filesystem::create_directories(dir_path);
152
153 ChiLogicalErrorIf(not root_dir_created,
154 "Failed to create directory " + dir_path.string());
155
156 const auto& vertex_subs = umesh.GetVertextCellSubscriptions();
157 const auto& raw_cells = umesh.GetRawCells();
158 const auto& raw_vertices = umesh.GetVertices();
159
160 auto& t_write =
161 Chi::log.CreateOrGetTimingBlock("FileMeshGenerator::WriteSplitMesh");
162 auto& t_sorting = Chi::log.CreateOrGetTimingBlock(
163 "Sorting data", "FileMeshGenerator::WriteSplitMesh");
164 auto& t_cells = Chi::log.CreateOrGetTimingBlock(
165 "WriteCells", "FileMeshGenerator::WriteSplitMesh");
166 auto& t_verts = Chi::log.CreateOrGetTimingBlock(
167 "WriteVerts", "FileMeshGenerator::WriteSplitMesh");
168 auto& t_serialize = Chi::log.CreateOrGetTimingBlock("Serialize");
169
170 uint64_t aux_counter = 0;
171 for (int pid = 0; pid < num_parts; ++pid)
172 {
173 t_write.TimeSectionBegin();
174 const std::filesystem::path file_path = dir_path.string() + "/" +
175 split_file_prefix_ + "_" +
176 std::to_string(pid) + ".cmesh";
177
178 std::ofstream ofile(file_path.string(),
179 std::ios_base::binary | std::ios_base::out);
180
181 ChiLogicalErrorIf(not ofile.is_open(),
182 "Failed to open " + file_path.string());
183
184 // Appropriate cells and vertices to the current part being writting
185 t_sorting.TimeSectionBegin();
186
187 std::vector<uint64_t> local_cells_needed;
188 std::set<uint64_t> cells_needed;
189 std::set<uint64_t> vertices_needed;
190 {
191 local_cells_needed.reserve(raw_cells.size() / num_parts);
192 {
193 uint64_t cell_global_id = 0;
194 for (auto cell_pid : cell_pids)
195 {
196 if (cell_pid == pid) local_cells_needed.push_back(cell_global_id);
197 ++cell_global_id;
198 }
199 }
200
201 for (uint64_t cell_global_id : local_cells_needed)
202 {
203 cells_needed.insert(cell_global_id);
204
205 const auto& raw_cell = *raw_cells[cell_global_id];
206
207 for (uint64_t vid : raw_cell.vertex_ids)
208 {
209 vertices_needed.insert(vid);
210 for (uint64_t ghost_gid : vertex_subs[vid])
211 {
212 if (ghost_gid == cell_global_id) continue;
213 cells_needed.insert(ghost_gid);
214
215 const auto& ghost_raw_cell = *raw_cells[ghost_gid];
216 for (uint64_t gvid : ghost_raw_cell.vertex_ids)
217 vertices_needed.insert(gvid);
218 }
219 }
220 }
221 }
222 t_sorting.TimeSectionEnd();
223
224 //================================================ Write mesh attributes
225 // and general info
226 const auto& mesh_options = umesh.GetMeshOptions();
227
228 chi::WriteBinaryValue(ofile, num_parts); // int
229
230 chi::WriteBinaryValue(ofile, scint(umesh.GetMeshAttributes())); // int
231 chi::WriteBinaryValue(ofile, mesh_options.ortho_Nx); // size_t
232 chi::WriteBinaryValue(ofile, mesh_options.ortho_Ny); // size_t
233 chi::WriteBinaryValue(ofile, mesh_options.ortho_Nz); // size_t
234
235 chi::WriteBinaryValue(ofile, raw_vertices.size()); // size_t
236
237 //================================================ Write the boundary map
238 const auto& bndry_map = mesh_options.boundary_id_map;
239 chi::WriteBinaryValue(ofile, bndry_map.size()); // size_t
240 for (const auto& [bid, bname] : bndry_map)
241 {
242 chi::WriteBinaryValue(ofile, bid); // uint64_t
243 const size_t num_chars = bname.size();
244 chi::WriteBinaryValue(ofile, num_chars); // size_t
245 ofile.write(bname.data(), scint(num_chars)); // characters
246 }
247
248 //================================================ Write how many cells
249 // and vertices in file
250 chi::WriteBinaryValue(ofile, cells_needed.size()); // size_t
251 chi::WriteBinaryValue(ofile, vertices_needed.size()); // size_t
252
253 //================================================ Write cells
254 const size_t BUFFER_SIZE = 4096 * 2;
255 chi_data_types::ByteArray serial_data;
256 serial_data.Data().reserve(BUFFER_SIZE * 2);
257 for (const auto& cell_global_id : cells_needed)
258 {
259 t_cells.TimeSectionBegin();
260 t_serialize.TimeSectionBegin();
261 const auto& cell = *raw_cells[cell_global_id];
262 serial_data.Write(static_cast<int>(cell_pids[cell_global_id])); // int
263 serial_data.Write(cell_global_id); // uint64_t
264 SerializeCell(cell, serial_data);
265 t_serialize.TimeSectionEnd();
266 if (serial_data.Size() > BUFFER_SIZE)
267 {
268 ofile.write((char*)serial_data.Data().data(),
269 scint(serial_data.Size()));
270 const size_t cap = serial_data.Data().capacity();
271 serial_data.Clear();
272 serial_data.Data().reserve(cap);
273 }
274 t_cells.TimeSectionEnd();
275 }
276 if (serial_data.Size() > 0)
277 {
278 ofile.write((char*)serial_data.Data().data(), scint(serial_data.Size()));
279 serial_data.Clear();
280 }
281
282 //================================================ Write vertices
283 t_verts.TimeSectionBegin();
284 for (const uint64_t vid : vertices_needed)
285 {
286 serial_data.Write(vid); // uint64_t
287 serial_data.Write(raw_vertices[vid]);
288 if (serial_data.Size() > BUFFER_SIZE)
289 {
290 ofile.write((char*)serial_data.Data().data(),
291 scint(serial_data.Size()));
292 serial_data.Clear();
293 }
294 }
295 if (serial_data.Size() > 0)
296 {
297 ofile.write((char*)serial_data.Data().data(), scint(serial_data.Size()));
298 serial_data.Clear();
299 }
300 t_verts.TimeSectionEnd();
301
302 ofile.close();
303 t_write.TimeSectionEnd();
304
305 const double fraction_complete =
306 static_cast<double>(pid) / static_cast<double>(num_parts);
307 if (fraction_complete >= static_cast<double>(aux_counter + 1) * 0.1)
308 {
309 if (verbose_)
310 Chi::log.Log() << Chi::program_timer.GetTimeString()
311 << " Surpassing part " << pid << " of " << num_parts
312 << " (" << (aux_counter + 1) * 10 << "%)";
313 ++aux_counter;
314 }
315 } // for p
316}

319void SplitFileMeshGenerator::SerializeCell(
320 const UnpartitionedMesh::LightWeightCell& cell,
321 chi_data_types::ByteArray& serial_buffer)
322{
323 serial_buffer.Write(cell.type);
324 serial_buffer.Write(cell.sub_type);
325 serial_buffer.Write(cell.centroid);
326 serial_buffer.Write(cell.material_id);
327 serial_buffer.Write(cell.vertex_ids.size());
328 for (uint64_t vid : cell.vertex_ids)
329 serial_buffer.Write(vid);
330 serial_buffer.Write(cell.faces.size());
331 for (const auto& face : cell.faces)
332 {
333 serial_buffer.Write(face.vertex_ids.size());
334 for (uint64_t vid : face.vertex_ids)
335 serial_buffer.Write(vid);
336 serial_buffer.Write(face.has_neighbor);
337 serial_buffer.Write(face.neighbor);
338 }

341SplitFileMeshGenerator::SplitMeshInfo SplitFileMeshGenerator::ReadSplitMesh()
342{
343 const int pid = Chi::mpi.location_id;
344 const std::filesystem::path dir_path =
345 std::filesystem::absolute(split_mesh_dir_path_);
346 const std::filesystem::path file_path = dir_path.string() + "/" +
347 split_file_prefix_ + "_" +
348 std::to_string(pid) + ".cmesh";
349
350 SplitMeshInfo info_block;
351 auto& cells = info_block.cells_;
352 auto& vertices = info_block.vertices_;
353 std::ifstream ifile(file_path, std::ios_base::binary | std::ios_base::in);
354
355 ChiLogicalErrorIf(not ifile.is_open(),
356 "Failed to open " + file_path.string());
357
358 //================================================== Read mesh attributes
359 // and general info
360 const size_t file_num_parts = chi::ReadBinaryValue<int>(ifile);
361
362 ChiLogicalErrorIf(Chi::mpi.process_count != file_num_parts,
363 "Split mesh files with prefix \"" + split_file_prefix_ +
364 "\" has been created with " +
365 std::to_string(file_num_parts) +
366 " parts but is now being read with " +
367 std::to_string(Chi::mpi.process_count) + " processes.");
368
369 info_block.mesh_attributes_ = chi::ReadBinaryValue<int>(ifile);
370 info_block.ortho_Nx_ = chi::ReadBinaryValue<size_t>(ifile);
371 info_block.ortho_Ny_ = chi::ReadBinaryValue<size_t>(ifile);
372 info_block.ortho_Nz_ = chi::ReadBinaryValue<size_t>(ifile);
373
374 info_block.num_global_vertices_ = chi::ReadBinaryValue<size_t>(ifile);
375
376 //================================================== Read boundary map
377 const size_t num_bndries = chi::ReadBinaryValue<size_t>(ifile);
378 for (size_t b = 0; b < num_bndries; ++b)
379 {
380 const uint64_t bid = chi::ReadBinaryValue<uint64_t>(ifile);
381 const size_t num_chars = chi::ReadBinaryValue<size_t>(ifile);
382 std::string bname(num_chars, ' ');
383 ifile.read(bname.data(), static_cast<int>(num_chars));
384
385 info_block.boundary_id_map_.insert(std::make_pair(bid, bname));
386 }
387
388 //================================================ Write how many cells
389 // and vertices in file
390 const size_t num_cells = chi::ReadBinaryValue<size_t>(ifile);
391 const size_t num_vertices = chi::ReadBinaryValue<size_t>(ifile);
392
393 //================================================== Read the cells
394 for (size_t c = 0; c < num_cells; ++c)
395 {
396 const int cell_pid = chi::ReadBinaryValue<int>(ifile);
397 const uint64_t cell_gid = chi::ReadBinaryValue<uint64_t>(ifile);
398 const CellType cell_type = chi::ReadBinaryValue<CellType>(ifile);
399 const CellType cell_sub_type = chi::ReadBinaryValue<CellType>(ifile);
400
401 UnpartitionedMesh::LightWeightCell new_cell(cell_type, cell_sub_type);
402
403 new_cell.centroid = chi::ReadBinaryValue<chi_mesh::Vector3>(ifile);
404 new_cell.material_id = chi::ReadBinaryValue<int>(ifile);
405
406 const size_t num_vids = chi::ReadBinaryValue<size_t>(ifile);
407 for (size_t v = 0; v < num_vids; ++v)
408 new_cell.vertex_ids.push_back(chi::ReadBinaryValue<uint64_t>(ifile));
409
410 const size_t num_faces = chi::ReadBinaryValue<size_t>(ifile);
411 for (size_t f = 0; f < num_faces; ++f)
412 {
413 UnpartitionedMesh::LightWeightFace new_face;
414 const size_t num_face_vids = chi::ReadBinaryValue<size_t>(ifile);
415 for (size_t v = 0; v < num_face_vids; ++v)
416 new_face.vertex_ids.push_back(chi::ReadBinaryValue<uint64_t>(ifile));
417
418 new_face.has_neighbor = chi::ReadBinaryValue<bool>(ifile);
419 new_face.neighbor = chi::ReadBinaryValue<uint64_t>(ifile);
420
421 new_cell.faces.push_back(std::move(new_face));
422 } // for f
423
424 cells.insert(
425 std::make_pair(CellPIDGID(cell_pid, cell_gid), std::move(new_cell)));
426 } // for cell c
427
428 //================================================== Read the vertices
429 for (size_t v = 0; v < num_vertices; ++v)
430 {
431 const uint64_t vid = chi::ReadBinaryValue<uint64_t>(ifile);
432 const chi_mesh::Vector3 vertex =
433 chi::ReadBinaryValue<chi_mesh::Vector3>(ifile);
434 vertices.insert(std::make_pair(vid, vertex));
435 } // for vertex v
436
437 ifile.close();
438
439 return info_block;
440}

442std::shared_ptr<MeshContinuum>
443SplitFileMeshGenerator::SetupLocalMesh(SplitMeshInfo& mesh_info)
444{
445 auto grid_ptr = chi_mesh::MeshContinuum::New();
446
447 grid_ptr->GetBoundaryIDMap() = mesh_info.boundary_id_map_;
448
449 auto& cells = mesh_info.cells_;
450 auto& vertices = mesh_info.vertices_;
451
452 for (const auto& [vid, vertex] : vertices)
453 grid_ptr->vertices.Insert(vid, vertex);
454
455 for (const auto& [pidgid, raw_cell] : cells)
456 {
457 const auto& [cell_pid, cell_global_id] = pidgid;
458 auto cell = SetupCell(
459 raw_cell, cell_global_id, cell_pid, STLVertexListHelper(vertices));
460
461 grid_ptr->cells.push_back(std::move(cell));
462 }
463
464 SetGridAttributes(
465 *grid_ptr,
466 static_cast<MeshAttributes>(mesh_info.mesh_attributes_),
467 {mesh_info.ortho_Nx_, mesh_info.ortho_Ny_, mesh_info.ortho_Nz_});
468
469 grid_ptr->SetGlobalVertexCount(mesh_info.num_global_vertices_);
470
471 ComputeAndPrintStats(*grid_ptr);
472
473 return grid_ptr;
474}
