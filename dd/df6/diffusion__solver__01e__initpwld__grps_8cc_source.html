566 DiffusionConvergenceTestNPT(KSP ksp, PetscInt n, PetscReal rnorm,
567  KSPConvergedReason* convergedReason, void *monitordestroy)
568 {
569  //======================================================= Compute rhs norm
570  Vec Rhs;
571  KSPGetRhs(ksp,&Rhs);
572  double rhs_norm;
573  VecNorm(Rhs,NORM_2,&rhs_norm);
574  if (rhs_norm < 1.0e-25)
575  rhs_norm = 1.0;
576 
577  //======================================================= Compute test criterion
578  double tol;
