126/**Private method for creating the spatial discretization method.*/
127chi_math::SDMPtr FieldFunctionGridBased::MakeSpatialDiscretization(
128 const chi::InputParameters& params)
129{
130 const auto& user_params = params.ParametersAtAssignment();
131 const auto& grid_ptr = chi_mesh::GetCurrentHandler().GetGrid();
132 const auto sdm_type = params.GetParamValue<std::string>("sdm_type");
133
134 typedef chi_math::spatial_discretization::FiniteVolume FV;
135 typedef chi_math::spatial_discretization::PieceWiseLinearContinuous PWLC;
136 typedef chi_math::spatial_discretization::PieceWiseLinearDiscontinuous PWLD;
137
138 if (sdm_type == "FiniteVolume") return FV::New(*grid_ptr);
139
140 chi_math::QuadratureOrder q_order = chi_math::QuadratureOrder::SECOND;
141 chi_math::CoordinateSystemType coordinate_system_type =
142 chi_math::CoordinateSystemType::CARTESIAN;
143
144 if (user_params.Has("quadrature_order"))
145 {
146 const uint32_t max_order =
147 static_cast<uint32_t>(chi_math::QuadratureOrder::FORTYTHIRD);
148 const uint32_t q_order_int =
149 params.GetParamValue<uint32_t>("quadrature_order");
150 ChiInvalidArgumentIf(q_order_int > max_order,
151 "Invalid quadature point order " +
152 std::to_string(q_order_int));
153 q_order = static_cast<chi_math::QuadratureOrder>(q_order_int);
154 }
155
156 if (user_params.Has("coordinate_system"))
157 {
158 const auto cs = params.GetParamValue<std::string>("coordinate_system");
159
160 if (cs == "cartesian")
161 coordinate_system_type = chi_math::CoordinateSystemType::CARTESIAN;
162 if (cs == "rz")
163 coordinate_system_type = chi_math::CoordinateSystemType::CYLINDRICAL;
164 if (cs == "1d_spherical")
165 coordinate_system_type = chi_math::CoordinateSystemType::SPHERICAL;
166 }
167
168 if (sdm_type == "PWLC")
169 return PWLC::New(*grid_ptr, q_order, coordinate_system_type);
170 if (sdm_type == "PWLD")
171 return PWLD::New(*grid_ptr, q_order, coordinate_system_type);
172
173 // If not returned by now
174 ChiInvalidArgument("Unsupported sdm_type \"" + sdm_type + "\"");
175}
