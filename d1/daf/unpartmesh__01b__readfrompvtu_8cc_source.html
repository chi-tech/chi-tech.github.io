27
28 //======================================== Attempt to open file
29 std::ifstream file;
30 file.open(options.file_name);
31 if (!file.is_open()) throw ErrorReadingFile(ReadFromVTU);
32 file.close();
33
34 //======================================== Read the file
35 mesh_options_ = options;
36 auto reader = vtkSmartPointer<vtkXMLPUnstructuredGridReader>::New();
37 reader->SetFileName(options.file_name.c_str());
38
39 if (not reader->CanReadFile(options.file_name.c_str()))
40 throw std::logic_error("Unable to read file-type with this routine");
41 reader->UpdateInformation();
42 reader->Update();
43
44 //======================================== Get all the grid blocks
45 // For vtu files this is very simple. The
46 // output of the reader is an UnstructuredGrid.
47 auto ugrid_main = vtkUGridPtr(reader->GetOutput());
48 std::vector<vtkUGridPtrAndName> grid_blocks = {{ugrid_main,""}};
49
50 //======================================== Get the main + bndry blocks
51 const int max_dimension = chi_mesh::FindHighestDimension(grid_blocks);
52 std::vector<vtkUGridPtrAndName> domain_grid_blocks =
53 chi_mesh::GetBlocksOfDesiredDimension(grid_blocks, max_dimension);
54 std::vector<vtkUGridPtrAndName> bndry_grid_blocks =
55 chi_mesh::GetBlocksOfDesiredDimension(grid_blocks, max_dimension-1);
56
57 //======================================== Process blocks
58 auto ugrid = chi_mesh::ConsolidateGridBlocks(domain_grid_blocks);
59
60 //======================================== Copy Data
61 CopyUGridCellsAndPoints(*ugrid, options.scale);
62
63 //======================================== Set material ids
64 const auto material_ids = chi_mesh::BuildCellMaterialIDsFromField(
65 ugrid, options.material_id_fieldname, options.file_name);
66 SetMaterialIDsFromList(material_ids);
67
68 //======================================== Always do this
69 chi_mesh::MeshAttributes dimension = NONE;
70 switch (max_dimension)
71 {
72 case 1: dimension = DIMENSION_1; break;
73 case 2: dimension = DIMENSION_2; break;
74 case 3: dimension = DIMENSION_3; break;
75 default: break;
76 }
77
78 attributes_ = dimension | UNSTRUCTURED;
79
80 ComputeCentroidsAndCheckQuality();
81 BuildMeshConnectivity();
82
83 chi::log.Log() << "Done reading PVTU file: " << options.file_name << ".";
