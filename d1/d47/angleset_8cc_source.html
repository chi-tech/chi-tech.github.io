45/**This function advances the work stages of an angleset.*/
46chi_mesh::sweep_management::AngleSetStatus
47chi_mesh::sweep_management::AngleSet::
48 AngleSetAdvance(chi_mesh::sweep_management::SweepChunk& sweep_chunk,
49 int angle_set_num,
50 const std::vector<size_t>& timing_tags,
51 chi_mesh::sweep_management::ExecutionPermission permission)
52{
53 typedef AngleSetStatus Status;
54
55 if (executed)
56 {
57 if (!sweep_buffer.DoneSending())
58 sweep_buffer.ClearDownstreamBuffers();
59 return AngleSetStatus::FINISHED;
60 }
61
62 //Check upstream data available
63 Status status = sweep_buffer.ReceiveUpstreamPsi(angle_set_num);
64
65 //Also check boundaries
66 for (auto& [bid,bndry] : ref_boundaries)
67 if (not bndry->CheckAnglesReadyStatus(angles,ref_subset))
68 {status = Status::RECEIVING; break;}
69
70 if (status == Status::RECEIVING) return status;
71 else if (status == Status::READY_TO_EXECUTE and
72 permission == ExecutionPermission::EXECUTE)
73 {
74 sweep_buffer.InitializeLocalAndDownstreamBuffers();
75
76 Chi::log.LogEvent(timing_tags[0], chi::ChiLog::EventType::EVENT_BEGIN);
77 sweep_chunk.Sweep(this); //Execute chunk
78 Chi::log.LogEvent(timing_tags[0], chi::ChiLog::EventType::EVENT_END);
79
80 //Send outgoing psi and clear local and receive buffers
81 sweep_buffer.SendDownstreamPsi(angle_set_num);
82 sweep_buffer.ClearLocalAndReceiveBuffers();
83
84 //Update boundary readiness
85 for (auto& [bid,bndry] : ref_boundaries)
86 bndry->UpdateAnglesReadyStatus(angles,ref_subset);
87
88 executed = true;
89 return AngleSetStatus::FINISHED;
90 }
91 else
92 return AngleSetStatus::READY_TO_EXECUTE;
93}
