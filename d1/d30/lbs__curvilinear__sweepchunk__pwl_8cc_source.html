127 const int ni_preloc_face_counter = preloc_face_counter;
128 const size_t as_num_angles = angle_set->angles.size();

130 {

132 preloc_face_counter = ni_preloc_face_counter;
133 const auto& angle_num = angle_set->angles[angle_set_index];
134 const auto& omega = groupset_.quadrature_->omegas_[angle_num];
135
136 const auto polar_level = map_polar_level_[angle_num];
137
138 const auto& fac_diamond_difference =
139 curvilinear_product_quadrature->GetDiamondDifferenceFactor()[angle_num];
140 const auto& fac_streaming_operator =
141 curvilinear_product_quadrature->GetStreamingOperatorFactor()[angle_num];

144 // ============================================ Gradient matrix
145 for (size_t i = 0; i < num_nodes; ++i)
146 for (size_t j = 0; j < num_nodes; ++j)
147 Amat_[i][j] = omega.Dot(G[i][j]) + fac_streaming_operator * Maux[i][j];

150 // ============================================ Source initialization

154 for (size_t i = 0; i < num_nodes; ++i)

158 for (int gsg = 0; gsg < gs_ss_size; ++gsg)

188 for (int gsg = 0; gsg < gs_ss_size; ++gsg)

190 }
191 }
192 }
