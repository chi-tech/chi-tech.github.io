125/**Applies diamond differencing on azimuthal directions.*/
126void SweepChunkPWLRZ::PostCellDirSweepCallback()
127{
128 const auto f0 = 1 / fac_diamond_difference_;
129 const auto f1 = f0 - 1;
130 for (size_t i = 0; i < cell_num_nodes_; ++i)
131 {
132 const auto ir = grid_fe_view_.MapDOFLocal(
133 *cell_, i, unknown_manager_, polar_level_, gs_gi_);
134 for (int gsg = 0; gsg < gs_ss_size_; ++gsg)
135 psi_sweep_[ir + gsg] = f0 * b_[gsg][i] - f1 * psi_sweep_[ir + gsg];
136 }
137}

139// ##################################################################
140/**Assembles the volumetric gradient term.*/
141void SweepChunkPWLRZ::KernelFEMRZVolumetricGradientTerm()
142{
143 const auto& G = *G_;
144 const auto& Maux = *Maux_;
145
146 for (int i = 0; i < cell_num_nodes_; ++i)
147 for (int j = 0; j < cell_num_nodes_; ++j)
148 {
149 Amat_[i][j] = omega_.Dot(G[i][j]) + fac_streaming_operator_ * Maux[i][j];
150 const auto jr = grid_fe_view_.MapDOFLocal(
151 *cell_, j, unknown_manager_, polar_level_, gs_gi_);
152 for (int gsg = 0; gsg < gs_ss_size_; ++gsg)
153 b_[gsg][i] +=
154 fac_streaming_operator_ * Maux[i][j] * psi_sweep_[jr + gsg];
155 }
156}

159/**Performs the integral over the surface of a face.*/
160void SweepChunkPWLRZ::KernelFEMRZUpwindSurfaceIntegrals()
161{
162 const size_t f = sweep_surface_status_info_.f;
163 if (sweep_surface_status_info_.on_boundary)
164 {
165 const auto& face_normal = cell_->faces_[f].normal_;
166 //------------------------------------------------------------------
167 // determine whether incoming direction is incident on the point
168 // of symmetry or on the axis of symmetry
169 // N.B.: a face is considered to be on the point/axis of symmetry
170 // if all are true:
171 // 1. the face normal is antiparallel to $\vec{e}_{d}$
172 // 2. all vertices of the face exhibit $v_{d} = 0$
173 // with $d = 2$ for 1D geometries and $d = 0$ for 2D geometries.
174 // Thanks to the verifications performed during initialisation,
175 // at this point it is necessary to confirm only the orientation.
176 const bool incident_on_symmetric_boundary =
177 (face_normal.Dot(normal_vector_boundary_) < -0.999999);
178 if (incident_on_symmetric_boundary) return;
179 }
180
181 const auto& M_surf_f = (*M_surf_)[f];
182 const double mu = face_mu_values_[f];
183 const size_t num_face_nodes = cell_mapping_->NumFaceNodes(f);
184 for (int fi = 0; fi < num_face_nodes; ++fi)
185 {
186 const int i = cell_mapping_->MapFaceNode(f, fi);
187 for (int fj = 0; fj < num_face_nodes; ++fj)
188 {
189 const int j = cell_mapping_->MapFaceNode(f, fj);
190
191 const double* psi = sweep_surface_status_info_.GetUpwindPsi(fj);
192
193 const double mu_Nij = -mu * M_surf_f[i][j];
194 Amat_[i][j] += mu_Nij;
195 for (int gsg = 0; gsg < gs_ss_size_; ++gsg)
196 b_[gsg][i] += psi[gsg] * mu_Nij;
197 } // for face node j
198 } // for face node i
199}
