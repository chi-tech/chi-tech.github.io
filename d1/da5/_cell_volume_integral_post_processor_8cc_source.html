80void CellVolumeIntegralPostProcessor::Execute(const Event& event_context)
81{
82 if (not initialized_) Initialize();
83
84 const auto* grid_field_function = GetGridBasedFieldFunction();
85
86 ChiLogicalErrorIf(not grid_field_function,
87 "Attempted to access invalid field"
88 "function");
89
90 const auto& ref_ff = *grid_field_function;
91 const auto& sdm = ref_ff.GetSpatialDiscretization();
92 const auto& grid = sdm.Grid();
93
94 const auto& uk_man = ref_ff.GetUnknownManager();
95 const auto uid = 0;
96 const auto cid = 0;
97
98 const auto field_data = ref_ff.GetGhostedFieldVector();
99
100 double local_integral = 0.0;
101 double local_volume = 0.0;
102 for (const uint64_t cell_local_id : cell_local_ids_)
103 {
104 const auto& cell = grid.local_cells[cell_local_id];
105 const auto& cell_mapping = sdm.GetCellMapping(cell);
106 const size_t num_nodes = cell_mapping.NumNodes();
107 const auto qp_data = cell_mapping.MakeVolumeQuadraturePointData();
108
109 std::vector<double> node_dof_values(num_nodes, 0.0);
110 for (size_t i = 0; i < num_nodes; ++i)
111 {
112 const int64_t imap = sdm.MapDOFLocal(cell, i, uk_man, uid, cid);
113 node_dof_values[i] = field_data[imap];
114 } // for i
115
116 for (const size_t qp : qp_data.QuadraturePointIndices())
117 {
118 // phi_h = sum_j b_j phi_j
119 double ff_value = 0.0;
120 for (size_t j = 0; j < num_nodes; ++j)
121 ff_value += qp_data.ShapeValue(j, qp) * node_dof_values[j];
122
123 local_integral += ff_value * qp_data.JxW(qp);
124 local_volume += qp_data.JxW(qp);
125 } // for qp
126 } // for cell-id
127
128 double globl_integral;
129 MPI_Allreduce(&local_integral, // sendbuf
130 &globl_integral, // recvbuf
131 1,
132 MPI_DOUBLE, // count + datatype
133 MPI_SUM, // operation
134 Chi::mpi.comm); // communicator
135 if (not compute_volume_average_) value_ = ParameterBlock("", globl_integral);
136 else
137 {
138 double globl_volume;
139 MPI_Allreduce(&local_volume, // sendbuf
140 &globl_volume, // recvbuf
141 1,
142 MPI_DOUBLE, // count + datatype
143 MPI_SUM, // operation
144 Chi::mpi.comm); // communicator
145
146 value_ = ParameterBlock("", globl_integral / globl_volume);
147 }
148
149 const int event_code = event_context.Code();
150 if (event_code == 32 /*SolverInitialized*/ or
151 event_code == 38 /*SolverAdvanced*/)
152 {
153 const auto& event_params = event_context.Parameters();
154
155 if (event_params.Has("timestep_index") and event_params.Has("time"))
156 {
157 const size_t index = event_params.GetParamValue<size_t>("timestep_index");
158 const double time = event_params.GetParamValue<double>("time");
159 TimeHistoryEntry entry{index, time, value_};
160 time_history_.push_back(std::move(entry));
161 }
162 }
163}
