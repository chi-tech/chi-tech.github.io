92/**Initialize function.*/
93void TransientSolver::Initialize()
94{
95 // Check size
96 if (lambdas_.size() != betas_.size())
97 throw std::logic_error(TextName() +
98 ": Number of precursors cannot be "
99 "deduced from precursor data because "
100 "the data lists are of different size.");
101
102 beta_ = std::accumulate(betas_.begin(), betas_.end(), /*init_val=*/0.0);
103
104 // ============================= Initializing linalg items
105 const auto& J = num_precursors_;
106 A_ = chi_math::DynamicMatrix<double>(J + 1, J + 1, 0.0);
107 I_ = A_;
108 I_.SetDiagonal(1.0);
109
110 x_t_ = chi_math::DynamicVector<double>(J + 1, 0.0);
111
112 // ============================= Assembling system
113 A_[0][0] = beta_ * (rho_ - 1.0) / gen_time_;
114 for (size_t j = 1; j <= J; ++j)
115 {
116 A_[0][j] = lambdas_[j - 1];
117 A_[j][j] = -lambdas_[j - 1];
118 A_[j][0] = betas_[j - 1] / gen_time_;
119 }
120
121 q_.resize(J + 1, 0.0);
122 q_[0] = source_strength_;
123
124 // ============================= Initializing x
125 // If there is a source and the reactivity is < 0 then
126 // there exists a unique solution.
127 if (source_strength_ > 0.0 and rho_ < 0.0)
128 {
129 const auto b_theta = -1.0 * q_;
130
131 x_t_ = A_.Inverse() * b_theta;
132 }
133 // Otherwise we initialize the system as a critical system with
134 // no source.
135 else
136 {
137 auto A_temp = A_;
138 auto b_temp = x_t_;
139 b_temp.Set(0.0);
140 b_temp[0] = 1.0;
141 for (auto& val : A_temp[0])
142 val = 0.0;
143 A_temp[0][0] = 1.0;
144
145 x_t_ = A_temp.Inverse() * b_temp;
146 }
147
148 Chi::log.Log() << "Final: " << x_t_.PrintStr();
149}

154/**Step function*/
155void TransientSolver::Step()
156{
157 A_[0][0] = beta_ * (rho_ - 1.0) / gen_time_;
158
159 if (time_integration_ == "implicit_euler" or
160 time_integration_ == "crank_nicolson")
161 {
162 double theta = 1.0;
163
164 if (time_integration_ == "implicit_euler") theta = 1.0;
165 else if (time_integration_ == "crank_nicolson")
166 theta = 0.5;
167
168 const double inv_tau = theta * dt_;
169
170 auto A_theta = I_ - inv_tau * A_;
171 auto b_theta = x_t_ + inv_tau * q_;
172
173 auto x_theta = A_theta.Inverse() * b_theta;
174
175 x_tp1_ = x_t_ + (1.0 / theta) * (x_theta - x_t_);
176 }
177 else if (time_integration_ == "explicit_euler")
178 {
179 x_tp1_ = x_t_ + dt_ * A_ * x_t_ + dt_ * q_;
180 }
181 else
182 ChiLogicalError("Unsupported time integration scheme.");
183
184 period_tph_ = dt_ / log(x_tp1_[0] / x_t_[0]);
185
186 if (period_tph_ > 0.0 and period_tph_ > 1.0e6) period_tph_ = 1.0e6;
187 if (period_tph_ < 0.0 and period_tph_ < -1.0e6) period_tph_ = -1.0e6;
188}
