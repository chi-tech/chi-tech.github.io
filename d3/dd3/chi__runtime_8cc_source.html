71// ############################################### Argument parser
72/**Parses input arguments.
73\param argc int Number of arguments supplied.
74\param argv char** Array of strings representing each argument.
75 */
76void Chi::run_time::ParseArguments(int argc, char** argv)
77{
78 bool input_file_found = false;
79 for (int i = 1; i < argc; i++)
80 {
81 std::string argument(argv[i]);
82
83 Chi::log.Log() << "Parsing argument " << i << " " << argument;
84
85 if (argument.find("-h") != std::string::npos or
86 argument.find("--help") != std::string::npos)
87 {
88 Chi::log.Log() << Chi::run_time::command_line_help_string_;
89 Chi::run_time::termination_posted_ = true;
90 }
91 else if (argument.find("--supress_beg_end_timelog") != std::string::npos)
92 {
93 Chi::run_time::supress_beg_end_timelog_ = true;
94 }
95 else if (argument.find("--allow_petsc_error_handler") != std::string::npos)
96 {
97 Chi::run_time::allow_petsc_error_handler_ = true;
98 }
99 else if (argument.find("--suppress_color") != std::string::npos)
100 {
101 Chi::run_time::suppress_color_ = true;
102 }
103 else if (argument.find("--dump-object-registry") != std::string::npos)
104 {
105 Chi::run_time::dump_registry_ = true;
106 Chi::run_time::termination_posted_ = true;
107 }
108 //================================================ No-graphics option
109 else if (argument.find("-b") != std::string::npos)
110 {
111 Chi::run_time::sim_option_interactive_ = false;
112 } //-b
113 //================================================ Verbosity
114 else if (argument.find("-v") != std::string::npos)
115 {
116 if ((i + 1) >= argc)
117 {
118 std::cerr << "Invalid option used with command line argument "
119 "-v. Options are 0,1 or 2."
120 << std::endl;
121 Chi::Exit(EXIT_FAILURE);
122 }
123 else
124 {
125 std::string v_option(argv[i + 1]);
126 try
127 {
128 int level = std::stoi(v_option);
129 Chi::log.SetVerbosity(level);
130 }
131 catch (const std::invalid_argument& e)
132 {
133 std::cerr << "Invalid option used with command line argument "
134 "-v. Options are 0,1 or 2."
135 << std::endl;
136 Chi::Exit(EXIT_FAILURE);
137 }
138 }
139
140 } //-v
141 else if ((argument.find('=') == std::string::npos) and (!input_file_found))
142 {
143 Chi::run_time::input_file_name_ = argument;
144 input_file_found = true;
145 Chi::run_time::sim_option_interactive_ = false;
146 } // no =
147 else if (argument.find('=') != std::string::npos)
148 {
149 Chi::console.GetCommandBuffer().push_back(argument);
150 } //=
151
152 } // for argument
153
154 if (Chi::run_time::dump_registry_)
155 {
156 ChiObjectMaker::GetInstance().DumpRegister();
157 Chi::console.DumpRegister();
158 }
159}

161// ############################################### Initialize ChiTech
162/**Initializes all necessary items for ChiTech.
163\param argc int Number of arguments supplied.
164\param argv char** Array of strings representing each argument.
165 */
166int Chi::Initialize(int argc,
167 char** argv,
168 MPI_Comm communicator)
169{
170 int location_id = 0, number_processes = 1;
171
172 MPI_Init(&argc, &argv); /* starts MPI */
173 MPI_Comm_rank(communicator, &location_id); /* get cur process id */
174 MPI_Comm_size(communicator, &number_processes); /* get num of processes */
175
176 mpi.SetCommunicator(communicator);
177 mpi.SetLocationID(location_id);
178 mpi.SetProcessCount(number_processes);
179
180 Chi::console.LoadRegisteredLuaItems();
181 Chi::console.PostMPIInfo(location_id, number_processes);
182
183 run_time::ParseArguments(argc, argv);
184
185 run_time::InitPetSc(argc, argv);
186
187 return 0;
188}

190/**Initializes PetSc for use by all entities.*/
191int Chi::run_time::InitPetSc(int argc, char** argv)
192{
193 PETSC_COMM_WORLD = mpi.comm;
194 PetscOptionsInsertString(nullptr, "-error_output_stderr");
195 if (not Chi::run_time::allow_petsc_error_handler_)
196 PetscOptionsInsertString(nullptr, "-no_signal_handler");
197
198 PetscCall(PetscInitialize(&argc, &argv, nullptr, nullptr));
199
200 return 0;
201}

223// ############################################### Interactive interface
224/**Runs the interactive chitech engine*/
225int Chi::RunInteractive(int argc, char** argv)
226{
227 if (not Chi::run_time::supress_beg_end_timelog_)
228 {
229 Chi::log.Log() << chi::ChiTimer::GetLocalDateTimeString()
230 << " Running ChiTech in interactive-mode with "
231 << Chi::mpi.process_count << " processes.";
232
233 Chi::log.Log() << "ChiTech version " << GetVersionStr();
234 }
235
236 Chi::log.Log() << "ChiTech number of arguments supplied: " << argc - 1;
237
238 Chi::log.LogAll();
239
240 Chi::console.FlushConsole();
241
242 const auto& input_fname = Chi::run_time::input_file_name_;
243
244 if (not input_fname.empty())
245 {
246 try
247 {
248 Chi::console.ExecuteFile(input_fname, argc, argv);
249 }
250 catch (const std::exception& excp)
251 {
252 Chi::log.LogAllError() << excp.what();
253 // No quitting if file execution fails
254 }
255 }
256
257 Chi::console.RunConsoleLoop();
258
259 if (not Chi::run_time::supress_beg_end_timelog_)
260 {
261 Chi::log.Log() << "Final program time " << program_timer.GetTimeString();
262 Chi::log.Log() << chi::ChiTimer::GetLocalDateTimeString()
263 << " ChiTech finished execution.";
264 }
265
266 return 0;
267}

269// ############################################### Batch interface
270/**Runs ChiTech in pure batch mode. Start then finish.*/
271int Chi::RunBatch(int argc, char** argv)
272{
273 if (not Chi::run_time::supress_beg_end_timelog_)
274 {
275 Chi::log.Log() << chi::ChiTimer::GetLocalDateTimeString()
276 << " Running ChiTech in batch-mode with "
277 << Chi::mpi.process_count << " processes.";
278
279 Chi::log.Log() << "ChiTech version " << GetVersionStr();
280 }
281
282 Chi::log.Log() << "ChiTech number of arguments supplied: " << argc - 1;
283
284 if (argc <= 1) Chi::log.Log() << Chi::run_time::command_line_help_string_;
285 Chi::console.FlushConsole();
286
287#ifndef NDEBUG
288 Chi::log.Log() << "Waiting...";
289 if (Chi::mpi.location_id == 0)
290 for (int k = 0; k < 2; ++k)
291 {
292 usleep(1000000);
293 Chi::log.Log() << k;
294 }
295
296 mpi.Barrier();
297#endif
298
299 const auto& input_fname = Chi::run_time::input_file_name_;
300 int error_code = 0;
301
302 if ((not input_fname.empty()) and (not Chi::run_time::termination_posted_))
303 {
304 try
305 {
306 error_code = Chi::console.ExecuteFile(input_fname, argc, argv);
307 }
308 catch (const std::exception& excp)
309 {
310 Chi::log.LogAllError() << excp.what();
311 Chi::Exit(EXIT_FAILURE);
312 }
313 }
314
315 if (not Chi::run_time::supress_beg_end_timelog_)
316 {
317 Chi::log.Log() << "\nFinal program time " << program_timer.GetTimeString();
318 Chi::log.Log() << chi::ChiTimer::GetLocalDateTimeString()
319 << " ChiTech finished execution of "
320 << Chi::run_time::input_file_name_;
321 }
322
323 return error_code;
324}
