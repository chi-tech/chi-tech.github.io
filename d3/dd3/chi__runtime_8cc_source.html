57/**Parses input arguments.
58\param argc int Number of arguments supplied.
59\param argv char** Array of strings representing each argument.
60 */
61void chi::run_time::ParseArguments(int argc, char** argv)
62{
63 bool input_file_found = false;
64 for (int i=1; i<argc; i++)
65 {
66 std::string argument(argv[i]);
67
68 chi::log.Log() << "Parsing argument " << i << " " << argument;
69
70 if (argument.find("-h")!=std::string::npos)
71 {
72 chi::log.Log()
73 << "\nUsage: exe inputfile [options values]\n"
74 << "\n"
75 << " -v Level of verbosity. Default 0. Can be either 0, 1 or 2.\n"
76 << " a=b Executes argument as a lua string. i.e. x=2 or y=[[\"string\"]]\n"
77 << " -allow_petsc_error_handler Allow petsc error handler.\n\n\n";
78
79 chi::log.Log() << "PETSc options:";
80 chi::run_time::termination_posted = true;
81 }
82 else if (argument.find("-allow_petsc_error_handler")!=std::string::npos)
83 {
84 chi::run_time::allow_petsc_error_handler = true;
85 }
86 //================================================ No-graphics option
87 else if (argument.find("-b")!=std::string::npos)
88 {
89 chi::run_time::sim_option_interactive = false;
90 }//-b
91 //================================================ Verbosity
92 else if (argument.find("-v") != std::string::npos)
93 {
94 if ((i+1) >= argc)
95 {
96 std::cerr << "Invalid option used with command line argument "
97 "-v. Options are 0,1 or 2." << std::endl;
98 chi::Exit(EXIT_FAILURE);
99 }
100 else
101 {
102 std::string v_option(argv[i+1]);
103 try {
104 int level = std::stoi(v_option);
105 chi::log.SetVerbosity(level);
106 }
107 catch (const std::invalid_argument& e)
108 {
109 std::cerr << "Invalid option used with command line argument "
110 "-v. Options are 0,1 or 2." << std::endl;
111 chi::Exit(EXIT_FAILURE);
112 }
113 }
114
115 }//-v
116 else if ((argument.find('=') == std::string::npos) && (!input_file_found) )
117 {
118 chi::run_time::input_file_name = argument;
119 input_file_found = true;
120 chi::run_time::sim_option_interactive = false;
121 }//no =
122 else if (argument.find('=') != std::string::npos)
123 {
124 chi::console.command_buffer.push_back(argument);
125 }//=
126
127 }//for argument
128}

130//############################################### Initialize ChiTech
131/**Initializes all necessary items for ChiTech.
132\param argc int Number of arguments supplied.
133\param argv char** Array of strings representing each argument.
134 */
135int chi::Initialize(int argc, char** argv)
136{
137 int location_id = 0, number_processes = 1;
138
139 MPI_Init (&argc, &argv); /* starts MPI */
140 MPI_Comm_rank (MPI_COMM_WORLD, &location_id); /* get current process id */
141 MPI_Comm_size (MPI_COMM_WORLD, &number_processes); /* get number of processes */
142
143// chi::mpi.SetLocationID(location_id);
144// chi::mpi.SetProcessCount(number_processes);
145
146 mpi.SetLocationID(location_id);
147 mpi.SetProcessCount(number_processes);
148
149 chi::console.PostMPIInfo(location_id, number_processes);
150
151 run_time::ParseArguments(argc, argv);
152
153 run_time::InitPetSc(argc,argv);
154
155 return 0;
156}
