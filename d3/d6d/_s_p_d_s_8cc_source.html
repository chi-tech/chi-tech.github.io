52 const chi_mesh::Vector3& omega,
53 std::set<int>& location_dependencies,
54 std::set<int>& location_successors,
55 std::vector<std::set<std::pair<int, double>>>& cell_successors)
56{
57 constexpr double tolerance = 1.0e-16;
58

60 constexpr auto FOINCOMING = FaceOrientation::INCOMING;
61 constexpr auto FOOUTGOING = FaceOrientation::OUTGOING;
62
63 cell_face_orientations_.assign(grid_.local_cells.size(), {});
64 for (auto& cell : grid_.local_cells)
65 cell_face_orientations_[cell.local_id_].assign(cell.faces_.size(),
66 FOPARALLEL);
67
68 for (auto& cell : grid_.local_cells)
69 {
70 size_t f = 0;
71 for (auto& face : cell.faces_)
72 {
73 //======================================= Determine if the face
74 // is incident
75 FaceOrientation orientation = FOPARALLEL;
76 const double mu = omega.Dot(face.normal_);
77
78 bool owns_face = true;
79 if (face.has_neighbor_ and cell.global_id_ > face.neighbor_id_ and
80 grid_.IsCellLocal(face.neighbor_id_))
81 owns_face = false;
82
83 if (owns_face)
84 {
85 // clang-format off
86 if (mu > tolerance) orientation = FOOUTGOING;
87 else if (mu < tolerance) orientation = FOINCOMING;
88
89 cell_face_orientations_[cell.local_id_][f] = orientation;
90
91 if (face.has_neighbor_ and grid_.IsCellLocal(face.neighbor_id_))
92 {
93 const auto& adj_cell = grid_.cells[face.neighbor_id_];
94 const auto ass_face = face.GetNeighborAssociatedFace(grid_);
95 auto& adj_face_ori =
96 cell_face_orientations_[adj_cell.local_id_][ass_face];
97
98 switch (orientation)
99 {
100 case FOPARALLEL: adj_face_ori = FOPARALLEL; break;
101 case FOINCOMING: adj_face_ori = FOOUTGOING; break;
102 case FOOUTGOING: adj_face_ori = FOINCOMING; break;
103 }
104 }
105 // clang-format on
106 } // if face owned
107 else if (face.has_neighbor_ and not grid_.IsCellLocal(face.neighbor_id_))
108 {
109 const auto& adj_cell = grid_.cells[face.neighbor_id_];
110 const auto ass_face = face.GetNeighborAssociatedFace(grid_);
111 const auto& adj_face = adj_cell.faces_[ass_face];
112
113 auto& cur_face_ori = cell_face_orientations_[cell.local_id_][f];
114
115 const double adj_mu = omega.Dot(adj_face.normal_);
116 if (adj_mu > tolerance) orientation = FOOUTGOING;
117 else if (adj_mu < tolerance)
118 orientation = FOINCOMING;
119
120 switch (orientation)
121 {
122 case FOPARALLEL:
123 cur_face_ori = FOPARALLEL;
124 break;
125 case FOINCOMING:
126 cur_face_ori = FOOUTGOING;
127 break;
128 case FOOUTGOING:
129 cur_face_ori = FOINCOMING;
130 break;
131 }
132 } // if not face owned locally at all
133
134 ++f;
135 } // for face
136 }
137
138 //============================================= Make directed connections
139 for (auto& cell : grid_.local_cells)
140 {
141 const uint64_t c = cell.local_id_;
142 size_t f = 0;
143 for (auto& face : cell.faces_)
144 {
145 const double mu = omega.Dot(face.normal_);
146 //======================================= If outgoing determine if
147 // it is to a local cell
148 if (cell_face_orientations_[cell.local_id_][f] == FOOUTGOING)
149 {
150 //================================ If it is a cell and not bndry
151 if (face.has_neighbor_)
152 {
153 //========================= If it is in the current location
154 if (face.IsNeighborLocal(grid_))
155 {
156 double weight = mu * face.ComputeFaceArea(grid_);
157 cell_successors[c].insert(
158 std::make_pair(face.GetNeighborLocalID(grid_), weight));
159 }
160 else
161 location_successors.insert(face.GetNeighborPartitionID(grid_));
162 }
163 }
164 //======================================= If not outgoing determine
165 // what it is dependent on
166 else
167 {
168 //================================if it is a cell and not bndry
169 if (face.has_neighbor_ and not face.IsNeighborLocal(grid_))
170 location_dependencies.insert(face.GetNeighborPartitionID(grid_));
171 }
172 ++f;
173 } // for face
174 } // for cell
175}
