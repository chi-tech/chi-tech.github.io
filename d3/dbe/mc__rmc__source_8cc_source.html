324 /**Executes a source sampling for the residual source.*/
325 chi_montecarlon::Particle chi_montecarlon::ResidualSource::
326  DirectSampling(chi_montecarlon::RandomNumberGenerator* rng)
327 {
328  int num_local_cells = grid->local_cell_glob_indices.size();
329  int lc = 0;
330  double sampling_normalization;
331  if (sample_uniformly) {
332  lc = std::floor( rng->Rand()*(num_local_cells) );
333  sampling_normalization = (cell_residuals[lc]/total_residual)*
334  num_local_cells*total_residual;
335  }
336  else {
337  lc = residual_sampler->Sample(rng->Rand());
338  sampling_normalization = total_residual;
339  }
340 
341  int cell_glob_index = grid->local_cell_glob_indices[lc];
342  auto cell = grid->cells[cell_glob_index];
343 
344  chi_montecarlon::Particle new_particle;
345 
346  //$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$ SLAB
347  if (cell->Type() == chi_mesh::CellType::SLAB)
348  {
349  auto slab_cell = (chi_mesh::CellSlab*)cell;
350 
351  int v0i = slab_cell->vertex_ids[0];
352  int v1i = slab_cell->vertex_ids[1];
353 
354  chi_mesh::Vertex v0 = *grid->nodes[v0i];
355  chi_mesh::Vertex v1 = *grid->nodes[v1i];
356 
357  double rn = rng->Rand();
358 
359  double center_res = std::fabs(cell_interior_residual[lc]);
360  double surfL_res = std::fabs(cell_surface_residualL[lc]);
361  double surfR_res = std::fabs(cell_surface_residualR[lc]);
362 
363  double cell_R = center_res + surfL_res + surfR_res;
364 
365  if (cell_R < 1.0e-16) {
366  new_particle.alive = false;
367  return new_particle;
368  }
369 
370 
371  //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% CENTER
372  if (rn < (center_res/cell_R))
373  {
374  //====================================== Sample direction
375  double costheta = 2.0*rng->Rand() - 1.0;
376  double theta = acos(costheta);
377  double varphi = rng->Rand()*2.0*M_PI;
378 
379  chi_mesh::Vector ref_dir;
380  ref_dir.x = sin(theta)*cos(varphi);
381  ref_dir.y = sin(theta)*sin(varphi);
382  ref_dir.z = cos(theta);
383 
384  //====================================== Sample position
385  double w = rng->Rand();
386  new_particle.pos = v0*w + v1*(1.0-w);
387 
388  //====================================== Set quantities
389  new_particle.dir = ref_dir;
390 
391  new_particle.egrp = 0;
392  new_particle.w = sampling_normalization*
393  ((cell_interior_residual[lc]<0.0) ? -1.0:1.0);
394 // new_particle.w = std::fabs(sampling_normalization);
395  new_particle.cur_cell_ind = cell_glob_index;
396 // new_particle.alive = false;
397  particles_C++;
398 
399  return new_particle;
400  }
401  //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% LEFT
402  else if (rn < ((center_res + surfL_res)/cell_R))
403 // if (rn < (0.5))
404  {
405  //====================================== Sample direction
406  double costheta = rng->Rand();
407  double theta = acos(sqrt(costheta));
408  double varphi = rng->Rand()*2.0*M_PI;
409 
410  chi_mesh::Vector ref_dir;
411  ref_dir.x = sin(theta)*cos(varphi);
412  ref_dir.y = sin(theta)*sin(varphi);
413  ref_dir.z = cos(theta);
414 
415  //====================================== Set position
416  new_particle.pos = v0+chi_mesh::Vector(0,0,1.0e-8);
417 
418  //====================================== Set quantities
419  new_particle.dir = ref_dir;
420 
421  new_particle.egrp = 0;
422  new_particle.w = sampling_normalization*
423  ((cell_surface_residualL[lc]<0.0) ? -1.0:1.0);
424 // new_particle.w = std::fabs(sampling_normalization);
425  new_particle.cur_cell_ind = cell_glob_index;
426 // new_particle.alive = false;
427  particles_L++;
428  weights_L += new_particle.w;
429 
430  return new_particle;
431  }
432  //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% RIGHT
433  else
434  {
435  //====================================== Sample direction
436  double costheta = rng->Rand();
437  double theta = acos(-sqrt(costheta));
438  double varphi = rng->Rand()*2.0*M_PI;
439 
440  chi_mesh::Vector ref_dir;
441  ref_dir.x = sin(theta)*cos(varphi);
442  ref_dir.y = sin(theta)*sin(varphi);
443  ref_dir.z = cos(theta);
444 
445  //====================================== Set position
446  new_particle.pos = v1-chi_mesh::Vector(0,0,1.0e-8);
447 
448  //====================================== Set quantities
449  new_particle.dir = ref_dir;
450 
451  new_particle.egrp = 0;
452  new_particle.w = sampling_normalization*
453  ((cell_surface_residualR[lc]<0.0) ? -1.0:1.0);
454 // new_particle.w = std::fabs(sampling_normalization);
455  new_particle.cur_cell_ind = cell_glob_index;
456 // new_particle.alive = false;
457  particles_R++;
458  weights_R += new_particle.w;
459 
460  return new_particle;
461  }
462 
463 
464  }
465 
466  return new_particle;
467 }
