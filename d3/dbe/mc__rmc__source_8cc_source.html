99  cell_surface_residualL.resize(num_local_cells,0.0);
100  cell_surface_residualR.resize(num_local_cells,0.0);
101  for (int lc=0; lc<num_local_cells; lc++)
102  {
103  int cell_glob_index = grid->local_cell_glob_indices[lc];
104  auto cell = grid->cells[cell_glob_index];
105 
106  chi_physics::Material* cell_mat =
107  chi_physics_handler.material_stack[cell->material_id];
108 
109  TrXS* xs = nullptr;
110  TrQ* source = nullptr;
111 
112  for (int p=0; p<cell_mat->properties.size(); p++)
113  {
114  if (typeid(*cell_mat->properties[p]) == typeid(TrXS))
115  xs = (TrXS*)cell_mat->properties[p];
116 
117  if (typeid(*cell_mat->properties[p]) == typeid(TrQ))
118  source = (TrQ*)cell_mat->properties[p];
119  }
120 
121  //$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$ SLAB
122  if (typeid(*cell) == typeid(chi_mesh::CellSlab))
123  {
124  chi_log.Log(LOG_0VERBOSE_1) << "Cell " << cell_glob_index;
125  auto slab_cell = (chi_mesh::CellSlab*)cell;
126  auto cell_fe_view = (SlabFEView*)resid_sdm_pwl->MapFeView(cell_glob_index);
127 
128  //==================================== Creating current cell dof-mapping
129  std::vector<int> dofs_to_map(cell_fe_view->dofs);
130  std::vector<int> cells_to_map;
131  std::vector<int> cur_cell_mapping;
132  for (int i=0; i<cell_fe_view->dofs; i++)
133  dofs_to_map[i] = i;
134  cells_to_map.resize(cell_fe_view->dofs,cell_glob_index);
135 
136  chi_log.Log(LOG_0VERBOSE_1) << "Mapping cell dofs ";
137  ff_interp.CreatePWLDMapping(
138  resid_ff->num_grps,resid_ff->num_moms,0,0,
139  dofs_to_map,cells_to_map,
140  *resid_ff->local_cell_dof_array_address,&cur_cell_mapping);
141 
142  chi_log.Log(LOG_0VERBOSE_1)
143  << "dof 0 phi=" << field[cur_cell_mapping[0]] << "\n"
144  << "dof 1 phi=" << field[cur_cell_mapping[1]];
145

147  chi_log.Log(LOG_0VERBOSE_1) << "Mapping adj cell dofs";
148  std::vector<std::vector<int>> adj_cell_mapping_f;
149 
150  int num_faces = 2;
151  for (int f=0; f<num_faces; f++)
152  {
153  int adj_cell_index = slab_cell->edges[f];
154 
155  std::vector<int> adj_mapping;
156 
157  if (adj_cell_index >= 0)
158  {
159  auto adj_cell = (chi_mesh::CellSlab*)grid->cells[adj_cell_index];
160 
161  int adj_num_dofs = 2;
162  for (int i=0; i<adj_num_dofs; i++)
163  {
164  dofs_to_map[i] = i;
165  cells_to_map[i] = adj_cell_index;
166  }
167 
168  ff_interp.CreatePWLDMapping(
169  resid_ff->num_grps,resid_ff->num_moms,0,0,
170  dofs_to_map,cells_to_map,
171  *resid_ff->local_cell_dof_array_address,&adj_mapping);
172 
173  chi_log.Log(LOG_0VERBOSE_1)
174  << "adj_cell " << adj_cell_index << "\n"
175  << "dof 0 phi=" << field[adj_mapping[0]] << "\n"
176  << "dof 1 phi=" << field[adj_mapping[1]];
177  }//if not bndry
178 
179  adj_cell_mapping_f.push_back(adj_mapping);
180  }//for faces
181 
182  //==================================== Computing interior residual
183  chi_log.Log(LOG_0VERBOSE_1) << "Computing cell interior residual";
184  double resid = 0.0;
185  for (int i=0; i<cell_fe_view->dofs; i++)
186  {
187  double phi = field[cur_cell_mapping[i]];
188  cell_interior_residual[lc] += source->source_value_g[0]*
189  cell_fe_view->IntV_shapeI[i];
190  cell_interior_residual[lc] -= xs->sigma_rg[0]*
191  cell_fe_view->IntV_shapeI[i]*phi;
192  }
193  chi_log.Log(LOG_0VERBOSE_1) << "Sigma a=" << xs->sigma_rg[0];
194 
195 
196  //==================================== Computing surface residuals
197  chi_log.Log(LOG_0VERBOSE_1) << "Computing cell surface residual";
198  for (int f=0; f<num_faces; f++)
199  {
200  int fmap = 0;
201  if (f == 0)
202  fmap = 1;
203 
204  int num_face_verts = 1;
205  for (int fi=0; fi<num_face_verts; fi++)
206  {
207  int i = f;
208  int imap = fmap;
209 
210  double phi = 0.5*field[cur_cell_mapping[0]] +
211  0.5*field[cur_cell_mapping[1]];
212  double phi_adj = 0.0;
213  if (slab_cell->edges[f] >= 0)
214  phi_adj = 0.5*field[adj_cell_mapping_f[f][0]] +
215  0.5*field[adj_cell_mapping_f[f][1]];
216  if (slab_cell->edges[f] == -1)
217  phi_adj = 0.5;
218 
219  chi_log.Log(LOG_0VERBOSE_1)
220  << "Face " << f << "\n"
221  << "phi=" << phi << " adj_phi="<<phi_adj;
222 
223  if (f == 0)
224  cell_surface_residualL[lc] -= (phi - phi_adj);
225  else
226  cell_surface_residualR[lc] -= (phi_adj - phi);
227 
228  }//for face verts
229  }//for face
230 
231  chi_log.Log(LOG_0VERBOSE_1)
232  << "Cell residuals "
233  << cell_interior_residual[lc] << " "
234  << cell_surface_residualL[lc] << " "
235  << cell_surface_residualR[lc];
236 
237  cell_residuals[lc] =
238  std::fabs(cell_interior_residual[lc]) +
239  std::fabs(cell_surface_residualL[lc]);
240 
241  if (lc == (grid->local_cell_glob_indices.size()-1))
242  cell_residuals[lc] += std::fabs(cell_surface_residualR[lc]);
243  }//slab
244  else
245  {
246  chi_log.Log(LOG_ALLERROR)
247  << "chi_montecarlon: Unsupported cell type encountered in "
248  << "call to ResidualSource::Initialize.";
249  exit(EXIT_FAILURE);
250  }
251  }//for local cells
252 
253  chi_log.Log(LOG_0) << "Computing total residual";
254  double total_resid = 0.0;
255  for (int lc=0; lc<num_local_cells; lc++)
256  total_resid += cell_residuals[lc];
257 
258  total_residual = total_resid;
259 
260  chi_log.Log(LOG_0) << "Total residual = " << total_residual;
261 
262  cell_residual_cdf.resize(num_local_cells,0.0);
263  double intgl = 0.0;
264  for (int lc=0; lc<num_local_cells; lc++)
265  {
266  intgl += cell_residuals[lc];
267  cell_residual_cdf[lc] = intgl/total_resid;
268  }
269 
270  residual_sampler = new chi_math::CDFSampler(cell_residual_cdf);
271 }

281  auto cell = grid->cells[cell_glob_index];
282 
283  chi_montecarlon::Particle new_particle;
284

286  if (lc == grid->local_cell_glob_indices.size())

292 
293  //$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$ SLAB

297

299  int v1i = slab_cell->v_indices[1];
300

302  chi_mesh::Vertex v1 = *grid->nodes[v1i];
303 
304  double rn = rng->Rand();
305 
306  double center_res = std::fabs(cell_interior_residual[lc]);
307  double surfL_res = std::fabs(cell_surface_residualL[lc]);
308  double surfR_res = std::fabs(cell_surface_residualR[lc]);
309 
310 
311  if (!sample_rite)
312  {
313  double cell_R = center_res + surfL_res;
314 
315  //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% CENTER
316  if (rn < center_res/cell_R)
317  {
318  //====================================== Sample direction

320  double theta = acos(costheta);
321  double varphi = rng->Rand()*2.0*M_PI;
322 
323  chi_mesh::Vector ref_dir;
324  ref_dir.x = sin(theta)*cos(varphi);
325  ref_dir.y = sin(theta)*sin(varphi);
326  ref_dir.z = cos(theta);
327 
328  //====================================== Sample position
329  double w = rng->Rand();
330  new_particle.pos = v0*w + v1*(1.0-w);
331 
332  //====================================== Set quantities
333  new_particle.dir = ref_dir;
334 
335  new_particle.egrp = 0;
336  new_particle.w = (cell_R/total_residual)*
337  cell_interior_residual[lc]/
