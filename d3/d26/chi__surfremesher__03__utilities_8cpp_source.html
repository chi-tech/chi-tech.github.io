309 void CHI_SURFACEREMESHER::FindCircumCircle(CHI_PATCH *patch, CST_FACE *tri, float *center, float *radius)
310 {
311  //===================================================== Get vertices
312  float* a = patch->Pstar.GetItem(tri->vertex[0]);
313  float* b = patch->Pstar.GetItem(tri->vertex[1]);
314  float* c = patch->Pstar.GetItem(tri->vertex[2]);
315  bool line1Vertical=false;
316  bool line2Vertical=false;
317 
318  //===================================================== Slope 1
319  float m1 = 0.0;
320  if (fabs(b[0]-a[0])>this->precision)
321  {
322  float mBA = (b[1]-a[1])/(b[0]-a[0]);
323  if (fabs(mBA)>this->precision)
324  {
325  m1 = -1.0/mBA;
326  } else
327  {
328  line1Vertical = true;
329  }
330 
331  }
332 
333  float x1 = (b[0]+a[0])/2.0;
334  float y1 = (b[1]+a[1])/2.0;
335 
336  //y = m*x +c
337  //c1 = y1-m1*x1
338  //yc = m1*xc +y1-m1*x1
339 
340  //===================================================== Slope 2
341  float m2 = 0.0;
342  if (fabs(c[0]-b[0])>this->precision)
343  {
344  float mCB = (c[1]-b[1])/(c[0]-b[0]);
345  if (fabs(mCB)>this->precision)
346  {
347  m2 = -1.0/mCB;
348  }
349  else
350  {
351  line2Vertical = true;
352  }
353  }
354 
355  float x2 = (c[0]+b[0])/2.0;
356  float y2 = (c[1]+b[1])/2.0;
357 
358  //y = m*x +c
359  //c2 = y2-m2*x2
360  //yc = m2*xc +y2-m2*x2
361 
362  //m1*xc +y1-m1*x1 = m2*xc +y2-m2*x2
363  //m1*xc -m2*xc = m1*x1-y1 - (m2*x2 - y2)
364 
365  if (line1Vertical)
366  {
367  center[0] = x1;
368  center[1] = m2*(center[0]-x2)+y2;
369  center[2] = 0.0;
370  }
371  else if (line2Vertical)
372  {
373  center[0] = x2;
374  center[1] = m1*(center[0]-x1)+y1;
375  center[2] = 0.0;
376  }
377  else
378  {
379  center[0] = ( (m1*x1 - y1) - (m2*x2-y2) )/(m1-m2);
380  center[1] = m2*(center[0]-x2)+y2;
381  center[2] = 0.0;
382  }
383 
384 
385  *radius = sqrt( (center[0]-b[0])*(center[0]-b[0]) + (center[1]-b[1])*(center[1]-b[1]) );
386 }

543 bool CHI_SURFACEREMESHER::FindIntersectingSimplices(CHI_PATCH *patch, int va, int vb, int vc)
544 {
545 
546  float* v1 = patch->Pstar.GetItem(va); if (v1==NULL){return false;}
547  float* v2 = patch->Pstar.GetItem(vb); if (v2==NULL){return false;}
548  float* v3 = patch->Pstar.GetItem(vc); if (v3==NULL){return false;}
549 
550  for (int e=0;e<patch->edgeLoops.itemCount;e++)
551  {
552  auto curLoop = patch->edgeLoops.GetItem(e);
553  for (int k=0;k<curLoop->edges.itemCount;k++)
554  {
555  auto curEdge = curLoop->edges.GetItem(k);
556  if (curEdge!=NULL)
557  {
558  //printf("Checking edge %d %d against triangle %d %d %d ...",curEdge[0],curEdge[1],va,vb,vc);
559  float* b = patch->Pstar.GetItem(curEdge[0]);
560  float* c = patch->Pstar.GetItem(curEdge[1]);
561 
562  float ori1 = Orient2D(v1,b,c)*Orient2D(v2,b,c);
563  float ori2 = Orient2D(v2,b,c)*Orient2D(v3,b,c);
564  float ori3 = Orient2D(v3,b,c)*Orient2D(v1,b,c);
565 
566  Eigen::Vector3f V1(v1[0],v1[1],0.0);
567  Eigen::Vector3f V2(v2[0],v2[1],0.0);
568  Eigen::Vector3f V3(v3[0],v3[1],0.0);
569 
570  Eigen::Vector3f B(b[0],b[1],0.0);
571  Eigen::Vector3f C(c[0],c[1],0.0);
572 
573  Eigen::Vector3f BC = C-B;
574  Eigen::Vector3f V12 = V2-V1;
575  Eigen::Vector3f V23 = V3-V2;
576  Eigen::Vector3f V31 = V1-V3;
577 
578  Eigen::Matrix<float,2,2> A;
579  Eigen::Vector2f rhs;
580  Eigen::Vector2f x;
581 
582  Eigen::Vector3f A12 = V12;
583 
584  //========================================= Triangle edge 1
585  A12 = V12;
586  if ( fabs(BC.cross(A12)(2)) > this->precision)
587  {
588  A(0,0) = BC(0); A(0,1) = -1*A12(0);
589  A(1,0) = BC(1); A(1,1) = -1*A12(1);
590  rhs(0) = V1(0)-B(0);
591  rhs(1) = V1(1)-B(1);
592 
593  x = A.colPivHouseholderQr().solve(rhs);
594  if ( ( (x(0)>(0.0+this->precision)) && (x(0)<(1.0-this->precision)) ) &&
595  ( (x(1)>(0.0+this->precision)) && (x(1)<(1.0-this->precision)) ) )
596  {
597 
598  //printf("intersecting V12, %6.3f %6.3f %.8f\n",x(0),x(1),fabs(BC.cross(A12)(2)));
599  return true;
600  }
601  }
602 
603  //========================================= Triangle edge 2
604  A12 = V23;
605  if ( fabs(BC.cross(A12)(2)) > this->precision)
606  {
607  A(0,0) = BC(0); A(0,1) = -1*A12(0);
608  A(1,0) = BC(1); A(1,1) = -1*A12(1);
609  rhs(0) = V2(0)-B(0);
610  rhs(1) = V2(1)-B(1);
611 
612  x = A.colPivHouseholderQr().solve(rhs);
613  if ( ( (x(0)>(0.0+this->precision)) && (x(0)<(1.0-this->precision)) ) &&
