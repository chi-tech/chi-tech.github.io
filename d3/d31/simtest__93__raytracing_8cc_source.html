50 for (int ell=0; ell<=scattering_order; ell++)

54 for (int ell=0; ell<=scattering_order; ell++)
55 for (int m=-ell; m<=ell; m++)

122 auto ContributePWLDTally = [&sdm,&grid,&phi_tally,&phi_uk_man,&sigma_t,
123 &num_moments,&m_to_ell_em_map](
124 const chi_mesh::Cell& cell,
125 const Vec3& positionA,
126 const Vec3& positionB,
127 const Vec3& omega,
128 const int g,
129 double weight)
130 {
131 const auto& cell_mapping = sdm.GetCellMapping(cell);
132 const size_t num_nodes = cell_mapping.NumNodes();
133
134 const auto phi_theta = chi_math::OmegaToPhiThetaSafe(omega);
135 const double phi = phi_theta.first;
136 const double theta = phi_theta.second;
137
138 std::vector<double> segment_lengths;
139 chi_mesh::PopulateRaySegmentLengths(grid, //input
140 cell, //input
141 positionA, //input
142 positionB, //input
143 omega, //input
144 segment_lengths); //output
145
146 std::vector<double> shape_values_k; //At s_k
147 std::vector<double> shape_values_kp1; //At s_{k+1}
148
149 cell_mapping.ShapeValues(positionA, //input
150 shape_values_k); //output
151
152 double d_run_sum = 0.0;
153 for (const auto& segment_length_k : segment_lengths)
154 {
155 d_run_sum += segment_length_k;
156 const double& d = d_run_sum;
157
158 cell_mapping.ShapeValues(positionA+omega*d, shape_values_kp1);
159
160 const auto& b_ik = shape_values_k;
161 const auto& b_ikp1 = shape_values_kp1;
162 const double& ell_k = segment_length_k;
163
164 for (size_t i=0; i<num_nodes; ++i)
165 {
166 const double C0 = b_ik[i] * ell_k;
167 const double C1 = b_ikp1[i] - b_ik[i];
168
169 for (size_t m=0; m < num_moments; ++m)
170 {
171 const int64_t dof_map = sdm.MapDOFLocal(cell, i, phi_uk_man, m, g);
172
173 //================= Apply harmonic weight
174 const auto& ell_em = m_to_ell_em_map.at(m);
175 const int ell = ell_em.first;
176 const int em = ell_em.second;
177
178 double w_harmonic = chi_math::Ylm(ell, em, phi, theta);
179
180 //================= Apply exponential attenuation weight
181 double w_exp = (C0 / sigma_t) * (1.0 - exp(-sigma_t * ell_k)) +
182 (C1 / (sigma_t * sigma_t)) *
183 (1.0 - (1 + sigma_t * ell_k) * exp(-sigma_t * ell_k));
184 w_exp *= weight / (ell_k * ell_k);
185
186 //================= Combine
187 double w_avg = w_harmonic * w_exp;
188
189 phi_tally[dof_map] += ell_k * w_avg ;
190 }//for moment m
191 }//for node i
192
193 shape_values_k = shape_values_kp1;
194 weight *= exp(-sigma_t * segment_length_k);
195 }//for d
196 };

198 auto GetCellApproximateSize = [&grid](const chi_mesh::Cell& cell)
199 {
200 const auto& v0 = grid.vertices[cell.vertex_ids_[0]];
201 double xmin = v0.x, xmax = v0.x;
202 double ymin = v0.y, ymax = v0.y;
203 double zmin = v0.z, zmax = v0.z;
204
205 for (uint64_t vid : cell.vertex_ids_)
206 {
207 const auto& v = grid.vertices[vid];
208
209 xmin = std::min(xmin, v.x); xmax = std::max(xmax, v.x);
210 ymin = std::min(ymin, v.y); ymax = std::max(ymax, v.y);
211 zmin = std::min(zmin, v.z); zmax = std::max(zmax, v.z);
212 }
213
214 return (chi_mesh::Vector3(xmin, ymin, zmin) -
215 chi_mesh::Vector3(xmax, ymax, zmax)).Norm();

293 //============================================= Post process tallies
294 for (const auto& cell : grid.local_cells)
295 {
296 //====================================== Compute mass matrix
297 // and its inverse
298 const auto& cell_mapping = sdm.GetCellMapping(cell);
299 const auto& qp_data = cell_mapping.MakeVolumeQuadraturePointData();
300 const size_t num_nodes = cell_mapping.NumNodes();
301
302 MatDbl M(num_nodes, VecDbl(num_nodes, 0.0));
303 for (auto qp : qp_data.QuadraturePointIndices())
304 for (size_t i=0; i<num_nodes; ++i)
305 for (size_t j=0; j<num_nodes; ++j)
306 M[i][j] += qp_data.ShapeValue(i,qp) * qp_data.ShapeValue(j, qp) *
307 qp_data.JxW(qp);
308
309 auto M_inv = chi_math::Inverse(M);
310
311 //====================================== Apply projection
312 VecDbl T(num_nodes, 0.0);
313 for (size_t m=0; m<num_moments; ++m)
314 for (size_t g=0; g<num_groups; ++g)

318 const int64_t imap = sdm.MapDOFLocal(cell, i, phi_uk_man, m, g);

321

325 {

327 phi_tally[imap] = phi_uc[i];
328 }
329 }//for group g

331 }//for cell
