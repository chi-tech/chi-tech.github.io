161 /**Attempts to retrieve an object of base-type `shared_ptr<T>` at the given
162 * handle. It then attempts to cast it to type `shared_ptr<R>` and, if
163 * successful, will return a reference of type R&.
164 * \n
165 * \n
166 * Example usage:
167 *
168 * \code
169 * const auto& surf_mesh = chi::GetStackItem<chi_mesh::SurfaceMesh>(
170 chi::surface_mesh_stack, surface_hndl);
171 * \endcode
172 * */
173 template <class R, class T>
174 static R& GetStackItem(std::vector<std::shared_ptr<T>>& stack,
175 const size_t handle,
176 const std::string& calling_function_name = "Unknown")
177 {
178 try
179 {
180 std::shared_ptr<T>& item = stack.at(handle);
181 std::shared_ptr<R> ret_item = std::dynamic_pointer_cast<R>(item);
182 if (not ret_item)
183 throw std::logic_error("chi::GetStackItem: Invalid return type used. "
184 "Calling function: " +
185 calling_function_name);
186 return *ret_item;
187 }
188 catch (const std::out_of_range& oor)
189 {
190 throw std::out_of_range("chi::GetStackItem: Invalid handle used. "
191 "Calling function: " +
192 calling_function_name);
193 }
194 }

196 /**Attempts to obtain object of type `shared_ptr<T>` at the given
197 * handle of a stack with parent type P.
198 * \n
199 * \n

201 *
202 * \code
203 * auto surf_mesh_ptr = chi::GetStackItemPtr<chi_mesh::SurfaceMesh>(

205 * \endcode

207 template <class T, class P>
208 static std::shared_ptr<T>
209 GetStackItemPtrAsType(std::vector<std::shared_ptr<P>>& stack,
210 const size_t handle,

213 std::shared_ptr<P> item_type_P;
