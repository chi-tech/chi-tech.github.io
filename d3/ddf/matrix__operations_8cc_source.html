275 //######################################################### Gauss Elimination
276 /** Gauss Elimination without pivoting.*/
277 void chi_math::GaussElimination(MatDbl &A,
278  VecDbl &b, int n)
279 {
280  // Forward elimination
281  for(int i = 0;i < n-1;++i)
282  {
283  const std::vector<double>& ai = A[i];
284  double bi = b[i];
285  double factor = 1.0/A[i][i];
286  for(int j = i+1;j < n;++j)
287  {
288  std::vector<double>& aj = A[j];
289  double val = aj[i] * factor;
290  b[j] -= val * bi;
291  for(int k = i+1;k < n;++k)
292  aj[k] -= val * ai[k];
293  }
294  }
295 
296  // Back substitution
297  for(int i = n-1;i >= 0;--i)
298  {
299  const std::vector<double>& ai = A[i];
300  double bi = b[i];
301  for(int j = i+1;j < n;++j)
302  bi -= ai[j] * b[j];
303  b[i] = bi/ai[i];
304  }
305 }

308 /** Computes the inverse of a matrix using Gauss-Elimination with pivoting.*/
309 MatDbl chi_math::InverseGEPivoting(const MatDbl &A)
310 {
311  assert(A.size());
312  assert(A.size() == A[0].size());
313 
314  unsigned int R = A.size();
315 
316  std::vector<std::vector<double> > M(R,std::vector<double>(R,0.));
317 
318  for(unsigned int i = 0; i < R; i++)
319  M[i][i] = 1.;
320 
321  std::vector<std::vector<double> > B = A;
322 
323  for (unsigned int c = 0; c < R; c++)
324  {
325  // Find a row which is nonzero on diagonal
326  unsigned int nonzerorow;
327  for (nonzerorow = c; nonzerorow < R && B[nonzerorow][c] == 0; ++nonzerorow) {}
328 
329  if ( nonzerorow != c )
330  {
331  SwapRow( nonzerorow , c, B );
332  SwapRow( nonzerorow , c, M );
333  }
334 
335  // Eliminate non-zero values
336  for (unsigned int r = 0; r < R; r++)
337  {
338  if ( r != c )
339  {
340  double g = B[r][c]/B[c][c];
341  if ( B[r][c] != 0 )
342  {
343  for(unsigned int k=0; k < R; k++)
344  {
345  B[r][k] -= B[c][k]*g;
346  M[r][k] -= M[c][k]*g;
347  }
348  B[r][c] = 0;
349  }
350  }
351  else
352  {
353  double g = 1/B[c][c];
354  for (unsigned int k = 0; k < R; k++)
355  {
356  B[r][k] *= g;
357  M[r][k] *= g;
358  }
359  }
360  }
361  }
362  return M;
363 }

366 /** Computes the inverse of a matrix.*/
367 MatDbl
368 chi_math::Inverse(const MatDbl &A)
369 {
370  size_t R = A.size();
371  std::vector<std::vector<double>> M(R,std::vector<double>(R,0.));
372  double f(0.);
373 
374  // Only calculate the determinant if matrix size is less than 5 since
375  // the inverse is directly calculated for larger matrices. Otherwise,
376  // the inverse routine spends all of its time sitting in the determinant
377  // function which is unnecessary.
378  if (R < 5)
379  {
380  f = Determinant(A);
381  assert(f != 0.);
382  f = 1./f;
383  }
384 
385  if (R==1)
386  M[0][0] = f;
387  else if (R==2)
388  {
389  M[0][0] = A[1][1];
390  M[0][1] = -A[0][1];
391  M[1][0] = -A[1][0];
392  M[1][1] = A[0][0];
393  Scale(M,f);
394  }
395  else if (R==3)
396  {
397  M[0][0] = A[2][2]*A[1][1]-A[2][1]*A[1][2];
398  M[0][1] = -(A[2][2]*A[0][1]-A[2][1]*A[0][2]);

404  M[2][1] = -(A[2][1]*A[0][0]-A[2][0]*A[0][1]);

439 //######################################################### Power iteration
440 /** Performs power iteration to obtain the fundamental eigen mode. The
441  * eigen-value of the fundamental mode is return whilst the eigen-vector
442  * is return via reference.*/
443 double chi_math::PowerIteration(
444  const MatDbl &A, VecDbl &e_vec, int max_it, double tol)
445 {
446  // Local Variables
447  unsigned int n = A.size();
448  int it_counter = 0;
449  VecDbl y(n,1.);
450  double lambda0 = 0.;
451 
452  // Perform initial iteration outside of loop
453  VecDbl Ay = MatMul(A, y);
454  double lambda = Dot( y, Ay );
455  y = VecMul( Ay, 1./Vec2Norm(Ay) );
456  if (lambda < 0.)
457  Scale(y, -1.0);
458 
459  // Perform convergence loop
460  bool converged = false;
461  while(!converged && it_counter < max_it)
462  {
463  // Update old eigenvalue
464  lambda0 = std::fabs(lambda);
465  // Calculate new eigenvalue/eigenvector
466  Ay = MatMul(A, y);
467  lambda = Dot( y, Ay );
468  y = VecMul( Ay, 1./Vec2Norm(Ay) );
469 
470  // Check if converged or not
471  if (std::fabs(std::fabs(lambda) - lambda0) <= tol)
472  converged = true;
473  // Update counter
474  ++it_counter;
475  }
476 
477  if (lambda < 0.)
478  Scale(y, -1.0);
479 
480  // Renormalize eigenvector for the last time
481  y = VecMul(Ay, 1./lambda);
482 
483  // Set eigenvector, return the eigenvalue
484  e_vec = std::move(y);
485 
486  return lambda;
487 }
