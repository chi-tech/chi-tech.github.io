128template<>
129void WGSLinearSolver<Mat, Vec, KSP>::SetRHS()
130{
131 auto gs_context_ptr = GetGSContextPtr(context_ptr_);
132
133 auto& groupset = gs_context_ptr->groupset_;
134 auto& lbs_solver = gs_context_ptr->lbs_solver_;
135
136 if (gs_context_ptr->log_info_)
137 chi::log.Log() << chi::program_timer.GetTimeString() << " Computing b";
138
139 //SetSource for RHS
140 saved_q_moments_local_ = lbs_solver.QMomentsLocal();
141 const int scope = gs_context_ptr->rhs_src_scope_;
142 gs_context_ptr->set_source_function_(groupset,
143 lbs_solver.QMomentsLocal(),
144 lbs_solver.PhiOldLocal(),
145 scope);
146
147 //=================================================== Apply transport operator
148 gs_context_ptr->ApplyInverseTransportOperator(scope);
149
150 //=================================================== Assemble PETSc vector
151 lbs_solver.SetGSPETScVecFromPrimarySTLvector(groupset,
152 b_,
153 PhiSTLOption::PHI_NEW);
154
155 //============================================= Compute RHS norm
156 VecNorm(b_, NORM_2, &context_ptr_->rhs_norm);
157
158 //============================================= Compute precondition RHS norm
159 PC pc;
160 KSPGetPC(solver_, &pc);
161 Vec temp_vec;
162 VecDuplicate(b_, &temp_vec);
163 PCApply(pc, b_, temp_vec);
164 VecNorm(temp_vec, NORM_2, &context_ptr_->rhs_preconditioned_norm);
165 VecDestroy(&temp_vec);
166}

169template<> void WGSLinearSolver<Mat, Vec, KSP>::PostSolveCallback()
170{
171 //============================================= Get convergence reason
172 KSPConvergedReason reason;
173 KSPGetConvergedReason(solver_,&reason);
174 if (reason != KSP_CONVERGED_RTOL and reason != KSP_DIVERGED_ITS)
175 chi::log.Log0Warning()
176 << "Krylov solver failed. "
177 << "Reason: " << chi_physics::GetPETScConvergedReasonstring(reason);
178
179 //============================================= Copy x to local solution
180 auto gs_context_ptr = GetGSContextPtr(context_ptr_);
181
182 auto& groupset = gs_context_ptr->groupset_;
183 auto& lbs_solver = gs_context_ptr->lbs_solver_;
184
185 lbs_solver.SetPrimarySTLvectorFromGSPETScVec(groupset, x_, PhiSTLOption::PHI_NEW);
186 lbs_solver.SetPrimarySTLvectorFromGSPETScVec(groupset, x_, PhiSTLOption::PHI_OLD);
187
188 //============================================= Restore saved q_moms
189 lbs_solver.QMomentsLocal() = saved_q_moments_local_;
190
191 //============================================= Context specific callback
192 gs_context_ptr->PostSolveCallback();
193}
