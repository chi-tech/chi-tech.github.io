89void DiscreteOrdinatesCurvilinearSolver::ComputeSecondaryUnitIntegrals()
90{
91 Chi::log.Log() << "Computing RZ secondary unit integrals.\n";
92 const auto& sdm = *discretization_;
93
94 //======================================== Define spatial weighting functions
95 struct SpatialWeightFunction // SWF
96 {
97 virtual double operator()(const chi_mesh::Vector3& pt) const { return 1.0; }
98 };
99
100 auto swf_ptr = std::make_shared<SpatialWeightFunction>();
101
102 //======================================== Define lambda for cell-wise comps
103 auto ComputeCellUnitIntegrals =
104 [&sdm](const chi_mesh::Cell& cell, const SpatialWeightFunction& swf)
105 {
106 const auto& cell_mapping = sdm.GetCellMapping(cell);
107 // const size_t cell_num_faces = cell.faces.size();
108 const size_t cell_num_nodes = cell_mapping.NumNodes();
109 const auto vol_qp_data = cell_mapping.MakeVolumeQuadraturePointData();
110
111 MatDbl IntV_shapeI_shapeJ(cell_num_nodes, VecDbl(cell_num_nodes));
112
113 // Volume integrals
114 for (unsigned int i = 0; i < cell_num_nodes; ++i)
115 {
116 for (unsigned int j = 0; j < cell_num_nodes; ++j)
117 {
118 for (const auto& qp : vol_qp_data.QuadraturePointIndices())
119 {
120 IntV_shapeI_shapeJ[i][j] +=
121 swf(vol_qp_data.QPointXYZ(qp)) * vol_qp_data.ShapeValue(i, qp) *
122 vol_qp_data.ShapeValue(j, qp) * vol_qp_data.JxW(qp); // M-matrix
123 } // for qp
124 } // for j
125 } // for i
126
127 return lbs::UnitCellMatrices{{}, // K-matrix
128 {}, // G-matrix
129 IntV_shapeI_shapeJ, // M-matrix
130 {}, // Vi-vectors
131
132 {}, // face M-matrices
133 {}, // face G-matrices
134 {}}; // face Si-vectors
135 };
136
137 const size_t num_local_cells = grid_ptr_->local_cells.size();
138 secondary_unit_cell_matrices_.resize(num_local_cells);
139
140 for (const auto& cell : grid_ptr_->local_cells)
141 secondary_unit_cell_matrices_[cell.local_id_] =
142 ComputeCellUnitIntegrals(cell, *swf_ptr);
143
144 Chi::mpi.Barrier();
145 Chi::log.Log()
146 << "Secondary Cell matrices computed. Process memory = "
147 << std::setprecision(3) << chi::Console::GetMemoryUsageInMB()
148 << " MB";
149}
