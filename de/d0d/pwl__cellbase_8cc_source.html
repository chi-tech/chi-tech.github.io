15/** This section just determines a mapping of face dofs
16to cell dofs. This is pretty simple since we can
17just loop over each face dof then subsequently
18loop over cell dofs, if the face dof node index equals
19the cell dof node index then the mapping is assigned.
20
21This mapping is not used by any of the methods in
22this class but is used by methods requiring the
23surface integrals of the shape functions.*/
24std::vector<std::vector<int>> chi_math::CellMappingFE_PWL::
25 MakeFaceNodeMapping(const chi_mesh::Cell &cell)
26{
27 const size_t num_faces = cell.faces_.size();
28 std::vector<std::vector<int>> mappings;
29 mappings.reserve(num_faces);
30 for (auto& face : cell.faces_)
31 {
32 std::vector<int> face_dof_mapping;
33 face_dof_mapping.reserve(face.vertex_ids_.size());
34 for (uint64_t fvid : face.vertex_ids_)
35 {
36 int mapping = -1;
37 for (size_t ci=0; ci<cell.vertex_ids_.size(); ci++)
38 {
39 if (fvid == cell.vertex_ids_[ci])
40 {
41 mapping = static_cast<int>(ci);
42 break;
43 }
44 }//for cell i
45 if (mapping<0)
46 {
47 Chi::log.LogAllError() << "Unknown face mapping encountered. "
48 "pwl_polyhedron.h";
49 Chi::Exit(EXIT_FAILURE);
50 }
51 face_dof_mapping.push_back(mapping);
52 }//for face i
53
54 mappings.push_back(face_dof_mapping);
55 }
56 return mappings;
57}

61 ComputeWeightedUnitIntegrals(chi_math::finite_element::UnitIntegralData& ui_data) const

63 // quadrature point data
64 chi_math::finite_element::InternalQuadraturePointData internal_data;
65 std::vector<chi_math::finite_element::FaceQuadraturePointData> faces_qp_data;
66 InitializeAllQuadraturePointData(internal_data, faces_qp_data);
67
68
69 // weighted JxW
70 std::vector<double> V_JxW;
71 for (const auto& qp : internal_data.QuadraturePointIndices())
72 {
73 const auto jxw =
74 SpatialWeightFunction(internal_data.QPointXYZ(qp)) *
75 internal_data.JxW(qp);
76 V_JxW.emplace_back(jxw);
77 }
78
79 std::vector<std::vector<double>> F_JxW(faces_qp_data.size());
80 for (size_t f = 0; f < faces_qp_data.size(); ++f)
81 for (const auto& qp : faces_qp_data[f].QuadraturePointIndices())
82 {
83 const auto jxw =
84 SpatialWeightFunction(faces_qp_data[f].QPointXYZ(qp)) *
85 faces_qp_data[f].JxW(qp);
86 F_JxW[f].emplace_back(jxw);
87 }
88
89
90 // integrals
91 using VecDbl = std::vector<double>;
92 using MatDbl = std::vector<VecDbl>;
93 using VecVec3 = std::vector<chi_mesh::Vector3>;
94 using MatVec3 = std::vector<VecVec3>;
95
96 const auto n_dof_per_cell = internal_data.NumNodes();
97
98 MatDbl IntV_gradshapeI_gradshapeJ(n_dof_per_cell, VecDbl(n_dof_per_cell));
99 MatVec3 IntV_shapeI_gradshapeJ(n_dof_per_cell, VecVec3(n_dof_per_cell));
100 MatDbl IntV_shapeI_shapeJ(n_dof_per_cell, VecDbl(n_dof_per_cell));
101 VecDbl IntV_shapeI(n_dof_per_cell);
102 VecVec3 IntV_gradshapeI(n_dof_per_cell);
103
104 std::vector<MatDbl> IntS_shapeI_shapeJ(faces_qp_data.size());
105 std::vector<VecDbl> IntS_shapeI(faces_qp_data.size());
106 std::vector<MatVec3> IntS_shapeI_gradshapeJ(faces_qp_data.size());
107
108
109 // volume integrals
110 for (unsigned int i = 0; i < n_dof_per_cell; ++i)
111 {
112 for (unsigned int j = 0; j < n_dof_per_cell; ++j)
113 {
114 for (const auto& qp : internal_data.QuadraturePointIndices())
115 {
116 IntV_gradshapeI_gradshapeJ[i][j]
117 += internal_data.ShapeGrad(i, qp).Dot(internal_data.ShapeGrad(j, qp)) *
118 V_JxW[qp];
119
120 IntV_shapeI_gradshapeJ[i][j]
121 += internal_data.ShapeValue(i, qp) *
122 internal_data.ShapeGrad(j, qp) *
123 V_JxW[qp];
124
125 IntV_shapeI_shapeJ[i][j]
126 += internal_data.ShapeValue(i, qp) *
127 internal_data.ShapeValue(j, qp) *
128 V_JxW[qp];
129 }// for qp
130 }// for j
131
132 for (const auto& qp : internal_data.QuadraturePointIndices())
133 {
134 IntV_shapeI[i]
135 += internal_data.ShapeValue(i, qp) * V_JxW[qp];
136
137 IntV_gradshapeI[i]
138 += internal_data.ShapeGrad(i, qp) * V_JxW[qp];
139 }// for qp
140 }//for i
141
142
143 // surface integrals
144 for (size_t f = 0; f < faces_qp_data.size(); ++f)
145 {
146 IntS_shapeI_shapeJ[f].resize(n_dof_per_cell, VecDbl(n_dof_per_cell));
147 IntS_shapeI[f].resize(n_dof_per_cell);
148 IntS_shapeI_gradshapeJ[f].resize(n_dof_per_cell, VecVec3(n_dof_per_cell));
149
150 for (unsigned int i = 0; i < n_dof_per_cell; ++i)
151 {
152 for (unsigned int j = 0; j < n_dof_per_cell; ++j)
153 {
154 for (const auto& qp : faces_qp_data[f].QuadraturePointIndices())
155 {
156 IntS_shapeI_shapeJ[f][i][j]
157 += faces_qp_data[f].ShapeValue(i, qp) *
158 faces_qp_data[f].ShapeValue(j, qp) *
159 F_JxW[f][qp];
160
161 IntS_shapeI_gradshapeJ[f][i][j]
162 += faces_qp_data[f].ShapeValue(i, qp) *
163 faces_qp_data[f].ShapeGrad(j, qp) *
164 F_JxW[f][qp];
165 }// for qp
166 }//for j
167
168 for (const auto& qp : faces_qp_data[f].QuadraturePointIndices())
169 {
170 IntS_shapeI[f][i]
171 += faces_qp_data[f].ShapeValue(i, qp) * F_JxW[f][qp];
172 }// for qp
173 }//for i
174 }//for f
175
176
177 // unit integral data
178 ui_data.Initialize(IntV_gradshapeI_gradshapeJ,
179 IntV_shapeI_gradshapeJ,
180 IntV_shapeI_shapeJ,
181 IntV_shapeI,
182 IntV_gradshapeI,
183 IntS_shapeI_shapeJ,
184 IntS_shapeI,
185 IntS_shapeI_gradshapeJ,
186 face_node_mappings_,
187 num_nodes_);
