24 /**Utility that converts a type to a byte vector provided that
25 * it has the sizeof() function defined for it.*/
26 template <typename T>
27 void PopulateRaw(const T& value)
28 {
29 auto src = reinterpret_cast<const std::byte*>(&value);
30
31 size_t num_bytes = sizeof(T);
32 raw_data_.resize(num_bytes);
33
34 std::copy(src, src + num_bytes, &raw_data_[0]);
35
36 data_initialized_ = true;
37 }
