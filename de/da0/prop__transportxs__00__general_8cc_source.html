46/**Makes a simple material with isotropic transfer matrix (L=0)
47 * and mostly down scattering but with a few of the last groups
48 * subject to up-scattering.*/
49void chi_physics::TransportCrossSections::
50 MakeSimple1(int in_G, double in_sigmat, double c)
51{
52 //======================================== Clear any previous data
53 Reset();
54
55 num_groups = in_G;
56 scattering_order = 0;
57
58 sigma_t.resize(in_G, in_sigmat);
59 sigma_f.resize(in_G, 0.0);
60 sigma_a.resize(in_G, 0.0);
61 chi.resize(in_G, 0.0);
62 chi_prompt.resize(in_G, 0.0);
63 nu_sigma_f.resize(in_G, 0.0);
64 nu_prompt_sigma_f.resize(in_G, 0.0);
65 nu_delayed_sigma_f.resize(in_G, 0.0);
66 inv_velocity.resize(in_G, 0.0);
67
68 transfer_matrices.emplace_back(in_G, in_G);
69
70 auto& ref_matrix = transfer_matrices.back();
71
72 if (num_groups == 1)
73 ref_matrix.SetDiagonal(std::vector<double>(in_G,in_sigmat*c));
74 else
75 ref_matrix.SetDiagonal(std::vector<double>(in_G,in_sigmat*c*2.0/4.0));
76
77 for (int g=0; g<in_G; g++)
78 {
79 //Downscattering
80 if (g>0)
81 ref_matrix.Insert(g,g-1,in_sigmat*c*2.0/4.0);
82
83 //Upscattering
84 if (g>(in_G/2))
85 {
86 if (g<(in_G-1))
87 {
88 ref_matrix.Insert(g,g-1,in_sigmat*c*1.0/4.0);
89 ref_matrix.Insert(g,g+1,in_sigmat*c*1.0/4.0);
90 }
91 else
92 ref_matrix.Insert(g,g-1,in_sigmat*c*2.0/4.0);
93 }
94 }//for g
95
96 sigma_a = ComputeAbsorptionXSFromTransfer();
97}

100/**Populates the cross-section from a combination of others.*/
101void chi_physics::TransportCrossSections::
102 MakeCombined(std::vector<std::pair<int, double> > &combinations)
103{
104 //======================================== Clear any previous data
105 Reset();
106
107 //======================================== Pickup all xs and make sure valid
108 std::vector<std::shared_ptr<chi_physics::TransportCrossSections>> cross_secs;
109 cross_secs.reserve(combinations.size());
110 size_t num_grps_G = 0;
111 size_t num_precursors_J = 0;
112
113 double N_total = 0.0; // total density
114 double Nf_total = 0.0; // total density of fissile materials
115 double Np_total = 0.0; // total density of materials with precursors
116
117 for (auto combo : combinations)
118 {
119 std::shared_ptr<chi_physics::TransportCrossSections> xs;
120 try
121 {
122 xs = chi::GetStackItemPtr(chi::trnsprt_xs_stack, combo.first);
123 }
124 catch(const std::out_of_range& o)
125 {
126 chi::log.LogAllError()
127 << "ERROR: Invalid cross-section handle"
128 << " in call to chiPhysicsMaterialSetProperty."
129 << std::endl;
130 chi::Exit(EXIT_FAILURE);
131 }
132
133 cross_secs.push_back(xs);
134
135 // Increment combo factor totals
136 N_total += combo.second;
137 if (xs->is_fissile)
138 {
139 this->is_fissile = true;
140 Nf_total += combo.second;
141
142 if (xs->num_precursors > 0)
143 Np_total += combo.second;
144 }
145
146 //============================ Define and check number of groups
147 if (cross_secs.size() == 1)
148 num_grps_G = xs->num_groups;
149 else if (xs->num_groups != num_grps_G)
150 {
151 chi::log.LogAllError()
152 << "In call to " << __FUNCTION__ << ": "
153 << "all cross-sections must have the same number of groups.";
154 chi::Exit(EXIT_FAILURE);
155 }
156
157 //============================ Increment number of precursors
158 if (not xs->is_fissile and xs->num_precursors > 0)
159 {
160 chi::log.LogAllError()
161 << "In call to " << __FUNCTION__ << ": "
162 << "only fissile materials are allowed to have delayed "
163 << "neutron precursors.";
164 chi::Exit(EXIT_FAILURE);
165 }
166 num_precursors_J += xs->num_precursors;
167 }//for cross-section
168
169 // Check that the fissile and precursor densities are greater than
170 // machine precision. If this condition is not met, the material is assumed
171 // to be either not fissile, have zero precursors, or both.
172 double eps = std::numeric_limits<double>::epsilon(); //machine precision
173 if (Nf_total < eps)
174 this->is_fissile = false;
175 if (Np_total < eps)
176 this->num_precursors = 0;
177
178 //======================================== Combine 1D cross-sections
179 this->num_groups = num_grps_G;
180 this->num_precursors = num_precursors_J;
181
182 sigma_t.resize(num_grps_G, 0.0);
183 sigma_f.resize(num_grps_G, 0.0);
184 sigma_a.resize(num_grps_G, 0.0);
185 chi.resize(num_grps_G, 0.0);
186 chi_prompt.resize(num_grps_G, 0.0);
187 nu.resize(num_grps_G,0.0);
188 nu_prompt.resize(num_grps_G,0.0);
189 nu_delayed.resize(num_grps_G,0.0);
190 nu_sigma_f.resize(num_grps_G, 0.0);
191 nu_prompt_sigma_f.resize(num_grps_G, 0.0);
192 nu_delayed_sigma_f.resize(num_grps_G, 0.0);
193 inv_velocity.resize(num_grps_G, 0.0);
194 precursor_lambda.resize(num_precursors_J, 0.0);
195 precursor_yield.resize(num_precursors_J, 0.0);
196 chi_delayed.resize(num_grps_G);
197 for (size_t g = 0; g < num_groups; ++g)
198 chi_delayed[g].resize(num_precursors_J, 0.0);
199
200 size_t precursor_count = 0;
201 for (size_t x = 0; x < cross_secs.size(); ++x)
202 {
203 scattering_order = std::max(this->scattering_order,
204 cross_secs[x]->scattering_order);
205
206 // Atom density
207 double N_i = combinations[x].second;
208
209 // Fraction of fissile density
210 double ff_i = 0.0;
211 if (cross_secs[x]->is_fissile)
212 ff_i = N_i / Nf_total;
213
214 // Fraction of precursor density
215 double pf_i = 0.0;
216 if (cross_secs[x]->num_precursors > 0)
217 pf_i = N_i / Np_total;
218
219 //======================================== Combine cross-sections
220 for (size_t g = 0; g < num_grps_G; ++g)
221 {
222 sigma_t [g] += cross_secs[x]->sigma_t [g] * N_i;
223 sigma_f [g] += cross_secs[x]->sigma_f [g] * N_i;
224 sigma_a [g] += cross_secs[x]->sigma_a [g] * N_i;
225
226 chi [g] += cross_secs[x]->chi [g] * ff_i;
227 chi_prompt [g] += cross_secs[x]->chi_prompt [g] * ff_i;
228
229 nu [g] += cross_secs[x]->nu [g] * ff_i;
230 nu_prompt [g] += cross_secs[x]->nu_prompt [g] * ff_i;
231 nu_delayed [g] += cross_secs[x]->nu_delayed [g] * ff_i;
232
233 nu_sigma_f [g] += cross_secs[x]->nu_sigma_f [g] * N_i;
234 nu_prompt_sigma_f [g] += cross_secs[x]->nu_prompt_sigma_f [g] * N_i;
235 nu_delayed_sigma_f[g] += cross_secs[x]->nu_delayed_sigma_f[g] * N_i;
236
237 if (x == 0)
238 inv_velocity[g] = cross_secs[x]->inv_velocity[g];
239 else if (inv_velocity[g] != cross_secs[x]->inv_velocity[g])
240 chi::log.LogAllWarning()
241 << "In call to " << __FUNCTION__
242 << ": all materials must have the same inv_velocity "
243 << "term per group. Invalid value encountered in "
244 << "material " << x << " group " << g << ". Using the "
245 << "value from the first cross-section set.";
246 }
247
248 //======================================== Compute precursor
249 // Here, all precursors across all materials are stored.
250 // The decay constants and delayed spectrum are what they are,
251 // however, some special treatment must be given to the yields.
252 // Because the yield tells us what fraction of delayed neutrons
253 // are produced from a given family, the sum over all families
254 // must yield unity. To achieve this end, we must scale all
255 // precursor yields based on the fraction of the total density
256 // of materials with precursors they make up.
257 if (cross_secs[x]->num_precursors > 0)
258 {
259 for (size_t j = 0; j < cross_secs[x]->num_precursors; ++j)
260 {
261 size_t count = precursor_count + j;
262 precursor_lambda[count] = cross_secs[x]->precursor_lambda[j];
263 precursor_yield [count] = cross_secs[x]->precursor_yield [j] * pf_i;
264 for (size_t g = 0; g < num_groups; ++g)
265 chi_delayed[g][count] = cross_secs[x]->chi_delayed[g][j];
266 }
267 precursor_count += cross_secs[x]->num_precursors;
268 }
269 }//for cross sections
270
271 //======================================== Combine transfer matrices
272 // This step is somewhat tricky. The cross-sections
273 // aren't guaranteed to have the same sparsity patterns
274 // and therefore simply adding them together has to take
275 // the sparse matrix's protection mechanisms into account.
276 transfer_matrices.clear();
277 transfer_matrices.resize(this->scattering_order + 1,
278 chi_math::SparseMatrix(num_grps_G,num_grps_G));
279 for (size_t x = 0; x < cross_secs.size(); ++x)
280 {
281 for (int m=0; m<(cross_secs[x]->scattering_order + 1); ++m)
282 {
283 auto& xs_tm = cross_secs[x]->transfer_matrices[m];
284 for (size_t i = 0; i < num_groups; ++i)
285 {
286 for (auto j : xs_tm.rowI_indices[i])
287 {
288 double value = xs_tm.ValueIJ(i,j)*combinations[x].second;
289 transfer_matrices[m].InsertAdd(i, j, value);
290 }
291 }//for i
292 }//for m
293 }//for xs
294}

298{
299 auto GetMatrixColumnSum = [](const chi_math::SparseMatrix& matrix, size_t col)
300 {
301 double sum = 0.0;
302
303 for (size_t row=0; row<matrix.NumRows(); ++row)
304 {
305 const auto& row_col_indices = matrix.rowI_indices[row];
306 const auto& row_values = matrix.rowI_values[row];
307 for (size_t j=0; j<row_col_indices.size(); ++j)
308 if (row_col_indices[j] == col)
309 sum += row_values[j];
310 }
311
312 return sum;
313 };
314
315 std::vector<GrpVal> temp_sigma_a(num_groups, 0.0);
316 if (not transfer_matrices.empty())
317 for (size_t g = 0; g < num_groups; ++g)
318 temp_sigma_a[g] = sigma_t[g] -
319 GetMatrixColumnSum(transfer_matrices[0], g);
320
321 return temp_sigma_a;
322}
