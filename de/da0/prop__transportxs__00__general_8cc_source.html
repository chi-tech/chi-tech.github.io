94/**
95 * Makes a simple material with isotropic transfer matrix (L=0)
96 * and mostly down scattering but with a few of the last groups
97 * subject to up-scattering.
98 */
99void chi_physics::TransportCrossSections::
100MakeSimple1(int n_grps, double sigma, double c)
101{
102 //clear any previous data
103 Reset();
104
105 //define the cross-section data
106 num_groups = n_grps;
107 scattering_order = 0;
108
109 sigma_t.resize(n_grps, sigma);
110 sigma_a.resize(n_grps, 0.0);
111 sigma_f.resize(n_grps, 0.0);
112
113 chi.resize(n_grps, 0.0);
114 chi_prompt.resize(n_grps, 0.0);
115
116 nu_sigma_f.resize(n_grps, 0.0);
117 nu_prompt_sigma_f.resize(n_grps, 0.0);
118 nu_delayed_sigma_f.resize(n_grps, 0.0);
119
120 inv_velocity.resize(n_grps, 0.0);
121
122 transfer_matrices.emplace_back(n_grps, n_grps);
123
124 auto& S = transfer_matrices.back();
125 double scale = (num_groups == 1)? 1.0 : 0.5;
126 S.SetDiagonal(std::vector<double>(n_grps, sigma * c * scale));
127
128 for (unsigned int g = 0; g < num_groups; ++g)
129 {
130 //Downscattering
131 if (g > 0)
132 S.Insert(g, g - 1, sigma * c * 0.5);
133
134 //Upscattering
135 if (g > num_groups / 2)
136 {
137 if (g < num_groups - 1)
138 {
139 S.Insert(g, g - 1, sigma * c * 0.25);
140 S.Insert(g, g + 1, sigma * c * 0.25);
141 }
142 else
143 S.Insert(g, g - 1, sigma * c * 0.5);
144 }
145 }//for g
146
147 ComputeAbsorption();
148 ComputeDiffusionParameters();
149}

153/** Populates the cross-section from a combination of others. */
154void chi_physics::TransportCrossSections::
155MakeCombined(std::vector<std::pair<int, double> > &combinations)
156{
157 //clear any previous data
158 Reset();
159
160 //pickup all xs and make sure valid
161 std::vector<std::shared_ptr<chi_physics::TransportCrossSections>> xsecs;
162 xsecs.reserve(combinations.size());
163
164 unsigned int n_grps = 0;
165 unsigned int n_precs = 0;
166
167 double Nf_total = 0.0; //total density of fissile materials
168 double Np_total = 0.0; //total density of materials with precursors
169
170 //loop over cross-sections
171 for (auto combo : combinations)
172 {
173 //get the cross-section from the lua stack
174 std::shared_ptr<chi_physics::TransportCrossSections> xs;
175 try
176 {
177 xs = chi::GetStackItemPtr(chi::trnsprt_xs_stack, combo.first);
178 }
179 catch(const std::out_of_range& o)
180 {
181 chi::log.LogAllError()
182 << "ERROR: Invalid cross-section handle"
183 << " in call to chiPhysicsMaterialSetProperty."
184 << std::endl;
185 chi::Exit(EXIT_FAILURE);
186 }
187 xsecs.push_back(xs);
188
189 //increment densities
190 if (xs->is_fissionable)
191 {
192 this->is_fissionable = true;
193 Nf_total += combo.second;
194
195 if (xs->num_precursors > 0)
196 Np_total += combo.second;
197 }
198
199 //define and check number of groups
200 if (xsecs.size() == 1)
201 n_grps = xs->num_groups;
202 else if (xs->num_groups != n_grps)
203 {
204 chi::log.LogAllError()
205 << "In call to " << __FUNCTION__ << ": "
206 << "all cross-sections must have the same number of groups.";
207 chi::Exit(EXIT_FAILURE);
208 }
209
210 //increment number of precursors
211 if (!xs->is_fissionable && xs->num_precursors > 0)
212 {
213 chi::log.LogAllError()
214 << "In call to " << __FUNCTION__ << ": "
215 << "Only fissionable materials are allowed to have delayed "
216 << "neutron precursors.";
217 chi::Exit(EXIT_FAILURE);
218 }
219 n_precs += xs->num_precursors;
220 }//for cross-section
221
222 // Check that the fissile and precursor densities are greater than
223 // machine precision. If this condition is not met, the material is assumed
224 // to be either not fissile, have zero precursors, or both.
225
226 double eps = 1.0e-12;
227 if (Nf_total < eps)
228 is_fissionable = false;
229 if (Np_total < eps)
230 num_precursors = 0;
231
232 //============================================================
233 // Initialize the data
234 //============================================================
235
236 num_groups = n_grps;
237 num_precursors = n_precs;
238
239 sigma_t.resize(n_grps, 0.0);
240 sigma_f.resize(n_grps, 0.0);
241 sigma_a.resize(n_grps, 0.0);
242 chi.resize(n_grps, 0.0);
243 chi_prompt.resize(n_grps, 0.0);
244 nu.resize(n_grps,0.0);
245 nu_prompt.resize(n_grps,0.0);
246 nu_delayed.resize(n_grps,0.0);
247 beta.resize(n_grps, 0.0);
248 nu_sigma_f.resize(n_grps, 0.0);
249 nu_prompt_sigma_f.resize(n_grps, 0.0);
250 nu_delayed_sigma_f.resize(n_grps, 0.0);
251 inv_velocity.resize(n_grps, 0.0);
252 precursor_lambda.resize(n_precs, 0.0);
253 precursor_yield.resize(n_precs, 0.0);
254 chi_delayed.resize(n_grps, std::vector<double>(n_precs, 0.0));
255
256 unsigned int precursor_count = 0;
257 for (size_t x = 0; x < xsecs.size(); ++x)
258 {
259 scattering_order = std::max(scattering_order,
260 xsecs[x]->scattering_order);
261
262 //atom density
263 double N_i = combinations[x].second;
264
265 //fraction of fissile density
266 double ff_i = 0.0;
267 if (xsecs[x]->is_fissionable)
268 ff_i = N_i / Nf_total;
269
270 //fraction of precursor density
271 double pf_i = 0.0;
272 if (xsecs[x]->num_precursors > 0)
273 pf_i = N_i / Np_total;
274
275 //============================================================
276 // Combine cross-sections
277 //============================================================
278
279 // Here, raw cross-sections are scaled by densities and
280 // spectra by fractional densities. The latter is done to
281 // preserve unit spectra. The inverse velocity term must be
282 // the same across all cross-section sets, so a check is
283 // performed to ensure this is the case.
284
285 for (unsigned int g = 0; g < n_grps; ++g)
286 {
287 sigma_t[g] += xsecs[x]->sigma_t[g] * N_i;
288 sigma_a[g] += xsecs[x]->sigma_a[g] * N_i;
289 sigma_f[g] += xsecs[x]->sigma_f[g] * N_i;
290
291 chi[g] += xsecs[x]->chi[g] * ff_i;
292 chi_prompt[g] += xsecs[x]->chi_prompt[g] * ff_i;
293
294 nu[g] += xsecs[x]->nu[g] * ff_i;
295 nu_prompt[g] += xsecs[x]->nu_prompt[g] * ff_i;
296 nu_delayed[g] += xsecs[x]->nu_delayed[g] * ff_i;
297 beta[g] += xsecs[x]->beta[g] * ff_i;
298
299 nu_sigma_f[g] += xsecs[x]->nu_sigma_f[g] * N_i;
300 nu_prompt_sigma_f[g] += xsecs[x]->nu_prompt_sigma_f[g] * N_i;
301 nu_delayed_sigma_f[g] += xsecs[x]->nu_delayed_sigma_f[g] * N_i;
302
303 if (x == 0)
304 inv_velocity[g] = xsecs[x]->inv_velocity[g];
305 else if (inv_velocity[g] != xsecs[x]->inv_velocity[g])
306 {
307 chi::log.LogAllError()
308 << "In call to " << __FUNCTION__ << ": "
309 << "All materials must have the same inv_velocity "
310 << "term per group. Invalid value encountered in "
311 << "material " << x << " group " << g << ". Using the "
312 << "value from the first cross-section set.";
313 chi::Exit(EXIT_FAILURE);
314 }
315 }
316
317 //============================================================
318 // Compute precursors
319 //============================================================
320
321 // Here, all precursors across all materials are stored.
322 // The decay constants and delayed spectrum are what they are,
323 // however, some special treatment must be given to the yields.
324 // Because the yield tells us what fraction of delayed neutrons
325 // are produced from a given family, the sum over all families
326 // must yield unity. To achieve this end, we must scale all
327 // precursor yields based on the fraction of the total density
328 // of materials with precursors they make up.
329
330 if (xsecs[x]->num_precursors > 0)
331 {
332 for (unsigned int j = 0; j < xsecs[x]->num_precursors; ++j)
333 {
334 unsigned int count = precursor_count + j;
335 precursor_lambda[count] = xsecs[x]->precursor_lambda[j];
336 precursor_yield [count] = xsecs[x]->precursor_yield [j] * pf_i;
337 for (size_t g = 0; g < num_groups; ++g)
338 chi_delayed[g][count] = xsecs[x]->chi_delayed[g][j];
339 }
340 precursor_count += xsecs[x]->num_precursors;
341 }
342 }//for cross sections
343
344 //============================================================
345 // Combine transfer matrices
346 //============================================================
347
348 // This step is somewhat tricky. The cross-sections
349 // aren't guaranteed to have the same sparsity patterns
350 // and therefore simply adding them together has to take
351 // the sparse matrix's protection mechanisms into account.
352
353 transfer_matrices.clear();
354 transfer_matrices.resize(scattering_order + 1,
355 chi_math::SparseMatrix(n_grps, n_grps));
356 for (size_t x = 0; x < xsecs.size(); ++x)
357 {
358 for (int m=0; m<(xsecs[x]->scattering_order + 1); ++m)
359 {
360 auto& xs_tm = xsecs[x]->transfer_matrices[m];
361 for (size_t i = 0; i < num_groups; ++i)
362 {
363 for (auto j : xs_tm.rowI_indices[i])
364 {
365 double value = xs_tm.ValueIJ(i,j) * combinations[x].second;
366 transfer_matrices[m].InsertAdd(i, j, value);
367 }
368 }//for i
369 }//for m
370 }//for xs
371
372 Finalize();
373}

377void chi_physics::TransportCrossSections::
378ComputeAbsorption()
379{
380 sigma_a.assign(num_groups, 0.0);
381
382 // compute for a pure absorber
383 if (transfer_matrices.empty())
384 for (size_t g = 0; g < num_groups; ++g)
385 sigma_a[g] = sigma_t[g];
386
387 // estimate from a transfer matrix
388 else
389 {
390 chi::log.LogAllWarning()
391 << "Estimating absorption from the transfer matrices.";
392
393 const auto& S0 = transfer_matrices[0];
394 for (size_t g = 0; g < num_groups; ++g)
395 {
396 // estimate the scattering cross-section
397 double sig_s = 0.0;
398 for (size_t row = 0; row < S0.NumRows(); ++row)
399 {
400 const auto& cols = S0.rowI_indices[row];
401 const auto& vals = S0.rowI_values[row];
402 for (size_t t = 0; t < cols.size(); ++t)
403 if (cols[t] == g)
404 {
405 sig_s += vals[t];
406 break;
407 }
408 }
409
410 sigma_a[g] = sigma_t[g] - sig_s;
411
412 // TODO: Decide whether this should be a warning or an error.
413 if (sigma_a[g] < 0.0)
414 chi::log.LogAllWarning()
415 << "Negative absorption cross-section encountered "
416 << "in group " << g << " when estimating from the "
417 << "transfer matrices";
418 }//for g
419 }//if scattering present
420
