275//######################################################### Gauss Elimination
276/** Gauss Elimination without pivoting.*/
277void chi_math::GaussElimination(MatDbl &A,
278 VecDbl &b, int n)
279{
280 // Forward elimination
281 for(int i = 0;i < n-1;++i)
282 {
283 const std::vector<double>& ai = A[i];
284 double bi = b[i];
285 double factor = 1.0/A[i][i];
286 for(int j = i+1;j < n;++j)
287 {
288 std::vector<double>& aj = A[j];
289 double val = aj[i] * factor;
290 b[j] -= val * bi;
291 for(int k = i+1;k < n;++k)
292 aj[k] -= val * ai[k];
293 }
294 }
295
296 // Back substitution
297 for(int i = n-1;i >= 0;--i)
298 {
299 const std::vector<double>& ai = A[i];
300 double bi = b[i];
301 for(int j = i+1;j < n;++j)
302 bi -= ai[j] * b[j];
303 b[i] = bi/ai[i];
304 }
305}

308/** Computes the inverse of a matrix using Gauss-Elimination with pivoting.*/
309MatDbl chi_math::InverseGEPivoting(const MatDbl &A)
310{
311 assert(A.size());
312 assert(A.size() == A[0].size());
313
314 const unsigned int R = A.size();
315
316 std::vector<std::vector<double> > M(R,std::vector<double>(R,0.));
317
318 for(unsigned int i = 0; i < R; i++)
319 M[i][i] = 1.;
320
321 std::vector<std::vector<double> > B = A;
322
323 for (unsigned int c = 0; c < R; c++)
324 {
325 // Find a row with the largest pivot value
326 unsigned int max_row=c; //nzr = non-zero row
327 for (unsigned int r = c; r < R; ++r)
328 if (std::fabs(B[r][c]) > std::fabs(B[max_row][c]))
329 max_row = r;
330
331 if (max_row != c )
332 {
333 SwapRow(max_row , c, B );
334 SwapRow(max_row , c, M );
335 }
336
337 // Eliminate non-zero values
338 for (unsigned int r = 0; r < R; r++)
339 {
340 if ( r != c )
341 {
342 double g = B[r][c]/B[c][c];
343 if ( B[r][c] != 0 )
344 {
345 for(unsigned int k=0; k < R; k++)
346 {
347 B[r][k] -= B[c][k]*g;
348 M[r][k] -= M[c][k]*g;
349 }
350 B[r][c] = 0;
351 }
352 }
353 else
354 {
355 double g = 1/B[c][c];
356 for (unsigned int k = 0; k < R; k++)
357 {
358 B[r][k] *= g;
359 M[r][k] *= g;
360 }
361 }
362 }
363 }
364 return M;
365}

374 double f(0.);
375
376 // Only calculate the determinant if matrix size is less than 5 since
377 // the inverse is directly calculated for larger matrices. Otherwise,
378 // the inverse routine spends all of its time sitting in the determinant
379 // function which is unnecessary.
380 if (R < 5)

382 f = Determinant(A);
383 assert(f != 0.);
384 f = 1./f;

441//######################################################### Power iteration
442/** Performs power iteration to obtain the fundamental eigen mode. The
443 * eigen-value of the fundamental mode is return whilst the eigen-vector
444 * is return via reference.*/
445double chi_math::PowerIteration(
446 const MatDbl &A, VecDbl &e_vec, int max_it, double tol)
447{
448 // Local Variables
449 unsigned int n = A.size();
450 int it_counter = 0;
451 VecDbl y(n,1.);
452 double lambda0 = 0.;
453
454 // Perform initial iteration outside of loop
455 VecDbl Ay = MatMul(A, y);
456 double lambda = Dot( y, Ay );
457 y = VecMul( Ay, 1./Vec2Norm(Ay) );
458 if (lambda < 0.)
459 Scale(y, -1.0);
460
461 // Perform convergence loop
462 bool converged = false;
463 while(!converged && it_counter < max_it)
464 {
465 // Update old eigenvalue
466 lambda0 = std::fabs(lambda);
467 // Calculate new eigenvalue/eigenvector
468 Ay = MatMul(A, y);
469 lambda = Dot( y, Ay );
470 y = VecMul( Ay, 1./Vec2Norm(Ay) );
471
472 // Check if converged or not
473 if (std::fabs(std::fabs(lambda) - lambda0) <= tol)
474 converged = true;
475 // Update counter
476 ++it_counter;
477 }
478
479 if (lambda < 0.)
480 Scale(y, -1.0);
481
482 // Renormalize eigenvector for the last time
483 y = VecMul(Ay, 1./lambda);
484
485 // Set eigenvector, return the eigenvalue
486 e_vec = std::move(y);
487
488 return lambda;
489}
