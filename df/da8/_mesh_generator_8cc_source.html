48MeshGenerator::MeshGenerator(const chi::InputParameters& params)
49 : ChiObject(params),
50 scale_(params.GetParamValue<double>("scale")),
51 replicated_(params.GetParamValue<bool>("replicated_mesh"))
52{
53 //============================================= Convert input handles
54 auto input_handles = params.GetParamVectorValue<size_t>("inputs");
55
56 for (const size_t input_handle : input_handles)

60 inputs_.push_back(&mesh_generator);

62

87void MeshGenerator::Execute()
88{
89 //======================================== Execute all input generators
90 // Note these could be empty
91 std::unique_ptr<UnpartitionedMesh> current_umesh = nullptr;
92 for (auto mesh_generator_ptr : inputs_)
93 {
94 auto new_umesh =
95 mesh_generator_ptr->GenerateUnpartitionedMesh(std::move(current_umesh));
96 current_umesh = std::move(new_umesh);
97 }
98
99 //======================================== Generate final umesh and convert it
100 current_umesh = GenerateUnpartitionedMesh(std::move(current_umesh));
101 auto grid_ptr = SetupMesh(std::move(current_umesh));
102
103 //======================================== Assign the mesh to a VolumeMesher
104 auto new_mesher =
105 std::make_shared<chi_mesh::VolumeMesher>(VolumeMesherType::UNPARTITIONED);
106 new_mesher->SetContinuum(grid_ptr);
107
108 if (Chi::current_mesh_handler < 0) chi_mesh::PushNewHandlerAndGetIndex();
109
110 auto& cur_hndlr = chi_mesh::GetCurrentHandler();
111 cur_hndlr.SetVolumeMesher(new_mesher);
112
113 Chi::mpi.Barrier();
114}
