material_id,

normal,
quadrature_angle_indices,

quadrature_phi_theta_angles,
group_indices,
time)

for gi=1,num_groups do
g = group_indices[gi]
value = 1.0

value = 0.0
end
dof_count = dof_count + 1
psi[dof_count] = value
end
end
return psi
end
