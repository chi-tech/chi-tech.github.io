115/**Returns an error string if the check has not passed. This method
116 * first checks whether all the required parameters are supplied. Then
117 * it checks that all the parameters supplied actually maps to valid parameters.
118 * */
119void InputParameters::AssignParameters(const ParameterBlock& params)
120{
121 param_block_at_assignment_ = params;
122 std::stringstream err_stream;
123
124 if (chi::log.GetVerbosity() >= 1)
125 chi::log.Log() << "Number of parameters " << params.NumParameters();
126
127 // ================================== Check required parameters
128 // Loops over all input-parameters that have been
129 // classed as being required. Input-parameters that
130 // have any form of deprecation is ignored.
131 for (const auto& [param_index, tag] : parameter_class_tags_)
132 {
133 if (tag != InputParameterTag::REQUIRED) continue;
134
135 const auto& req_param = GetParam(param_index);
136 const auto& req_param_name = req_param.Name();
137
138 if (deprecation_warning_tags_.count(req_param_name) > 0 or
139 deprecation_error_tags_.count(req_param_name) > 0 or
140 renamed_error_tags_.count(req_param_name) > 0)
141 continue;
142
143 if (not params.Has(req_param_name))
144 err_stream << "Required param \"" << req_param_name
145 << "\" not supplied.\ndoc-string: "
146 << GetParameterDocString(req_param_name);
147 }
148
149 if (not err_stream.str().empty()) ThrowInputError;
150
151 // ================================== Check unused parameters
152 // Loops over all candidate-parameters and
153 // checks whether they have an assignable
154 // input-parameter or if they have been renamed.
155 {
156 for (const auto& param : params.Parameters())
157 {
158 const auto& param_name = param.Name();
159 if (IsParameterIgnored(param_name)) continue;
160 if (not this->Has(param_name))
161 err_stream << "Invalid param \"" << param_name << "\" supplied.\n";
162 else if (renamed_error_tags_.count(param_name) > 0)
163 {
164 err_stream << "Invalid param \"" << param_name << "\" supplied. ";
165 err_stream << "The parameter has been renamed. ";
166 err_stream << renamed_error_tags_.at(param_name);
167 }
168 }
169
170 if (not err_stream.str().empty()) ThrowInputError;
171 }
172
173 // ================================== Check deprecation warnings
174 // Loops over all candidate-parameters and
175 // checks whether they have deprecation warnings.
176 {
177 const auto& dep_warns = deprecation_warning_tags_;
178 for (const auto& param : params.Parameters())
179 {
180 const auto& param_name = param.Name();
181
182 if (IsParameterIgnored(param_name)) continue;
183
184 if (this->Has(param_name) and (dep_warns.count(param_name) > 0))
185 chi::log.Log0Warning()
186 << "Parameter \"" << param_name << "\" has been deprecated "
187 << "and will be removed soon.\n"
188 << dep_warns.at(param_name);
189 }
190 }
191
192 // ================================== Check deprecation errors
193 // Loops over all candidate-parameters and
194 // checks whether they have deprecation errors.
195 {
196 const auto& dep_errs = deprecation_error_tags_;
197 for (const auto& param : params.Parameters())
198 {
199 const auto& param_name = param.Name();
200
201 if (IsParameterIgnored(param_name)) continue;
202
203 if (this->Has(param_name) and (dep_errs.count(param_name) > 0))
204 {
205 chi::log.Log0Error()
206 << "Parameter \"" << param_name << "\" has been deprecated.\n"
207 << dep_errs.at(param_name);
208 chi::Exit(EXIT_FAILURE);
209 }
210 }
211 }
212
213 // ================================== Now attempt to assign values
214 for (auto& param : params.Parameters())
215 {
216 const auto& param_name = param.Name();
217
218 if (IsParameterIgnored(param_name)) continue;
219
220 auto& input_param = GetParam(param_name);
221
222 // ====================== Check types match
223 if (param.Type() != input_param.Type())
224 {
225 if (type_mismatch_allowed_tags_.count(param_name) == 0)
226 {
227 err_stream << "Invalid parameter type \""
228 << ParameterBlockTypeName(param.Type())
229 << "\" for parameter \"" << param_name
230 << "\". Expecting type \""
231 << ParameterBlockTypeName(input_param.Type()) << "\".\n"
232 << "doc-string: " << GetParameterDocString(param_name);
233 continue;
234 } // if not mismatch allowed
235 } // if type mismatch
236
237 // ====================== Check constraint
238 if (constraint_tags_.count(input_param.Name()) != 0)
239 {
240 const auto& constraint = constraint_tags_.at(input_param.Name());
241 if (not constraint->IsAllowable(param.Value()))
242 {
243 err_stream << constraint->OutOfRangeString(input_param.Name(),
244 param.Value());
245 err_stream << "\n";
246 continue;
247 }
248 } // if constraint
249
250 if (chi::log.GetVerbosity() >= 1)
251 chi::log.Log() << "Setting parameter " << param_name;
252 input_param = param;
253 } // for input params
254
255 if (not err_stream.str().empty()) ThrowInputError;
256}

296{
297 if (Has(param_name))
298 {

324/**Dumps the input parameters to stdout.*/
325void InputParameters::DumpParameters() const
326{
327
328 chi::log.Log() << "DESCRIPTION_BEGIN";
329 std::cout << GetGeneralDescription() << "\n";
330 chi::log.Log() << "DESCRIPTION_END\n";
331
332 const std::string sp2 = " ";
333 const std::string sp4 = " ";
334 for (const auto& param : Parameters())
335 {
336 const auto& param_name = param.Name();
337 chi::log.Log() << sp2 << "PARAM_BEGIN " << param_name;
338
339 const auto type = param.Type();
340
341 chi::log.Log() << sp4 << "TYPE " << ParameterBlockTypeName(type);
342
343 if (parameter_class_tags_.at(param_name) == InputParameterTag::OPTIONAL)
344 {
345 chi::log.Log() << sp4 << "TAG OPTIONAL";
346 if (type != ParameterBlockType::BLOCK and
347 type != ParameterBlockType::ARRAY)
348 chi::log.Log() << sp4 << "DEFAULT_VALUE " << param.Value().PrintStr();
349 else if (type == ParameterBlockType::ARRAY)
350 {
351 std::stringstream outstr;
352 outstr << sp4 << "DEFAULT_VALUE ";
353 for (size_t k = 0; k < param.NumParameters(); ++k)
354 {
355 const auto& sub_param = param.GetParam(k);
356 outstr << sub_param.Value().PrintStr() << ", ";
357 }
358 chi::log.Log() << outstr.str();
359 }
360 }
361 else
362 chi::log.Log() << sp4 << "TAG REQUIRED";
363
364 if (constraint_tags_.count(param_name) != 0)
365 chi::log.Log() << sp4 << "CONSTRAINTS "
366 << constraint_tags_.at(param_name)->PrintRange();
367
368 if (parameter_doc_string_.count(param_name) != 0)
369 {
370 chi::log.Log() << "DOC_STRING_BEGIN";
371 std::cout << parameter_doc_string_.at(param_name) << "\n";
372 chi::log.Log() << "DOC_STRING_END";
373 }
374
375 chi::log.Log() << sp2 << "PARAM_END";
376 }
