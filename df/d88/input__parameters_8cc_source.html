135/**Returns an error string if the check has not passed. This method
136 * first checks whether all the required parameters are supplied. Then
137 * it checks that all the parameters supplied actually maps to valid parameters.
138 * */
139void InputParameters::AssignParameters(const ParameterBlock& params)
140{
141 param_block_at_assignment_ = params;
142 std::stringstream err_stream;
143
144 if (Chi::log.GetVerbosity() >= 1)
145 Chi::log.Log() << "Number of parameters " << params.NumParameters();
146
147 // ================================== Check required parameters
148 // Loops over all input-parameters that have been
149 // classed as being required. Input-parameters that
150 // have any form of deprecation is ignored.
151 for (const auto& [param_index, tag] : parameter_class_tags_)
152 {
153 if (tag != InputParameterTag::REQUIRED) continue;
154
155 const auto& req_param = GetParam(param_index);
156 const auto& req_param_name = req_param.Name();
157
158 if (deprecation_warning_tags_.count(req_param_name) > 0 or
159 deprecation_error_tags_.count(req_param_name) > 0 or
160 renamed_error_tags_.count(req_param_name) > 0)
161 continue;
162
163 if (not params.Has(req_param_name))
164 err_stream << "Required param \"" << req_param_name
165 << "\" not supplied.\ndoc-string: "
166 << GetParameterDocString(req_param_name);
167 }
168
169 if (not err_stream.str().empty()) ThrowInputError;
170
171 // ================================== Check unused parameters
172 // Loops over all candidate-parameters and
173 // checks whether they have an assignable
174 // input-parameter or if they have been renamed.
175 {
176 for (const auto& param : params.Parameters())
177 {
178 const auto& param_name = param.Name();
179 if (IsParameterIgnored(param_name)) continue;
180 if (not this->Has(param_name))
181 err_stream << "Invalid param \"" << param_name << "\" supplied.\n";
182 else if (renamed_error_tags_.count(param_name) > 0)
183 {
184 err_stream << "Invalid param \"" << param_name << "\" supplied. ";
185 err_stream << "The parameter has been renamed. ";
186 err_stream << renamed_error_tags_.at(param_name);
187 }
188 }
189
190 if (not err_stream.str().empty()) ThrowInputError;
191 }
192
193 // ================================== Check deprecation warnings
194 // Loops over all candidate-parameters and
195 // checks whether they have deprecation warnings.
196 {
197 const auto& dep_warns = deprecation_warning_tags_;
198 for (const auto& param : params.Parameters())
199 {
200 const auto& param_name = param.Name();
201
202 if (IsParameterIgnored(param_name)) continue;
203
204 if (this->Has(param_name) and (dep_warns.count(param_name) > 0))
205 Chi::log.Log0Warning()
206 << "Parameter \"" << param_name << "\" has been deprecated "
207 << "and will be removed soon.\n"
208 << dep_warns.at(param_name);
209 }
210 }
211
212 // ================================== Check deprecation errors
213 // Loops over all candidate-parameters and
214 // checks whether they have deprecation errors.
215 {
216 const auto& dep_errs = deprecation_error_tags_;
217 for (const auto& param : params.Parameters())
218 {
219 const auto& param_name = param.Name();
220
221 if (IsParameterIgnored(param_name)) continue;
222
223 if (this->Has(param_name) and (dep_errs.count(param_name) > 0))
224 {
225 Chi::log.Log0Error()
226 << "Parameter \"" << param_name << "\" has been deprecated.\n"
227 << dep_errs.at(param_name);
228 Chi::Exit(EXIT_FAILURE);
229 }
230 }
231 }
232
233 // ================================== Now attempt to assign values
234 for (auto& param : params.Parameters())
235 {
236 const auto& param_name = param.Name();
237
238 if (IsParameterIgnored(param_name)) continue;
239
240 auto& input_param = GetParam(param_name);
241
242 // ====================== Check types match
243 if (param.Type() != input_param.Type())
244 {
245 if (type_mismatch_allowed_tags_.count(param_name) == 0)
246 {
247 err_stream << "Invalid parameter type \""
248 << ParameterBlockTypeName(param.Type())
249 << "\" for parameter \"" << param_name
250 << "\". Expecting type \""
251 << ParameterBlockTypeName(input_param.Type()) << "\".\n"
252 << "doc-string: " << GetParameterDocString(param_name);
253 continue;
254 } // if not mismatch allowed
255 } // if type mismatch
256
257 // ====================== Check constraint
258 if (constraint_tags_.count(input_param.Name()) != 0)
259 {
260 const auto& constraint = constraint_tags_.at(input_param.Name());
261 if (not constraint->IsAllowable(param.Value()))
262 {
263 err_stream << constraint->OutOfRangeString(input_param.Name(),
264 param.Value());
265 err_stream << "\n";
266 continue;
267 }
268 } // if constraint
269
270 if (Chi::log.GetVerbosity() >= 1)
271 Chi::log.Log() << "Setting parameter " << param_name;
272 input_param = param;
273 } // for input params
274
275 if (not err_stream.str().empty()) ThrowInputError;
276}

344/**Dumps the input parameters to stdout.*/
345void InputParameters::DumpParameters() const
346{
347 Chi::log.Log() << "CLASS_NAME " << class_name_;
348
349 Chi::log.Log() << "DESCRIPTION_BEGIN";
350 std::cout << GetGeneralDescription() << "\n";
351 Chi::log.Log() << "DESCRIPTION_END\n";
352
353 Chi::log.Log() << "DOC_GROUP " << doc_group_;
354
355 const std::string sp2 = " ";
356 const std::string sp4 = " ";
357 const auto params = Parameters();
358 for (const auto& param : params)
359 {
360 const auto& param_name = param.Name();
361 Chi::log.Log() << sp2 << "PARAM_BEGIN " << param_name;
362
363 const auto type = param.Type();
364
365 Chi::log.Log() << sp4 << "TYPE " << ParameterBlockTypeName(type);
366
367 if (parameter_class_tags_.at(param_name) == InputParameterTag::OPTIONAL)
368 {
369 Chi::log.Log() << sp4 << "TAG OPTIONAL";
370 if (type != ParameterBlockType::BLOCK and
371 type != ParameterBlockType::ARRAY)
372 Chi::log.Log() << sp4 << "DEFAULT_VALUE " << param.Value().PrintStr();
373 else if (type == ParameterBlockType::ARRAY)
374 {
375 std::stringstream outstr;
376 outstr << sp4 << "DEFAULT_VALUE ";
377 for (size_t k = 0; k < param.NumParameters(); ++k)
378 {
379 const auto& sub_param = param.GetParam(k);
380 outstr << sub_param.Value().PrintStr() << ", ";
381 }
382 Chi::log.Log() << outstr.str();
383 }
384 }
385 else
386 Chi::log.Log() << sp4 << "TAG REQUIRED";
387
388 if (constraint_tags_.count(param_name) != 0)
389 Chi::log.Log() << sp4 << "CONSTRAINTS "
390 << constraint_tags_.at(param_name)->PrintRange();
391
392 if (parameter_doc_string_.count(param_name) != 0)
393 {
394 Chi::log.Log() << sp4 << "DOC_STRING_BEGIN";
395 std::cout << parameter_doc_string_.at(param_name) << "\n";
396 Chi::log.Log() << sp4 << "DOC_STRING_END";
397 }
398
399 const auto& linkage = GetParameterDocumentationLink(param_name);
400 if (not linkage.empty())
401 {
402 Chi::log.Log() << sp4 << "LINKS " << linkage;
403 }
404
405 Chi::log.Log() << sp2 << "PARAM_END";
406 }
