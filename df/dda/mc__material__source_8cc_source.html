54 
55  if (cell->Type() == chi_mesh::CellType::SLAB)
56  {
57  auto v0 = *ref_grid->nodes[cell->vertex_ids[0]];

206 chi_montecarlon::Particle chi_montecarlon::MaterialSource::
207  CreateParticle(chi_montecarlon::RandomNumberGenerator *rng)
208 {
209  Particle new_particle;
210 
211  //======================================== Sample group
212  int g = std::lower_bound(
213  group_cdf.begin(),
214  group_cdf.end(),
215  rng->Rand()) - group_cdf.begin();
216 
217  new_particle.egrp = g;
218 
219  if (group_element_cdf[g].empty())
220  {
221  new_particle.alive = false;
222  return new_particle;
223  }
224  if (g>0)
225  chi_log.Log(LOG_0) << g;
226 
227  //======================================== Sample element
228  int elem = std::lower_bound(
229  group_element_cdf[g].begin(),
230  group_element_cdf[g].end(),
231  rng->Rand()) - group_element_cdf[g].begin();
232 
233  auto& src_element = group_sources[g][elem];
234 
235  //======================================== Sample position
236  if (src_element.geom_legs.size() == 1) //Slab
237  {
238  double u = rng->Rand();
239  new_particle.pos = src_element.ref_point +
240  src_element.geom_legs[0]*u;
241  }
242  else if (src_element.geom_legs.size() == 2) //Triangle
243  {
244  double u=rng->Rand();
245  double v=rng->Rand();
246 
247  while ((u+v)>1.0)
248  {u=rng->Rand(); v=rng->Rand();}
249 
250  new_particle.pos = src_element.ref_point +
251  src_element.geom_legs[0]*u +
252  src_element.geom_legs[1]*v;
253  }
254  else if (src_element.geom_legs.size() == 3) //Tetrahedron
255  {
256  double u=rng->Rand();
257  double v=rng->Rand();
258  double w=rng->Rand();
259 
260  while ((u+v+w)>1.0)
261  {u=rng->Rand(); v=rng->Rand(); w=rng->Rand();}
262 
263  new_particle.pos = src_element.ref_point +
264  src_element.geom_legs[0]*u +
265  src_element.geom_legs[1]*v +
266  src_element.geom_legs[2]*w;
267  }
268 
269  //======================================== Sample direction
270  double costheta = 2.0*rng->Rand() - 1.0;
271  double theta = acos(costheta);
272  double varphi = rng->Rand()*2.0*M_PI;
273 
274  chi_mesh::Vector ref_dir;
275  ref_dir.x = sin(theta)*cos(varphi);
276  ref_dir.y = sin(theta)*sin(varphi);
277  ref_dir.z = cos(theta);
278 
279  new_particle.dir = ref_dir;
280 
281  //======================================== Determine weight
282  new_particle.w = 1.0;
283 
284  new_particle.cur_cell_ind = src_element.cell_g_index;
285 
286  return new_particle;
