48void lbs::SweepChunkPWL::
49 Sweep(chi_mesh::sweep_management::AngleSet *angle_set)
50{
51 if (!a_and_b_initialized)
52 {
53 Amat.resize(max_num_cell_dofs, std::vector<double>(max_num_cell_dofs));
54 Atemp.resize(max_num_cell_dofs, std::vector<double>(max_num_cell_dofs));
55 b.resize(num_grps, std::vector<double>(max_num_cell_dofs, 0.0));
56 source.resize(max_num_cell_dofs, 0.0);
57 a_and_b_initialized = true;
58 }
59
60 const auto spds = angle_set->GetSPDS();
61 const auto fluds = angle_set->fluds;
62 const bool surface_source_active = IsSurfaceSourceActive();
63 std::vector<double>& output_phi = GetDestinationPhi();
64 std::vector<double>& output_psi = GetDestinationPsi();
65
66 const SubSetInfo& grp_ss_info = groupset.grp_subset_infos[angle_set->ref_subset];
67 const size_t gs_ss_size = grp_ss_info.ss_size;
68 const size_t gs_ss_begin = grp_ss_info.ss_begin;
69 const int gs_gi = groupset.groups[gs_ss_begin].id; // Groupset subset first group number
70
71 int deploc_face_counter = -1;
72 int preloc_face_counter = -1;
73
74 auto const& d2m_op = groupset.quadrature->GetDiscreteToMomentOperator();
75 auto const& m2d_op = groupset.quadrature->GetMomentToDiscreteOperator();
76
77 // ========================================================== Loop over each cell
78 size_t num_loc_cells = spds->spls.item_id.size();
79 for (size_t spls_index = 0; spls_index < num_loc_cells; ++spls_index)
80 {
81 const int cell_local_id = spds->spls.item_id[spls_index];
82 const auto& cell = grid_view->local_cells[cell_local_id];
83 const auto& fe_intgrl_values = grid_fe_view.GetUnitIntegrals(cell);
84 const auto num_faces = cell.faces.size();
85 const int num_nodes = static_cast<int>(fe_intgrl_values.NumNodes());
86 auto& transport_view = grid_transport_view[cell.local_id];
87 const auto& sigma_tg = transport_view.XS().sigma_t;
88 std::vector<bool> face_incident_flags(num_faces, false);
89 std::vector<double> face_mu_values(num_faces, 0.0);
90
91 // =================================================== Get Cell matrices
92 const auto& G = fe_intgrl_values.GetIntV_shapeI_gradshapeJ();
93 const auto& M = fe_intgrl_values.GetIntV_shapeI_shapeJ();
94 const auto& M_surf = fe_intgrl_values.GetIntS_shapeI_shapeJ();
95 const auto& IntS_shapeI = fe_intgrl_values.GetIntS_shapeI();
96
97 // =================================================== Loop over angles in set
98 const int ni_deploc_face_counter = deploc_face_counter;
99 const int ni_preloc_face_counter = preloc_face_counter;
100 const size_t as_num_angles = angle_set->angles.size();
101 for (size_t angle_set_index = 0; angle_set_index<as_num_angles; ++angle_set_index)
102 {
103 deploc_face_counter = ni_deploc_face_counter;
104 preloc_face_counter = ni_preloc_face_counter;
105 const int angle_num = angle_set->angles[angle_set_index];
106 const chi_mesh::Vector3& omega = groupset.quadrature->omegas[angle_num];
107 const double wt = groupset.quadrature->weights[angle_num];
108
109 // ============================================ Gradient matrix
110 for (int i = 0; i < num_nodes; ++i)
111 for (int j = 0; j < num_nodes; ++j)
112 Amat[i][j] = omega.Dot(G[i][j]);
113
114 for (int gsg = 0; gsg < gs_ss_size; ++gsg)
115 b[gsg].assign(num_nodes, 0.0);
116
117 // ============================================ Upwinding structure
118 Upwinder upwind{fluds, angle_set, spls_index, angle_set_index,
119 /*in_face_counter*/0,
120 /*preloc_face_counter*/0,
121 /*out_face_counter*/0,
122 /*deploc_face_counter*/0,
123 /*bndry_id*/0, angle_num, cell.local_id,
124 /*f*/0,gs_gi, gs_ss_begin,
125 surface_source_active};
126
127 // ============================================ Surface integrals
128 int in_face_counter = -1;
129 for (int f = 0; f < num_faces; ++f)
130 {
131 const auto& face = cell.faces[f];
132 const double mu = omega.Dot(face.normal);
133 face_mu_values[f] = mu;
134
135 if (mu >= 0.0) continue;
136
137 face_incident_flags[f] = true;
138 const bool local = transport_view.IsFaceLocal(f);
139 const bool boundary = not face.has_neighbor;
140 const uint64_t bndry_id = face.neighbor_id;
141
142 if (local) ++in_face_counter;
143 else if (not boundary) ++preloc_face_counter;
144
145 upwind.in_face_counter = in_face_counter;
146 upwind.preloc_face_counter = preloc_face_counter;
147 upwind.bndry_id = bndry_id;
148 upwind.f = f;
149
150 const size_t num_face_indices = face.vertex_ids.size();
151 for (int fi = 0; fi < num_face_indices; ++fi)
152 {
153 const int i = fe_intgrl_values.FaceDofMapping(f,fi);
154 for (int fj = 0; fj < num_face_indices; ++fj)
155 {
156 const int j = fe_intgrl_values.FaceDofMapping(f,fj);
157
158 const double* psi = upwind.GetUpwindPsi(fj, local, boundary);
159
160 const double mu_Nij = -mu * M_surf[f][i][j];
161 Amat[i][j] += mu_Nij;
162 for (int gsg = 0; gsg < gs_ss_size; ++gsg)
163 b[gsg][i] += psi[gsg]*mu_Nij;
164
165 }//for face j
166 }//for face i
167 } // for f
168
169 // ========================================== Looping over groups
170 for (int gsg = 0; gsg < gs_ss_size; ++gsg)
171 {
172 const int g = gs_gi+gsg;
173
174 // ============================= Contribute source moments
175 // q = M_n^T * q_moms
176 for (int i = 0; i < num_nodes; ++i)
177 {
178 double temp_src = 0.0;
179 for (int m = 0; m < num_moms; ++m)
180 {
181 const size_t ir = transport_view.MapDOF(i, m, g);
182 temp_src += m2d_op[m][angle_num]*q_moments[ir];
183 }//for m
184 source[i] = temp_src;
185 }//for i
186
187 // ============================= Mass Matrix and Source
188 // Atemp = Amat + sigma_tgr * M
189 // b += M * q
190 const double sigma_tgr = sigma_tg[g];
191 for (int i = 0; i < num_nodes; ++i)
192 {
193 double temp = 0.0;
194 for (int j = 0; j < num_nodes; ++j)
195 {
196 const double Mij = M[i][j];
197 Atemp[i][j] = Amat[i][j] + Mij*sigma_tgr;
198 temp += Mij*source[j];
199 }//for j
200 b[gsg][i] += temp;
201 }//for i
202
203 // ============================= Solve system
204 chi_math::GaussElimination(Atemp, b[gsg], num_nodes);
205 }
206
207 // ============================= Accumulate flux
208 for (int m = 0; m < num_moms; ++m)
209 {
210 const double wn_d2m = d2m_op[m][angle_num];
211 for (int i = 0; i < num_nodes; ++i)
212 {
213 const size_t ir = transport_view.MapDOF(i, m, gs_gi);
214 for (int gsg = 0; gsg < gs_ss_size; ++gsg)
215 output_phi[ir + gsg] += wn_d2m * b[gsg][i];
216 }
217 }
218
219 for (auto& callback : moment_callbacks)
220 callback(this, angle_set);
221
222 // ============================= Save angular fluxes if needed
223 if (save_angular_flux)
224 {
225 const auto& psi_uk_man = groupset.psi_uk_man;
226 for (int i = 0; i < num_nodes; ++i)
227 {
228 int64_t ir = grid_fe_view.MapDOFLocal(cell,i,psi_uk_man,angle_num,0);
229 for (int gsg = 0; gsg < gs_ss_size; ++gsg)
230 output_psi[ir + gsg] = b[gsg][i];
231 }//for i
232 }//if save psi
233
234 int out_face_counter = -1;
235 for (int f = 0; f < num_faces; ++f)
236 {
237 if (face_incident_flags[f]) continue;
238 double mu = face_mu_values[f];
239
240 // ============================= Set flags and counters
241 out_face_counter++;
242 const auto& face = cell.faces[f];
243 const bool local = transport_view.IsFaceLocal(f);
244 const bool boundary = not face.has_neighbor;
245 const size_t num_face_indices = face.vertex_ids.size();
246 const std::vector<double>& IntF_shapeI = IntS_shapeI[f];
247 const uint64_t bndry_id = face.neighbor_id;
248
249 bool reflecting_bndry = false;
250 if (boundary)
251 if (angle_set->ref_boundaries[bndry_id]->IsReflecting())
252 reflecting_bndry = true;
253
254 if (not boundary and not local) ++deploc_face_counter;
255
256 upwind.out_face_counter = out_face_counter;
257 upwind.deploc_face_counter = deploc_face_counter;
258 upwind.bndry_id = bndry_id;
259 upwind.f = f;
260
261 for (int fi = 0; fi < num_face_indices; ++fi)
262 {
263 const int i = fe_intgrl_values.FaceDofMapping(f,fi);
264
265 double* psi = upwind.GetDownwindPsi(fi, local, boundary, reflecting_bndry);
266
267 if (not boundary or reflecting_bndry)
268 for (int gsg = 0; gsg < gs_ss_size; ++gsg)
269 psi[gsg] = b[gsg][i];
270 if (boundary and not reflecting_bndry)
271 for (int gsg = 0; gsg < gs_ss_size; ++gsg)
272 transport_view.AddOutflow(gs_gi + gsg,
273 wt*mu*b[gsg][i]*IntF_shapeI[i]);
274 }//for fi
275 }//for face
276 } // for n
277 } // for cell
278}//Sweep
