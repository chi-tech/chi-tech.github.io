16 const double scale)
17{
18 const std::string fname =
19 "chi_mesh::UnpartitionedMesh::CopyUGridCellsAndPoints";
20
21 uint64_t total_cell_count = ugrid.GetNumberOfCells();
22 uint64_t total_point_count = ugrid.GetNumberOfPoints();
23
24 chi::log.Log()
25 << "Clean grid num cells and points: "
26 << total_cell_count << " "
27 << total_point_count;
28
29 //======================================== Push cells
30 for (size_t c=0; c<total_cell_count; ++c)
31 {
32 auto vtk_cell = ugrid.GetCell(static_cast<vtkIdType>(c));
33 auto vtk_celldim = vtk_cell->GetCellDimension();
34
35 if (vtk_celldim == 3)
36 raw_cells_.push_back(CreateCellFromVTKPolyhedron(vtk_cell));
37 else if (vtk_celldim == 2)
38 raw_cells_.push_back(CreateCellFromVTKPolygon(vtk_cell));
39 else if (vtk_celldim == 1)
40 raw_cells_.push_back(CreateCellFromVTKLine(vtk_cell));
41 else if (vtk_celldim == 0)
42 raw_cells_.push_back(CreateCellFromVTKVertex(vtk_cell));
43 else
44 throw std::logic_error(fname + ": Unsupported cell dimension.");
45 }//for c
46
47 //======================================== Push points
48 for (size_t p=0; p<total_point_count; ++p)
49 {
50 auto point = ugrid.GetPoint(static_cast<vtkIdType>(p));
51
52 point[0] = point[0]*scale;
53 point[1] = point[1]*scale;
54 point[2] = point[2]*scale;
55
56 vertices_.emplace_back(point[0], point[1], point[2]);
57
58 if (not bound_box_)
59 bound_box_ = std::shared_ptr<BoundBox>(new BoundBox{point[0],point[0],
60 point[1],point[1],
61 point[2],point[2]});
62
63 bound_box_->xmin = std::min(bound_box_->xmin, point[0]);
64 bound_box_->xmax = std::max(bound_box_->xmax, point[0]);
65 bound_box_->ymin = std::min(bound_box_->ymin, point[1]);
66 bound_box_->ymax = std::max(bound_box_->ymax, point[1]);
67 bound_box_->zmin = std::min(bound_box_->zmin, point[2]);
68 bound_box_->zmax = std::max(bound_box_->zmax, point[2]);
69 }

74/**Set material-ids based on block-wise material ids.*/
75void chi_mesh::UnpartitionedMesh::
76 SetMaterialIDsFromBlocks(const std::vector<uint64_t> &block_mat_ids)
77{
78 size_t total_cell_count = raw_cells_.size();
79 for (size_t c=0; c<total_cell_count; ++c)
80 {
81 auto cell = raw_cells_[c];
82
83 int mat_id=-1;
84 uint64_t prev_block_lim = 0;
85 for (auto block_lim : block_mat_ids)
86 {
87 ++mat_id;
88 if (c<block_lim and c>=prev_block_lim) break;
89
90 prev_block_lim=block_lim;
91 }
92
93 cell->material_id = mat_id;
94 }
95}

116 const double EPSILON = 1.0e-12;
117 //======================================== Build boundary faces
118 std::vector<LightWeightFace*> bndry_faces;
119 for (auto& cell_ptr : raw_cells_)
120 for (auto& face : cell_ptr->faces)
121 if (not face.has_neighbor)
122 bndry_faces.push_back(&face);
123
124 chi::log.Log() << "Number of boundary faces: " << bndry_faces.size();
125
126 //======================================== Build boundary vertex ids
127 std::set<uint64_t> bndry_vids_set;
128 for (const auto& face_ptr : bndry_faces)
129 for (const auto vid : face_ptr->vertex_ids)
130 bndry_vids_set.insert(vid);
131
132 //======================================== Process each boundary block
133 uint64_t bndry_id = 0;
134 for (const auto& ugrid_name : bndry_grid_blocks)
135 {
136 auto ugrid = ugrid_name.first;
137
138 mesh_options_.boundary_id_map[bndry_id] = ugrid_name.second;
139
140 //================================= Build vertex map
141 bool mapping_failed = false;
142 std::vector<size_t> vertex_map(ugrid->GetNumberOfPoints(),0);
143 for (size_t p=0; p<ugrid->GetNumberOfPoints(); ++p)
144 {
145 chi_mesh::Vector3 point;
146 ugrid->GetPoint(static_cast<vtkIdType>(p), &point.x);
147
148 bool map_found = false;
149 for (const auto vid : bndry_vids_set)
150 if ((point - vertices_[vid]).NormSquare() < EPSILON)
151 {
152 vertex_map[p] = vid;
153 map_found = true;
154 break;
155 }

157 if (not map_found)

159 chi::log.Log0Warning() << "chi_mesh::UnpartitionedMesh::"
160 "SetBoundaryIDsFromBlocks: Failed to map a vertex. " +
161 point.PrintStr() + " for boundary " + ugrid_name.second +
162 " therefore the boundary assignment was skipped.";
163 mapping_failed = true;
164 break;
165 }
166 }//for point in boundary block
167
168 if (mapping_failed) continue;
169
170 //================================= Build vertex subscriptions
171 std::map<uint64_t , std::set<size_t>> vertex_face_subs;
172 for (size_t f=0; f<bndry_faces.size(); ++f)
173 for (const auto vid : bndry_faces[f]->vertex_ids)
174 vertex_face_subs[vid].insert(f);

176 //================================= Process each cell in bndry block
177 size_t num_faces_boundarified = 0;
178 auto& bndry_block = ugrid_name.first;

180 for (size_t bc=0; bc<num_bndry_block_cells; ++bc)

182 auto bndry_cell = bndry_block->GetCell(static_cast<vtkIdType>(bc));
183
184 //========================== Build list of face candidates
185 // and vertex set
186 std::set<size_t> face_ids_short_list;
187 std::set<uint64_t> bndry_cell_id_set;
188 size_t num_points = bndry_cell->GetNumberOfPoints();
189 for (size_t p=0; p<num_points; ++p)
190 {
191 auto point_id = bndry_cell->GetPointId(static_cast<int>(p));
192 bndry_cell_id_set.insert(vertex_map[point_id]);
193 for (size_t face_id : vertex_face_subs[vertex_map[point_id]])
194 face_ids_short_list.insert(face_id);
195 }//for point p
196
197 for (size_t face_id : face_ids_short_list)
198 {
199 auto& face = bndry_faces[face_id];
200 const auto& face_vids = face->vertex_ids;
201 std::set<uint64_t> face_id_set(face_vids.begin(), face_vids.end());
202
203 if (face_id_set == bndry_cell_id_set)
204 {
205 face->neighbor = bndry_id;
206 ++num_faces_boundarified;
207 }
208 }//for face_id
209 }//for boundary cell bc
