27int lbs::DiscOrdKEigenvalueSolver::NonLinearKEigen()
28{
29 chi::log.Log()
30 << "\n********** Solving k-eigenvalue problem with "
31 << "Non-Linear iteration.\n";
32
33 const auto& sdm = *discretization_;
34
35 std::vector<int> groupset_ids;
36 for (auto& groupset : groupsets_)
37 groupset_ids.push_back(groupset.id_);
38
39 const size_t num_local_phi_dofs = sdm.GetNumLocalDOFs(flux_moments_uk_man_);
40 const size_t num_globl_phi_dofs = sdm.GetNumGlobalDOFs(flux_moments_uk_man_);
41
42 size_t num_local_psi_dofs = 0;
43 size_t num_globl_psi_dofs = 0;
44 for (auto& groupset : groupsets_)
45 {
46 const auto num_delayed_psi_info =
47 groupset.angle_agg_.GetNumDelayedAngularDOFs();
48 num_local_psi_dofs += num_delayed_psi_info.first;
49 num_globl_psi_dofs += num_delayed_psi_info.second;
50 }
51
52 const size_t num_local_dofs = num_local_phi_dofs + num_local_psi_dofs;
53 const size_t num_globl_dofs = num_globl_phi_dofs + num_globl_psi_dofs;
54
55 chi::log.Log()
56 << "Total number of angular unknowns: "
57 << num_globl_psi_dofs;
58
59 KResidualFunctionContext residual_function_context{TextName(), 1.0};
60
61 //============================================= Create the vectors
62 Vec phi, r; /* solution, residual vectors */
63 phi = chi_math::PETScUtils::CreateVector(sc_int64_t(num_local_dofs),
64 sc_int64_t(num_globl_dofs));
65 VecDuplicate(phi, &r);
66
67 //============================================= Create SNES
68 SNES snes; /* nonlinear solver context */
69
70 SNESCreate(PETSC_COMM_WORLD, &snes);
71 SNESSetType(snes, SNESNEWTONLS);
72 SNESSetApplicationContext(snes, &(*primary_ags_solver_->GetContext()));
73// SNESSetFromOptions(snes);
74
75 SNESSetTolerances(snes,
76 tolerance_, //absolute tolerance
77 1.0e-50, //relative tolerance
78 tolerance_, //solution tolerance
79 static_cast<int>(max_iterations_), //max iterations
80 -1); //max r-evals
81
82 SNESSetMaxLinearSolveFailures(snes, 1000);
83
84 SNESMonitorSet(snes, &lbs::KEigenSNESMonitor,
85 &residual_function_context, nullptr);
86
87 //============================================= Set the residual function
88 SNESSetFunction(snes, r, SNESKResidualFunction, &residual_function_context);
89
90 //============================================= Setup Matrix-Free Jacobian
91 Mat J; /* Jacobian matrix */
92
93 MatCreateSNESMF(snes, &J);
94 SNESSetJacobian(snes, J, J, MatMFFDComputeJacobian, nullptr);
95
96 //============================================= Set linear solver and
97 // preconditioner
98 KSP ksp; /* linear solver context */
99 PC pc; /* preconditioner context */
100
101 SNESGetKSP(snes, &ksp);
102 KSPSetType(ksp, KSPGMRES);
103
104 auto& first_groupset = groupsets_.front();
105 KSPSetTolerances(ksp,
106 1.0e-50, //relative tol
107 first_groupset.residual_tolerance_, //absolute tol
108 1.0e6, //divergence tol
109 first_groupset.max_iterations_); //max iters
110
111 KSPGMRESSetRestart(ksp, first_groupset.gmres_restart_intvl_);
112
113 KSPSetOptionsPrefix(ksp, (" " + TextName() + "_NonLinearK_Inner").c_str());
114 if (options_.verbose_inner_iterations)
115 KSPMonitorSet(ksp, &chi_math::PETScUtils::KSPMonitorRelativeToRHS, nullptr, nullptr);
116
117 KSPGetPC(ksp, &pc);
118 PCSetType(pc, PCNONE);
119
120 //============================================= Compute initial guess
121 SetPhiVectorScalarValues(phi_old_local_, 1.0);
122
123 DiscOrdSteadyStateSolver::SetMultiGSPETScVecFromPrimarySTLvector(
124 groupset_ids, phi, PhiSTLOption::PHI_OLD);
125
126 //============================================= Solve
127 SNESSolve(snes, nullptr, phi);
128
129 //============================================= Unpack solution
130 SetPrimarySTLvectorFromGroupScopedPETScVec(
131 groups_.front().id_, groups_.back().id_, phi, phi_old_local_);
132
133 //============================================= Compute final k_eff
134 k_eff_ = ComputeFissionProduction(phi_old_local_);
135
136 //================================================== Print summary
137 chi::log.Log() << "\n";
138 chi::log.Log()
139 << " Final k-eigenvalue : "
140 << std::setprecision(7) << k_eff_;
141 chi::log.Log() << "\n";
142
143 return 0;
144}
