109 double chi_mesh::SurfaceMesherDelaunay::
110  DelaunayPatch::InCircle(chi_mesh::Vertex a,
111  chi_mesh::Vertex b,
112  chi_mesh::Vertex c,
113  chi_mesh::Vertex d)
114 {
115  Eigen::Matrix4d m;
116  m(0,0) = a.x; m(0,1) = a.y; m(0,2) = a.x*a.x + a.y*a.y; m(0,3) = 1.0;
117  m(1,0) = b.x; m(1,1) = b.y; m(1,2) = b.x*b.x + b.y*b.y; m(1,3) = 1.0;
118  m(2,0) = c.x; m(2,1) = c.y; m(2,2) = c.x*c.x + c.y*c.y; m(2,3) = 1.0;
119  m(3,0) = d.x; m(3,1) = d.y; m(3,2) = d.x*d.x + d.y*d.y; m(3,3) = 1.0;
120 
121  return m.determinant();
122 }

455 FindCircumCircle(chi_mesh::Vertex a, chi_mesh::Vertex b, chi_mesh::Vertex c,
456  chi_mesh::Vertex &center,
457  double &radius)
458 {
459  bool line1Vertical=false;
460  bool line2Vertical=false;
461 
462  //===================================================== Slope 1
463  float m1 = 0.0;
464  if (fabs(b.x-a.x)>tolerance)
465  {
466  float mBA = (b.y-a.y)/(b.x-a.x);
467  if (fabs(mBA)>tolerance)
468  {
469  m1 = -1.0/mBA;
470  } else
471  {
472  line1Vertical = true;
473  }
474 
475  }
476 
477  float x1 = (b.x+a.x)/2.0;
478  float y1 = (b.y+a.y)/2.0;
479 
480  //y = m*x +c
481  //c1 = y1-m1*x1
482  //yc = m1*xc +y1-m1*x1
483 
484  //===================================================== Slope 2
485  float m2 = 0.0;
486  if (fabs(c.x-b.x)>tolerance)
487  {
488  float mCB = (c.y-b.y)/(c.x-b.x);
489  if (fabs(mCB)>tolerance)
490  {
491  m2 = -1.0/mCB;
492  }
493  else
494  {
495  line2Vertical = true;
496  }
497  }
498 
499  float x2 = (c.x+b.x)/2.0;
500  float y2 = (c.y+b.y)/2.0;
501 
502  //y = m*x +c
503  //c2 = y2-m2*x2
504  //yc = m2*xc +y2-m2*x2
505 
506  //m1*xc +y1-m1*x1 = m2*xc +y2-m2*x2
507  //m1*xc -m2*xc = m1*x1-y1 - (m2*x2 - y2)
508 
509  if (line1Vertical)
510  {
511  center.x = x1;
512  center.y = m2*(center.x-x2)+y2;
513  center.z = 0.0;
514  }
515  else if (line2Vertical)
516  {
517  center.x = x2;
518  center.y = m1*(center.x-x1)+y1;
519  center.z = 0.0;
520  }
521  else
522  {
523  center.x = ( (m1*x1 - y1) - (m2*x2-y2) )/(m1-m2);
524  center.y = m2*(center.x-x2)+y2;
525  center.z = 0.0;
526  }
527 
528 
529  radius = sqrt( (center.x-b.x)*(center.x-b.x) + (center.y-b.y)*(center.y-b.y) );
