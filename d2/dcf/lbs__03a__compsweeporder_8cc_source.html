27/**Initializes the sweep ordering for the given groupset.*/
28void lbs::SteadySolver::ComputeSweepOrderings(LBSGroupset& groupset) const
29{
30 if (options.verbose_inner_iterations)
31 chi::log.Log()
32 << chi::program_timer.GetTimeString()
33 << " Computing Sweep ordering.\n";
34
35 //============================================= Clear sweep ordering
36 groupset.sweep_orderings.clear();
37 groupset.sweep_orderings.shrink_to_fit();
38
39 auto& mesh_handler = chi_mesh::GetCurrentHandler();
40 auto mesher = mesh_handler.volume_mesher;
41
42 const auto parmetis_partitioning = chi_mesh::VolumeMesher::PartitionType::PARMETIS;
43
44 bool no_cycles_parmetis_partitioning =
45 (mesher->options.partition_type == parmetis_partitioning and
46 (not groupset.allow_cycles));
47 bool is_1D_geometry = options.geometry_type == GeometryType::ONED_SLAB;
48
49 //============================================= Check possibility of cycles
50 if (no_cycles_parmetis_partitioning and not is_1D_geometry and chi::mpi.process_count>1)
51 {
52 chi::log.LogAllError()
53 << "When using PARMETIS type partitioning then groupset iterative method"
54 " must be NPT_CLASSICRICHARDSON_CYCLES or NPT_GMRES_CYCLES";
55 chi::Exit(EXIT_FAILURE);
56 }
57
58 //================================================== Compute sweep ordering
59 switch (groupset.angleagg_method)
60 {
61 case AngleAggregationType::SINGLE:
62 ComputeSweepOrderingsAngleAggSingle(groupset); break;
63 case AngleAggregationType::POLAR:
64 ComputeSweepOrderingsAngleAggPolar(groupset); break;
65 case AngleAggregationType::AZIMUTHAL:
66 ComputeSweepOrderingsAngleAggAzimuthal(groupset); break;
67 default:
68 throw std::logic_error(std::string(__FUNCTION__) +
69 " Invalid angle aggregation type.");
70 }//switch on method
71
72 if (options.verbose_inner_iterations)
73 chi::log.Log()
74 << chi::program_timer.GetTimeString()
75 << " Done computing sweep orderings. Process memory = "
76 << std::setprecision(3)
77 << chi::console.GetMemoryUsageInMB() << " MB";
78
79}

101void lbs::SteadySolver::ComputeSweepOrderingsAngleAggPolar(LBSGroupset& groupset) const
102{
103 if (options.verbose_inner_iterations)
104 chi::log.Log()
105 << chi::program_timer.GetTimeString()
106 << " Computing Sweep ordering - Angle aggregation: Polar";
107
108 const auto mesher = chi_mesh::GetCurrentHandler().volume_mesher;
109
110 if (options.geometry_type == GeometryType::ONED_SLAB ||
111 options.geometry_type == GeometryType::TWOD_CARTESIAN ||
112 (options.geometry_type == GeometryType::THREED_CARTESIAN &&
113 typeid(*mesher) == typeid(chi_mesh::VolumeMesherExtruder)))
114 {
115 if (groupset.quadrature->type == chi_math::AngularQuadratureType::ProductQuadrature)
116 {
117 const auto product_quadrature =
118 std::static_pointer_cast<chi_math::ProductQuadrature>(groupset.quadrature);
119
120 const auto num_azi = product_quadrature->azimu_ang.size();
121 const auto num_pol = product_quadrature->polar_ang.size();
122
123 if (options.geometry_type == GeometryType::ONED_SLAB)
124 {
125 if (num_azi != 1)
126 {
127 chi::log.LogAllError()
128 << "Incompatible number of azimuthal angles in quadrature set "
129 << "for a 1D simulation.";
130 chi::Exit(EXIT_FAILURE);
131 }
132 }
133 else
134 {
135 if (num_azi < 4)
136 {
137 chi::log.LogAllError()
138 << "Incompatible number of azimuthal angles in quadrature set "
139 << "for a 2D or 3D simulation.";
140 chi::Exit(EXIT_FAILURE);
141 }
142 }
143
144 if ((num_pol < 2) || (num_pol % 2 != 0))
145 {
146 chi::log.LogAllError()
147 << "Incompatible number of polar angles in quadrature set.";
148 chi::Exit(EXIT_FAILURE);
149 }
150
151 //============================================= Create sweep ordering
152 // per azimuthal angle
153 // per hemisphere
154 const unsigned int pa = num_pol/2;
155
156 //=========================================== TOP HEMISPHERE
157 for (unsigned int i = 0; i < num_azi; ++i)
158 {
159 const auto dir_idx = product_quadrature->GetAngleNum(pa-1, i);
160 const auto new_swp_order =
161 chi_mesh::sweep_management::
162 CreateSweepOrder(product_quadrature->omegas[dir_idx],
163 this->grid,
164 groupset.allow_cycles);
165 groupset.sweep_orderings.emplace_back(new_swp_order);
166 }
167 //=========================================== BOTTOM HEMISPHERE
168 for (unsigned int i = 0; i < num_azi; ++i)
169 {
170 const auto dir_idx = product_quadrature->GetAngleNum(pa, i);
171 const auto new_swp_order =
172 chi_mesh::sweep_management::
173 CreateSweepOrder(product_quadrature->omegas[dir_idx],
174 this->grid,
175 groupset.allow_cycles);
176 groupset.sweep_orderings.emplace_back(new_swp_order);
177 }
178 }//if product quadrature
179 else
180 {
181 chi::log.LogAllError()
182 << "The simulation is not using \"LBSGroupset.ANGLE_AGG_SINGLE\", "
183 "and therefore only certain angular quadrature types are supported. "
184 "i.e., for now just AngularQuadratureType::ProductQuadrature.";
185 chi::Exit(EXIT_FAILURE);
186 }
187 }
188 else
189 {
190 chi::log.LogAllError()
191 << "The simulation is not using \"LBSGroupset.ANGLE_AGG_SINGLE\", "
192 "and therefore only certain geometry types are supported. i.e., "
193 "GeometryType::ONED_SLAB, GeometryType::TWOD_CARTESIAN, "
194 "GeometryType::THREED_CARTESIAN.";
195 chi::Exit(EXIT_FAILURE);
196 }
