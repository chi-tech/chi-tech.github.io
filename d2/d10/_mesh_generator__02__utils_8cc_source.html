7/**Determines if a cells needs to be included as a ghost or as a local cell.*/
8bool chi_mesh::MeshGenerator::CellHasLocalScope(
9 const chi_mesh::UnpartitionedMesh::LightWeightCell& lwcell,
10 uint64_t cell_global_id,
11 const std::vector<std::set<uint64_t>>& vertex_subscriptions,
12 const std::vector<int64_t>& cell_partition_ids)
13{
14 if (replicated_)
15 return true;
16 // First determine if the cell is a local cell
17 int cell_pid = static_cast<int>(cell_partition_ids[cell_global_id]);
18 if (cell_pid == Chi::mpi.location_id) return true;
19
20 // Now determine if the cell is a ghost cell
21 for (uint64_t vid : lwcell.vertex_ids)
22 for (uint64_t cid : vertex_subscriptions[vid])
23 {
24 if (cid == cell_global_id) continue;
25 int adj_pid = static_cast<int>(cell_partition_ids[cid]);
26 if (adj_pid == Chi::mpi.location_id) return true;
27 }
28
29 return false;
30}

33/**Converts a light-weight cell to a real cell.*/
34std::unique_ptr<chi_mesh::Cell>
35MeshGenerator::SetupCell(const UnpartitionedMesh::LightWeightCell& raw_cell,
36 uint64_t global_id,
37 uint64_t partition_id,
38 const std::vector<chi_mesh::Vector3>& vertices)
39{
40 auto cell =
41 std::make_unique<chi_mesh::Cell>(raw_cell.type, raw_cell.sub_type);
42 cell->centroid_ = raw_cell.centroid;
43 cell->global_id_ = global_id;
44 cell->partition_id_ = partition_id;
45 cell->material_id_ = raw_cell.material_id;
46
47 cell->vertex_ids_ = raw_cell.vertex_ids;
48
49 size_t face_counter = 0;
50 for (auto& raw_face : raw_cell.faces)
51 {
52 chi_mesh::CellFace newFace;
53
54 newFace.has_neighbor_ = raw_face.has_neighbor;
55 newFace.neighbor_id_ = raw_face.neighbor;
56
57 newFace.vertex_ids_ = raw_face.vertex_ids;
58 auto vfc = chi_mesh::Vertex(0.0, 0.0, 0.0);
59 for (auto fvid : newFace.vertex_ids_)
60 vfc = vfc + vertices[fvid];
61 newFace.centroid_ = vfc / static_cast<double>(newFace.vertex_ids_.size());
62
63 if (cell->Type() == CellType::SLAB)
64 {
65 // A slab face is very easy. If it is the first face
66 // the normal is -khat. If it is the second face then
67 // it is +khat.
68 if (face_counter == 0)
69 newFace.normal_ = chi_mesh::Vector3(0.0, 0.0, -1.0);
70 else
71 newFace.normal_ = chi_mesh::Vector3(0.0, 0.0, 1.0);
72 }
73 else if (cell->Type() == CellType::POLYGON)
74 {
75 // A polygon face is just a line so we can just grab
76 // the first vertex and form a vector with the face
77 // centroid. The normal is then just khat
78 // cross-product with this vector.
79 uint64_t fvid = newFace.vertex_ids_[0];
80 auto vec_vvc = vertices[fvid] - newFace.centroid_;
81
82 newFace.normal_ = chi_mesh::Vector3(0.0, 0.0, 1.0).Cross(vec_vvc);
83 newFace.normal_.Normalize();
84 }
85 else if (cell->Type() == CellType::POLYHEDRON)
86 {
87 // A face of a polyhedron can itself be a polygon
88 // which can be multifaceted. Here we need the
89 // average normal over all the facets computed
90 // using an area-weighted average.
91 const size_t num_face_verts = newFace.vertex_ids_.size();
92 double total_area = 0.0;
93 for (size_t fv = 0; fv < num_face_verts; ++fv)
94 {
95 size_t fvp1 = (fv < (num_face_verts - 1)) ? fv + 1 : 0;
96
97 uint64_t fvid_m = newFace.vertex_ids_[fv];
98 uint64_t fvid_p = newFace.vertex_ids_[fvp1];
99
100 auto leg_m = vertices[fvid_m] - newFace.centroid_;
101 auto leg_p = vertices[fvid_p] - newFace.centroid_;
102
103 auto vn = leg_m.Cross(leg_p);
104
105 double area = 0.5 * vn.Norm();
106 total_area += area;
107
108 newFace.normal_ = newFace.normal_ + area * vn.Normalized();
109 }
110 newFace.normal_ = newFace.normal_ / total_area;
111 newFace.normal_.Normalize();
112 }
113 ++face_counter;
114
115 cell->faces_.push_back(newFace);
116 }
117
118 return cell;
119}
