38double* chi_mesh::sweep_management::PRIMARY_FLUDS::
39OutgoingPsi(int cell_so_index, int outb_face_counter,
40 int face_dof, int n)
41{
42 // Face category
43 int fc = so_cell_outb_face_face_category[cell_so_index][outb_face_counter];
44
45 if (fc >= 0)
