137void PostProcessor::ReceiveEventUpdate(const Event& event)
138{
139 auto it = std::find(subscribed_events_for_execution_.begin(),
140 subscribed_events_for_execution_.end(),
141 event.Name());
142
143 if (it != subscribed_events_for_execution_.end())
144 {
145 if (event.Code() >= 31 and event.Code() <= 38 and
146 not solvername_filter_.empty())
147 {
148 if (event.Parameters().GetParamValue<std::string>("solver_name") !=
149 solvername_filter_)
150 return;
151 }
152
153 Execute(event);
154 if (Chi::log.GetVerbosity() >= 1)
155 Chi::log.Log0Verbose1() << "Post processor \"" << Name()
156 << "\" executed on "
157 "event \""
158 << event.Name() << "\".";
159 }

175/**Converts a scalar value into a string format based on this post-processor's
176 * numeric specifications.*/
177std::string
178PostProcessor::ConvertScalarValueToString(const ParameterBlock& value) const
179{
180 std::string value_string;
181 if (value.Type() == ParameterBlockType::BOOLEAN)
182 {
183 value_string = value.GetValue<bool>() ? "true" : "false";
184 }
185 else if (value.Type() == ParameterBlockType::FLOAT)
186 {
187 const auto dblval = value.GetValue<double>();
188 char buffer[30];
189 const auto numeric_format = NumericFormat();
190 const size_t precision = NumericPrecision();
191 if (numeric_format == PPNumericFormat::SCIENTIFIC)
192 {
193 const std::string format_spec = "%." + std::to_string(precision) + "e";
194 snprintf(buffer, 30, format_spec.c_str(), dblval);
195 }
196 else if (numeric_format == PPNumericFormat::FLOATING_POINT)
197 {
198 const std::string format_spec = "%." + std::to_string(precision) + "f";
199 snprintf(buffer, 30, format_spec.c_str(), dblval);
200 }
201 else // GENERAL
202 {
203 if (dblval < 1.0e-4)
204 {
205 const std::string format_spec = "%." + std::to_string(precision) + "e";
206 snprintf(buffer, 30, format_spec.c_str(), dblval);
207 }
208 else if (dblval >= 1.0e-4 and dblval < 1.0e6)
209 {
210 const std::string format_spec = "%." + std::to_string(precision) + "f";
211 snprintf(buffer, 30, format_spec.c_str(), dblval);
212 }
213 else
214 {
215 const std::string format_spec = "%." + std::to_string(precision) + "e";
216 snprintf(buffer, 30, format_spec.c_str(), dblval);
217 }
218 } // if num_format
219
220 value_string = buffer;
221 }
222 else if (value.Type() == ParameterBlockType::STRING)
223 {
224 value_string = value.GetValue<std::string>();
225 }
226 else if (value.Type() == ParameterBlockType::INTEGER)
227 {
228 const auto intval = value.GetValue<int64_t>();
229 char buffer[30];
230 snprintf(buffer, 30, "%lld", intval);
231 value_string = buffer;
232 }
233
234 return value_string;
235}

237std::string
238PostProcessor::ConvertValueToString(const ParameterBlock& value) const
239{
240 const PPType type = FigureTypeFromValue(value);
241 if (type == PPType::SCALAR) return ConvertScalarValueToString(value);
242 else if (type == PPType::VECTOR)
243 {
244 if (value.NumParameters() == 0) return "";
245 const auto& first_entry = value.GetParam(0);
246 const auto first_entry_type = first_entry.Type();
247
248 ChiLogicalErrorIf(FigureTypeFromValue(first_entry) != PPType::SCALAR,
249 "The entries of the vector value of post-processor \"" +
250 Name() + "\" must all be SCALAR.");
251
252 std::string output;
253 for (const auto& entry : value)
254 {
255 ChiLogicalErrorIf(
256 entry.Type() != first_entry_type,
257 "Mixed typed encountered in the vector values of post-processor \"" +
258 Name() + "\"");
259 output.append(ConvertScalarValueToString(entry) + " ");
260 }
261
262 return output;
263 }
264 else
265 {
266 std::string outstr;
267 value.RecursiveDumpToString(outstr);
268 std::replace(outstr.begin(), outstr.end(), '\n', ' ');
269 return outstr;
270 }

273PPType PostProcessor::FigureTypeFromValue(const ParameterBlock& value)
274{
275 const std::vector<ParameterBlockType> scalar_types = {
276 ParameterBlockType::BOOLEAN,
277 ParameterBlockType::FLOAT,
278 ParameterBlockType::STRING,
279 ParameterBlockType::INTEGER};
280
281 /**Lambda to check if this is a scalar*/
282 auto IsScalar = [&scalar_types](const ParameterBlockType& block_type)
283 {
284 return std::find(scalar_types.begin(), scalar_types.end(), block_type) !=
285 scalar_types.end();
286 };
287
288 if (not value.HasValue() and value.NumParameters() == 0)
289 return PPType::NO_VALUE;
290 else if (IsScalar(value.Type()))
291 return PPType::SCALAR;
292 else if (value.Type() == ParameterBlockType::ARRAY)
293 {
294 if (value.NumParameters() == 0) return PPType::NO_VALUE;
295 else
296 {
297 if (IsScalar(value.GetParam(0).Type())) return PPType::VECTOR;
298 else
299 return PPType::ARBITRARY;
300 }
301 }
302 else if (value.Type() == ParameterBlockType::BLOCK)
303 return PPType::ARBITRARY;
304 else
305 ChiLogicalError("Unsupported type");
306}
