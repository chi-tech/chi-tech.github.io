22void chi_math::CellMapping::ComputeCellVolumeAndAreas(
23 const chi_mesh::MeshContinuum& grid,
24 const chi_mesh::Cell& cell,
25 double& volume,
26 std::vector<double>& areas)
27{
28 switch (cell.Type())
29 {
30 case chi_mesh::CellType::SLAB:
31 {
32 const auto& v0 = grid.vertices[cell.vertex_ids_[0]];
33 const auto& v1 = grid.vertices[cell.vertex_ids_[1]];
34
35 volume = (v1-v0).Norm();
36 areas = {1.0,1.0};
37 break;
38 }
39 case chi_mesh::CellType::POLYGON:
40 {
41 volume = 0.0;
42 const auto& v2 = cell.centroid_;
43
44 size_t num_faces = cell.faces_.size();
45 areas.reserve(num_faces);
46
47 for (size_t f=0; f<num_faces; ++f)
48 {
49 const uint64_t v0i = cell.faces_[f].vertex_ids_[0];
50 const uint64_t v1i = cell.faces_[f].vertex_ids_[1];
51
52 const auto& v0 = grid.vertices[v0i];
53 const auto& v1 = grid.vertices[v1i];
54
55 areas.push_back((v1-v0).Norm());
56
57 const chi_mesh::Vector3 sidev01 = v1 - v0;
58 const chi_mesh::Vector3 sidev02 = v2 - v0;
59
60 double sidedetJ = ((sidev01.x)*(sidev02.y) -
61 (sidev02.x)*(sidev01.y));
62
63 volume += sidedetJ/2.0;
64 }//for face
65
66 break;
67 }
68 case chi_mesh::CellType::POLYHEDRON:
69 {
70 volume = 0.0;
71 const auto& vcc = cell.centroid_;
72
73 size_t num_faces = cell.faces_.size();
74 areas.assign(num_faces, 0.0);
75 for (size_t f=0; f<num_faces; f++)
76 {
77 const auto& face = cell.faces_[f];
78 const size_t num_edges = face.vertex_ids_.size();
79 for (size_t e=0; e<num_edges; ++e)
80 {
81 size_t ep1 = (e < (num_edges-1))? e+1 : 0;
82 uint64_t v0i = face.vertex_ids_[e ];
83 uint64_t v1i = face.vertex_ids_[ep1];
84
85 const auto& v0 = grid.vertices[v0i];
86 const auto& v1 = cell.faces_[f].centroid_;
87 const auto& v2 = grid.vertices[v1i];
88 const auto& v3 = vcc;
89
90 const auto sidev01 = v1-v0;
91 const auto sidev02 = v2-v0;
92 const auto sidev03 = v3-v0;
93
94 chi_mesh::Matrix3x3 J;
95
96 J.SetColJVec(0,sidev01);
97 J.SetColJVec(1,sidev02);
98 J.SetColJVec(2,sidev03);
99
100 areas[f] += (sidev01.Cross(sidev02)).Norm()/2.0;
101 volume += J.Det()/6.0;
102 }//for edge
103 }//for face
104 break;
105 }
106 default:
107 throw std::logic_error("chi_math::CellMapping::ComputeCellVolume: "
108 "Unsupported cell type.");
109 }
110}

216 += faces_qp_data[f].ShapeValue(i, qp) *

226 }// for qp
227 }//for i
228 }//for f
