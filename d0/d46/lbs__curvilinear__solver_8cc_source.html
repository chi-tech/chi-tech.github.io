201 if (n_dot_e > 0.999999)
202 {
203 face_orthogonal = true;
204 break;
205 }
206 else if (n_dot_e < -0.999999)
207 {
208 for (const auto& v_id : face.vertex_ids)
209 {

211 if (std::abs(vertex[d]) > 1.0e-12)
212 {

241void
242lbs_curvilinear::Solver::InitializeSpatialDiscretization()
243{
244 chi::log.Log() << "Initializing spatial discretization.\n";
245
246 const auto setup_flags =
247 chi_math::finite_element::COMPUTE_CELL_MAPPINGS |
248 chi_math::finite_element::COMPUTE_UNIT_INTEGRALS;
249 auto qorder = chi_math::QuadratureOrder::INVALID_ORDER;
250 auto system = chi_math::CoordinateSystemType::UNDEFINED;
251
252 // primary discretisation
253 switch (options.geometry_type)
254 {
255 case lbs::GeometryType::ONED_SPHERICAL:
256 {
257 qorder = chi_math::QuadratureOrder::FOURTH;
258 system = chi_math::CoordinateSystemType::SPHERICAL;
259 break;
260 }
261 case lbs::GeometryType::ONED_CYLINDRICAL:
262 case lbs::GeometryType::TWOD_CYLINDRICAL:
263 {
264 qorder = chi_math::QuadratureOrder::THIRD;
265 system = chi_math::CoordinateSystemType::CYLINDRICAL;
266 break;
267 }
268 default:
269 {
270 chi::log.LogAllError()
271 << "LBSCurvilinear::Solver::InitializeSpatialDiscretization : "
272 << "invalid geometry, static_cast<int>(type) = "
273 << static_cast<int>(options.geometry_type);
274 chi::Exit(EXIT_FAILURE);
275 }
276 }
277
278 typedef chi_math::SpatialDiscretization_PWLD SDM_PWLD;
279 discretization = SDM_PWLD::New(grid, setup_flags, qorder, system);
280
281 ComputeUnitIntegrals();
282
283 // secondary discretisation
284 // system - manipulated such that the spatial discretisation returns
285 // a cell view of the same type but with weighting of degree one less
286 // than the primary discretisation
287 switch (options.geometry_type)
288 {
289 case lbs::GeometryType::ONED_SPHERICAL:
290 {
291 qorder = chi_math::QuadratureOrder::THIRD;
292 system = chi_math::CoordinateSystemType::CYLINDRICAL;
293 break;
294 }
295 case lbs::GeometryType::ONED_CYLINDRICAL:
296 case lbs::GeometryType::TWOD_CYLINDRICAL:
297 {
298 qorder = chi_math::QuadratureOrder::SECOND;
299 system = chi_math::CoordinateSystemType::CARTESIAN;
300 break;
301 }
302 default:
303 {
304 chi::log.LogAllError()
305 << "LBSCurvilinear::Solver::InitializeSpatialDiscretization : "
306 << "invalid geometry, static_cast<int>(type) = "
307 << static_cast<int>(options.geometry_type);
308 chi::Exit(EXIT_FAILURE);
309 }
310 }
311
312 discretization_secondary = SDM_PWLD::New(grid, setup_flags, qorder, system);
313
314
315 MPI_Barrier(MPI_COMM_WORLD);
316 chi::log.Log()
317 << "Cell matrices computed. Process memory = "
318 << std::setprecision(3)
319 << chi_objects::ChiConsole::GetMemoryUsageInMB() << " MB";
320}
