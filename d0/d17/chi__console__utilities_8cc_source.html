16// Execute file
17/** Executes the given file in the Lua engine.
18\author Jan*/
19int chi::Console::ExecuteFile(const std::string& fileName,
20 int argc,
21 char** argv) const
22{
23 lua_State* L = this->console_state_;
24 if (not fileName.empty())
25 {
26 if (argc > 0)
27 {
28 lua_newtable(L);
29 for (int i = 1; i <= argc; i++)
30 {
31 lua_pushnumber(L, i);
32 lua_pushstring(L, argv[i - 1]);
33 lua_settable(L, -3);
34 }
35 lua_setglobal(L, "chiArgs");
36 }
37 int error = luaL_dofile(this->console_state_, fileName.c_str());
38
39 if (error > 0)
40 {
41 Chi::log.LogAllError()
42 << "LuaError: " << lua_tostring(this->console_state_, -1);
43 return EXIT_FAILURE;
44 }
45 }
46 return EXIT_SUCCESS;
47}

67 lua_CFunction function_ptr)
68{
69 auto& console = GetInstance();

166char chi::Console::AddWrapperToRegistryInNamespaceWithName(

200/**Sets/Forms a lua function in the state using a namespace structure.*/
201void chi::Console::SetLuaFuncNamespaceTableStructure(
202 const std::string& full_lua_name, lua_CFunction function_ptr)
203{
204 auto L = GetInstance().console_state_;
205 const auto lua_name_split = chi::StringSplit(full_lua_name, "::");
206
207 if (lua_name_split.size() == 1)
208 {
209 lua_pushcfunction(L, function_ptr);
210 lua_setglobal(L, lua_name_split.back().c_str());
211 return;
212 }
213
214 const std::vector<std::string> table_names(lua_name_split.begin(),
215 lua_name_split.end() - 1);
216
217 FleshOutLuaTableStructure(table_names);
218
219 lua_pushstring(L, lua_name_split.back().c_str());
220 lua_pushcfunction(L, function_ptr);
221 lua_settable(L, -3);
222
223 lua_pop(L, lua_gettop(L));
224}

226// ###################################################################
227/**Sets/Forms a table structure that mimics the namespace structure of
228 * a string. For example the string "sing::sob::nook::Tigger" will be
229 * assigned a table structure
230 * `sing.sob.nook.Tigger = "sing::sob::nook::Tigger"`. Then finally assigns
231 * lua call to this table.*/
232void chi::Console::SetLuaFuncWrapperNamespaceTableStructure(
233 const std::string& full_lua_name)
234{
235 auto L = GetInstance().console_state_;
236
237 /**Lambda for making a chunk*/
238 auto MakeChunk = [&L, &full_lua_name]()
239 {
240 std::string chunk_code = "local params = ...; ";
241 chunk_code +=
242 "return chi_console.LuaWrapperCall(\"" + full_lua_name + "\", ...)";
243
244 luaL_loadstring(L, chunk_code.c_str());
245 };
246
247 const auto table_names = chi::StringSplit(full_lua_name, "::");
248 std::vector<std::string> namespace_names;
249 for (const auto& table_name : table_names)
250 if (table_name != table_names.back()) namespace_names.push_back(table_name);
251
252 const auto& function_name = table_names.back();
253
254 if (not namespace_names.empty())
255 {
256 FleshOutLuaTableStructure(namespace_names);
257 lua_pushstring(L, function_name.c_str());
258 MakeChunk();
259 lua_settable(L, -3);
260 }
261 else
262 {
263 MakeChunk();
264 lua_setglobal(L, function_name.c_str());
265 }
266
267 lua_pop(L, lua_gettop(L));
268}

304// ##################################################################
305/**Fleshes out a path in a table tree. For example, given
306 * "fee::foo::fah::koo, this routine will make sure that
307 * fee.foo.fah.koo is defined as a table tree structure. The routine will
308 * create a table structure where one is needed and leave existing ones alone.
309 *
310 * At the end of the routine the last table in the structure will be on top
311 * of the stack.*/
312void chi::Console::FleshOutLuaTableStructure(
313 const std::vector<std::string>& table_names)
314{
315 auto L = GetInstance().console_state_;
316
317 for (const auto& table_name : table_names)
318 {
319 // The first entry needs to be in lua's global scope,
320 // so it looks a little different
321 if (table_name == table_names.front())
322 {
323 lua_getglobal(L, table_name.c_str());
324 if (not lua_istable(L, -1))
325 {
326 lua_pop(L, 1);
327 lua_newtable(L);
328 lua_setglobal(L, table_name.c_str());
329 lua_getglobal(L, table_name.c_str());
330 }
331 }
332 else
333 {
334 lua_getfield(L, -1, table_name.c_str());
335 if (not lua_istable(L, -1))
336 {
337 lua_pop(L, 1);
338 lua_pushstring(L, table_name.c_str());
339 lua_newtable(L);
340 lua_settable(L, -3);
341 lua_getfield(L, -1, table_name.c_str());
342 }
343 }
344 } // for table_key in table_keys
345}

348/**Assumes a table is on top of the stack, then loads the table
349 * with chunks that call registered methods of the class with the first
350 * argument being the object's handle.*/
351void chi::Console::SetObjectMethodsToTable(const std::string& class_name,
352 size_t handle)
353{
354 auto& console = GetInstance();
355 auto L = console.console_state_;
356
357 const auto& class_method_registry = console.class_method_registry_;
358 if (class_method_registry.count(class_name) == 0) return;
359
360 const auto& method_list = class_method_registry.at(class_name);
361 for (const auto& method_name : method_list)
362 {
363 std::string function_name = class_name;
364 function_name += "::";
365 function_name += method_name;
366
367 const auto path_names = chi::StringSplit(function_name, "::");
368 std::string path_dot_name;
369 for (const auto& name : path_names)
370 {
371 path_dot_name += name;
372 if (name != path_names.back()) path_dot_name += ".";
373 }
374
375 std::string chunk_code = "local params = ...; ";
376 chunk_code +=
377 "return " + path_dot_name + "(" + std::to_string(handle) + ", ...)";
378
379 lua_pushstring(L, method_name.c_str());
380 luaL_loadstring(L, chunk_code.c_str());
381 lua_settable(L, -3);
382 }
383}

386/**Sets a lua constant in the lua state.*/
387void chi::Console::SetLuaConstant(const std::string& constant_name,
388 const chi_data_types::Varying& value)
389{
390 auto& console = GetInstance();
391 auto L = console.console_state_;
392 const auto path_names = chi::StringSplit(constant_name, "::");
393
394 auto PushVaryingValue = [&L](const chi_data_types::Varying& var_value)
395 {
396 if (var_value.Type() == chi_data_types::VaryingDataType::BOOL)
397 lua_pushboolean(L, var_value.BoolValue());
398 else if (var_value.Type() == chi_data_types::VaryingDataType::STRING)
399 lua_pushstring(L, var_value.StringValue().c_str());
400 else if (var_value.Type() == chi_data_types::VaryingDataType::INTEGER)
401 lua_pushinteger(L, static_cast<lua_Integer>(var_value.IntegerValue()));
402 else if (var_value.Type() == chi_data_types::VaryingDataType::FLOAT)
403 lua_pushnumber(L, var_value.FloatValue());
404 else
405 ChiInvalidArgument("Unsupported value type. Only bool, string, int and "
406 "double is supported");
407 };
408
409 if (path_names.size() == 1)
410 {
411 PushVaryingValue(value);
412 lua_setglobal(L, path_names.front().c_str());
413 }
414 else
415 {
416 std::vector<std::string> namespace_names;
417 for (const auto& table_name : path_names)
418 if (table_name != path_names.back())
419 {
420 namespace_names.push_back(table_name);
421 }
422
423 FleshOutLuaTableStructure(namespace_names);
424 lua_pushstring(L, path_names.back().c_str());
425 PushVaryingValue(value);
426 lua_settable(L, -3);
427 }
428
429 lua_pop(L, lua_gettop(L));
430}

459/**Given an old status, will update the bindings for only newly registered

461void chi::Console::UpdateConsoleBindings(
462 const chi::RegistryStatuses& old_statuses)
463{
464 auto ListHasValue =

466 { return std::find(list.begin(), list.end(), value) != list.end(); };
467
468 const auto& object_factory = ChiObjectFactory::GetInstance();
469 for (const auto& [key, _] : object_factory.Registry())
470 if (not ListHasValue(old_statuses.objfactory_keys_, key))
471 SetObjectNamespaceTableStructure(key);
472
473 for (const auto& [key, entry] : lua_function_registry_)
474 if (not ListHasValue(old_statuses.objfactory_keys_, key))
475 SetLuaFuncNamespaceTableStructure(key, entry.function_ptr);
476
477 for (const auto& [key, entry] : function_wrapper_registry_)
478 if (not ListHasValue(old_statuses.objfactory_keys_, key))
479 if (entry.call_func) SetLuaFuncWrapperNamespaceTableStructure(key);
