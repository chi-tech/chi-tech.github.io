14// Execute file
15/** Executes the given file in the Lua engine.
16\author Jan*/
17int chi::ChiConsole::ExecuteFile(const std::string& fileName,
18 int argc,
19 char** argv) const
20{
21 lua_State* L = this->console_state_;
22 if (not fileName.empty())
23 {
24 if (argc > 0)
25 {
26 lua_newtable(L);
27 for (int i = 1; i <= argc; i++)
28 {
29 lua_pushnumber(L, i);
30 lua_pushstring(L, argv[i - 1]);
31 lua_settable(L, -3);
32 }
33 lua_setglobal(L, "chiArgs");
34 }
35 int error = luaL_dofile(this->console_state_, fileName.c_str());
36
37 if (error > 0)
38 {
39 Chi::log.LogAllError()
40 << "LuaError: " << lua_tostring(this->console_state_, -1);
41 return EXIT_FAILURE;
42 }
43 }
44 return EXIT_SUCCESS;
45}

62/**Basic addition to registry. Used by the other public methods
63 * to registry a text-key to a lua function.*/
64void chi::ChiConsole::AddFunctionToRegistry(
65 const std::string& name_in_lua, lua_CFunction function_ptr)
66{
67 auto& console = GetInstance();
68
69 // Check if the function name is already there
70 if (console.lua_function_registry_.count(name_in_lua) > 0)
71 {
72 const auto& current_entry = console.lua_function_registry_.at(name_in_lua);
73
74 throw std::logic_error(std::string(__PRETTY_FUNCTION__) +
75 ": Attempted "
76 "to register lua function \"" +
77 name_in_lua +
78 "\" but the function "
79 "is already taken by " +
80 current_entry.function_raw_name);
81 }
82
83 console.lua_function_registry_.insert(std::make_pair(
84 name_in_lua, LuaFunctionRegistryEntry{function_ptr, name_in_lua}));
85}

136/**Wrapper functions operate with input and output parameters, essentially
137 * hiding the lua interface.*/
138char chi::ChiConsole::AddWrapperToRegistryInNamespaceWithName(
139 const std::string& namespace_name,
140 const std::string& name_in_lua,
141 WrapperGetInParamsFunc syntax_function,
142 WrapperCallFunc actual_function,
143 bool ignore_null_call_func /*=false*/)
144{
145 const std::string name = (namespace_name.empty())
146 ? name_in_lua
147 : namespace_name + "::" + name_in_lua;
148
149 auto& console = GetInstance();

151
152 ChiLogicalErrorIf(

154 std::string("Attempted to register lua-function wrapper \"") + name +
155 "\" but a wrapper with the same name already exists");
156
157 if (not syntax_function)
158 syntax_function = DefaultGetInParamsFunc;
159

163 LuaFuncWrapperRegEntry reg_entry;
164 reg_entry.get_in_params_func = syntax_function;
165 reg_entry.call_func = actual_function;
166
167 registry.insert(std::make_pair(name, reg_entry));
168
169 return 0;
170}

173/**Sets/Forms a lua function in the state using a namespace structure.*/
174void chi::ChiConsole::SetLuaFuncNamespaceTableStructure(
175 const std::string& full_lua_name, lua_CFunction function_ptr)
176{
177 auto L = GetInstance().console_state_;
178 const auto lua_name_split = chi_misc_utils::StringSplit(full_lua_name, "::");
179
180 if (lua_name_split.size() == 1)
181 {
182 lua_pushcfunction(L, function_ptr);
183 lua_setglobal(L, lua_name_split.back().c_str());
184 return;
185 }
186
187 const std::vector<std::string> table_names(lua_name_split.begin(),
188 lua_name_split.end() - 1);
189
190 FleshOutLuaTableStructure(table_names);
191
192 lua_pushstring(L, lua_name_split.back().c_str());
193 lua_pushcfunction(L, function_ptr);
194 lua_settable(L, -3);
195
196 lua_pop(L, lua_gettop(L));
197}

199// ###################################################################
200/**Sets/Forms a table structure that mimics the namespace structure of
201 * a string. For example the string "sing::sob::nook::Tigger" will be
202 * assigned a table structure
203 * `sing.sob.nook.Tigger = "sing::sob::nook::Tigger"`. Then finally assigns
204 * lua call to this table.*/
205void chi::ChiConsole::SetLuaFuncWrapperNamespaceTableStructure(
206 const std::string& full_lua_name)
207{
208 auto L = GetInstance().console_state_;
209
210 /**Lambda for making a chunk*/
211 auto MakeChunk = [&L, &full_lua_name]()
212 {
213 std::string chunk_code = "local params = ...; ";
214 chunk_code +=
215 "return chi_console.LuaWrapperCall(\"" + full_lua_name + "\", ...)";
216
217 luaL_loadstring(L, chunk_code.c_str());
218 };
219
220 const auto table_names = chi_misc_utils::StringSplit(full_lua_name, "::");
221 std::vector<std::string> namespace_names;
222 for (const auto& table_name : table_names)
223 if (table_name != table_names.back()) namespace_names.push_back(table_name);
224
225 const auto& function_name = table_names.back();
226
227 if (not namespace_names.empty())
228 {
229 FleshOutLuaTableStructure(namespace_names);
230 lua_pushstring(L, function_name.c_str());
231 MakeChunk();
232 lua_settable(L, -3);
233 }
234 else
235 {
236 MakeChunk();
237 lua_setglobal(L, function_name.c_str());
238 }
239
240 lua_pop(L, lua_gettop(L));
241}

278/**Fleshes out a path in a table tree. For example, given
279 * "fee::foo::fah::koo, this routine will make sure that
280 * fee.foo.fah.koo is defined as a table tree structure. The routine will
281 * create a table structure where one is needed and leave existing ones alone.
282 *
283 * At the end of the routine the last table in the structure will be on top
284 * of the stack.*/
285void chi::ChiConsole::FleshOutLuaTableStructure(
286 const std::vector<std::string>& table_names)
287{
288 auto L = GetInstance().console_state_;
289
290 for (const auto& table_name : table_names)
291 {
292 // The first entry needs to be in lua's global scope,
293 // so it looks a little different
294 if (table_name == table_names.front())
295 {
296 lua_getglobal(L, table_name.c_str());
297 if (not lua_istable(L, -1))
298 {
299 lua_pop(L, 1);
300 lua_newtable(L);
301 lua_setglobal(L, table_name.c_str());
302 lua_getglobal(L, table_name.c_str());
303 }
304 }
305 else
306 {
307 lua_getfield(L, -1, table_name.c_str());
308 if (not lua_istable(L, -1))
309 {
310 lua_pop(L, 1);
311 lua_pushstring(L, table_name.c_str());
312 lua_newtable(L);
313 lua_settable(L, -3);
314 lua_getfield(L, -1, table_name.c_str());
315 }
316 }
317 } // for table_key in table_keys
318}

321/**Assumes a table is on top of the stack, then loads the table
322 * with chunks that call registered methods of the class with the first
323 * argument being the object's handle.*/
324void chi::ChiConsole::SetObjectMethodsToTable(
325 const std::string& class_name, size_t handle)
326{
327 auto& console = GetInstance();
328 auto L = console.console_state_;
329
330 const auto& class_method_registry = console.class_method_registry_;
331 if (class_method_registry.count(class_name) == 0) return;
332
333 const auto& method_list = class_method_registry.at(class_name);
334 for (const auto& method_name : method_list)
335 {
336 std::string function_name = class_name;
337 function_name += "::";
338 function_name += method_name;
339
340 const auto path_names = chi_misc_utils::StringSplit(function_name, "::");
341 std::string path_dot_name;
342 for (const auto& name : path_names)
343 {
344 path_dot_name += name;
345 if (name != path_names.back()) path_dot_name += ".";
346 }
347
348 std::string chunk_code = "local params = ...; ";
349 chunk_code +=
350 "return " + path_dot_name + "(" + std::to_string(handle) + ", ...)";
351
352 lua_pushstring(L, method_name.c_str());
353 luaL_loadstring(L, chunk_code.c_str());
354 lua_settable(L, -3);
355 }
356}
