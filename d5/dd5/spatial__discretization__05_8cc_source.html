38mapping[102][3][1] = 1
39\endverbatim
40
41*/
42std::vector<std::vector<std::vector<int>>>
43chi_math::SpatialDiscretization::MakeInternalFaceNodeMappings(
44 const double tolerance /*=1.0e-12*/) const
45{
46 typedef std::vector<int> FaceAdjMapping;
47 typedef std::vector<FaceAdjMapping> PerFaceAdjMapping;
48 typedef std::vector<PerFaceAdjMapping> CellAdjMapping;
49
50 const auto& grid = this->ref_grid_;
51
52 CellAdjMapping cell_adj_mapping;
53 for (const auto& cell : grid.local_cells)
54 {
55 const auto& cell_mapping = this->GetCellMapping(cell);
56 const auto& node_locations = cell_mapping.GetNodeLocations();
57 const size_t num_faces = cell.faces_.size();
58
59 PerFaceAdjMapping per_face_adj_mapping;
60
61 for (size_t f = 0; f < num_faces; ++f)
62 {
63 const auto& face = cell.faces_[f];
64 const auto num_face_nodes = cell_mapping.NumFaceNodes(f);
65 FaceAdjMapping face_adj_mapping(num_face_nodes, -1);
66 if (face.has_neighbor_)
67 {
68 const auto& adj_cell = grid.cells[face.neighbor_id_];
69 const auto& adj_cell_mapping = this->GetCellMapping(adj_cell);
70 const auto& adj_node_locations = adj_cell_mapping.GetNodeLocations();
71 const size_t adj_num_nodes = adj_cell_mapping.NumNodes();
72
73 for (size_t fi = 0; fi < num_face_nodes; ++fi)
74 {
75 const int i = cell_mapping.MapFaceNode(f, fi);
76 const auto& ivec3 = node_locations[i];
77
78 for (size_t ai = 0; ai < adj_num_nodes; ++ai)
79 {
80 const auto& aivec3 = adj_node_locations[ai];
81 if ((ivec3 - aivec3).NormSquare() < tolerance)
82 {
83 face_adj_mapping[fi] = static_cast<int>(ai);
84 break;
85 }
86 } // for ai
87 if (face_adj_mapping[fi] < 0)
88 throw std::logic_error("Face node mapping failed");
89 } // for fi
90 } // if internal face
91
92 per_face_adj_mapping.push_back(std::move(face_adj_mapping));
93 } // for face
94
95 cell_adj_mapping.push_back(std::move(per_face_adj_mapping));
96 } // for cell
97
98 return cell_adj_mapping;
99}

102/**Copy part of vector A to vector B. Suppose vector A's entries are
103 * managed `chi_math::UnknownManager` A (`uk_manA`) and that the
104 * entries of the vector B are managed by `chi_math::UnknownManager` B
105 * (`uk_manB`). This function copies the entries associated with an unknown with
106 * id `uk_id_A` in `uk_manA` from vector A to vector B such that the entries

108 * `uk_manB`. All the components are copied.
109 *
110\param from_vector Vector to copy from.
111\param to_vector Vector to copy to.
112\param from_vec_uk_structure Unknown manager for vector A.
113\param from_vec_uk_id Unknown-id in unknown manager A.
114\param to_vec_uk_structure Unknown manager for vector B.
115\param to_vec_uk_id Unknown-id in unknown manager B.
116 */
117void chi_math::SpatialDiscretization::CopyVectorWithUnknownScope(
118 const std::vector<double>& from_vector,
119 std::vector<double>& to_vector,

122 const UnknownManager& to_vec_uk_structure,

124{

126 "CopyVectorWithUnknownScope";

128 const auto& ukmanT = to_vec_uk_structure;
129 const auto& ukidF = from_vec_uk_id;
130 const auto& ukidT = to_vec_uk_id;
131 try
132 {
133 const auto& ukA = from_vec_uk_structure.unknowns_.at(from_vec_uk_id);

136 if (ukA.num_components_ != ukB.num_components_)

141
142 for (const auto& cell : ref_grid_.local_cells)
143 {
144 const auto& cell_mapping = this->GetCellMapping(cell);
145 const size_t num_nodes = cell_mapping.NumNodes();
146
147 for (size_t i = 0; i < num_nodes; ++i)
148 {
149 for (size_t c = 0; c < num_comps; ++c)
150 {
151 const int64_t fmap = MapDOFLocal(cell, i, ukmanF, ukidF, c);

154 to_vector[imap] = from_vector[fmap];

157 } // for cell
