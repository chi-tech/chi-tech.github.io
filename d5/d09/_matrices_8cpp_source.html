53 // inverse of 2x2 matrix
54 // If cannot find inverse, set identity matrix
56 Matrix2& Matrix2::invert()
57 {
58  float determinant = getDeterminant();
59  if(fabs(determinant) <= EPSILON)
60  {
61  return identity();
62  }
63 
64  float tmp = m[0]; // copy the first element
65  float invDeterminant = 1.0f / determinant;
66  m[0] = invDeterminant * m[3];
67  m[1] = -invDeterminant * m[1];
68  m[2] = -invDeterminant * m[2];
69  m[3] = invDeterminant * tmp;
70 
71  return *this;
72 }

122 // inverse 3x3 matrix
123 // If cannot find inverse (det=0), set identity matrix
124 // M^-1 = adj(M) / det(M)
125 // | m4m8-m5m7 m5m6-m3m8 m3m7-m4m6 |
126 // = | m7m2-m8m1 m0m8-m2m6 m6m1-m7m0 | / det(M)
127 // | m1m5-m2m4 m2m3-m0m5 m0m4-m1m3 |
129 Matrix3& Matrix3::invert()
130 {
131  float determinant, invDeterminant;
132  float tmp[9];
133 
134  tmp[0] = m[4] * m[8] - m[5] * m[7];
135  tmp[1] = m[7] * m[2] - m[8] * m[1];
136  tmp[2] = m[1] * m[5] - m[2] * m[4];
137  tmp[3] = m[5] * m[6] - m[3] * m[8];
138  tmp[4] = m[0] * m[8] - m[2] * m[6];
139  tmp[5] = m[2] * m[3] - m[0] * m[5];
140  tmp[6] = m[3] * m[7] - m[4] * m[6];
141  tmp[7] = m[6] * m[1] - m[7] * m[0];
142  tmp[8] = m[0] * m[4] - m[1] * m[3];
143 
144  // check determinant if it is 0
145  determinant = m[0] * tmp[0] + m[1] * tmp[3] + m[2] * tmp[6];
146  if(fabs(determinant) <= EPSILON)
147  {
148  return identity(); // cannot inverse, make it idenety matrix
149  }
150 
151  // divide by the determinant
152  invDeterminant = 1.0f / determinant;
153  m[0] = invDeterminant * tmp[0];
154  m[1] = invDeterminant * tmp[1];
155  m[2] = invDeterminant * tmp[2];
156  m[3] = invDeterminant * tmp[3];
157  m[4] = invDeterminant * tmp[4];
158  m[5] = invDeterminant * tmp[5];
159  m[6] = invDeterminant * tmp[6];
160  m[7] = invDeterminant * tmp[7];
161  m[8] = invDeterminant * tmp[8];
162 
163  return *this;
164 }

169 // retrieve angles in degree from rotation matrix, M = Rx*Ry*Rz
170 // Rx: rotation about X-axis, pitch
171 // Ry: rotation about Y-axis, yaw(heading)
172 // Rz: rotation about Z-axis, roll
173 // Rx Ry Rz
174 // |1 0 0| | Cy 0 Sy| |Cz -Sz 0| | CyCz -CySz Sy |
175 // |0 Cx -Sx|*| 0 1 0|*|Sz Cz 0| = | SxSyCz+CxSz -SxSySz+CxCz -SxCy|
176 // |0 Sx Cx| |-Sy 0 Cy| | 0 0 1| |-CxSyCz+SxSz CxSySz+SxCz CxCy|
177 //
178 // Pitch: atan(-m[7] / m[8]) = atan(SxCy/CxCy)
179 // Yaw : asin(m[6]) = asin(Sy)
180 // Roll : atan(-m[3] / m[0]) = atan(SzCy/CzCy)
182 Vector3 Matrix3::getAngle() const
183 {
184  float pitch, yaw, roll; // 3 angles
185 
186  // find yaw (around y-axis) first
187  // NOTE: asin() returns -90~+90, so correct the angle range -180~+180
188  // using z value of forward vector
189  yaw = RAD2DEG * asinf(m[6]);
190  if(m[8] < 0)
191  {
192  if(yaw >= 0) yaw = 180.0f - yaw;
193  else yaw =-180.0f - yaw;
194  }
195 
196  // find roll (around z-axis) and pitch (around x-axis)
197  // if forward vector is (1,0,0) or (-1,0,0), then m[0]=m[4]=m[9]=m[10]=0
198  if(m[0] > -EPSILON && m[0] < EPSILON)
199  {
200  roll = 0; //@@ assume roll=0
201  pitch = RAD2DEG * atan2f(m[1], m[4]);
202  }
203  else
204  {
205  roll = RAD2DEG * atan2f(-m[3], m[0]);
206  pitch = RAD2DEG * atan2f(-m[7], m[8]);
207  }
208 
209  return Vector3(pitch, yaw, roll);
210 }

262 // compute the inverse of 4x4 Euclidean transformation matrix
263 //
264 // Euclidean transformation is translation, rotation, and reflection.
265 // With Euclidean transform, only the position and orientation of the object
266 // will be changed. Euclidean transform does not change the shape of an object
267 // (no scaling). Length and angle are reserved.
268 //
269 // Use inverseAffine() if the matrix has scale and shear transformation.
270 //
271 // M = [ R | T ]
272 // [ --+-- ] (R denotes 3x3 rotation/reflection matrix)
273 // [ 0 | 1 ] (T denotes 1x3 translation matrix)
274 //
275 // y = M*x -> y = R*x + T -> x = R^-1*(y - T) -> x = R^T*y - R^T*T
276 // (R is orthogonal, R^-1 = R^T)
277 //
278 // [ R | T ]-1 [ R^T | -R^T * T ] (R denotes 3x3 rotation matrix)
279 // [ --+-- ] = [ ----+--------- ] (T denotes 1x3 translation)
280 // [ 0 | 1 ] [ 0 | 1 ] (R^T denotes R-transpose)
282 Matrix4& Matrix4::invertEuclidean()
283 {
284  // transpose 3x3 rotation matrix part
285  // | R^T | 0 |
286  // | ----+-- |
287  // | 0 | 1 |
288  float tmp;
289  tmp = m[1]; m[1] = m[4]; m[4] = tmp;
290  tmp = m[2]; m[2] = m[8]; m[8] = tmp;
291  tmp = m[6]; m[6] = m[9]; m[9] = tmp;
292 
293  // compute translation part -R^T * T
294  // | 0 | -R^T x |
295  // | --+------- |
296  // | 0 | 0 |
297  float x = m[12];
298  float y = m[13];
299  float z = m[14];
300  m[12] = -(m[0] * x + m[4] * y + m[8] * z);
301  m[13] = -(m[1] * x + m[5] * y + m[9] * z);
302  m[14] = -(m[2] * x + m[6] * y + m[10]* z);
303 
304  // last row should be unchanged (0,0,0,1)
305 
306  return *this;
307 }

312 // compute the inverse of a 4x4 affine transformation matrix
313 //
314 // Affine transformations are generalizations of Euclidean transformations.
315 // Affine transformation includes translation, rotation, reflection, scaling,
316 // and shearing. Length and angle are NOT preserved.
317 // M = [ R | T ]
318 // [ --+-- ] (R denotes 3x3 rotation/scale/shear matrix)
319 // [ 0 | 1 ] (T denotes 1x3 translation matrix)
320 //
321 // y = M*x -> y = R*x + T -> x = R^-1*(y - T) -> x = R^-1*y - R^-1*T
322 //
323 // [ R | T ]-1 [ R^-1 | -R^-1 * T ]
324 // [ --+-- ] = [ -----+---------- ]
325 // [ 0 | 1 ] [ 0 + 1 ]
327 Matrix4& Matrix4::invertAffine()
328 {
329  // R^-1
330  Matrix3 r(m[0],m[1],m[2], m[4],m[5],m[6], m[8],m[9],m[10]);
331  r.invert();
332  m[0] = r[0]; m[1] = r[1]; m[2] = r[2];
333  m[4] = r[3]; m[5] = r[4]; m[6] = r[5];
334  m[8] = r[6]; m[9] = r[7]; m[10]= r[8];
335 
336  // -R^-1 * T
337  float x = m[12];
338  float y = m[13];
339  float z = m[14];
340  m[12] = -(r[0] * x + r[3] * y + r[6] * z);
341  m[13] = -(r[1] * x + r[4] * y + r[7] * z);
342  m[14] = -(r[2] * x + r[5] * y + r[8] * z);
343 
344  // last row should be unchanged (0,0,0,1)
345  //m[3] = m[7] = m[11] = 0.0f;
346  //m[15] = 1.0f;
347 
348  return * this;
349 }

354 // inverse matrix using matrix partitioning (blockwise inverse)
355 // It devides a 4x4 matrix into 4 of 2x2 matrices. It works in case of where
356 // det(A) != 0. If not, use the generic inverse method
357 // inverse formula.
358 // M = [ A | B ] A, B, C, D are 2x2 matrix blocks
359 // [ --+-- ] det(M) = |A| * |D - ((C * A^-1) * B)|
360 // [ C | D ]
361 //
362 // M^-1 = [ A' | B' ] A' = A^-1 - (A^-1 * B) * C'
363 // [ ---+--- ] B' = (A^-1 * B) * -D'
364 // [ C' | D' ] C' = -D' * (C * A^-1)
365 // D' = (D - ((C * A^-1) * B))^-1
366 //
367 // NOTE: I wrap with () if it it used more than once.
368 // The matrix is invertable even if det(A)=0, so must check det(A) before
369 // calling this function, and use invertGeneric() instead.
371 Matrix4& Matrix4::invertProjective()
372 {
373  // partition
374  Matrix2 a(m[0], m[1], m[4], m[5]);
375  Matrix2 b(m[8], m[9], m[12], m[13]);
376  Matrix2 c(m[2], m[3], m[6], m[7]);
377  Matrix2 d(m[10], m[11], m[14], m[15]);
378 
379  // pre-compute repeated parts
380  a.invert(); // A^-1
381  Matrix2 ab = a * b; // A^-1 * B
382  Matrix2 ca = c * a; // C * A^-1
383  Matrix2 cab = ca * b; // C * A^-1 * B
384  Matrix2 dcab = d - cab; // D - C * A^-1 * B
385 
386  // check determinant if |D - C * A^-1 * B| = 0
387  //NOTE: this function assumes det(A) is already checked. if |A|=0 then,
388  // cannot use this function.
389  float determinant = dcab[0] * dcab[3] - dcab[1] * dcab[2];
390  if(fabs(determinant) <= EPSILON)
391  {
392  return identity();
393  }
394 
395  // compute D' and -D'
396  Matrix2 d1 = dcab; // (D - C * A^-1 * B)
397  d1.invert(); // (D - C * A^-1 * B)^-1
398  Matrix2 d2 = -d1; // -(D - C * A^-1 * B)^-1
399 
400  // compute C'
401  Matrix2 c1 = d2 * ca; // -D' * (C * A^-1)
402 
403  // compute B'
404  Matrix2 b1 = ab * d2; // (A^-1 * B) * -D'
405 
406  // compute A'
407  Matrix2 a1 = a - (ab * c1); // A^-1 - (A^-1 * B) * C'
408 
409  // assemble inverse matrix
410  m[0] = a1[0]; m[4] = a1[2]; /*|*/ m[8] = b1[0]; m[12]= b1[2];
411  m[1] = a1[1]; m[5] = a1[3]; /*|*/ m[9] = b1[1]; m[13]= b1[3];
412  /*-----------------------------+-----------------------------*/
413  m[2] = c1[0]; m[6] = c1[2]; /*|*/ m[10]= d1[0]; m[14]= d1[2];
414  m[3] = c1[1]; m[7] = c1[3]; /*|*/ m[11]= d1[1]; m[15]= d1[3];
415 
416  return *this;
417 }

422 // compute the inverse of a general 4x4 matrix using Cramer's Rule
423 // If cannot find inverse, return indentity matrix
424 // M^-1 = adj(M) / det(M)
426 Matrix4& Matrix4::invertGeneral()
427 {
428  // get cofactors of minor matrices
429  float cofactor0 = getCofactor(m[5],m[6],m[7], m[9],m[10],m[11], m[13],m[14],m[15]);
430  float cofactor1 = getCofactor(m[4],m[6],m[7], m[8],m[10],m[11], m[12],m[14],m[15]);
431  float cofactor2 = getCofactor(m[4],m[5],m[7], m[8],m[9], m[11], m[12],m[13],m[15]);
432  float cofactor3 = getCofactor(m[4],m[5],m[6], m[8],m[9], m[10], m[12],m[13],m[14]);
433 
434  // get determinant
435  float determinant = m[0] * cofactor0 - m[1] * cofactor1 + m[2] * cofactor2 - m[3] * cofactor3;
436  if(fabs(determinant) <= EPSILON)
437  {
438  return identity();
439  }
440 
441  // get rest of cofactors for adj(M)
442  float cofactor4 = getCofactor(m[1],m[2],m[3], m[9],m[10],m[11], m[13],m[14],m[15]);
443  float cofactor5 = getCofactor(m[0],m[2],m[3], m[8],m[10],m[11], m[12],m[14],m[15]);
444  float cofactor6 = getCofactor(m[0],m[1],m[3], m[8],m[9], m[11], m[12],m[13],m[15]);
445  float cofactor7 = getCofactor(m[0],m[1],m[2], m[8],m[9], m[10], m[12],m[13],m[14]);
446 
447  float cofactor8 = getCofactor(m[1],m[2],m[3], m[5],m[6], m[7], m[13],m[14],m[15]);
448  float cofactor9 = getCofactor(m[0],m[2],m[3], m[4],m[6], m[7], m[12],m[14],m[15]);
449  float cofactor10= getCofactor(m[0],m[1],m[3], m[4],m[5], m[7], m[12],m[13],m[15]);
450  float cofactor11= getCofactor(m[0],m[1],m[2], m[4],m[5], m[6], m[12],m[13],m[14]);
451 
452  float cofactor12= getCofactor(m[1],m[2],m[3], m[5],m[6], m[7], m[9], m[10],m[11]);
453  float cofactor13= getCofactor(m[0],m[2],m[3], m[4],m[6], m[7], m[8], m[10],m[11]);
454  float cofactor14= getCofactor(m[0],m[1],m[3], m[4],m[5], m[7], m[8], m[9], m[11]);
455  float cofactor15= getCofactor(m[0],m[1],m[2], m[4],m[5], m[6], m[8], m[9], m[10]);
456 
457  // build inverse matrix = adj(M) / det(M)
458  // adjugate of M is the transpose of the cofactor matrix of M
459  float invDeterminant = 1.0f / determinant;
460  m[0] = invDeterminant * cofactor0;
461  m[1] = -invDeterminant * cofactor4;
462  m[2] = invDeterminant * cofactor8;
463  m[3] = -invDeterminant * cofactor12;
464 
465  m[4] = -invDeterminant * cofactor1;
466  m[5] = invDeterminant * cofactor5;
467  m[6] = -invDeterminant * cofactor9;
468  m[7] = invDeterminant * cofactor13;
469 
470  m[8] = invDeterminant * cofactor2;
471  m[9] = -invDeterminant * cofactor6;
472  m[10]= invDeterminant * cofactor10;
473  m[11]= -invDeterminant * cofactor14;
474 
475  m[12]= -invDeterminant * cofactor3;
476  m[13]= invDeterminant * cofactor7;
477  m[14]= -invDeterminant * cofactor11;
478  m[15]= invDeterminant * cofactor15;
479 
480  return *this;
481 }

560 Matrix4& Matrix4::rotate(float angle, float x, float y, float z)
561 {
562  float c = cosf(angle * DEG2RAD); // cosine
563  float s = sinf(angle * DEG2RAD); // sine
564  float c1 = 1.0f - c; // 1 - c
565  float m0 = m[0], m4 = m[4], m8 = m[8], m12= m[12],
566  m1 = m[1], m5 = m[5], m9 = m[9], m13= m[13],
567  m2 = m[2], m6 = m[6], m10= m[10], m14= m[14];
568 
569  // build rotation matrix
570  float r0 = x * x * c1 + c;
571  float r1 = x * y * c1 + z * s;
572  float r2 = x * z * c1 - y * s;
573  float r4 = x * y * c1 - z * s;
574  float r5 = y * y * c1 + c;
575  float r6 = y * z * c1 + x * s;
576  float r8 = x * z * c1 + y * s;
577  float r9 = y * z * c1 - x * s;
578  float r10= z * z * c1 + c;
579 
580  // multiply rotation matrix
581  m[0] = r0 * m0 + r4 * m1 + r8 * m2;
582  m[1] = r1 * m0 + r5 * m1 + r9 * m2;
583  m[2] = r2 * m0 + r6 * m1 + r10* m2;

777 // retrieve angles in degree from rotation matrix, M = Rx*Ry*Rz
778 // Rx: rotation about X-axis, pitch
779 // Ry: rotation about Y-axis, yaw(heading)
780 // Rz: rotation about Z-axis, roll
781 // Rx Ry Rz
782 // |1 0 0| | Cy 0 Sy| |Cz -Sz 0| | CyCz -CySz Sy |
783 // |0 Cx -Sx|*| 0 1 0|*|Sz Cz 0| = | SxSyCz+CxSz -SxSySz+CxCz -SxCy|
784 // |0 Sx Cx| |-Sy 0 Cy| | 0 0 1| |-CxSyCz+SxSz CxSySz+SxCz CxCy|
785 //
786 // Pitch: atan(-m[9] / m[10]) = atan(SxCy/CxCy)
787 // Yaw : asin(m[8]) = asin(Sy)
788 // Roll : atan(-m[4] / m[0]) = atan(SzCy/CzCy)
790 Vector3 Matrix4::getAngle() const
791 {
792  float pitch, yaw, roll; // 3 angles
793 
794  // find yaw (around y-axis) first
795  // NOTE: asin() returns -90~+90, so correct the angle range -180~+180
796  // using z value of forward vector
797  yaw = RAD2DEG * asinf(m[8]);
798  if(m[10] < 0)
799  {
800  if(yaw >= 0) yaw = 180.0f - yaw;
801  else yaw =-180.0f - yaw;
802  }
803 
804  // find roll (around z-axis) and pitch (around x-axis)
805  // if forward vector is (1,0,0) or (-1,0,0), then m[0]=m[4]=m[9]=m[10]=0
806  if(m[0] > -EPSILON && m[0] < EPSILON)
807  {
808  roll = 0; //@@ assume roll=0
809  pitch = RAD2DEG * atan2f(m[1], m[5]);
810  }
811  else
812  {
813  roll = RAD2DEG * atan2f(-m[4], m[0]);
814  pitch = RAD2DEG * atan2f(-m[9], m[10]);
815  }
816 
817  return Vector3(pitch, yaw, roll);
818 }
