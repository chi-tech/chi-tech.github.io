31/**Creates a raw polyhedron cell from a vtk-polyhedron.*/
32chi_mesh::UnpartitionedMesh::LightWeightCell* chi_mesh::UnpartitionedMesh::
33CreateCellFromVTKPolyhedron(vtkCell *vtk_cell)
34{
35 auto polyh_cell = new LightWeightCell(CellType::POLYHEDRON,
36 CellType::POLYHEDRON);
37
38 auto vtk_polyh = vtkPolyhedron::SafeDownCast(vtk_cell);
39 auto num_cpoints = vtk_polyh->GetNumberOfPoints();
40 auto num_cfaces = vtk_polyh->GetNumberOfFaces();
41
42 polyh_cell->vertex_ids.reserve(num_cpoints);
43 auto point_ids = vtk_polyh->GetPointIds();
44 for (int p=0; p<num_cpoints; ++p)
45 {
46 uint64_t point_id = point_ids->GetId(p);
47 polyh_cell->vertex_ids.push_back(point_id);
48 }//for p
49
50 polyh_cell->faces.reserve(num_cfaces);
51 for (int f=0; f<num_cfaces; ++f)
52 {
53 LightWeightFace face;
54 auto vtk_face = vtk_polyh->GetFace(f);
55 auto num_face_points = vtk_face->GetNumberOfPoints();
56
57 face.vertex_ids.reserve(num_face_points);
58 auto face_point_ids = vtk_face->GetPointIds();
59 for (int p = 0; p < num_face_points; ++p) {
60 uint64_t point_id = face_point_ids->GetId(p);
61 face.vertex_ids.push_back(point_id);
62 }
63
64 polyh_cell->faces.push_back(face);
65 }
66
67 return polyh_cell;
68}

151/**Creates a raw polygon cell from a vtk-polygon.*/
152chi_mesh::UnpartitionedMesh::LightWeightCell* chi_mesh::UnpartitionedMesh::
153CreateCellFromVTKPolygon(vtkCell *vtk_cell)
154{
155 auto poly_cell = new LightWeightCell(CellType::POLYGON,
156 CellType::POLYGON);
157
158 auto vtk_polygon = vtkPolygon::SafeDownCast(vtk_cell);
159 auto num_cpoints = vtk_polygon->GetNumberOfPoints();
160 auto num_cfaces = num_cpoints;
161
162 poly_cell->vertex_ids.reserve(num_cpoints);
163 auto point_ids = vtk_polygon->GetPointIds();
164 for (int p=0; p<num_cpoints; ++p)
165 {
166 uint64_t point_id = point_ids->GetId(p);
167 poly_cell->vertex_ids.push_back(point_id);
168 }//for p
169
170 poly_cell->faces.reserve(num_cfaces);
171 for (int f=0; f<num_cfaces; ++f)
172 {
173 LightWeightFace face;
174
175 auto v0_id = poly_cell->vertex_ids[f];
176 auto v1_id = (f<(num_cfaces-1))? poly_cell->vertex_ids[f+1] :
177 poly_cell->vertex_ids[0];
178
179 face.vertex_ids.reserve(2);
180 face.vertex_ids.push_back(v0_id);
181 face.vertex_ids.push_back(v1_id);
182
183 poly_cell->faces.push_back(face);
184 }
185
186 return poly_cell;
187}

326/**Reads a VTK unstructured mesh.*/
327void chi_mesh::UnpartitionedMesh::
328 ReadFromVTU(const chi_mesh::UnpartitionedMesh::Options &options)
329{
330 //======================================== Attempt to open file
331 std::ifstream file;
332 file.open(options.file_name);
333
334 if (!file.is_open())
335 {
336 chi::log.LogAllError()
337 << "Failed to open file: "<< options.file_name <<" in call "
338 << "to ReadFromVTU \n";
339 chi::Exit(EXIT_FAILURE);
340 }
341 file.close();
342
343 //======================================== Read the file
344 auto reader = vtkSmartPointer<vtkXMLUnstructuredGridReader>::New();
345 reader->SetFileName(options.file_name.c_str());
346 chi::log.Log()
347 << "Reading VTU file : \""
348 << options.file_name << "\".";
349
350 reader->Update();
351
352 chi::log.Log()
353 << "Done reading VTU file: \""
354 << options.file_name << "\".";
355
356 //======================================== Remove duplicate vertices
357 auto cleaner = vtkSmartPointer<vtkCleanUnstructuredGrid>::New();
358 cleaner->SetInputData(reader->GetOutput());
359 cleaner->Update();
360 auto ugrid = cleaner->GetOutput();
361
362 auto total_cell_count = ugrid->GetNumberOfCells();
363 auto total_point_count = ugrid->GetNumberOfPoints();
364
365 chi::log.Log()
366 << "Clean grid num cells and points: "
367 << total_cell_count << " "
368 << total_point_count;
369
370 //======================================== Determine if reading
371 // cell identifiers
372 if (options.material_id_fieldname != options.boundary_id_fieldname)
373 {
374 chi::log.LogAllError()
375 << "The VTU reader expects material identifiers and boundary identifiers "
376 << "to be defined in the same field.";
377 chi::Exit(EXIT_FAILURE);
378 }
379
380 vtkDataArray* cell_id_array_ptr = nullptr;
381 if (options.material_id_fieldname.empty())
382 {
383 chi::log.Log()
384 << "A user-supplied field name from which to recover cell identifiers "
385 << "has not been provided. Only the mesh will be read.";
386 }
387 else
388 {
389 chi::log.Log()
390 << "A user-supplied field name from which to recover cell identifiers "
391 << "has been provided. The mesh will be read and both material ID and "
392 << "boundary ID will be read from the vtkCellData field with name : \""
393 << options.material_id_fieldname << "\".";
394
395 const auto vtk_abstract_array_ptr =
396 ugrid->GetCellData()->GetAbstractArray(options.material_id_fieldname.c_str());
397 if (!vtk_abstract_array_ptr)
398 {
399 chi::log.LogAllError()
400 << "The VTU file : \"" << options.file_name << "\" "
401 << "does not contain a vtkCellData field of name : \""
402 << options.material_id_fieldname << "\".";
403 chi::Exit(EXIT_FAILURE);
404 }
405
406 cell_id_array_ptr = vtkArrayDownCast<vtkDataArray>(vtk_abstract_array_ptr);
407 if (!cell_id_array_ptr)
408 {
409 chi::log.LogAllError()
410 << "The VTU file : \"" << options.file_name << "\" "
411 << "with vtkCellData field of name : \""
412 << options.material_id_fieldname << "\" "
413 << "cannot be downcast to vtkDataArray";
414 chi::Exit(EXIT_FAILURE);
415 }
416
417 const auto cell_id_n_tup = cell_id_array_ptr->GetNumberOfTuples();
418 if (cell_id_n_tup != total_cell_count)
419 {
420 chi::log.LogAllError()
421 << "The VTU file : \"" << options.file_name << "\" "
422 << "with vtkCellData field of name : \""
423 << options.material_id_fieldname << "\" has n. tuples : "
424 << cell_id_n_tup << ", but differs from the value expected : "
425 << total_cell_count << ".";
426 chi::Exit(EXIT_FAILURE);
427 }
428
429 const auto cell_id_n_val = cell_id_array_ptr->GetNumberOfValues();
430 if (cell_id_n_val != total_cell_count)
431 {
432 chi::log.LogAllError()
433 << "The VTU file : \"" << options.file_name << "\" "
434 << "with vtkCellData field of name : \""
435 << options.material_id_fieldname << "\" has n. values : "
436 << cell_id_n_val << ", but differs from the value expected : "
437 << total_cell_count << ".";
438 chi::Exit(EXIT_FAILURE);
439 }
440 }
441
442 //======================================== Scan for mesh dimension
443 int mesh_dim = 0;
444 for (unsigned int c = 0; c < total_cell_count; ++c)
445 {
446 const auto vtk_cell = ugrid->GetCell(c);
447 const auto vtk_cell_dim = vtk_cell->GetCellDimension();
448 mesh_dim = std::max(vtk_cell_dim, mesh_dim);
449 }
450
451 if (mesh_dim < 1 || mesh_dim > 3)
452 {
453 chi::log.LogAllError()
454 << "The VTU file : \"" << options.file_name << "\" "
455 << "does not identify a mesh of valid dimension.";
456 chi::Exit(EXIT_FAILURE);
457 }
458
459 //======================================== Push cells
460 size_t num_polyhedrons = 0;
461 size_t num_hexahedrons = 0;
462 size_t num_tetrahedrons = 0;
463 size_t num_polygons = 0;
464 size_t num_quads = 0;
465 size_t num_triangles = 0;
466 size_t num_lines = 0;
467 size_t num_vertices = 0;
468 for (unsigned int c = 0; c < total_cell_count; ++c)
469 {
470 const auto vtk_cell = ugrid->GetCell(c);
471 const auto vtk_celltype = vtk_cell->GetCellType();
472 const auto vtk_cell_dim = vtk_cell->GetCellDimension();
473
474 // construct cell
475 LightWeightCell* chi_lwc = nullptr;
476 if (vtk_celltype == VTK_POLYHEDRON)
477 {
478 chi_lwc = CreateCellFromVTKPolyhedron(vtk_cell);
479 ++num_polyhedrons;
480 }
481 else if (vtk_celltype == VTK_HEXAHEDRON)
482 {
483 chi_lwc = CreateCellFromVTKHexahedron(vtk_cell);
484 ++num_hexahedrons;
485 }
486 else if (vtk_celltype == VTK_TETRA)
487 {
488 chi_lwc = CreateCellFromVTKTetrahedron(vtk_cell);
489 ++num_tetrahedrons;
490 }
491 else if (vtk_celltype == VTK_POLYGON)
492 {
493 chi_lwc = CreateCellFromVTKPolygon(vtk_cell);
494 ++num_polygons;
495 }
496 else if (vtk_celltype == VTK_QUAD)
497 {
498 chi_lwc = CreateCellFromVTKQuad(vtk_cell);
499 ++num_quads;
500 }
501 else if (vtk_celltype == VTK_TRIANGLE)
502 {
503 chi_lwc = CreateCellFromVTKTriangle(vtk_cell);
504 ++num_triangles;
505 }
506 else if (vtk_celltype == VTK_LINE)
507 {
508 chi_lwc = CreateCellFromVTKLine(vtk_cell);
509 ++num_lines;
510 }
511 else if (vtk_celltype == VTK_VERTEX)
512 {
513 chi_lwc = CreateCellFromVTKVertex(vtk_cell);
514 ++num_vertices;
515 }
516 else
517 throw std::invalid_argument(std::string(__FUNCTION__) +
518 ": Unsupported cell type.");
519
520 // append to appropriate collection
521 if (vtk_cell_dim == mesh_dim)
522 raw_cells.emplace_back(chi_lwc);
523 else if (vtk_cell_dim == mesh_dim-1)
524 raw_boundary_cells.emplace_back(chi_lwc);
525
526 // apply cell identifier
527 if (cell_id_array_ptr)
528 {
529 std::vector<double> cell_id_vec(1);
530 cell_id_array_ptr->GetTuple(c, cell_id_vec.data());
531 const auto cell_id = (int)cell_id_vec.front();
532
533 if (vtk_cell_dim == mesh_dim)
534 raw_cells.back()->material_id = cell_id;
535 else if (vtk_cell_dim == mesh_dim-1)
536 raw_boundary_cells.back()->material_id = cell_id;
537 }
538 }//for c
539 chi::log.Log() << "Number cells read: " << total_cell_count << "\n"
540 << "polyhedrons : " << num_polyhedrons << "\n"
541 << "hexahedrons : " << num_hexahedrons << "\n"
542 << "tetrahedrons : " << num_tetrahedrons << "\n"
543 << "polygons : " << num_polygons << "\n"
544 << "quads : " << num_quads << "\n"
545 << "triangles : " << num_triangles << "\n"
546 << "lines : " << num_lines << "\n"
547 << "vertices : " << num_vertices;
548
549 //======================================== Push points
550 for (int p=0; p<total_point_count; ++p)
551 {
552 auto point = ugrid->GetPoint(p);
553 vertices.emplace_back(point[0],point[1],point[2]);
554
555 if (point[0] < bound_box.xmin) bound_box.xmin = point[0];
556 if (point[0] > bound_box.xmax) bound_box.xmax = point[0];
557 if (point[1] < bound_box.ymin) bound_box.ymin = point[1];
558 if (point[1] > bound_box.ymax) bound_box.ymax = point[1];
559 if (point[2] < bound_box.zmin) bound_box.zmin = point[2];
560 if (point[2] > bound_box.zmax) bound_box.zmax = point[2];
561 }
562
563 //======================================== Always do this
564 chi_mesh::MeshAttributes dimension = NONE;
565 if (mesh_dim == 1) dimension = DIMENSION_1;
566 if (mesh_dim == 2) dimension = DIMENSION_2;
567 if (mesh_dim == 3) dimension = DIMENSION_3;
568
569 attributes = dimension | UNSTRUCTURED;
570
571 ComputeCentroidsAndCheckQuality();
572 BuildMeshConnectivity();
573}
