37 //============================================= Get grid
38 grid_ptr_ = chi_mesh::GetCurrentHandler().GetGrid();
39 const auto& grid = *grid_ptr_;
40 if (grid_ptr_ == nullptr)
41 throw std::logic_error(std::string(__PRETTY_FUNCTION__) +

45
46 //============================================= BIDs
47 auto globl_unique_bndry_ids = grid.GetDomainUniqueBoundaryIDs();
48
49 const auto& grid_boundary_id_map = grid_ptr_->GetBoundaryIDMap();
50 for (uint64_t bndry_id : globl_unique_bndry_ids)
51 {
52 if (grid_boundary_id_map.count(bndry_id) == 0)
53 throw std::logic_error(fname + ": Boundary id " +
54 std::to_string(bndry_id) + " does not have a name-assignment.");
55
56 const auto& bndry_name = grid_boundary_id_map.at(bndry_id);
57 if (boundary_preferences_.find(bndry_name) != boundary_preferences_.end())
58 {
59 BoundaryInfo bndry_info = boundary_preferences_.at(bndry_name);
60 auto& bndry_vals = bndry_info.second;
61 switch (bndry_info.first)
62 {
63 case BoundaryType::Reflecting:
64 {
65 boundaries_.insert(std::make_pair(
66 bndry_id,Boundary{BoundaryType::Reflecting, {0., 0., 0.}}));
67 Chi::log.Log() << "Boundary " << bndry_name << " set to reflecting.";
68 break;
69 }
70 case BoundaryType::Dirichlet:
71 {
72 if (bndry_vals.empty()) bndry_vals.resize(1,0.0);
73 boundaries_.insert(std::make_pair(
74 bndry_id,Boundary{BoundaryType::Dirichlet, {bndry_vals[0], 0., 0.}}));
75 Chi::log.Log() << "Boundary " << bndry_name << " set to dirichlet.";
76 break;
77 }
78 case BoundaryType::Robin:
79 {
80 if (bndry_vals.size()!=3)
81 throw std::logic_error(std::string(__PRETTY_FUNCTION__) +
82 " Robin needs 3 values in bndry vals.");
83 boundaries_.insert(std::make_pair(
84 bndry_id,Boundary{BoundaryType::Robin, {bndry_vals[0],
85 bndry_vals[1],
86 bndry_vals[2]}}));
87 Chi::log.Log() << "Boundary " << bndry_name
88 << " set to robin." << bndry_vals[0]<<","
89 << bndry_vals[1]<<","<<bndry_vals[2];
90 break;
91 }
92 case BoundaryType::Vacuum:
93 {
94 boundaries_.insert(std::make_pair(
95 bndry_id,Boundary{BoundaryType::Robin, {0.25, 0.5, 0.}}));
96 Chi::log.Log() << "Boundary " << bndry_name << " set to vacuum.";
97 break;
98 }
99 case BoundaryType::Neumann:
100 {
101 if (bndry_vals.size()!=3)
102 throw std::logic_error(std::string(__PRETTY_FUNCTION__) +
103 " Neumann needs 3 values in bndry vals.");
104 boundaries_.insert(std::make_pair(
105 bndry_id,Boundary{BoundaryType::Robin, {0., bndry_vals[0],
106 bndry_vals[1]}}));
107 Chi::log.Log() << "Boundary " << bndry_name
108 << " set to neumann." << bndry_vals[0];
109 break;
110 }
111 }//switch boundary type
112 }
113 else
114 {
115 boundaries_.insert(std::make_pair(
116 bndry_id,Boundary{BoundaryType::Dirichlet, {0., 0., 0.}}));
117 Chi::log.Log0Verbose1()
118 << "No boundary preference found for boundary index " << bndry_name
119 << "Dirichlet boundary added with zero boundary value.";
120 }
121 }//for bndry
122

126
127 const auto& OneDofPerNode = sdm.UNITARY_UNKNOWN_MANAGER;

129 num_globl_dofs_ = sdm.GetNumGlobalDOFs(OneDofPerNode);
130
131 Chi::log.Log() << "Num local DOFs: " << num_local_dofs_;
132 Chi::log.Log() << "Num globl DOFs: " << num_globl_dofs_;
133

140 b_ = chi_math::PETScUtils::CreateVector(n, N);
141
142 std::vector<int64_t> nodal_nnz_in_diag;
143 std::vector<int64_t> nodal_nnz_off_diag;
144 sdm.BuildSparsityPattern(nodal_nnz_in_diag,nodal_nnz_off_diag, OneDofPerNode);
145
146 chi_math::PETScUtils::InitMatrixSparsity(A_,
147 nodal_nnz_in_diag,
148 nodal_nnz_off_diag);

152 std::string solver_name;
153 if (not TextName().empty()) solver_name = TextName() + "-";
154
155 std::string text_name = solver_name + "phi";

159 std::make_shared<chi_physics::FieldFunctionGridBased>(

171void fv_diffusion::Solver::Execute()
172{
173 Chi::log.Log() << "\nExecuting CFEM Diffusion solver";
174
175 const auto& grid = *grid_ptr_;
176 const auto& sdm = *sdm_ptr_;
177
178 lua_State* L = Chi::console.GetConsoleState();
179
180 //============================================= Assemble the system
181 // P ~ Present cell
182 // N ~ Neighbor cell
183 Chi::log.Log() << "Assembling system: ";
184 for (const auto& cell_P : grid.local_cells)
185 {
186 const auto& cell_mapping = sdm.GetCellMapping(cell_P);
187 const double volume_P = cell_mapping.CellVolume(); //Volume of present cell
188 const auto& x_cc_P = cell_P.centroid_;
189
190 const auto imat = cell_P.material_id_;
191
192 const double sigma_a = CallLua_iXYZFunction(L, "Sigma_a",imat,x_cc_P);
193 const double q_ext = CallLua_iXYZFunction(L, "Q_ext" ,imat,x_cc_P);
194 const double D_P = CallLua_iXYZFunction(L, "D_coef" ,imat,x_cc_P);
195
196 const int64_t imap = sdm.MapDOF(cell_P, 0);
197 MatSetValue(A_, imap, imap, sigma_a * volume_P, ADD_VALUES);
198 VecSetValue(b_, imap, q_ext * volume_P, ADD_VALUES);
199
200 for (size_t f=0; f < cell_P.faces_.size(); ++f)
201 {
202 const auto& face = cell_P.faces_[f];
203 const auto& x_fc = face.centroid_;
204 const auto x_PF = x_fc - x_cc_P;
205 const auto A_f = cell_mapping.FaceArea(f);
206 const auto A_f_n = A_f * face.normal_;
207
208 if (face.has_neighbor_)
209 {
210 const auto& cell_N = grid.cells[face.neighbor_id_];
211 const int jmat = cell_N.material_id_;
212 const auto& x_cc_N = cell_N.centroid_;
213 const auto x_PN = x_cc_N - x_cc_P;
214
215 const double D_N = CallLua_iXYZFunction(L, "D_coef",jmat,x_cc_N);
216
217 const double w = x_PF.Norm()/x_PN.Norm();
218 const double D_f = 1.0/(w/D_P + (1.0-w)/D_N);
219
220 const double entry_ii = A_f_n.Dot(D_f * x_PN/x_PN.NormSquare());
221 const double entry_ij = - entry_ii;
222
223 const int64_t jmap = sdm.MapDOF(cell_N, 0);
224 MatSetValue(A_, imap, imap, entry_ii, ADD_VALUES);
225 MatSetValue(A_, imap, jmap, entry_ij, ADD_VALUES);
226 }//internal face
227 else
228 {
229 const auto& bndry = boundaries_[face.neighbor_id_];
230
231 if (bndry.type_ == BoundaryType::Robin)
232 {
233 const auto& aval = bndry.values_[0];
234 const auto& bval = bndry.values_[1];
235 const auto& fval = bndry.values_[2];
236
237 if (std::fabs(bval) < 1e-8)
238 throw std::logic_error("if b=0, this is a Dirichlet BC, not a Robin BC");
239
240 if (std::fabs(aval) > 1.0e-8)
241 MatSetValue(A_, imap, imap, A_f * aval / bval, ADD_VALUES);
242 if (std::fabs(fval) > 1.0e-8)
243 VecSetValue(b_, imap, A_f * fval / bval, ADD_VALUES);
244 }//if Robin
245
246 if (bndry.type_ == BoundaryType::Dirichlet)
247 {
248 const auto& boundary_value = bndry.values_[0];
249
250 const auto& x_cc_N = x_cc_P + 2.0*x_PF;
251 const auto x_PN = x_cc_N - x_cc_P;
252
253 const double D_f = D_P;
254 const double entry_ii = A_f_n.Dot(D_f * x_PN/x_PN.NormSquare());
255
256 MatSetValue(A_, imap, imap, entry_ii, ADD_VALUES);
257 VecSetValue(b_, imap, entry_ii * boundary_value, ADD_VALUES);
258 }//if Dirichlet
259 }//bndry face
260 }//for f
261 }//for cell
262
263 Chi::log.Log() << "Global assembly";
264
265 MatAssemblyBegin(A_, MAT_FINAL_ASSEMBLY);
266 MatAssemblyEnd(A_, MAT_FINAL_ASSEMBLY);
267 VecAssemblyBegin(b_);
268 VecAssemblyEnd(b_);
269
270 Chi::log.Log() << "Done global assembly";
271
272 //============================================= Create Krylov Solver
273 Chi::log.Log() << "Solving: ";
274 auto petsc_solver =
275 chi_math::PETScUtils::CreateCommonKrylovSolverSetup(
276 A_, //Matrix
277 TextName(), //Solver name
278 KSPCG, //Solver type
279 PCGAMG, //Preconditioner type
280 basic_options_("residual_tolerance").FloatValue(), //Relative residual tolerance
281 basic_options_("max_iters").IntegerValue() //Max iterations
282 );
283
284 //============================================= Solve
285 KSPSolve(petsc_solver.ksp, b_, x_);
286
287 UpdateFieldFunctions();
288
289 Chi::log.Log() << "Done solving";
290
