100 chi_montecarlon::Particle chi_montecarlon::ResidualSource2::
101  CreateBndryParticle(chi_montecarlon::RandomNumberGenerator* rng)
102 {
103  chi_montecarlon::Particle new_particle;
104 
105  if (source_patch_cdf.empty())
106  {
107  new_particle.alive = false;
108  return new_particle;
109  }
110 
111  //======================================== Sample source patch
112  int source_patch_sample = std::lower_bound(
113  source_patch_cdf.begin(),
114  source_patch_cdf.end(),
115  rng->Rand()) - source_patch_cdf.begin();
116 
117  auto& source_patch = source_patches[source_patch_sample];
118 
119  //======================================== Get references
120  int cell_local_id = std::get<0>(source_patch);
121  int f = std::get<1>(source_patch);
122  auto& RotationMatrix = std::get<2>(source_patch);
123  auto cell = &grid->local_cells[cell_local_id];
124  auto& face = cell->faces[f];
125  auto fv_view = fv_sdm->MapFeView(cell_local_id);
126 
127  //======================================== Sample position
128  if (cell->Type() == chi_mesh::CellType::SLAB)
129  new_particle.pos = *grid->vertices[face.vertex_ids[0]];
130  else if (cell->Type() == chi_mesh::CellType::POLYGON)
131  {
132  chi_mesh::Vertex& v0 = *grid->vertices[face.vertex_ids[0]];
133  chi_mesh::Vertex& v1 = *grid->vertices[face.vertex_ids[1]];
134  double w = rng->Rand();
135  new_particle.pos = v0*w + v1*(1.0-w);
136  }
137  else if (cell->Type() == chi_mesh::CellType::POLYHEDRON)
138  {
139  auto polyh_cell = (chi_mesh::CellPolyhedron*)cell;
140  auto polyh_fv_view = (PolyhedronFVView*)fv_view;
141 
142  auto edges = polyh_cell->GetFaceEdges(f);
143 
144  //===================== Sample side
145  double rn = rng->Rand();
146  int s = -1;
147  double cumulated_area = 0.0;
148  for (auto face_side_area : polyh_fv_view->face_side_area[f])
149  {
150  s++;
151  if (rn < ((face_side_area+cumulated_area)/polyh_fv_view->face_area[f]))
152  break;
153  cumulated_area+=face_side_area;
154  }
155 
156  double w0 = rng->Rand();
157  double w1 = rng->Rand()*(1.0-w0);
158  chi_mesh::Vector3& v0 = *grid->vertices[edges[s][0]];
159  new_particle.pos = v0 + polyh_fv_view->face_side_vectors[f][s][0]*w0 +
160  polyh_fv_view->face_side_vectors[f][s][1]*w1;
161  }
162 
163  //======================================== Sample direction
164  double costheta = rng->Rand(); //Sample half-range only
165  double theta = acos(sqrt(costheta));
166  double varphi = rng->Rand()*2.0*M_PI;
167 
168  chi_mesh::Vector3 ref_dir;
169  ref_dir.x = sin(theta)*cos(varphi);
170  ref_dir.y = sin(theta)*sin(varphi);
171  ref_dir.z = cos(theta);
172 
173  new_particle.dir = RotationMatrix*ref_dir;
174 
175  //======================================== Determine weight
176  // Interpolate phi
177  auto pwl_view = ref_solver->pwl_discretization->MapFeViewL(cell_local_id);
178  std::vector<double> shape_values(pwl_view->dofs,0.0);
179  pwl_view->ShapeValues(new_particle.pos,shape_values);
180  auto ff_dof_vals = (*resid_ff).GetCellDOFValues(cell_local_id,
181  new_particle.egrp,
182  0);
183 
184  double cell_phi = 0.0;
185  for (int dof=0; dof<pwl_view->dofs; dof++)
186  cell_phi += ff_dof_vals[dof]*shape_values[dof];
187 
188  //============================== Get boundary flux
189  double bndry_phi = 0.0;
190  if (ref_bndry == abs(face.neighbor) and (face.neighbor<0))
191  bndry_phi = 2.0*ref_bndry_val;
192 
193  new_particle.w = bndry_phi - cell_phi;
194 
195  //======================================== Sample energy
196  new_particle.egrp = 0;
197 
198  new_particle.cur_cell_global_id = cell->global_id;
199  new_particle.cur_cell_local_id = cell_local_id;
200 
201  return new_particle;
202 }

206 chi_montecarlon::Particle chi_montecarlon::ResidualSource2::
207  CreateParticle(chi_montecarlon::RandomNumberGenerator* rng)

210

212  int num_loc_cells = ref_solver->grid->local_cell_glob_indices.size();
213  int lc = std::floor((num_loc_cells)*rng->Rand());
214 
215  int cell_glob_index = ref_solver->grid->local_cell_glob_indices[lc];
216  auto cell = &ref_solver->grid->local_cells[lc];
217  auto pwl_view = ref_solver->pwl_discretization->MapFeViewL(lc);
218  int map = ref_solver->local_cell_pwl_dof_array_address[lc];
219 
220  int mat_id = cell->material_id;

224  auto xs = (chi_physics::TransportCrossSections*)mat->properties[xs_id];
225

227  double sigt = xs->sigma_tg[0];
228  double sigs = sigt-siga;
229 
230  //======================================== Sample position
231  new_particle.pos = GetRandomPositionInCell(rng, cell_vol_info[lc]);
232 
233  //======================================== Sample direction
234  double costheta = 2.0*rng->Rand()-1.0;
235  double theta = acos(costheta);
236  double varphi = rng->Rand()*2.0*M_PI;
237 
238  chi_mesh::Vector3 ref_dir;
239  ref_dir.x = sin(theta)*cos(varphi);
240  ref_dir.y = sin(theta)*sin(varphi);
241  ref_dir.z = cos(theta);
242 
243  new_particle.dir = ref_dir;
244 
245  //======================================== Sample uncollided
246  std::vector<double> shape_values;
247  pwl_view->ShapeValues(new_particle.pos,shape_values);
248 
249  double domain_volume = ref_solver->domain_volume;
250  double tmf = ref_solver->tally_multipl_factor;
251 
252  double weight = 0.0;
253  for (int dof=0; dof<pwl_view->dofs; ++dof)
254  {
255  int ir = map + dof*ref_solver->num_grps*ref_solver->num_moms +
256  ref_solver->num_grps*0 + 0;
257  weight += shape_values[dof]*ref_solver->phi_pwl_uncollided_rmc[ir];
258  }
259  weight *= sigs*domain_volume/tmf;
260 
261  new_particle.w = weight;
262  new_particle.egrp = 0;
263 
264  new_particle.alive = true;
265 
266  if (std::fabs(weight) < 1.0e-12)
267  new_particle.alive = false;
268 
269  new_particle.cur_cell_global_id = cell_glob_index;
270  new_particle.cur_cell_local_id = cell->local_id;
271 
272  return new_particle;
