49 /**Initializes an rmc source.
50  *
51  * This process involves numerous steps. One of the first steps is
52  * to */
53 void chi_montecarlon::ResidualSource2::
54 Initialize(chi_mesh::MeshContinuum *ref_grid,
55  SpatialDiscretization_FV *ref_fv_sdm,
56  chi_montecarlon::Solver* ref_solver)
57 {
58  chi_log.Log(LOG_0) << "Initializing Residual Bndry Source";
59  grid = ref_grid;
60  fv_sdm = ref_fv_sdm;
61  this->ref_solver = ref_solver;
62 
63  const int ALL_BOUNDRIES = -1;
64 
65  //============================================= Build surface src patchess
66  // The surface points
67  double total_patch_area = 0.0;
68  for (auto& cell_glob_index : grid->local_cell_glob_indices)
69  {
70  auto cell = grid->cells[cell_glob_index];
71  auto fv_view = fv_sdm->MapFeView(cell_glob_index);
72 
73 
74  int f=0;
75  for (auto& face : cell->faces)
76  {
77  if (not grid->IsCellBndry(face.neighbor)) {++f; continue;}
78 
79  //Determine if face will be sampled
80  bool sample_face = true;
81 
82 
83  if (sample_face)
84  {
85  double face_area = fv_view->face_area[f];
86  chi_mesh::Matrix3x3 R;
87 
88  chi_mesh::Vector n = cell->faces[f].normal*-1.0;
89  chi_mesh::Vector khat(0.0,0.0,1.0);
90 
91  if (n.Dot(khat) > 0.9999999)
92  R.SetDiagonalVec(1.0,1.0,1.0);
93  else if (n.Dot(khat) < -0.9999999)
94  R.SetDiagonalVec(1.0,1.0,-1.0);
95  else
96  {
97  chi_mesh::Vector binorm = khat.Cross(n);
98  binorm = binorm/binorm.Norm();
99 
100  chi_mesh::Vector tangent = binorm.Cross(n);
101  tangent = tangent/tangent.Norm();
102 
103  R.SetColJVec(0,tangent);
104  R.SetColJVec(1,binorm);
105  R.SetColJVec(2,n);
106  }
107 
108  total_patch_area += face_area;
109  source_patches.emplace_back(cell_glob_index,f,R,face_area);
110  }
111  ++f;
112  }//for face
113  }//for cell
114 
115  //============================================= Build source cdf
116  source_patch_cdf.resize(source_patches.size(),0.0);
117  double cumulative_value = 0.0;
118  int p = 0;
119  for (auto& source_patch : source_patches)
120  {
121  cumulative_value += std::get<3>(source_patch);
122  source_patch_cdf[p++] = cumulative_value/total_patch_area;
123  }
124 }

127 /**Executes a source sampling for the residual source.*/
128 chi_montecarlon::Particle chi_montecarlon::ResidualSource2::
129 CreateBndryParticle(chi_montecarlon::RandomNumberGenerator* rng)
130 {
131  chi_montecarlon::Particle new_particle;
132 
133  //======================================== Sample source patch
134  int source_patch_sample = std::lower_bound(
135  source_patch_cdf.begin(),
136  source_patch_cdf.end(),
137  rng->Rand()) - source_patch_cdf.begin();
138 
139  auto& source_patch = source_patches[source_patch_sample];
140 
141  //======================================== Get references
142  int cell_glob_index = std::get<0>(source_patch);
143  int f = std::get<1>(source_patch);
144  auto& RotationMatrix = std::get<2>(source_patch);
145  auto cell = grid->cells[cell_glob_index];
146  auto face = cell->faces[f];
147  auto fv_view = fv_sdm->MapFeView(cell_glob_index);
148 
149  //======================================== Sample position
150  if (cell->Type() == chi_mesh::CellType::SLAB)
151  new_particle.pos = *grid->nodes[face.vertex_ids[0]];
152  else if (cell->Type() == chi_mesh::CellType::POLYGON)
153  {
154  chi_mesh::Vertex& v0 = *grid->nodes[face.vertex_ids[0]];
155  chi_mesh::Vertex& v1 = *grid->nodes[face.vertex_ids[1]];
156  double w = rng->Rand();
157  new_particle.pos = v0*w + v1*(1.0-w);
158  }
159  else if (cell->Type() == chi_mesh::CellType::POLYHEDRON)
160  {
161  auto polyh_cell = (chi_mesh::CellPolyhedron*)cell;
162  auto polyh_fv_view = (PolyhedronFVView*)fv_view;
163 
164  auto edges = polyh_cell->GetFaceEdges(f);
165 
166  //===================== Sample side
167  double rn = rng->Rand();
168  int s = -1;
169  double cumulated_area = 0.0;
170  for (auto face_side_area : polyh_fv_view->face_side_area[f])
171  {
172  s++;
173  if (rn < ((face_side_area+cumulated_area)/polyh_fv_view->face_area[f]))
174  break;
175  cumulated_area+=face_side_area;
176  }
177 
178  double w0 = rng->Rand();
179  double w1 = rng->Rand()*(1.0-w0);
180  chi_mesh::Vector& v0 = *grid->nodes[edges[s][0]];
181  new_particle.pos = v0 + polyh_fv_view->face_side_vectors[f][s][0]*w0 +
182  polyh_fv_view->face_side_vectors[f][s][1]*w1;
183  }
184 
185  //======================================== Sample direction
186  double costheta = rng->Rand(); //Sample half-range only
187  double theta = acos(sqrt(costheta));
188 // double theta = acos(std::max(costheta,0.001));
189  double varphi = rng->Rand()*2.0*M_PI;
190 
191  chi_mesh::Vector ref_dir;
192  ref_dir.x = sin(theta)*cos(varphi);
193  ref_dir.y = sin(theta)*sin(varphi);
194  ref_dir.z = cos(theta);
195 
196  new_particle.dir = RotationMatrix*ref_dir;
197 
198  //======================================== Determine weight
199  //============================== Interpolate phi
200  auto pwl_view = ref_solver->pwl_discretization->MapFeView(cell_glob_index);
201  std::vector<double> shape_values(pwl_view->dofs,0.0);
202  pwl_view->ShapeValues(new_particle.pos,shape_values);
203 
204  //TODO: Make proper mappings here
205  double cell_phi = 0.0;
206  for (int dof=0; dof<pwl_view->dofs; dof++)
207  {
208  int map = (*resid_ff->local_cell_dof_array_address)[cell->cell_local_id];
209  int mom = 0;
210  int ir = map + dof*resid_ff->num_grps*resid_ff->num_moms +
211  resid_ff->num_grps*mom + new_particle.egrp;
212  cell_phi += (*resid_ff->field_vector_local)[ir]*shape_values[dof];
213  }
214  //============================== Get boundary flux
215  double bndry_phi = 0.0;
216  if (ref_bndry == abs(face.neighbor))
217  bndry_phi = 2.0*ref_bndry_val;
218 
219  new_particle.w = bndry_phi - cell_phi;
220 
221  //======================================== Sample energy
222  new_particle.egrp = 0;
223 
224  new_particle.cur_cell_ind = cell_glob_index;
225 
226  return new_particle;
227 }

288

290  double tmf = ref_solver->tally_multipl_factor;
291

293  for (int dof=0; dof<pwl_view->dofs; ++dof)
294  {
295  int ir = map + dof*ref_solver->num_grps*ref_solver->num_moms +
296  ref_solver->num_grps*0 + 0;
297  weight += shape_values[dof]*ref_solver->phi_pwl_uncollided_rmc[ir];
298  }
299  weight *= sigs*domain_volume/tmf;
300 
301  new_particle.w = weight;
302  new_particle.egrp = 0;
303 
304  new_particle.alive = true;
305
