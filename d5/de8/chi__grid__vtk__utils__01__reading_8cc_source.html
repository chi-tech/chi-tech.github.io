13/**Finds the highest dimension across all the grid blocks. This

15 * are boundary faces.*/
16int chi_mesh::FindHighestDimension(std::vector<vtkUGridPtrAndName> &ugrid_blocks)
17{
18
19 int max_dim = 0;
20 for (auto& ugrid : ugrid_blocks)
21 if (ugrid.first->GetNumberOfCells() != 0)

82/**Given several unstructured grid blocks, each denoting a material id,
83 * this function sets material ids accordingly.*/
84std::vector<uint64_t> chi_mesh::
85 BuildBlockCellExtents(std::vector<vtkUGridPtrAndName> &ugrid_blocks,
86 const int desired_dimension)
87{
88 std::vector<uint64_t> block_mat_ids;
89 size_t total_cells = 0;
90
91 for (auto& ugrid : ugrid_blocks)
92 {
93 uint64_t num_cells = ugrid.first->GetNumberOfCells();
94
95 if (num_cells == 0) continue;
96
97 if (ugrid.first->GetCell(0)->GetCellDimension() == desired_dimension)
98 {
99 total_cells += num_cells;
100 block_mat_ids.push_back(total_cells);
101 }
102 }
103 return block_mat_ids;
104}
