13/**Finds the highest dimension across all the grid blocks. This
14 * is useful when a vtk-read mesh contains multiple blocks. Some of which
15 * are boundary faces.*/
16int chi_mesh::FindHighestDimension(std::vector<vtkUGridPtrAndName> &ugrid_blocks)
17{
18
19 int max_dim = 0;
20 for (auto& ugrid : ugrid_blocks)
21 if (ugrid.first->GetNumberOfCells() != 0)
22 max_dim = std::max(max_dim, ugrid.first->GetCell(0)->GetCellDimension());
23
24 return max_dim;
25}

29/**Consolidates all blocks containing cells with the desired dimension.
30 * Thereafter it removes duplicate vertices.*/
31chi_mesh::vtkUGridPtr chi_mesh::
32 ConsolidateGridBlocks(std::vector<vtkUGridPtrAndName> &ugrid_blocks)
33{
34 const std::string fname = "chi_mesh::ConsolidateGridBlocks";
35
36 //======================================== Determine if all blocks have
37 // global-ids
38 bool has_global_ids = true;
39 for (auto& ugrid_name : ugrid_blocks)
40 {
41 auto& ugrid = ugrid_name.first;
42 const bool has_cell_gids = ugrid->GetCellData()->GetGlobalIds();
43 const bool has_pnts_gids = ugrid->GetPointData()->GetGlobalIds();
44 const bool has_block_ids = ugrid->GetCellData()->GetArray("BlockID");
45
46 if ((not has_cell_gids) or (not has_pnts_gids))
47 has_global_ids = false;
48
49 if (not has_block_ids)
50 throw std::logic_error(fname + ": Grid block " + ugrid_name.second +
51 " has not \"BlockID\" array.");
52 }//for grid_name pairs
53
54 if (has_global_ids)
55 chi::log.Log() << fname << ": blocks have global-id arrays";
56
57 //======================================== Consolidate the blocks
58 auto append = vtkSmartPointer<vtkAppendFilter>::New();
59 for (auto& ugrid : ugrid_blocks)
60 append->AddInputData(ugrid.first);
61
62 append->MergePointsOn();
63 append->Update();
64
65 auto consolidated_ugrid = vtkSmartPointer<vtkUnstructuredGrid>(
66 vtkUnstructuredGrid::SafeDownCast(append->GetOutput()));
67
68 chi::log.Log0Verbose1()
69 << "Consolidated grid num cells and points: "
70 << consolidated_ugrid->GetNumberOfCells() << " "
71 << consolidated_ugrid->GetNumberOfPoints();
72
73 if (has_global_ids)
74 {
75 const vtkIdType num_points = consolidated_ugrid->GetNumberOfPoints();
76 vtkIdType min_id = num_points;
77 vtkIdType max_id = 0;
78 for (vtkIdType p=0; p<num_points; ++p)
79 {
80 auto point_gids = vtkIdTypeArray::SafeDownCast(consolidated_ugrid->GetPointData()->GetGlobalIds());
81 auto point_gid = point_gids->GetValue(p);
82
83 min_id = std::min(min_id, point_gid);
84 max_id = std::max(max_id, point_gid);
85 }
86
87 chi::log.Log()
88 << "Minimum and Maximum node-ids " << min_id << " " << max_id;
89 }
90
91 return consolidated_ugrid;
92}

96/**Provides a map of the different grids that have the
97 * requested dimension.*/
98std::vector<chi_mesh::vtkUGridPtrAndName> chi_mesh::
99 GetBlocksOfDesiredDimension(std::vector<vtkUGridPtrAndName> &ugrid_blocks,
100 int desired_dimension)
101{
102 std::vector<chi_mesh::vtkUGridPtrAndName> desired_blocks;
103 for (auto& ugrid : ugrid_blocks)
104 {
105 if (ugrid.first->GetNumberOfCells() == 0) continue;
106
107 if (ugrid.first->GetCell(0)->GetCellDimension() == desired_dimension)
108 desired_blocks.push_back(ugrid);
109 }
110
111 return desired_blocks;
112}

116/**Given several unstructured grid blocks, each denoting a material id,
117 * this function sets material ids accordingly.*/
118std::vector<uint64_t> chi_mesh::
119 BuildBlockCellExtents(std::vector<vtkUGridPtrAndName> &ugrid_blocks,
120 const int desired_dimension)
121{
122 std::vector<uint64_t> block_mat_ids;
123 size_t total_cells = 0;
124
125 for (auto& ugrid : ugrid_blocks)
126 {
127 uint64_t num_cells = ugrid.first->GetNumberOfCells();
128
129 if (num_cells == 0) continue;
130
131 if (ugrid.first->GetCell(0)->GetCellDimension() == desired_dimension)
132 {
133 total_cells += num_cells;
134 block_mat_ids.push_back(total_cells);
135 }
136 }
137 return block_mat_ids;
138}

141/**Given several unstructured grid blocks, each denoting a material id,
142 * this function creates a VTK cell-data array called "BlockID" that holds
143 * this information.*/
144void chi_mesh::
145 SetBlockIDArrays(std::vector<vtkUGridPtrAndName> &ugrid_blocks)
146{
147 int block_id = 0;
148 for (auto& ugrid : ugrid_blocks)
149 {
150 const vtkIdType num_cells = ugrid.first->GetNumberOfCells();
151
152 if (num_cells == 0) continue;
153
154 vtkNew<vtkIntArray> block_id_list;
155 block_id_list->SetName("BlockID");
156
157 for (vtkIdType c=0; c<num_cells; ++c)
158 block_id_list->InsertNextValue(block_id);
159
160 auto arr = ugrid.first->GetCellData()->GetArray("BlockID");
161 if (not arr)
162 ugrid.first->GetCellData()->RemoveArray("BlockID");
163
164 ugrid.first->GetCellData()->AddArray(block_id_list);
165 ++block_id;
166 }
167}

171/**Retrieves material-ids from a field.*/
172std::vector<int> chi_mesh::
173 BuildCellMaterialIDsFromField(vtkUGridPtr &ugrid,
174 const std::string& field_name,
175 const std::string& file_name)
176{
177 const size_t total_cell_count = ugrid->GetNumberOfCells();
178 std::vector<int> material_ids(total_cell_count, -1);
179
180 //======================================== Determine if reading
181 // cell identifiers
182 vtkDataArray* cell_id_array_ptr;
183 if (field_name.empty())
184 {
185 chi::log.Log0Warning()
186 << "A user-supplied field name from which to recover material identifiers "
187 << "has not been found. Material-ids will be left unassigned.";
188 goto end_error_checks;
189 }
190 else
191 {
192 auto cell_data = ugrid->GetCellData();
193 const auto vtk_abstract_array_ptr =
194 cell_data->GetAbstractArray(field_name.c_str());
195
196 if (!vtk_abstract_array_ptr)
197 {
198 chi::log.Log0Warning()
199 << "The VTU file : \"" << file_name << "\" "
200 << "does not contain a vtkCellData field of name : \""
201 << field_name << "\". Material-ids will be left unassigned.";
202 goto end_error_checks;
203 }
204
205 cell_id_array_ptr = vtkArrayDownCast<vtkDataArray>(vtk_abstract_array_ptr);
206 if (!cell_id_array_ptr)
207 {
208 chi::log.Log0Warning()
209 << "The VTU file : \"" << file_name << "\" "
210 << "with vtkCellData field of name : \""
211 << field_name << "\" "
212 << "cannot be downcast to vtkDataArray. Material-ids will be left "
213 "unassigned.";
214 goto end_error_checks;
215 }
216
217 const auto cell_id_n_tup = cell_id_array_ptr->GetNumberOfTuples();
218 if (cell_id_n_tup != total_cell_count)
219 {
220 chi::log.Log0Warning()
221 << "The VTU file : \"" << file_name << "\" "
222 << "with vtkCellData field of name : \""
223 << field_name << "\" has n. tuples : "
224 << cell_id_n_tup << ", but differs from the value expected : "
225 << total_cell_count << ". Material-ids will be left unassigned.";
226 goto end_error_checks;
227 }
228
229 const auto cell_id_n_val = cell_id_array_ptr->GetNumberOfValues();
230 if (cell_id_n_val != total_cell_count)
231 {
232 chi::log.Log0Warning()
233 << "The VTU file : \"" << file_name << "\" "
234 << "with vtkCellData field of name : \""
235 << field_name << "\" has n. values : "
236 << cell_id_n_val << ", but differs from the value expected : "
237 << total_cell_count << ". Material-ids will be left unassigned.";
238 goto end_error_checks;
239 }
240 }
241
242 // apply cell identifier
243 for (size_t c = 0; c < total_cell_count; ++c)
244 {
245 std::vector<double> cell_id_vec(1);
246 cell_id_array_ptr->GetTuple(static_cast<vtkIdType>(c), cell_id_vec.data());
247 const auto mat_id = (int) cell_id_vec.front();
248
249 material_ids[c] = mat_id;
250 }
251
252 end_error_checks:
253 return material_ids;
254}
