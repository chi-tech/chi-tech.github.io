16VectorGhostCommunicator::VectorGhostCommunicator(
17 const uint64_t local_size,
18 const uint64_t global_size,
19 const std::vector<int64_t>& ghost_ids,
20 const MPI_Comm communicator)
21 : local_size_(local_size),
22 global_size_(global_size),
23 ghost_ids_(ghost_ids),
24 comm_(communicator)
25{
26 // Get the process ID and total process count
27 MPI_Comm_rank(comm_, &location_id_);
28 MPI_Comm_size(comm_, &process_count_);
29
30 // Get the local vector sizes per process
31 std::vector<uint64_t> local_sizes(process_count_, 0);
32 MPI_Allgather(&local_size_, //sendbuf
33 1, MPI_UINT64_T, //sendcount + sendtype
34 local_sizes.data(), //recvbuf
35 1, MPI_UINT64_T, //recvcount + recvtype
36 comm_); //communicator
37
38 // With the vector sizes per processor, now the offsets for each
39 // processor can be defined using a cumulative sum per processor.
40 // This allows for the determination of whether a global index is
41 // locally owned or not.
42 extents_.assign(process_count_ + 1, 0);
43 for (size_t locJ = 1; locJ < process_count_; ++locJ)
44 extents_[locJ] = extents_[locJ - 1] + local_sizes[locJ - 1];
45 extents_[process_count_] =
46 extents_[process_count_ - 1] + local_sizes.back();
47
48 // Construct a mapping between processes and the ghost indices
49 // that belong to them. This information yields what this process
50 // needs to receive from other processes.
51 std::map<int, std::vector<int64_t>> recv_map;
52 for (int64_t ghost_id : ghost_ids)
53 recv_map[FindOwnerPID(ghost_id)].push_back(ghost_id);
54
55 // This process will receive data in process-contiguous manner,
56 // so a mapping needs to be developed to map each ghost id to
57 // its respective ordering in the received data.
58 size_t count = 0;
59 for (const auto& [pid, gids] : recv_map)
60 for (const int64_t gid : gids)
61 ghost_to_recv_map_[gid] = count++;
62
63 // Now, the structure of the data being received from communication
64 // is developed. This involves determining the amount of data
65 // being sent per process and the starting position of the data in
66 // the receive buffer per process.
67 int total_recvcounts = 0;
68 recvcounts_.assign(process_count_, 0);
69 recvdispls_.assign(process_count_, 0);
70 for (const auto& [pid, gids] : recv_map)
71 {
72 recvcounts_[pid] = static_cast<int>(gids.size());
73 recvdispls_[pid] = total_recvcounts;
74 total_recvcounts += gids.size();
75 }
76
77 // For communication, each process must also know what it is
78 // sending to other processors. If each processor sends each
79 // other process the global ids it needs to receive, then each
80 // process will know what other processes need from it. The
81 // MPI utility MapAllToAll in Chi-Tech accomplishes this task,
82 // returning a mapping of processes to the global ids that this
83 // process needs to send.
84 std::map<int, std::vector<int64_t>> send_map =
85 chi_mpi_utils::MapAllToAll(recv_map, MPI_INT64_T, comm_);
86
87 // With this information, the amount of information that needs
88 // to be sent can be determined.
89 size_t send_size = 0;
90 for (const auto& [pid, gids] : send_map)
91 send_size += gids.size();
92
93 // Next, the local ids on this process that need to be
94 // communicated to other processes can be determined and stored.
95 local_ids_to_send_.reserve(send_size);
96 for (const auto& [pid, gids] : send_map)
97 for (const int64_t gid : gids)
98 {
99 ChiLogicalErrorIf(
100 gid < extents_[location_id_] or
101 gid >= extents_[location_id_ + 1],
102 std::string(__FUNCTION__) + ": " +
103 "Problem determining communication pattern. Process " +
104 std::to_string(pid) + " determined that process " +
105 std::to_string(location_id_) + " needs to communicate global id " +
106 std::to_string(gid) + " to it, but this id is not locally owned.");
107
108 local_ids_to_send_.push_back(gid - extents_[location_id_]);
109 }
110
111 // Finally, the communication pattern for the data being sent
112 // can be constructed similarly to that for the received data.
113 int total_sendcounts = 0;
114 sendcounts_.assign(process_count_, 0);
115 senddispls_.assign(process_count_, 0);
116 for (const auto& [pid, gids] : send_map)
117 {
118 sendcounts_[pid] = static_cast<int>(gids.size());
119 senddispls_[pid] = total_sendcounts;
120 total_sendcounts += gids.size();
121 }
122}

126int64_t VectorGhostCommunicator::
127MapGhostToLocal(const int64_t ghost_id) const
128{
129 ChiInvalidArgumentIf(
130 ghost_to_recv_map_.count(ghost_id) == 0,
131 "The given ghost id does not belong to this communicator.");
132
133 // Get the position within the ghost id vector of the given ghost id
134 const auto k = std::find(ghost_ids_.begin(),
135 ghost_ids_.end(), ghost_id) - ghost_ids_.begin();
136
137 // Local index is local size plus the position in the ghost id vector
138 return local_size_ + k;
139}

143void VectorGhostCommunicator::
144CommunicateGhostEntries(std::vector<double>& ghosted_vector) const
145{
146 ChiInvalidArgumentIf(
147 ghosted_vector.size() != local_size_ + ghost_ids_.size(),
148 std::string(__FUNCTION__) + ": Vector size mismatch.");
149
150 // Serialize the data that needs to be sent
151 const size_t send_size = local_ids_to_send_.size();
152 std::vector<double> send_data;
153 send_data.reserve(send_size);
154 for (const int64_t local_id : local_ids_to_send_)
155 send_data.push_back(ghosted_vector[local_id]);
156
157 // Create serialized storage for the data to be received
158 const size_t recv_size = ghost_ids_.size();
159 std::vector<double> recv_data(recv_size, 0.0);
160
161 // Communicate the ghost data
162 MPI_Alltoallv(send_data.data(),
163 sendcounts_.data(),
164 senddispls_.data(),
165 MPI_DOUBLE,
166 recv_data.data(),
167 recvcounts_.data(),
168 recvdispls_.data(),
169 MPI_DOUBLE,
170 comm_);
171
172 // Lastly, populate the local vector with ghost data. All ghost data is
173 // appended to the back of the local vector. Using the mapping between
174 // ghost indices and the relative ghost index position along with the
175 // ordering of the ghost indices, this can be accomplished.
176 for (size_t k = 0; k < recv_size; ++k)
177 ghosted_vector[local_size_ + k] =
178 recv_data[ghost_to_recv_map_.at(ghost_ids_[k])];
179}

191std::vector<double> VectorGhostCommunicator::
192MakeGhostedVector(const std::vector<double>& local_vector) const
193{
194 ChiInvalidArgumentIf(
195 local_vector.size() != local_size_,
196 std::string(__FUNCTION__) + ": Incompatible unghosted vector." +
197 "unghosted_vector.size() != local_size_");
198
199 // Add ghost indices to the back of the unghosted vector
200 std::vector<double> vec = local_vector;
201 for (size_t i = 0; i < ghost_ids_.size(); ++i)
202 vec.emplace_back(0.0);
203 return vec;
204}

207int VectorGhostCommunicator::
208FindOwnerPID(const int64_t global_id) const
209{
210 ChiInvalidArgumentIf(
211 global_id < 0 or global_id >= global_size_,
212 std::string(__FUNCTION__) + ": Invalid global id." +
213 "Global ids must be in [0, global_size_).");
214
215 for (int p = 0; p < process_count_; ++p)
216 if (global_id >= extents_[p] and global_id < extents_[p + 1])
217 return p;
218 return -1;
219}
