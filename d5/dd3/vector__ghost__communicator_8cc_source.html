43 // The next steps can be a little confusing. What we essentially want to

45 // multiple times and as efficiently as possible. This information includes
46 // what information to send to other locations AS WELL AS what information

59 //Buy building the query-map we essentially

63 //reconfiguration.
64 //This map will be used during a communication.

66 for (const auto& pid_list : giver_query_map)
67 for (int64_t gid : pid_list.second)

74 //Since the giver query-map contains the ids of information to be obtained
75 //from the givers we can also build recvcounts and recvdispls from it
76 //since, during each scatter, this is the configuration in which the info
77 //will come from the givers. This will be used in a call to
78 //MPI_Alltoallv.

86 m_recvcounts[pid] = static_cast<int>(num_counts);
87 m_recvdispls[pid] = static_cast<int>(total_recvcounts);
88 total_recvcounts += num_counts;
89 }

93 //a taker_query_map. This is now a different perspective
94 //and this location now has a map of what data is needed
95 //by each dependent process.

103 // local indices
104 //First determine how much data we will
105 //eventually have to send. This allows us
106 //to reserve some space.
107 size_t amount_to_give = 0;
108 for (const auto& pid_list_pair : taker_query_map)
109 amount_to_give += pid_list_pair.second.size();

111 //======================================== Premap local ids
112 //These will be used during a communication.
113 m_takers_local_ids.clear();
114 m_takers_local_ids.reserve(amount_to_give);
115 for (const auto& pid_list_pair : taker_query_map)
116 {
117 const auto& list = pid_list_pair.second;
118 for (int64_t gid : list)
119 {
120 const uint64_t local_block_beg = m_locI_extents[m_location_id];
121 const uint64_t local_block_end = m_locI_extents[m_location_id+1];
122 if (not (gid >= local_block_beg and gid < local_block_end))
123 throw std::logic_error(fname + ": Error mapping index in "
124 "takers_query_map.");
125
126 m_takers_local_ids.push_back(gid - static_cast<int64_t>(local_block_beg));
127 }//for gid
128 }
129
130 //======================================== Build sendcounts and senddispls
131 //Since we now know what the structure will
132 //of sending the data we can compute m_sendcounts
133 //and m_senddispls, which will be used in a call
134 //to MPI_Alltoallv
135 m_sendcounts.assign(m_process_count, 0);
136 m_senddispls.assign(m_process_count, 0);
137
138 uint64_t total_sendcounts = 0;
139 for (const auto& pid_vec_pair : taker_query_map)
140 {
141 const int pid = pid_vec_pair.first;
142 const size_t num_counts = pid_vec_pair.second.size();
143 m_sendcounts[pid] = static_cast<int>(num_counts);
144 m_senddispls[pid] = static_cast<int>(total_sendcounts);
145 total_sendcounts += num_counts;
146 }

151int chi_math::VectorGhostCommunicator::FindOwnerPID(uint64_t global_id) const
152{
153 if (global_id >= m_globl_size)
154 throw std::logic_error(
155 "chi_math::VectorWithGhosts:FindOwnerPID global_id >= m_globl_size");
156 for (int locI=0; locI<static_cast<int>(m_process_count); ++locI)
157 if (global_id >= m_locI_extents[locI] and
158 global_id < m_locI_extents[locI+1])
159 return locI;
160 return int(-1);
161}

164void chi_math::VectorGhostCommunicator::
165 CommunicateGhostEntries(std::vector<double> &local_vector) const
166{
167 if (local_vector.size() != (m_local_size + m_ghost_indices.size()))
168 throw std::logic_error(
169 "chi_math::VectorWithGhosts::CommunicateGhostEntries: Vector size "
170 "mismatch.");
171
172 //======================================== Build data to be sent
173 const size_t amount_to_send = m_takers_local_ids.size();
174 std::vector<double> send_data;
175 send_data.reserve(amount_to_send);
176 for (const int64_t local_id : m_takers_local_ids)
177 send_data.push_back(local_vector[local_id]);
178
179 //======================================== Allocate receive buffer
180 const size_t amount_to_recv = m_ghost_indices.size();
181 std::vector<double> recv_data(amount_to_recv, 0.0);
182
183 //======================================== Communicate
184 MPI_Alltoallv(send_data.data(),
185 m_sendcounts.data(),
186 m_senddispls.data(),
187 MPI_DOUBLE,
188 recv_data.data(),
189 m_recvcounts.data(),
190 m_recvdispls.data(),
191 MPI_DOUBLE,
192 m_comm);
193
194 //======================================== Populate local vector with ghost
195 // data
196 for (size_t k=0; k<amount_to_recv; ++k)
197 {
198 const int64_t gid = m_ghost_indices[k];
199 const size_t rid = m_ghost_ids_to_recv_map.at(gid);
200 local_vector[m_local_size + k] = recv_data[rid];
201 }
202}
