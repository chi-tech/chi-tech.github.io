295 bool CHI_TEXTURE::LoadTGA(CST_TGA_TEXTURE* texture, char* filename) // Loads A TGA File Into Memory
296 {
297  GLubyte TGAheader[12] = { 0, 0, 2, 0, 0, 0, 0, 0, 0, 0, 0, 0 }; // Uncompressed TGA Header
298  GLubyte TGAcompare[12]; // Used To Compare TGA Header
299  GLubyte header[6]; // First 6 Useful Bytes From The Header
300  GLuint bytesPerPixel; // Holds Number Of Bytes Per Pixel Used In The TGA File
301  GLuint imageSize; // Used To Store The Image Size When Setting Aside Ram
302  GLuint temp; // Temporary Variable
303  FILE *file;
304 
305  file = fopen(filename, "rb"); // Open The TGA File
306  if (file == NULL || // Does File Even Exist?
307  fread(TGAcompare, 1, sizeof(TGAcompare), file) != sizeof(TGAcompare) || // Are There 12 Bytes To Read?
308  memcmp(TGAheader, TGAcompare, sizeof(TGAheader)) != 0 || // Does The Header Match What We Want?
309  fread(header, 1, sizeof(header), file) != sizeof(header)) // If So Read Next 6 Header Bytes
310  {
311  if (file == NULL) // Did The File Even Exist? *Added Jim Strong*
312  {
313  return false; // Return False
314  }
315  else
316  {
317  fclose(file); // If Anything Failed, Close The File
318  return false; // Return False
319  }
320  }
321 
322  texture->width = header[1] * 256 + header[0]; // Determine The TGA Width (highbyte*256+lowbyte)
323  texture->height = header[3] * 256 + header[2]; // Determine The TGA Height (highbyte*256+lowbyte)
324 
325  if (texture->width <= 0 || // Is The Width Less Than Or Equal To Zero
326  texture->height <= 0 || // Is The Height Less Than Or Equal To Zero
327  (header[4] != 24 && header[4] != 32)) // Is The TGA 24 or 32 Bit?
328  {
329  fclose(file); // If Anything Failed, Close The File
330  return false; // Return False
331  }
332 
333  texture->bpp = header[4]; // Grab The TGA's Bits Per Pixel (24 or 32)
334  bytesPerPixel = texture->bpp / 8; // Divide By 8 To Get The Bytes Per Pixel
335  imageSize = texture->width*texture->height*bytesPerPixel; // Calculate The Memory Required For The TGA Data
336 
337  texture->imageData = (GLubyte *)malloc(imageSize); // Reserve Memory To Hold The TGA Data
338 
339  if (texture->imageData == NULL || // Does The Storage Memory Exist?
340  fread(texture->imageData, 1, imageSize, file) != imageSize) // Does The Image Size Match The Memory Reserved?
341  {
342  if (texture->imageData != NULL) // Was Image Data Loaded
343  {
344  free(texture->imageData); // If So, Release The Image Data
345  }
346 
347  fclose(file); // Close The File
348 
349  return false; // Return False
350  }
351 
352 
353  for (GLuint i = 0; i<(GLuint)int(imageSize); i += bytesPerPixel) // Loop Through The Image Data
354  { // Swaps The 1st And 3rd Bytes ('R'ed and 'B'lue)
355  temp = texture->imageData[i]; // Temporarily Store The Value At Image Data 'i'
356  texture->imageData[i] = texture->imageData[i + 2]; // Set The 1st Byte To The Value Of The 3rd Byte
357  texture->imageData[i + 2] = temp; // Set The 3rd Byte To The Value In 'temp' (1st Byte Value)
358  }
359 
360  fclose(file); // Close The File
361  return true; // Texture Building Went Ok, Return True
362 }
