33 
34  double avg_weight = 0.0;
35 
36  //--------------------------------- Get Src and mappings
37  auto src = (chi_montecarlon::ResidualSource2*)sources.back();
38  int map = local_cell_pwl_dof_array_address[cell_local_ind];
39  int rmap = (*src->resid_ff->local_cell_dof_array_address)[cell_local_ind];

41 
42  //======================================== Contribute PWLD
43  //--------------------------------- Develop segments
44  segment_lengths.clear();
45  segment_lengths.push_back(tracklength);
46  chi_mesh::PopulateRaySegmentLengths(grid, cell,
47  segment_lengths,
48  prtcl.pos, pf,prtcl.dir);
49 
50 
51  //--------------------------------- Lambda getting phi
52  auto phi_s = [](std::vector<double>& N_in,
53  int dofs, int rmap,
54  chi_montecarlon::ResidualSource2* rsrc,
55  int egrp)
56  {
57  double phi = 0.0;
58  for (int dof=0; dof<dofs; dof++)
59  {
60  int ir = rmap +
61  dof*rsrc->resid_ff->num_grps*rsrc->resid_ff->num_moms +
62  rsrc->resid_ff->num_grps*0 +
63  egrp;
64  phi += (*rsrc->resid_ff->field_vector_local)[ir]*N_in[dof];
65  }//for dof
66 
67  return phi;
68  };
69 
70  //--------------------------------- Lambda getting gradphi
71  auto gradphi_s = [](std::vector<chi_mesh::Vector>& Grad_in,
72  int dofs, int rmap,
73  chi_montecarlon::ResidualSource2* rsrc,
74  int egrp)
75  {
76  chi_mesh::Vector gradphi;
77  for (int dof=0; dof<dofs; dof++)
78  {
79  int ir = rmap +
80  dof*rsrc->resid_ff->num_grps*rsrc->resid_ff->num_moms +
81  rsrc->resid_ff->num_grps*0 +
82  egrp;
83  gradphi = gradphi + (Grad_in[dof]*(*rsrc->resid_ff->field_vector_local)[ir]);
84  }//for dof
85 
86  return gradphi;
87  };
88 
89  //--------------------------------- Lambda computing q
90  auto q_s = [](double siga,double phi, double Q,
91  chi_mesh::Vector& omega,chi_mesh::Vector& nabla_phi)
92  {
93  double retval = Q - siga*phi - omega.Dot(nabla_phi);
94 
95  return retval;
96  };
97 
98  //================================= Loop over segments
99  chi_mesh::Vector p_i = prtcl.pos;
100  chi_mesh::Vector p_f = prtcl.pos;
101 
102  //========================== Compute q_i
103  cell_pwl_view->ShapeValues(p_i,N);
104  cell_pwl_view->GradShapeValues(p_i,Grad);
105  double phi_i = phi_s(N,cell_pwl_view->dofs,rmap,src,prtcl.egrp);
106  auto gradphi_i = gradphi_s(Grad,cell_pwl_view->dofs,rmap,src,prtcl.egrp);
107 
108  double q_i = q_s(siga,phi_i,0.0,prtcl.dir,gradphi_i);
109  double q_f = q_i;
110 
111  for (auto seg_len : segment_lengths) //segment_length
112  {
113 // int n_s = std::max(std::ceil(5*seg_len*sigt),4.0);
114  int n_s = 2;
115  double s_L = seg_len/n_s;
116 
117  for (int subdiv=0; subdiv<n_s; subdiv++)
118  {
119  p_i = p_f; //Only p_f gets changed below
120  q_i = q_f;
121 
122  //========================== Compute q_f
123  p_f = p_i + prtcl.dir*s_L;
124 
125  cell_pwl_view->ShapeValues(p_f,N);
126  cell_pwl_view->GradShapeValues(p_f,Grad);
127  double phi_f = phi_s(N,cell_pwl_view->dofs,rmap,src,prtcl.egrp);
128  auto gradphi_f = gradphi_s(Grad,cell_pwl_view->dofs,rmap,src,prtcl.egrp);
129 
130  q_f = q_s(siga,phi_f,0.0,prtcl.dir,gradphi_f);
131 
132  //========================== Computing a and b
133  double a = q_i;
134  double b = (q_f-q_i)/s_L;
135 
136  //========================== Computing Iq
137  double Iq = (a/sigt)*(exp(sigt*s_L)-1.0);
138  Iq += (b/sigt/sigt)*(sigt*s_L - 1.0)*exp(sigt*s_L);
139  Iq -= (b/sigt/sigt)*(0.0 - 1.0)*1.0;
140 
141  //========================== Computing w_f
142  double w_f = exp(-sigt*s_L)*prtcl.w + exp(-sigt*s_L)*Iq;
143 
144  double w_avg = 0.5*(w_f + prtcl.w);
145  prtcl.w = w_f;
146  avg_weight += s_L*w_avg;
147 
148  //========================== Contribute to tally
149 
150  //Get shape values at center of contrib
151  auto p_c = (p_f+p_i)*0.5;
152  cell_pwl_view->ShapeValues(p_c,N);
153 
154  for (int dof=0; dof<cell_pwl_view->dofs; dof++)
155  {
156  int ir = map + dof*num_grps*num_moms + num_grps*0 + prtcl.egrp;
157  double pwl_tally_contrib = s_L*w_avg*N[dof];
158 
159  phi_pwl_tally[ir] += pwl_tally_contrib;
160  phi_pwl_tally_sqr[ir] += pwl_tally_contrib*pwl_tally_contrib;
161  }//for dof
162 
163  }//for subdivision
164 
165  }//for segment_length
166  avg_weight/=tracklength;
167 
168  //======================================== Contribute avg tally
169  int ir = cell_local_ind*num_grps + prtcl.egrp;
170 
171  double tally_contrib = tracklength*avg_weight;
172 
173  phi_tally[ir] += tally_contrib;
174  phi_tally_sqr[ir] += tally_contrib*tally_contrib;

183  }
184 
185  //======================================== Adjust weight for face jump
186  cell_pwl_view->ShapeValues(pf,N);
187  double phi_neg = phi_s(N,cell_pwl_view->dofs,rmap,src,prtcl.egrp);
188 
189  double phi_pos = 0.0;
190  int neighbor = ray_dest_info.destination_face_neighbor;
191  if (neighbor>0)
192  {
193  auto adj_cell = grid->cells[neighbor];
194  int adj_cell_local_ind = adj_cell->cell_local_id;
195  auto adj_cell_pwl_view = pwl_discretization->MapFeView(neighbor);
196 
197  adj_cell_pwl_view->ShapeValues(pf,N);
198  int armap =
199  (*src->resid_ff->local_cell_dof_array_address)[adj_cell_local_ind];
200  phi_pos = phi_s(N,adj_cell_pwl_view->dofs,armap,src,prtcl.egrp);
201 
202  prtcl.w += (phi_neg - phi_pos);
203  }
