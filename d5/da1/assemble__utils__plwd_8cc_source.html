116 /**Still searching for a reference for this.
117  *
118  * For Polygons:
119  * Defined from paper \n
120  * Turcksin B, Ragusa J, "Discontinuous diffusion synthetic acceleration
121  * for S_n transport on 2D arbitrary polygonal meshes", Journal of
122  * Computational Physics 274, pg 356-369, 2014.\n
123  * \n
124  * Nv = Number of vertices. If Nv <= 4 then the perimeter parameter
125  * should be replaced by edge length.*/
126 double chi_diffusion::Solver::HPerpendicular(chi_mesh::Cell* cell,
127  CellFEView* fe_view,
128  int f)
129 {
130  double hp = 1.0;
131 
132  int Nf = cell->faces.size();
133  int Nv = cell->vertex_ids.size();
134 
135  //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% SLAB
136  if (cell->Type() == chi_mesh::CellType::SLAB)
137  {
138  auto slab_fe_view = (SlabFEView*)fe_view;
139  hp = slab_fe_view->h/2.0;
140  }
141  //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% POLYGON
142  else if (cell->Type() == chi_mesh::CellType::POLYGON)
143  {
144  Nv = 4;
145  chi_mesh::CellFace& face = cell->faces[f];
146 
147  int v0i = face.vertex_ids[0];
148  int v1i = face.vertex_ids[1];
149 
150  chi_mesh::Vertex& v0 = *grid->vertices[v0i];
151  chi_mesh::Vertex& v1 = *grid->vertices[v1i];
152 
153  double perimeter = (v1 - v0).Norm();
154 
155  double area = 0.0;
156  for (int i=0; i<fe_view->dofs; i++)
157  area += fe_view->IntV_shapeI[i];
158 
159  if (Nv == 3)
160  hp = 2*area/perimeter;
161  else if (Nv == 4)
162  hp = area/perimeter;
163  else //Nv > 4
164  {
165  if (Nv%2 == 0)
166  hp = 4*area/perimeter;
167  else
168  {
169  hp = 2*area/perimeter;
170  hp += sqrt(2*area / Nv*sin(2*M_PI/Nv));
171  }
172  }
173  }
174  //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% POLYHEDRON
175  else if (cell->Type() == chi_mesh::CellType::POLYHEDRON)
176  {
177  double volume = 0.0;
178  for (int i=0; i<fe_view->dofs; i++)
179  volume += fe_view->IntV_shapeI[i];
180 
181  double area = 0.0;
182  for (int fr=0; fr<Nf; fr++)
183  for (int i=0; i<Nv; i++)
184  area += fe_view->IntS_shapeI[i][fr];
185 
186  if (Nf == 4) //Tet
187  hp = 3*volume/area;
188  else if (Nf == 6 && Nv == 8) //Hex
189  hp = volume/area;
190  else //Polyhedron
191  hp = 6*volume/area;
192  }//Polyhedron
193  else
194  {
195  chi_log.Log(LOG_ALLERROR)
196  << "Unsupported cell type in call to HPerpendicular";
197  exit(EXIT_FAILURE);
198  }
199 
200 
201  return hp;
202 }
