67
68 std::vector<uint64_t> snapped_vertex_ids;
69 std::set<uint64_t> cell_ids_modified;

73 {
74 for (const auto& face : cell.faces_)

76 if (boundary_nodes_only_ and face.has_neighbor_) continue;

78 bool matches = true;
