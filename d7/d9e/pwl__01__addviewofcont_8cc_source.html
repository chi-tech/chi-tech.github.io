114/**Adds a PWL Finite Element for each cell of the local problem.*/
115void SpatialDiscretization_PWLD::PreComputeCellSDValues()
116{
117 size_t num_local_cells = ref_grid->local_cells.size();
118
119 //================================================== Create empty view
120 // for each cell
121 {
122 using namespace chi_math::finite_element;
123 if (setup_flags & SetupFlags::COMPUTE_CELL_MAPPINGS)
124 {
125 if (!mapping_initialized)
126 {
127 chi_log.Log() << chi_program_timer.GetTimeString()
128 << " Computing cell views";
129 for (const auto& cell : ref_grid->local_cells)
130 cell_mappings.push_back(MakeCellMappingFE(cell));
131
132 mapping_initialized = true;
133 }
134 }
135 }
136 MPI_Barrier(MPI_COMM_WORLD);
137
138 //============================================= Unit integrals
139 {
140 using namespace chi_math::finite_element;
141 if (setup_flags & SetupFlags::COMPUTE_UNIT_INTEGRALS)
142 {
143 if (not integral_data_initialized)
144 {
145 chi_log.Log() << chi_program_timer.GetTimeString()
146 << " Computing unit integrals.";
147 fe_unit_integrals.reserve(num_local_cells);
148 for (size_t lc=0; lc<num_local_cells; ++lc)
149 {
150 UIData ui_data;
151
152 auto cell_fe_view = GetCellMappingFE(lc);
153 cell_fe_view->ComputeUnitIntegrals(ui_data);
154
155 fe_unit_integrals.push_back(std::move(ui_data));
156 }
157
158 integral_data_initialized = true;
159 }
160 }//if compute unit intgrls
161 }
162 MPI_Barrier(MPI_COMM_WORLD);
163
164
165 //============================================= Quadrature data
166 {
167 using namespace chi_math::finite_element;
168 if (setup_flags & SetupFlags::COMPUTE_QP_DATA)
169 {
170 if (not qp_data_initialized)
171 {
172 chi_log.Log() << chi_program_timer.GetTimeString()
173 << " Computing quadrature data.";
174 fe_vol_qp_data.reserve(num_local_cells);
175 fe_srf_qp_data.reserve(num_local_cells);
176 for (size_t lc=0; lc<num_local_cells; ++lc)
177 {
178 fe_vol_qp_data.emplace_back();
179 fe_srf_qp_data.emplace_back();
180
181 auto cell_fe_view = GetCellMappingFE(lc);
182 cell_fe_view->InitializeAllQuadraturePointData(fe_vol_qp_data.back(),
183 fe_srf_qp_data.back());
184 }
185
186 qp_data_initialized = true;
187 }
188 }//if init qp data
189 }

193/**Adds a PWL Finite Element for each cell of the neighboring cells.*/
194void SpatialDiscretization_PWLD::PreComputeNeighborCellSDValues()
195{
196 //================================================== Populate cell fe views

209 neighbor_cell_fe_views.insert(

303/**Maps a neigboring cell from a global cell index. The spatial discretizations
304 * maintains a non-ghost version of all neighboring cells.*/
305chi_mesh::Cell& SpatialDiscretization_PWLD::
306 GetNeighborCell(uint64_t cell_glob_index)
307{
308 //=================================== First check locally
309 if (ref_grid->IsCellLocal(cell_glob_index))
310 return ref_grid->cells[cell_glob_index];
311
312 //=================================== Now check neighbor cells
313 auto neighbor_location = neighbor_cells.find(cell_glob_index);
314
315 if (neighbor_location != neighbor_cells.end())
316 return *neighbor_cells.at(cell_glob_index);
317 else
318 throw std::logic_error(std::string(__FUNCTION__) +
319 " Mapping of neighbor cell failed.");
320}

323/**Maps a neigboring cell's fe view from a global cell index.*/
324std::shared_ptr<CellMappingFE_PWL> SpatialDiscretization_PWLD::
325 GetNeighborCellMappingFE(uint64_t cell_glob_index)
326{
327 //=================================== First check locally
328 if (ref_grid->IsCellLocal(cell_glob_index))
329 {
330 auto& neighbor_cell = ref_grid->cells[cell_glob_index];
331 return GetCellMappingFE(neighbor_cell.local_id);
332 }
333
334 //=================================== Now check neighbor cells
335 if (nb_mapping_initialized)
336 {
337 auto neighbor_location = neighbor_cell_fe_views.find(cell_glob_index);
338
339 if (neighbor_location != neighbor_cell_fe_views.end())
340 return neighbor_cell_fe_views.at(cell_glob_index);
341 else
342 throw std::logic_error(std::string(__FUNCTION__) +
343 " Mapping of neighbor cell failed.");
344 }
345 else
346 {
347 return MakeCellMappingFE(GetNeighborCell(cell_glob_index));
348 }
