253 /**Maps a neigboring cell from a global cell index. The spatial discretizations
254  * maintains a non-ghost version of all neighboring cells.*/
255 chi_mesh::Cell& SpatialDiscretization_PWLD::
256  GetNeighborCell(int cell_glob_index)
257 {
258  //=================================== First check locally
259  if (ref_grid->IsCellLocal(cell_glob_index))
260  return ref_grid->cells[cell_glob_index];
261 
262  //=================================== Now check neighbor cells
263  auto neighbor_location = neighbor_cells.find(cell_glob_index);
264

267  else
268  throw std::logic_error(std::string(__FUNCTION__) +
269  " Mapping of neighbor cell failed.");
