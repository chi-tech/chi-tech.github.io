38/**This routine groups angle-indices to groups sharing the same sweep
39 * ordering. It also takes geometry into account.*/
40std::pair<UniqueSOGroupings, DirIDToSOMap> LBSDiscreteOrdinatesSolver::
41 AssociateSOsAndDirections(const chi_mesh::MeshContinuum &grid,
42 const chi_math::AngularQuadrature& quadrature,
43 const AngleAggregationType agg_type,
44 const lbs::GeometryType lbs_geo_type)
45{
46 const std::string fname = __FUNCTION__;
47
48 //================================================== Checks
49 LogicCheck(quadrature.omegas_.empty(),
50 ": Quadrature with no omegas cannot be used.")
51 LogicCheck(quadrature.weights_.empty(),
52 ": Quadrature with no weights cannot be used.")
53
54 //================================================== Build groupings
55 UniqueSOGroupings unq_so_grps;
56 switch (agg_type)
57 {
58 //=========================================== Single
59 // The easiest aggregation type. Every direction
60 // either has/is assumed to have a unique sweep
61 // ordering. Hence there is only group holding ALL
62 // the direction indices.
63 case AngleAggregationType::SINGLE:
64 {
65 const size_t num_dirs = quadrature.omegas_.size();
66 for (size_t n=0; n<num_dirs; ++n)
67 unq_so_grps.push_back({n});
68 break;
69 }//case agg_type SINGLE
70
71 //=========================================== Polar
72 // The following conditions allow for polar
73 // angle aggregation.
74 case AngleAggregationType::POLAR:
75 {
76 //Check geometry types
77 const auto grid_attribs = grid.Attributes();
78 if (not(grid_attribs & chi_mesh::ORTHOGONAL or
79 grid_attribs & chi_mesh::DIMENSION_2 or
80 grid_attribs & chi_mesh::EXTRUDED))
81 throw std::logic_error(POLAR_ILLEGAL_GEOTYPE);
82
83 //Check quadrature type
84 const auto quad_type = quadrature.type_;
85 if (quad_type != chi_math::AngularQuadratureType::ProductQuadrature)
86 throw std::logic_error(POLAR_ONLY_PRODUCT);
87
88 //Process Product Quadrature
89 try
90 {
91 typedef chi_math::ProductQuadrature ProdQuadType;
92 const auto& product_quad = dynamic_cast<const ProdQuadType&>(quadrature);
93
94 const auto num_azi = product_quad.azimu_ang_.size();
95 const auto num_pol = product_quad.polar_ang_.size();
96
97 //Make two separate list of polar angles
98 //One upward-pointing and one downward
99 std::vector<size_t> upward_polar_ids;
100 std::vector<size_t> dnward_polar_ids;
101 for (size_t p=0; p<num_pol; ++p)
102 if (product_quad.polar_ang_[p] > M_PI_2)
103 upward_polar_ids.push_back(p);
104 else
105 dnward_polar_ids.push_back(p);
106
107 //Define lambda working for both upward and dnward polar-ids
108 /**Lambda to convert indices and push it onto unq_so_grps.*/
109 auto MapPolarAndAzimuthalIDs = [&product_quad, &unq_so_grps](
110 const DirIDs& polar_ids,const size_t azimuthal_id)
111 {
112 DirIDs dir_ids;
113 dir_ids.reserve(polar_ids.size());
114 for (const size_t p : polar_ids)
115 dir_ids.push_back(product_quad.GetAngleNum(p, azimuthal_id));
116 unq_so_grps.push_back(std::move(dir_ids));
117 };
118
119 //Stack id's for all azimuthal angles
120 for (size_t a=0; a<num_azi; ++a)
121 {
122 if (not upward_polar_ids.empty())
123 MapPolarAndAzimuthalIDs(upward_polar_ids, a);
124 if (not dnward_polar_ids.empty())
125 MapPolarAndAzimuthalIDs(dnward_polar_ids, a);
126 }//for azi-id a
127
128 }//try product quadrature
129 catch (const std::bad_cast& bc)
130 {
131 throw std::runtime_error(PRODUCT_QUAD_CASTING_FAILED);
132 }
133
134 break;
135 }//case agg_type POLAR
136
137 //====================================== Azimuthal
138 case AngleAggregationType::AZIMUTHAL:
139 {
140 //Check geometry types
141 if (not (lbs_geo_type == GeometryType::ONED_SPHERICAL or
142 lbs_geo_type == GeometryType::TWOD_CYLINDRICAL))
143 throw std::logic_error(AZIMUTHAL_ILLEGAL_GEOTYPE);
144
145 //Check quadrature type
146 const auto quad_type = quadrature.type_;
147 if (quad_type != chi_math::AngularQuadratureType::ProductQuadrature)
148 throw std::logic_error(AZIMUTHAL_ONLY_PRODUCT);
149
150 //Process Product Quadrature
151 try
152 {
153 typedef chi_math::ProductQuadrature ProdQuadType;
154 const auto& product_quad = dynamic_cast<const ProdQuadType&>(quadrature);
155
156 for (const auto& dir_set : product_quad.GetDirectionMap())
157 {
158 std::vector<unsigned int> group1;
159 std::vector<unsigned int> group2;
160 for (const auto& dir_id : dir_set.second)
161 if (quadrature.abscissae_[dir_id].phi > M_PI_2)
162 group1.push_back(dir_id);
163 else
164 group2.push_back(dir_id);
165
166 DirIDs group1_ids(group1.begin(),group1.end());
167 DirIDs group2_ids(group2.begin(),group2.end());
168
169 unq_so_grps.push_back(std::move(group1_ids));
170 unq_so_grps.push_back(std::move(group2_ids));
171 }
172 }//try product quadrature
173 catch (const std::bad_cast& bc)
174 {
175 throw std::runtime_error(PRODUCT_QUAD_CASTING_FAILED);
176 }
177
178 break;
179 }
180 default:
181 throw std::invalid_argument(fname + ": Called with UNDEFINED angle "
182 "aggregation type.");
183 }//switch angle aggregation type
184
185 //================================================== Map directions to sweep
186 // orderings
187 DirIDToSOMap dir_id_to_so_map;
188 {
189 size_t so_grouping_id = 0;
190 for (const auto& so_grouping : unq_so_grps)
191 {
192 for (const size_t dir_id: so_grouping)
193 dir_id_to_so_map[dir_id] = so_grouping_id;
194
195 ++so_grouping_id;
196 }//for so_grouping
197 }//map scope
198
199 return {unq_so_grps, dir_id_to_so_map};
200}
