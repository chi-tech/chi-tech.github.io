23/**Solves a groupset using GMRES.*/
24bool lbs::SteadySolver::GMRES(LBSGroupset& groupset,
25 MainSweepScheduler& sweep_scheduler,
26 SourceFlags lhs_src_scope,
27 SourceFlags rhs_src_scope,
28 bool log_info /* = true*/)
29{
30 constexpr bool WITH_DELAYED_PSI = true;
31 if (log_info)
32 {
33 chi_log.Log(LOG_0)
34 << "\n\n";
35 chi_log.Log(LOG_0)
36 << "********** Solving groupset " << groupset.id
37 << " with GMRES.\n\n";
38 chi_log.Log(LOG_0)
39 << "Quadrature number of angles: "
40 << groupset.quadrature->abscissae.size() << "\n"
41 << "Groups " << groupset.groups.front().id << " "
42 << groupset.groups.back().id << "\n\n";
43 }
44
45 //================================================== Get groupset dof sizes
46 size_t groupset_numgrps = groupset.groups.size();
47 auto num_delayed_ang_DOFs = groupset.angle_agg.GetNumDelayedAngularDOFs();
48 size_t local_size = local_node_count * num_moments * groupset_numgrps +
49 num_delayed_ang_DOFs.first;
50 size_t globl_size = glob_node_count * num_moments * groupset_numgrps +
51 num_delayed_ang_DOFs.second;
52
53 if (log_info)
54 chi_log.Log(LOG_0)
55 << "Number of lagged angular unknowns: " << num_delayed_ang_DOFs.second;
56
57 //================================================== Create PETSc vectors
58 phi_new = chi_math::PETScUtils::CreateVector(static_cast<int64_t>(local_size),
59 static_cast<int64_t>(globl_size));
60 Vec x_temp;
61 VecSet(phi_new,0.0);
62 VecDuplicate(phi_new,&phi_old);
63 VecDuplicate(phi_new,&q_fixed);
64 VecDuplicate(phi_new,&x_temp);
65
66 //=================================================== Create Data context
67 // available inside
68 // Action
69 KSPDataContext data_context(*this, groupset, x_temp,
70 sweep_scheduler, lhs_src_scope);
71
72 //=================================================== Create the matrix-shell
73 Mat A;
74 MatCreateShell(PETSC_COMM_WORLD,static_cast<int64_t>(local_size),
75 static_cast<int64_t>(local_size),
76 static_cast<int64_t>(globl_size),
77 static_cast<int64_t>(globl_size),
78 &data_context,&A);
79
80 //================================================== Set the action-operator
81 MatShellSetOperation(A, MATOP_MULT, (void (*)()) LBSMatrixAction_Ax);
82
83 //================================================== Create Krylov Solver
84 KSP ksp;
85 KSPCreate(PETSC_COMM_WORLD, &ksp);
86 KSPSetType(ksp,KSPGMRES);
87 KSPSetOperators(ksp,A,A);
88
89 KSPSetTolerances(ksp,1.e-50,
90 groupset.residual_tolerance,1.0e50,
91 groupset.max_iterations);
92 KSPGMRESSetRestart(ksp, groupset.gmres_restart_intvl);
93 KSPSetApplicationContext(ksp, &data_context);
94 KSPSetConvergenceTest(ksp, &KSPConvergenceTestNPT, nullptr, nullptr);
95 KSPSetInitialGuessNonzero(ksp, PETSC_TRUE);
96 KSPSetUp(ksp);
97
98 //================================================== Compute b
99 if (log_info)
100 {
101 chi_log.Log(LOG_0) << chi_program_timer.GetTimeString() << " Computing b";
102 }
103
104 //SetSource for RHS
105 auto init_q_moments_local = q_moments_local;
106 SetSource(groupset, q_moments_local, rhs_src_scope);
107
108 //Tool the sweep chunk
109 auto& sweep_chunk = sweep_scheduler.sweep_chunk;
110 bool use_surface_source_flag = (rhs_src_scope & APPLY_MATERIAL_SOURCE) and
111 (not options.use_src_moments);
112 sweep_chunk.SetSurfaceSourceActiveFlag(use_surface_source_flag);
113 sweep_chunk.ZeroIncomingDelayedPsi();
114
115 //Sweep
116 sweep_chunk.ZeroFluxDataStructures();
117 sweep_scheduler.Sweep();
118
119 //=================================================== Apply DSA
120 if (groupset.apply_wgdsa)
121 {
122 AssembleWGDSADeltaPhiVector(groupset, phi_old_local.data(), phi_new_local.data());
123 ((chi_diffusion::Solver*)groupset.wgdsa_solver)->ExecuteS(true,false);
124 DisAssembleWGDSADeltaPhiVector(groupset, phi_new_local.data());
125 }
126 if (groupset.apply_tgdsa)
127 {
128 AssembleTGDSADeltaPhiVector(groupset, phi_old_local.data(), phi_new_local.data());
129 ((chi_diffusion::Solver*)groupset.tgdsa_solver)->ExecuteS(true,false);
130 DisAssembleTGDSADeltaPhiVector(groupset, phi_new_local.data());
131 }
132
133 //=================================================== Assemble vectors
134 SetPETScVecFromSTLvector(groupset, q_fixed, phi_new_local, WITH_DELAYED_PSI);
135 SetPETScVecFromSTLvector(groupset, phi_old, phi_old_local, WITH_DELAYED_PSI);
136
137 //=================================================== Retool for GMRES
138 sweep_chunk.SetSurfaceSourceActiveFlag(lhs_src_scope & APPLY_MATERIAL_SOURCE);
139 sweep_chunk.SetDestinationPhi(phi_new_local);
140
141 double phi_old_norm=0.0;
142 VecNorm(phi_old,NORM_2,&phi_old_norm);
143
144 if (phi_old_norm > 1.0e-10)
145 {
146 VecCopy(phi_old,phi_new);
147 if (log_info)
148 chi_log.Log(LOG_0) << "Using phi_old as initial guess.";
149 }
150
151
152 //**************** CALL GMRES SOLVE ******************
153 if (log_info)
154 {
155 chi_log.Log(LOG_0)
156 << chi_program_timer.GetTimeString() << " Starting iterations";
157 }
158 KSPSolve(ksp,q_fixed,phi_new);
159 //****************************************************
160
161 KSPConvergedReason reason;
162 KSPGetConvergedReason(ksp,&reason);
163 if (reason != KSP_CONVERGED_RTOL)
164 chi_log.Log(LOG_0WARNING)
165 << "GMRES solver failed. "
166 << "Reason: " << chi_physics::GetPETScConvergedReasonstring(reason);
167
168 SetSTLvectorFromPETScVec(groupset, phi_new, phi_new_local, WITH_DELAYED_PSI);
169 SetSTLvectorFromPETScVec(groupset, phi_new, phi_old_local, WITH_DELAYED_PSI);
170
171 //================================================== Perform final sweep
172 // with converged phi and
173 // delayed psi dofs
174 ZeroOutflowBalanceVars(groupset);
175
176 sweep_chunk.SetDestinationPhi(phi_new_local);
177 sweep_chunk.SetSurfaceSourceActiveFlag(rhs_src_scope & APPLY_MATERIAL_SOURCE);
178
179 q_moments_local = init_q_moments_local;
180 SetSource(groupset, q_moments_local, lhs_src_scope | rhs_src_scope);
181
182 sweep_chunk.ZeroDestinationPhi();
183 sweep_scheduler.Sweep();
184
185 ScopedCopySTLvectors(groupset, phi_new_local, phi_old_local);
186
187 //==================================================== Clean up
188 KSPDestroy(&ksp);
189 VecDestroy(&phi_new);
190 VecDestroy(&phi_old);
191 VecDestroy(&q_fixed);
192 VecDestroy(&x_temp);
193 MatDestroy(&A);
194
195 //==================================================== Print solution info
196 {
197 double sweep_time = sweep_scheduler.GetAverageSweepTime();
198 double chunk_overhead_ratio = 1.0 - sweep_scheduler.GetAngleSetTimings()[2];
199 double source_time=
200 chi_log.ProcessEvent(source_event_tag,
201 ChiLog::EventOperation::AVERAGE_DURATION);
202 size_t num_angles = groupset.quadrature->abscissae.size();
203 size_t num_unknowns = glob_node_count *
204 num_angles *
205 groupset.groups.size();
206
207 if (log_info)
208 {
209 chi_log.Log(LOG_0)
210 << "\n\n";
211 chi_log.Log(LOG_0)
212 << " Set Src Time/sweep (s): "
213 << source_time;
214 chi_log.Log(LOG_0)
215 << " Average sweep time (s): "
216 << sweep_time;
217 chi_log.Log(LOG_0)
218 << " Chunk-Overhead-Ratio : "
219 << chunk_overhead_ratio;
220 chi_log.Log(LOG_0)
221 << " Sweep Time/Unknown (ns): "
222 << sweep_time*1.0e9*chi_mpi.process_count/
223 static_cast<double>(num_unknowns);
224 chi_log.Log(LOG_0)
225 << " Number of unknowns per sweep: " << num_unknowns;
226 chi_log.Log(LOG_0)
227 << "\n\n";
228
229 std::string sweep_log_file_name =
230 std::string("GS_") + std::to_string(groupset.id) +
231 std::string("_SweepLog_") + std::to_string(chi_mpi.location_id) +
232 std::string(".log");
233 groupset.PrintSweepInfoFile(sweep_scheduler.sweep_event_tag,
234 sweep_log_file_name);
235 }
236 }
237
238 return reason == KSP_CONVERGED_RTOL;
239}
