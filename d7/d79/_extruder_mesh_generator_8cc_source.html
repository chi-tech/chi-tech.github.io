69ExtruderMeshGenerator::ExtruderMeshGenerator(const chi::InputParameters& params)
70 : MeshGenerator(params),
71 top_boundary_name_(params.GetParamValue<std::string>("top_boundary_name")),
72 bottom_boundary_name_(
73 params.GetParamValue<std::string>("bottom_boundary_name"))
74{
75 const auto& layers_param = params.GetParam("layers");
76
77 double current_z_level = 0.0;
78 for (const auto& layer_block : layers_param)
79 {
80 auto valid_params = ExtrusionLayer::GetInputParameters();
81 valid_params.SetErrorOriginScope("ExtruderMeshGenerator:\"layers\"");
82 valid_params.AssignParameters(layer_block);
83
84 const int h_and_z_config =
85 int(layer_block.Has("h")) + int(layer_block.Has("z"));
86
87 if (h_and_z_config != 1)
88 ChiInvalidArgument("For an ExtrusionLayer either \"h\" or \"z\" must"
89 "be specified and also not both.");
90
91 auto n = valid_params.GetParamValue<uint32_t>("n");
92 double h;
93 if (layer_block.Has("h")) h = valid_params.GetParamValue<double>("h");
94 else
95 {
96 double z = valid_params.GetParamValue<double>("z");
97 ChiInvalidArgumentIf(z <= current_z_level,
98 "For extrusion layers, the \"z\" coordinates must "
99 "be monotonically increasing.");
100 h = z - current_z_level;
101 }
102
103 current_z_level += h;
104
105 layers_.push_back(ExtrusionLayer{h, n});
106
107 Chi::log.Log0Verbose1()
108 << "Layer " << layer_block.Name() << " height=" << h
109 << " num_sub_layers=" << n << " top-z=" << current_z_level;
110 } // layer_block in layers_param
111}

114std::unique_ptr<UnpartitionedMesh>
115ExtruderMeshGenerator::GenerateUnpartitionedMesh(
116 std::unique_ptr<UnpartitionedMesh> input_umesh)
117{
118 Chi::log.Log0Verbose1() << "ExtruderMeshGenerator::GenerateUnpartitionedMesh";
119 const chi_mesh::Vector3 khat(0.0, 0.0, 1.0);
120
121 ChiInvalidArgumentIf(
122 not(input_umesh->GetMeshAttributes() & DIMENSION_2),
123 "Input mesh is not 2D. A 2D mesh is required for extrusion");
124
125 const auto& template_vertices = input_umesh->GetVertices();
126 const auto& template_cells = input_umesh->GetRawCells();
127
128 const size_t num_template_vertices = template_vertices.size();
129
130 ChiLogicalErrorIf(template_vertices.empty(), "Input mesh has no vertices.");
131 ChiLogicalErrorIf(template_cells.empty(), "Input mesh has no cells.");
132
133 //============================================= Check cells
134 for (const auto& template_cell_ptr : template_cells)
135 {
136 const auto& template_cell = *template_cell_ptr;
137 if (template_cell.type != chi_mesh::CellType::POLYGON)
138 throw std::logic_error("ExtruderMeshGenerator: "
139 "Template cell error. Not of base type POLYGON");
140
141 // Check cell not inverted
142 const auto& v0 = template_cell.centroid;
143 const auto& v1 = template_vertices[template_cell.vertex_ids[0]];
144 const auto& v2 = template_vertices[template_cell.vertex_ids[1]];
145
146 auto v01 = v1 - v0;
147 auto v02 = v2 - v0;
148
149 if (v01.Cross(v02).Dot(khat) < 0.0)
150 throw std::logic_error("Extruder attempting to extrude a template"
151 " cell with a normal pointing downward. This"
152 " causes erratic behavior and needs to be"
153 " corrected.");
154 }
155
156 auto umesh = std::make_unique<UnpartitionedMesh>();
157
158 //============================================= Update boundary maps
159 auto& umesh_bndry_map = umesh->GetMeshOptions().boundary_id_map;
160 umesh_bndry_map = input_umesh->GetMeshOptions().boundary_id_map;
161
162 const uint64_t zmax_bndry_id = umesh->MakeBoundaryID(top_boundary_name_);
163 umesh_bndry_map[zmax_bndry_id] = top_boundary_name_;
164 const uint64_t zmin_bndry_id = umesh->MakeBoundaryID(bottom_boundary_name_);
165 umesh_bndry_map[zmin_bndry_id] = bottom_boundary_name_;
166
167 //============================================= Setup z-levels
168 double current_z = 0.0;
169 std::vector<double> z_levels = {current_z};
170 for (const auto& layer : layers_)
171 {
172 const double dz = layer.height_ / layer.num_sub_layers_;
173 for (uint32_t i = 0; i < layer.num_sub_layers_; ++i)
174 z_levels.push_back(current_z += dz);
175 }
176
177 //============================================= Build vertices
178 typedef chi_mesh::Vector3 Vec3;
179 auto& extruded_vertices = umesh->GetVertices();
180 for (const double z_level : z_levels)
181 for (const auto& template_vertex : template_vertices)
182 extruded_vertices.push_back(
183 Vec3(template_vertex.x, template_vertex.y, z_level));
184
185 //============================================= Build cells
186 size_t k = 0;
187 for (const auto& layer : layers_)
188 {
189 for (uint32_t n = 0; n < layer.num_sub_layers_; ++n)
190 {
191 for (const auto& template_cell : template_cells)
192 {
193 //================================== Determine cell sub-type
194 CellType extruded_subtype;
195 // clang-format off
196 switch (template_cell->sub_type)
197 {
198 case CellType::TRIANGLE: extruded_subtype = CellType::WEDGE; break;
199 case CellType::QUADRILATERAL: extruded_subtype = CellType::HEXAHEDRON; break;
200 default: extruded_subtype = CellType::POLYHEDRON;
201 }
202 // clang-format on
203
204 //================================== Create new cell
205 auto new_cell_ptr = new UnpartitionedMesh::LightWeightCell(
206 CellType::POLYHEDRON, extruded_subtype);
207 auto& new_cell = *new_cell_ptr;
208
209 new_cell.material_id = template_cell->material_id;
210
211 //================================== Build vertices
212 const size_t tc_num_verts = template_cell->vertex_ids.size();
213 new_cell.vertex_ids.reserve(2 * tc_num_verts);
214 for (const auto tc_vid : template_cell->vertex_ids)
215 new_cell.vertex_ids.push_back(tc_vid + k * num_template_vertices);
216 for (const auto tc_vid : template_cell->vertex_ids)
217 new_cell.vertex_ids.push_back(tc_vid +
218 (k + 1) * num_template_vertices);
219
220 //================================== Create side faces
221 for (const auto& tc_face : template_cell->faces)
222 {
223 UnpartitionedMesh::LightWeightFace new_face;
224
225 // clang-format off
226 new_face.vertex_ids.resize(4, -1);
227 new_face.vertex_ids[0] = tc_face.vertex_ids[0] + k * num_template_vertices;
228 new_face.vertex_ids[1] = tc_face.vertex_ids[1] + k * num_template_vertices;
229 new_face.vertex_ids[2] = tc_face.vertex_ids[1] + (k + 1) * num_template_vertices;
230 new_face.vertex_ids[3] = tc_face.vertex_ids[0] + (k + 1) * num_template_vertices;
231 // clang-format on
232
233 if (not tc_face.has_neighbor) new_face.neighbor = tc_face.neighbor;
234
235 new_cell.faces.push_back(std::move(new_face));
236 } // for tc face
237
238 //================================== Create top and bottom faces
239 // Top face
240 {
241 UnpartitionedMesh::LightWeightFace new_face;
242
243 new_face.vertex_ids.reserve(template_cell->vertex_ids.size());
244 for (auto vid : template_cell->vertex_ids)
245 new_face.vertex_ids.push_back(vid +
246 (k + 1) * num_template_vertices);
247
248 if ((k + 1) == z_levels.size()) new_face.neighbor = zmax_bndry_id;
249
250 new_cell.faces.push_back(std::move(new_face));
251 }
252 // Bottom face
253 {
254 UnpartitionedMesh::LightWeightFace new_face;
255
256 new_face.vertex_ids.reserve(template_cell->vertex_ids.size());
257 auto& vs = template_cell->vertex_ids;
258 for (auto vid = vs.rbegin(); vid != vs.rend(); ++vid)
259 new_face.vertex_ids.push_back((*vid) + k * num_template_vertices);
260
261 if (k == 0) new_face.neighbor = zmin_bndry_id;
262
263 new_cell.faces.push_back(std::move(new_face));
264 }
265 umesh->GetRawCells().push_back(new_cell_ptr);
266 } // for template cell
267 ++k;
268 } // for sub-layer n
269 } // for layer
270
271 umesh->SetAttributes(DIMENSION_3 | EXTRUDED);
272
273 umesh->ComputeCentroidsAndCheckQuality();
274 umesh->BuildMeshConnectivity();
275
276 Chi::log.Log0Verbose1()
277 << "ExtruderMeshGenerator::GenerateUnpartitionedMesh Done";
278 return umesh;
279}
