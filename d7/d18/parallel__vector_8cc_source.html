105void ParallelVector::Assemble()
106{
107 // Define the local operation mode.
108 // 0=Do Nothing, 1=Set, 2=Add, 3=INVALID (mixed set/add ops)
109 const short local_mode =
110 short(not set_cache_.empty()) + short(not add_cache_.empty()) * 2;
111 ChiLogicalErrorIf(local_mode == 3, "Invalid operation mode.");
112
113 // Now, determine the global operation mode
114 short global_mode;
115 MPI_Allreduce(&local_mode, &global_mode, 1,
116 MPI_SHORT, MPI_MAX, comm_);
117
118 // If the mode is to do nothing, exit
119 if (global_mode == 0)
120 return;
121
122 // Next, ensure that all operation types are compatible
123 ChiLogicalErrorIf(
124 local_mode != 0 and local_mode != global_mode,
125 "The operation on each process must be either 0 (do nothing),"
126 "or the same across all processes.");
127
128 // Now, store the global operation type and get the appropriate cache
129 using OpType = VecOpType;
130 const auto global_op_type = static_cast<OpType>(global_mode);
131 auto& op_cache = global_op_type == OpType ::SET_VALUE ?
132 set_cache_ : add_cache_;
133
134
135 // First, segregate the local and non-local operations
136 std::vector<std::pair<int64_t, double>> local_cache;
137 std::vector<std::pair<int64_t, double>> nonlocal_cache;
138 for (const auto& op : op_cache)
139 {
140 const int op_pid = FindOwnerPID(op.first);
141 if (op_pid == location_id_) local_cache.emplace_back(op);
142 else nonlocal_cache.emplace_back(op);
143 }
144
145 // The local operations can be handled immediately
146 for (const auto& [global_id, value] : local_cache)
147 {
148 const int64_t local_id = global_id - extents_[location_id_];
149 ChiLogicalErrorIf(
150 local_id < 0 or local_id >= local_size_,
151 "Invalid mapping from global to local.");
152
153 if (global_op_type == OpType::SET_VALUE) values_[local_id] = value;
154 else values_[local_id] += value;
155 }
156
157 // With this, the data that needs to be sent to other processes must be
158 // determined. Here, a mapping is developed between the processes that
159 // need to be sent information, and the serialized operations that need
160 // to be sent. The operations are serialized by converting the
161 // int64_t-double pair to bytes.
162 std::map<int, chi_data_types::ByteArray> pid_send_map;
163 for (const auto& [global_id, value] : nonlocal_cache)
164 {
165 const int pid = FindOwnerPID(global_id);
166 auto& byte_array = pid_send_map[pid];
167 byte_array.Write(global_id);
168 byte_array.Write(value);
169 }
170
171 // For use with MPI, the byte arrays from above is converted to an
172 // STL vector of bytes.
173 std::map<int, std::vector<std::byte>> pid_send_map_bytes;
174 for (const auto& [pid, byte_array] : pid_send_map)
175 pid_send_map_bytes[pid] = byte_array.Data();
176
177 // With the information that needs to be sent to other processes obtained,
178 // now, the information to be received from other processes is needed.
179 // To do this, each process must send to each other process the information
180 // that it needs. With each process knowing what each other process needs
181 // from it, a map of information to be sent is obtained.
182 std::map<int, std::vector<std::byte>> pid_recv_map_bytes =
183 chi_mpi_utils::MapAllToAll(pid_send_map_bytes, MPI_BYTE);
184
185 // The received information is now processed, unpacked, and the
186 // necessary operations performed
187 for (const auto& [pid, byte_vector] : pid_recv_map_bytes)
188 {
189 const auto packet_size = sizeof(std::pair<int64_t, double>);
190
191 ChiLogicalErrorIf(
192 byte_vector.size() % packet_size != 0,
193 "Unrecognized received operations. Operations are serialized with "
194 "an int64_t and double, but the received packet from process " +
195 std::to_string(pid) + " on process " + std::to_string(location_id_) +
196 " is not an integer multiple of the size of an int64_t and double.");
197
198 const size_t num_ops = byte_vector.size() / packet_size;
199 chi_data_types::ByteArray byte_array(byte_vector);
200 for (size_t k = 0; k < num_ops; ++k)
201 {
202 const int64_t global_id = byte_array.Read<int64_t>();
203 const double value = byte_array.Read<double>();
204
205 // Check that the global ID is in fact valid for this process
206 const int64_t local_id = global_id - extents_[location_id_];
207
208 ChiLogicalErrorIf(
209 local_id < 0 or local_id >= local_size_,
210 "A non-local global ID was received by process " +
211 std::to_string(location_id_) + " by process " + std::to_string(pid) +
212 " during vector assembly.");
213
214 // Contribute to the local vector
215 if (global_op_type == OpType ::SET_VALUE) values_[local_id] = value;
216 else values_[local_id] += value;
217 }
218 }
219
220 // Finally, clear the operation cache
221 op_cache.clear();
222}

225void ParallelVector::DefineParallelStructure()
226{
227 // Get the local vector sizes per processor
228 std::vector<uint64_t> local_sizes(process_count_, 0);
229 MPI_Allgather(&local_size_, //sendbuf
230 1, MPI_UINT64_T, //sendcount + sendtype
231 local_sizes.data(), //recvbuf
232 1, MPI_UINT64_T, //recvcount + recvtype
233 comm_); //communicator
234
235 // With the vector sizes per processor, now the offsets for each
236 // processor can be defined using a cumulative sum per processor.
237 // This allows for the determination of whether a global index is
238 // locally owned or not.
239 extents_.assign(process_count_ + 1, 0);
240 for (size_t p = 1; p < process_count_; ++p)
241 extents_[p] = extents_[p - 1] + local_sizes[p - 1];
242 extents_[process_count_] =
243 extents_[process_count_ - 1] + local_sizes.back();
244}

247int ParallelVector::FindOwnerPID(const uint64_t global_id) const
248{
249 ChiInvalidArgumentIf(global_id >= global_size_,
250 "Invalid global id specified.");
251
252 for (int p = 0; p < process_count_; ++p)
253 if (global_id >= extents_[p] and
254 global_id < extents_[p + 1])
255 return p;
256 return -1;
257
258}
