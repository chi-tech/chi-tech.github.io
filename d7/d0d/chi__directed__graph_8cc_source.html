26/** Removes a vertex from the graph.*/
27void chi_graph::DirectedGraph::
28 VertexAccessor::RemoveVertex(int v)
29{
30 //=================================== Check "from" is in range
31 if ((v<0) || (v>=vertices.size()))
32 {
33 chi_log.Log(LOG_ALL)
34 << "chi_graph::DirectedGraph::VertexAccessor: "
35 << "Error removing vertex " << v;
36 exit(EXIT_FAILURE);
37 }
38
39 auto& vertex = vertices[v];
40
41 //=================================== Get adjacent vertices
42 auto num_us = vertex.us_edge.size();
43 auto num_ds = vertex.ds_edge.size();
44 std::vector<int> adj_verts;
45 adj_verts.reserve(num_us+num_ds);
46
47 for (int u : vertex.us_edge)
48 adj_verts.push_back(u);
49
50 for (int u : vertex.ds_edge)
51 adj_verts.push_back(u);
52
53 //=================================== Remove v from all u
54 for (int u : adj_verts)
55 {
56 vertices[u].us_edge.erase(v);
57 vertices[u].ds_edge.erase(v);
58 }
59
60 vertex_valid_flags[v] = false;
61}

150/**SCC main recursive algorithm. This is the recursive call for the
151 * method defined below this one
152 * (chi_graph::DirectedGraph::FindStronglyConnectedConnections).*/
153void chi_graph::DirectedGraph::SCCAlgorithm(
154 int u, int& time,
155 std::vector<int>& disc,
156 std::vector<int>& low,
157 std::vector<bool>& on_stack,
158 std::stack<int>& stack,
159 std::vector<std::vector<int>>& SCCs)
160{
161 // Init discovery time and low value
162 disc[u] = low[u] = ++time;
163 stack.push(u);
164 on_stack[u] = true;
165
166 for (auto v : vertices[u].ds_edge)
167 {
168 if (disc[v] == -1)
169 {
170 SCCAlgorithm(v,time,disc,low,on_stack,stack,SCCs);
171 low[u] = std::min(low[u],low[v]);
172 }
173 else if (on_stack[v] == true)
174 low[u] = std::min(low[u],disc[v]);
175 }
176
177 int w=0;
178 if (low[u] == disc[u])
179 {
180 std::vector<int> sub_SCC;
181 while (stack.top() != u)
182 {
183 w = stack.top();
184 sub_SCC.push_back(w);
185 on_stack[w] = false;
186 stack.pop();
187 }
188 w = stack.top();
189 sub_SCC.push_back(w);
190 if (sub_SCC.size() > 1) SCCs.push_back(sub_SCC);
191 on_stack[w] = false;
192 stack.pop();
193 }
194
195}

198/**Find strongly connected components. This method is the implementation
199 * of Tarjan's algorithm [1].
200 *
201 * [1] Tarjan R.E. "Depth-first search and linear graph algorithms",
202 * SIAM Journal on Computing, 1972.
203 *
204 * It returns collections of vertices that form strongly connected
205 * components excluding singletons.*/
206std::vector<std::vector<int>> chi_graph::DirectedGraph::
207 FindStronglyConnectedComponents()
208{
209 size_t V = vertices.size();
210
211 std::vector<int> disc(V,-1); // Discovery times
212 std::vector<int> low(V,-1); // Earliest visited vertex
213 std::vector<bool> on_stack(V,false); // On stack flags
214 std::stack<int> stack; // Stack
215
216 std::vector<std::vector<int>> SCCs; // Collection of SCCs
217
218 int time = 0;
219
220 for (int v=0; v<V; ++v)
221 if (disc[v]==-1)
222 SCCAlgorithm(v,time,disc,low,on_stack,stack,SCCs);
223
224
225 return SCCs;
226}

230/** Generates a topological sort. This method is the implementation
231 * of Kahn's algorithm [1].
232 *
233 * [1] Kahn, Arthur B. (1962), "Topological sorting of large networks",
234 * Communications of the ACM, 5 (11): 558â€“562
235 *
236 * \return Returns the vertex ids sorted topologically. If this
237 * vector is empty the algorithm failed because it detected
238 * cyclic dependencies.*/
239std::vector<int> chi_graph::DirectedGraph::GenerateTopologicalSort()
240{
241 bool has_cycles=false;
242 std::vector<int> L;
243 std::vector<GraphVertex*> S;
244
245 L.reserve(vertices.size());
246 S.reserve(vertices.size());
247
248 //======================================== Make a copy of the graph
249 auto cur_vertices = vertices;
250
251 //======================================== Identify vertices that
252 // have no incoming edge
253 for (auto& vertex : cur_vertices)
254 if (vertex.us_edge.empty())
255 S.push_back(&vertex);
256
257 if (S.empty())
258 {
259 has_cycles = true;
260 goto endofalgo;
261 }
262
263 //======================================== Repeatedly remove
264 // vertices
265 while (not S.empty())
266 {
267 GraphVertex* node_n = S.back(); int n=node_n->id;
268 S.erase(S.end()-1);
269
270 L.push_back(n);
271 auto nodes_m = node_n->ds_edge;
272 for (int m : nodes_m)
273 {
274 GraphVertex* node_m = &cur_vertices[m];
275
276 //remove edge from graph
277 node_n->ds_edge.erase(m);
278 node_m->us_edge.erase(n);
279
280 if (node_m->us_edge.empty())
281 S.push_back(node_m);
282 }
283 }
284
285 endofalgo:
286 {
287 if (has_cycles)
288 return std::vector<int>();
289 if (L.size() != vertices.size())
290 return std::vector<int>();
291 }
292
293 return L;
294}

297/**Finds a sequence that minimizes the Feedback Arc Set (FAS). This
298 * algorithm implements the algorithm depicted in [1].
299 *
300 * [1] Eades P., Lin X., Smyth W.F., "Fast & Effective heuristic for
301 * the feedback arc set problem", Information Processing Letters,
302 * Volume 47. 1993.*/
303std::vector<int> chi_graph::DirectedGraph::
304 FindApproxMinimumFAS()
305{
306 auto GetVertexDelta = [](chi_graph::GraphVertex& vertex)
307 {
308// double delta = vertex.ds_edge.size() - vertex.us_edge.size();
309// return delta;
310
311 double delta = 0.0;
312 for (int ds : vertex.ds_edge)
313 delta += 1.0*vertex.ds_weights[ds];
314
315 for (int us : vertex.us_edge)
316 delta -= 1.0*vertex.us_weights[us];
317
318 return delta;
319 };
320
321 auto& TG = *this;
322
323 //==================================== Execute GR-algorithm
324 std::vector<int> s1,s2,s;
325 while (TG.vertices.GetNumValid()>0)
326 {
327 //======================== Remove sinks
328 while (TG.GetNumSinks()>0)
329 {
330 for (auto& u : TG.vertices)
331 if (u.ds_edge.empty())
332 {
333 TG.RemoveVertex(u.id);
334 s2.push_back(u.id);
335 break;
336 }
337 }//G contains sinks
338
339 //======================== Remove sources
340 while (TG.GetNumSources()>0)
341 {
342 for (auto& u : TG.vertices)
343 if (u.us_edge.empty())
344 {
345 TG.RemoveVertex(u.id);
346 s1.push_back(u.id);
347 break;
348 }
349 }//G contains sinks
350
351 //======================== Get max delta
352 std::pair<int,double> max_delta(-1,-100.0);
353 for (auto& u : TG.vertices)
354 {
355 double delta = GetVertexDelta(u);
356 if (delta > max_delta.second)
357 max_delta = std::make_pair(u.id,delta);
358 }
359
360 //======================== Remove max delta
361 TG.RemoveVertex(max_delta.first);
362 s1.push_back(max_delta.first);
363 }
364
365
366 //========================== Make appr. minimum FAS sequence
367 s.reserve(s1.size() + s2.size());
368 for (int u : s1) s.push_back(u);
369 for (int u : s2) s.push_back(u);
370
371 return s;
372}

441std::vector<std::pair<int,int>>
442chi_graph::DirectedGraph::RemoveCyclicDependencies()
443{
444 std::vector<std::pair<int,int>> edges_to_remove;
445
446 //============================================= Utility lambdas
447 auto IsInList = [](std::vector<int>& list, int val)
448 {
449 return std::find(list.begin(),list.end(),val) != list.end();
450 };
451
452
453 //============================================= Find initial SCCs
454 auto SCCs = FindStronglyConnectedComponents();
455
456 int iter=0;
457 while (not SCCs.empty())
458 {
459 if (chi_log.GetVerbosity() >= LOG_0VERBOSE_2)
460 chi_log.Log(LOG_ALL)
461 << "Inter cell cyclic dependency removal. Iteration " << ++iter;
462
463 //============================================= Remove bi-connected then
464 // tri-connected SCCs then
465 // n-connected
466 for (auto& subDG : SCCs)
467 {
468 //====================================== If bi-connected
469 if (subDG.size()==2)
470 {
471 RemoveEdge(subDG.front(), subDG.back());
472 edges_to_remove.emplace_back(subDG.front(), subDG.back());
473 }//bi-connected
474 //====================================== If tri-connected
475 else if (subDG.size()==3)
476 {
477 bool found=false;
478 for (int u : subDG)
479 {
480 for (int v : vertices[u].ds_edge)
481 if (IsInList(subDG,v))
482 {
483 found=true;
484 RemoveEdge(u, v);
485 edges_to_remove.emplace_back(u, v);
486 break;
487 }
488 if (found) break;
489 }//for u
490 }//tri-connected
491 //====================================== If n-connected
492 else
493 {
494 //=============================== Add vertices to temporary graph
495 chi_graph::DirectedGraph TG; //Temp Graph
496 for (auto u : subDG)
497 TG.AddVertex();
498
499 //=============================== Add local connectivity
500 int mapping_u = 0;
501 for (auto u : subDG)
502 {
503 for (auto v : vertices[u].ds_edge)
504 {
505 auto mapv = std::find(subDG.begin(),subDG.end(),v);
506 if (mapv != subDG.end())
507 {
508 int mapping_v = mapv - subDG.begin();
509 TG.AddEdge(mapping_u,mapping_v,vertices[u].ds_weights[v]);
510 }
511 }//for v
512
513 ++mapping_u;
514 }//for u
515
516 //=============================== Make a copy of the graph verts
517 std::vector<chi_graph::GraphVertex> verts_copy;
518 verts_copy.reserve(TG.vertices.size());
519 for (auto& v : TG.vertices)
520 verts_copy.push_back(v);
521
522 //=============================== Solve the minimum Feedback
523 // Arc Set (FAS) problem
524 auto s = TG.FindApproxMinimumFAS();
525
526 //========================== Build a sequence map
527 // This maps original sequence
528 // to minFAS sequence. i.e. originally
529 // we had v=0,1,2,3... and afterwards we
530 // something like s=7,3,1,5,0,....
531 // smap[v] then gives the position of v in s
532 std::vector<int> smap(s.size(),-1);
533 int count=0;
534 for (int u: s)
535 smap[u] = count++;
536
537 //========================== Build edges to remove
538 std::vector<std::pair<int,int>> edges_to_rem;
539 for (auto& u : verts_copy)
540 {
541 int cur_map = smap[u.id];
542 for (int v : u.ds_edge)
543 {
544 int adj_map = smap[v];
545 if (adj_map < cur_map)
546 edges_to_rem.emplace_back(u.id,v);
547 }
548 }
549
550 for (auto& edge : edges_to_rem)
551 {
552 int u = subDG[edge.first];
553 int v = subDG[edge.second];
554 RemoveEdge(u, v);
555 edges_to_remove.emplace_back(u, v);
556 }
557
558 }//n-connected
559 }//for sub-DG
560
561
562 //============================================= Find SSCs again
563 // This step is like an insurance policy for if
564 // something came through. There should be no SSCs
565 // after the minFAS process, however, we just look
566 // again in-case.
567 SCCs = FindStronglyConnectedComponents();
568 }
569
570 return edges_to_remove;
571}
