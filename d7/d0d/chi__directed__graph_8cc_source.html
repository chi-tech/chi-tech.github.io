149 /**SCC main recursive algorithm. This is the recursive call for the
150  * method defined below this one
151  * (chi_graph::DirectedGraph::FindStronglyConnectedConnections).*/
152 void chi_graph::DirectedGraph::SCCAlgorithm(
153  int u, int& time,
154  std::vector<int>& disc,
155  std::vector<int>& low,
156  std::vector<bool>& on_stack,
157  std::stack<int>& stack,
158  std::vector<std::vector<int>>& SCCs)
159 {
160  // Init discovery time and low value
161  disc[u] = low[u] = ++time;
162  stack.push(u);
163  on_stack[u] = true;
164 
165  for (auto v : vertices[u].ds_edge)
166  {
167  if (disc[v] == -1)
168  {
169  SCCAlgorithm(v,time,disc,low,on_stack,stack,SCCs);
170  low[u] = std::min(low[u],low[v]);
171  }
172  else if (on_stack[v] == true)
173  low[u] = std::min(low[u],disc[v]);
174  }
175 
176  int w=0;
177  if (low[u] == disc[u])
178  {
179  std::vector<int> sub_SCC;
180  while (stack.top() != u)
181  {
182  w = stack.top();
183  sub_SCC.push_back(w);
184  on_stack[w] = false;
185  stack.pop();
186  }
187  w = stack.top();
188  sub_SCC.push_back(w);
189  if (sub_SCC.size() > 1) SCCs.push_back(sub_SCC);
190  on_stack[w] = false;
191  stack.pop();
192  }
193 
194 }

229 /** Generates a topological sort. This method is the implementation
230  * of Kahn's algorithm [1].
231  *
232  * [1] Kahn, Arthur B. (1962), "Topological sorting of large networks",
233  * Communications of the ACM, 5 (11): 558–562
234  *
235  * \return Returns the vertex ids sorted topologically. If this
236  * vector is empty the algorithm failed because it detected
237  * cyclic dependencies.*/
238 std::vector<int> chi_graph::DirectedGraph::GenerateTopologicalSort()
239 {
240  bool has_cycles=false;
241  std::vector<int> L;
242  std::vector<GraphVertex*> S;
243 
244  L.reserve(vertices.size());
245  S.reserve(vertices.size());
246 
247  //======================================== Make a copy of the graph
248  auto cur_vertices = vertices;
249 
250  //======================================== Identify vertices that
251  // have no incoming edge
252  for (auto& vertex : cur_vertices)
253  if (vertex.us_edge.empty())
254  S.push_back(&vertex);
255 
256  if (S.empty())
257  {
258  has_cycles = true;
259  goto endofalgo;
260  }
261 
262  //======================================== Repeatedly remove
263  // vertices
264  while (not S.empty())
265  {
266  GraphVertex* node_n = S.back(); int n=node_n->id;
267  S.erase(S.end()-1);
268 
269  L.push_back(n);

271  for (int m : nodes_m)
272  {
273  GraphVertex* node_m = &cur_vertices[m];
274 
275  //remove edge from graph
276  node_n->ds_edge.erase(m);
277  node_m->us_edge.erase(n);
278 
279  if (node_m->us_edge.empty())

281  }
282  }
283

293 }

296 /**Finds a sequence that minimizes the Feedback Arc Set (FAS). This
297  * algorithm implements the algorithm depicted in [1].
298  *
299  * [1] Eades P., Lin X., Smyth W.F., "Fast & Effective heuristic for
300  * the feedback arc set problem", Information Processing Letters,
301  * Volume 47. 1993.*/
302 std::vector<int> chi_graph::DirectedGraph::
303  FindApproxMinimumFAS()
304 {
305  auto GetVertexDelta = [](chi_graph::GraphVertex& vertex)
306  {
307 // double delta = vertex.ds_edge.size() - vertex.us_edge.size();
308 // return delta;
309 
310  double delta = 0.0;
311  for (int ds : vertex.ds_edge)
312  delta += 1.0*vertex.ds_weights[ds];
313 
314  for (int us : vertex.us_edge)
315  delta -= 1.0*vertex.us_weights[us];
316 
317  return delta;
318  };
319 
320  auto& TG = *this;
321 
322  //==================================== Execute GR-algorithm
323  std::vector<int> s1,s2,s;
324  while (TG.vertices.GetNumValid()>0)
325  {
326  //======================== Remove sinks
327  while (TG.GetNumSinks()>0)
328  {
329  for (auto& u : TG.vertices)
330  if (u.ds_edge.empty())
331  {
332  TG.RemoveVertex(u.id);
333  s2.push_back(u.id);
334  break;
335  }
336  }//G contains sinks
337 
338  //======================== Remove sources
339  while (TG.GetNumSources()>0)
340  {
341  for (auto& u : TG.vertices)
342  if (u.us_edge.empty())
343  {
344  TG.RemoveVertex(u.id);
345  s1.push_back(u.id);
346  break;
347  }
348  }//G contains sinks
349 
350  //======================== Get max delta
351  std::pair<int,double> max_delta(-1,-100.0);
352  for (auto& u : TG.vertices)
353  {
354  double delta = GetVertexDelta(u);
355  if (delta > max_delta.second)
356  max_delta = std::make_pair(u.id,delta);
357  }
358 
359  //======================== Remove max delta
360  TG.RemoveVertex(max_delta.first);
361  s1.push_back(max_delta.first);
362  }
363 
364 
365  //========================== Make appr. minimum FAS sequence
366  s.reserve(s1.size() + s2.size());
367  for (int u : s1) s.push_back(u);
368  for (int u : s2) s.push_back(u);
369 
370  return s;
371 }
