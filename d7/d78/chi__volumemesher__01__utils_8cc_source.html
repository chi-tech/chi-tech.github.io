59 void chi_mesh::VolumeMesher::
60 CreatePolygonCells(chi_mesh::SurfaceMesh *surface_mesh,
61  chi_mesh::MeshContinuum *vol_continuum)
62 {
63  //============================================= Get current mesh handler
64  chi_mesh::MeshHandler* handler = chi_mesh::GetCurrentHandler();
65 
66  //============================================= Copy nodes
67  std::vector<chi_mesh::Vertex>::iterator vertex;
68  for (vertex = surface_mesh->vertices.begin();
69  vertex != surface_mesh->vertices.end();
70  vertex++)
71  {
72  chi_mesh::Node* node = new chi_mesh::Node;
73  *node = (*vertex.base());
74 
75  vol_continuum->nodes.push_back(node);
76  }
77 
78  //============================================= Process faces
79  std::vector<chi_mesh::Face>::iterator face;
80  for (face = surface_mesh->faces.begin();
81  face != surface_mesh->faces.end();
82  face++)
83  {
84  chi_mesh::CellPolygon* cell = new chi_mesh::CellPolygon;
85 
86  for (int k=0;k<3;k++)
87  {
88  cell->v_indices.push_back(face->v_index[k]);
89 
90  int* edge_indices = new int[5];
91 
92  edge_indices[0]=face->e_index[k][0];
93  edge_indices[1]=face->e_index[k][1];
94  edge_indices[2]=face->e_index[k][2];
95  edge_indices[3]=face->e_index[k][3];
96 
97  //auto adj_cell = vol_continuum->item_id[edge_indices[2]];
98  //edge_indices[4] = GetCellXYPartitionID(adj_cell);

101  //chi_mesh::Vertex v1 = *vol_continuum->nodes[edge_indices[1]];
102  chi_mesh::Vertex v0 = surface_mesh->vertices[edge_indices[0]];
103  chi_mesh::Vertex v1 = surface_mesh->vertices[edge_indices[1]];
104  chi_mesh::Vector vk = chi_mesh::Vector(0.0,0.0,1.0);
105 
106  chi_mesh::Vector va = v1-v0;
107  chi_mesh::Vector vn = va.Cross(vk);
108  vn = vn/vn.Norm();
109  cell->edgenormals.push_back(vn);
110 
111  cell->edges.push_back(edge_indices);
112 
113 
114  cell->centroid = cell->centroid +
115  surface_mesh->vertices[face->v_index[k]];
116  }
117  cell->centroid = cell->centroid/3;
118 
119  //====================================== Compute xy partition id
120  cell->xy_partition_indices = GetCellXYPartitionID(cell);
121  cell->partition_id = cell->xy_partition_indices.second*
122  handler->surface_mesher->partitioning_x +
123  cell->xy_partition_indices.first;
124 
125  cell->cell_global_id = vol_continuum->cells.size();
126 
127  vol_continuum->cells.push_back(cell);
128  }
129  for (int f=0; f<surface_mesh->poly_faces.size(); f++)
130  {
131  chi_mesh::PolyFace* face = surface_mesh->poly_faces[f];
132 
133  chi_mesh::CellPolygon* cell = new chi_mesh::CellPolygon;
134 
135  //====================================== Copy vertices
136  for (int v=0; v<face->v_indices.size();v++)
137  {
138  cell->v_indices.push_back(face->v_indices[v]);
139  cell->centroid = cell->centroid +
140  surface_mesh->vertices[face->v_indices[v]];
141  }
142  cell->centroid = cell->centroid/cell->v_indices.size();
143 
144  //====================================== Compute partition id
145  cell->xy_partition_indices = GetCellXYPartitionID(cell);
146  cell->partition_id = cell->xy_partition_indices.second*
147  handler->surface_mesher->partitioning_x +
148  cell->xy_partition_indices.first;
149 
150  //====================================== Copy edges
151  for (int e=0; e<face->edges.size(); e++)
152  {
153  int* src_side = face->edges[e];
154  int* des_side = new int[5];
155 
156  for (int k=0;k<4;k++)
157  {
158  des_side[k] = src_side[k];
159  }
160 
161  chi_mesh::Vertex v0 = surface_mesh->vertices[des_side[0]];
162  chi_mesh::Vertex v1 = surface_mesh->vertices[des_side[1]];
163  chi_mesh::Vector vk = chi_mesh::Vector(0.0,0.0,1.0);
164 
165  chi_mesh::Vector va = v1-v0;
166  chi_mesh::Vector vn = va.Cross(vk);
167  vn = vn/vn.Norm();
168  cell->edgenormals.push_back(vn);
169 
170  cell->edges.push_back(des_side);
171  }
172 
173  cell->cell_global_id = vol_continuum->cells.size();
174 
175  vol_continuum->cells.push_back(cell);
176  }
177 }

186 std::pair<int,int> chi_mesh::VolumeMesher::
187  GetCellXYPartitionID(chi_mesh::Cell *cell)
188 {
189  std::pair<int,int> ij_id(0,0);
190  bool found_partition = false;
191 
192  if (chi_mpi.process_count == 1){return ij_id;}
193 
194  //================================================== Get the current handler
195  chi_mesh::MeshHandler* mesh_handler = chi_mesh::GetCurrentHandler();
196  chi_mesh::SurfaceMesher* surf_mesher = mesh_handler->surface_mesher;
197 
198  if ((typeid(*surf_mesher) == typeid(chi_mesh::SurfaceMesherTriangle)) ||
199  (typeid(*surf_mesher) == typeid(chi_mesh::SurfaceMesherPredefined)))
200  {
201  //====================================== Sanity check on partitioning
202  int num_x_subsets = surf_mesher->xcuts.size()+1;
203  int num_y_subsets = surf_mesher->ycuts.size()+1;
204 
205  int x_remainder = num_x_subsets%surf_mesher->partitioning_x;
206  int y_remainder = num_y_subsets%surf_mesher->partitioning_y;
207 
208  if (x_remainder != 0)
209  {
210  chi_log.Log(LOG_ALLERROR)
211  << "When specifying x-partitioning, the number of grp_subsets in x "
212  "needs to be divisible by the number of partitions in x.";
213  exit(EXIT_FAILURE);
214  }
215 
216  if (y_remainder != 0)
217  {
218  chi_log.Log(LOG_ALLERROR)
219  << "When specifying y-partitioning, the number of grp_subsets in y "
220  "needs to be divisible by the number of partitions in y.";
221  exit(EXIT_FAILURE);
222  }
223 
224  int subsets_per_partitionx = num_x_subsets/surf_mesher->partitioning_x;
225  int subsets_per_partitiony = num_y_subsets/surf_mesher->partitioning_y;
226 
227 // chi_log.Log(LOG_0ERROR) << num_x_subsets;
228 // chi_log.Log(LOG_0ERROR) << num_y_subsets;
229 // chi_log.Log(LOG_0ERROR) << subsets_per_partitionx;
230 // chi_log.Log(LOG_0ERROR) << subsets_per_partitiony;
231 
232 
233  //====================================== Determine x-partition
234  int x=-1;
235  int xcount=-1;
236  for (int i = subsets_per_partitionx-1;
237  i < surf_mesher->xcuts.size();
238  i += subsets_per_partitionx)
239  {
240  xcount++;
241  if (cell->centroid.x <= surf_mesher->xcuts[i])
242  {
243  x = xcount;
244  break;
245  }
246  }
247  if (x<0)
248  {
249  x = surf_mesher->partitioning_x-1;
250  }
251 
252  //====================================== Determine y-partition
253  int y=-1;
254  int ycount=-1;
255  for (int i = subsets_per_partitiony-1;
256  i < surf_mesher->ycuts.size();
257  i += subsets_per_partitiony)
258  {
259  ycount++;
260  if (cell->centroid.y <= surf_mesher->ycuts[i])
261  {
262  y = ycount;
263  break;
264  }
265  }
266  if (y<0)
267  {
268  y = surf_mesher->partitioning_y - 1;
269  }
270 
271  //====================================== Set partitioning
272  ij_id.first = x;
273  ij_id.second= y;
274 
275  }//if typeid
276 
277  return ij_id;
278 }

284 std::tuple<int,int,int> chi_mesh::VolumeMesher::
285 GetCellXYZPartitionID(chi_mesh::Cell *cell)
286 {
287  std::tuple<int,int,int> ijk_id(0,0,0);
288  bool found_partition = false;
289 
290  if (chi_mpi.process_count == 1){return ijk_id;}
291 
292  //================================================== Get ij indices
293  std::pair<int,int> ij_id = GetCellXYPartitionID(cell);
294 
295 
296  //================================================== Get the current handler
297  chi_mesh::MeshHandler* mesh_handler = chi_mesh::GetCurrentHandler();
298  chi_mesh::VolumeMesher* vol_mesher = mesh_handler->volume_mesher;
299 
300  //&&&&&&&&&&&&&&&&&&&&&&&&&&&&&&&&&&&&&&&&&&&&&&&&&& SLAB
301  if (typeid(*vol_mesher) == typeid(chi_mesh::VolumeMesherLinemesh1D))
302  {
303  auto line_mesher = (chi_mesh::VolumeMesherLinemesh1D*)vol_mesher;
304 
305  if (chi_mpi.process_count != options.partition_z and !options.mesh_global)
306  {
307  chi_log.Log(LOG_ALLERROR)
308  << "Number of process requested, " << options.partition_z
309  << ", in PARTITION_Z does not match the amount of processes "
310  << "available " << chi_mpi.process_count;
311  exit(EXIT_FAILURE);
312  }
313 
314  int cells_per_loc =
315  ceil(line_mesher->num_slab_cells/(double)options.partition_z);
316 
317  int cur_loc = 0;
318  for (int k=0; k<chi_mpi.process_count; k++)
319  {
320  if (cell->cell_global_id < ((k+1)*cells_per_loc))
321  {
322  cur_loc = k;
323  found_partition = true;
324  break;
325  }
326  }//for k
327 
328  std::get<0>(ijk_id) = ij_id.first;
329  std::get<1>(ijk_id) = ij_id.second;
330  std::get<2>(ijk_id) = cur_loc;
331  }
332  //&&&&&&&&&&&&&&&&&&&&&&&&&&&&&&&&&&&&&&&&&&&&&&&&&& POLYHEDRON
333  else if (typeid(*vol_mesher) == typeid(chi_mesh::VolumeMesherExtruder))
334  {
335  auto extruder = (chi_mesh::VolumeMesherExtruder*)vol_mesher;
336 
337  //====================================== Create virtual cuts
338  if (vol_mesher->zcuts.size()==0)
339  {
340  int num_sub_layers = extruder->vertex_layers.size()-1;
341 
342  if ((num_sub_layers%vol_mesher->options.partition_z) != 0)
343  {
344  chi_log.Log(LOG_ALLERROR)
345  << "Number of sub-layers in extruded mesh is not divisible "
346  << "by the requested number of z-partitions.";
347  exit(EXIT_FAILURE);
348  }
349 
350  int delta_zk = num_sub_layers/
351  vol_mesher->options.partition_z;
352  for (int k=0; k<(vol_mesher->options.partition_z); k++)
353  {
354  int layer_index = k*delta_zk + delta_zk;
355  if (layer_index > (extruder->vertex_layers.size()-1))
356  {
357  layer_index = (int)extruder->vertex_layers.size()-1;
358  vol_mesher->zcuts.push_back(extruder->vertex_layers[layer_index]);
359  }
360  else
361  {
362  vol_mesher->zcuts.push_back(extruder->vertex_layers[layer_index]);
363 
364  if (chi_log.GetVerbosity()==LOG_0VERBOSE_2)
365  {
366  printf("Z-Cut %d, %g\n",vol_mesher->zcuts.size(),
367  extruder->vertex_layers[layer_index]);
368  }
369  }
370  }
371  }
372 
373 
374  //====================================== Scan cuts for location
375  double zmin = -1.0e-16;
376  double zmax = 1.0e-16;
377  for (int k=0; k<(vol_mesher->zcuts.size()); k++)
378  {
379  zmax = vol_mesher->zcuts[k];
380 
381  double z = cell->centroid.z;
382 
383  if (chi_log.GetVerbosity()==LOG_0VERBOSE_2)
384  {
385  printf("zmax = %g, zmin = %g, cell_z = %g\n",zmax,zmin,z);
386  }
387 
388 
389  if ((z > zmin) && (z < zmax))
390  {
391  std::get<0>(ijk_id) = ij_id.first;
392  std::get<1>(ijk_id) = ij_id.second;
393  std::get<2>(ijk_id) = k;
394 
395  found_partition = true;
396  break;
397  }
398  zmin = zmax;
399  }
400  }//if typeid
401  else if (typeid(*vol_mesher) == typeid(chi_mesh::VolumeMesherLinemesh1D))
402  {
403  found_partition = true;
404  std::get<0>(ijk_id) = ij_id.first;
405  std::get<1>(ijk_id) = ij_id.second;
406  std::get<2>(ijk_id) = 0;
407  }
408 
409  //================================================== Report unallocated item_id
410  if (!found_partition)
411  {
412  chi_log.Log(LOG_0ERROR)
413  << "A cell was encountered for which "
414  "no zpartition id was found";
415  exit(EXIT_FAILURE);
416  }
417 
418  return ijk_id;
419 }

456

458  {
459  if (polyh_cell->faces[f]->face_indices[0]<0)
460  {
461  vol_continuum->boundary_cell_indices.push_back(c);
462  break;
463  }
464  }
465  }//if typeid
