10 /**Initializes the data structures necessary for interpolation. This is
11  * independent of the physics and hence is a routine on its own.
12  *
13  * The first step of this initialization is to determine which cells
14  * are intersected by this plane. For polyhedrons this is evaluated
15  * tet-by-tet.
16  *
17  * The second step is find where face-edges are intersected. This will
18  * effectively create intersection polygons.*/

23  //================================================== Check grid available
24  if (field_functions.size() == 0)

28  exit(EXIT_FAILURE);

32  }
33

38  for (int lc=0; lc<num_local_cells; lc++)

84  }

88  }//if polyh

101  {

103 
104  //========================================= Initialize cell intersection
105  // data structure

107  cell_isds->cell_global_index = cell_glob_index;
108  cell_intersections.push_back(cell_isds);
109 
110  //========================================= Loop over vertices
111  for (int v=0; v<poly_cell->v_indices.size(); v++)
112  {

114  new FFIFaceEdgeIntersection;
115 
116  int v0gi = poly_cell->v_indices[v];
117 
118  face_isds->v0_g_index = v0gi;
119  face_isds->v1_g_index = v0gi;
120 
121  face_isds->v0_dofindex_cell = v;
122  face_isds->v1_dofindex_cell = v;
123 
124  face_isds->weights = std::pair<double,double>(0.5,0.5);
125  face_isds->point = *grid_view->nodes[v0gi];
126 
127  cell_isds->intersections.push_back(face_isds);
128  }

132 
133  //========================================= Set straight 2D center
134  // This is normally transformed for the 3D case
135  cell_isds->intersection_2d_centre = cell_isds->intersection_centre;

140  {
