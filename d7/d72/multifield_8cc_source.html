141/**Constructor.*/
142void MultiFieldOperation::Execute()
143{
144 typedef unsigned int uint;
145 typedef const int64_t cint64_t;
146 const auto& sdm = primary_ff_->SDM();
147 const auto& uk_man = primary_ff_->UnkManager();
148 const auto& grid = sdm.Grid();
149
150 const size_t num_deps = dependent_ffs_.size();
151
152 for (const auto& cell : grid.local_cells)
153 {
154 const auto& cell_mapping = sdm.GetCellMapping(cell);
155 const size_t num_nodes = cell_mapping.NumNodes();
156
157 const auto nodes_xyz = cell_mapping.GetNodeLocations();
158
159 for (size_t i = 0; i < num_nodes; ++i)
160 {
161 std::vector<double> input_params;
162 const auto& node_i_xyz = nodes_xyz[i];
163 for (size_t d = 0; d < 3; ++d)
164 input_params.push_back(node_i_xyz[d]);
165 input_params.push_back(static_cast<double>(cell.material_id_));
166
167 for (size_t k = 0; k < num_deps; ++k)
168 {
169 const auto& dep_ff = dependent_ffs_[k];
170 const double value =
171 dep_ff->Evaluate(cell, node_i_xyz, dependent_field_ref_component_[k]);
172 input_params.push_back(value);
173 }
174
175 std::vector<double> output_params = function_ptr_->Evaluate(input_params);
176
177 ChiLogicalErrorIf(
178 output_params.size() != result_component_references_.size(),
179 "Calling function number of output values not matching");
180
181 size_t k = 0;
182 for (uint c : result_component_references_)
183 {
184 cint64_t dof_map = sdm.MapDOFLocal(cell, i, uk_man, 0, c);
185
186 primary_ff_->FieldVector()[dof_map] = output_params[k++];
187 }
188
189 } // for node i
190 } // for cell
191}
