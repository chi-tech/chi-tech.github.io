74void AggregateNodalValuePostProcessor::Execute(const Event& event_context)
75{
76 if (not initialized_) Initialize();
77
78 const auto* grid_field_function = GetGridBasedFieldFunction();
79
80 ChiLogicalErrorIf(not grid_field_function,
81 "Attempted to access invalid field"
82 "function");
83
84 const auto& ref_ff = *grid_field_function;
85 const auto& sdm = ref_ff.SDM();
86 const auto& grid = sdm.Grid();
87
88 const auto& uk_man = ref_ff.UnkManager();
89 const auto uid = 0;
90 const auto cid = 0;
91
92 const auto field_data = ref_ff.GetGhostedFieldVector();
93 const size_t num_local_dofs =
94 ref_ff.SDM().GetNumLocalDOFs(ref_ff.UnkManager());
95
96 double local_max_value = 0.0;
97 double local_min_value = 0.0;
98 double local_accumulation = 0.0;
99 bool first_local = true;
100 for (const uint64_t cell_local_id : cell_local_ids_)
101 {
102 const auto& cell = grid.local_cells[cell_local_id];
103 const auto& cell_mapping = sdm.GetCellMapping(cell);
104 const size_t num_nodes = cell_mapping.NumNodes();
105
106 for (size_t i = 0; i < num_nodes; ++i)
107 {
108 const int64_t imap = sdm.MapDOFLocal(cell, i, uk_man, uid, cid);
109 if (imap >= 0 and imap < num_local_dofs)
110 {
111 const double field_value = field_data[imap];
112 if (first_local)
113 {
114 local_max_value = field_value;
115 local_min_value = field_value;
116 first_local = false;
117 }
118
119 local_max_value = std::max(local_max_value, field_value);
120 local_min_value = std::min(local_min_value, field_value);
121 local_accumulation += field_value;
122 }
123 } // for i
124 } // for cell-id
125
126 if (operation_ == "max")
127 {
128 double globl_max_value;
129 MPI_Allreduce(&local_max_value, // sendbuf
130 &globl_max_value, // recvbuf
131 1,
132 MPI_DOUBLE, // count + datatype
133 MPI_MAX, // op
134 Chi::mpi.comm); // communicator
135
136 value_ = ParameterBlock("", globl_max_value);
137 }
138 else if (operation_ == "min")
139 {
140 double globl_min_value;
141 MPI_Allreduce(&local_min_value, // sendbuf
142 &globl_min_value, // recvbuf
143 1,
144 MPI_DOUBLE, // count + datatype
145 MPI_MIN, // op
146 Chi::mpi.comm); // communicator
147
148 value_ = ParameterBlock("", globl_min_value);
149 }
150 else if (operation_ == "avg")
151 {
152 double globl_accumulation;
153 MPI_Allreduce(&local_accumulation, // sendbuf
154 &globl_accumulation, // recvbuf
155 1,
156 MPI_DOUBLE, // count + datatype
157 MPI_SUM, // op
158 Chi::mpi.comm); // communicator
159
160 const size_t num_globl_dofs =
161 ref_ff.SDM().GetNumGlobalDOFs(ref_ff.UnkManager());
162 value_ = ParameterBlock("", globl_accumulation / double(num_globl_dofs));
163 }
164 else
165 ChiLogicalError("Unsupported operation type \"" + operation_ + "\".");
166}
