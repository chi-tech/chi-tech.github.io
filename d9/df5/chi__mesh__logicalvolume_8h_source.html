123/**Right Circular Cylinder (RCC) logical volume.
124 *
125 * Determining whether a point is within an RCC is tricky.
126 * */
127class chi_mesh::RCCLogicalVolume : public LogicalVolume
128{
129public:
130 double x0,y0,z0;
131 double vx,vy,vz;
132 double r;
133
134 RCCLogicalVolume()
135 {
136 type_index = RCC;
137 x0 = 0.0; y0 = 0.0; z0 = 0.0;
138 vx = 0.0; vy = 0.0; vz = 1.0;
139 r = 1.0;
140 }
141
142 RCCLogicalVolume(double ix0, double iy0, double iz0,
143 double ivx, double ivy, double ivz,
144 double ir)
145 {
146 type_index = RCC;
147 x0 = ix0; y0 = iy0; z0 = iz0;
148 vx = ivx; vy = ivy; vz = ivz;
149 r = ir;
150 }
151
152 bool Inside(const chi_mesh::Vector3& point) const override
153 {
154 typedef chi_mesh::Vector3 Vec3;
155
156 const auto& pr = point; //reference point
157 const Vec3 p0(x0, y0, z0); //cylinder root
158 const Vec3 cyl_dir_vec(vx, vy, vz); //cylinder direction vector
159 const Vec3 k_hat(0.0, 0.0, 1.0); //k_hat
160
161 const Vec3 p0r = pr - p0;
162 const Vec3 cyl_unit_dir = cyl_dir_vec.Normalized(); //aka cud
163 const double cyl_length = cyl_dir_vec.Norm();
164
165 //====================================== Check if point is within
166 // normal extents
167 const double p0r_dot_cud = p0r.Dot(cyl_unit_dir);
168 if (p0r_dot_cud < 0.0 or p0r_dot_cud > cyl_length)
169 return false;
170
171 //====================================== Building rotation matrix
172 //This rotation matrix must be such that,
173 //when a coordinate system is rotated with it,
174 //the new normal vector points along the
175 Vec3 binorm;
176 Vec3 tangent;
177 if (std::abs(cyl_dir_vec.Dot(k_hat) / cyl_dir_vec.Norm()) > (1.0 - 1.0e-12))
178 {
179 binorm = Vec3(0.0, 1.0, 0.0);
180 tangent = Vec3(1.0, 0.0, 0.0);
181 }
182 else
183 {
184 binorm = k_hat.Cross(cyl_dir_vec);
185 binorm = binorm/binorm.Norm();
186 tangent = binorm.Cross(cyl_dir_vec);
187 tangent = tangent/tangent.Norm();
188 }
189
190 //====================================== Project p0r onto the binorm and
191 // tangent
192 const Vec3 p0r_projected(p0r.Dot(tangent), p0r.Dot(binorm), 0.0);
193
194 //====================================== Determine if point is within cylinder
195 if (p0r_projected.NormSquare() <= r*r)
196 return true;
197 else
198 return false;
199 }
200};
