63 LogicalVolume(LogicalVolumeType::LVSPHERE)

67 y0_ = in_y;
68 z0_ = in_z;

70
71 bool Inside(const chi_mesh::Vector3& point) const override
72 {
73 double dx = point.x - x0_;
74 double dy = point.y - y0_;
75 double dz = point.z - z0_;
76
77 double R2 = dx*dx + dy*dy + dz*dz;
78
79 if (R2<= (r_ * r_))
80 return true;
81 else
82 return false;
83 }

126/**Right Circular Cylinder (RCC) logical volume.
127 *
128 * Determining whether a point is within an RCC is tricky.
129 * */
130class chi_mesh::RCCLogicalVolume : public LogicalVolume
131{
132public:
133 double x0_,y0_,z0_;
134 double vx_,vy_,vz_;
135 double r_;
136
137 RCCLogicalVolume() : LogicalVolume(LogicalVolumeType::LVRCC)
138 {
139 x0_ = 0.0; y0_ = 0.0; z0_ = 0.0;
140 vx_ = 0.0; vy_ = 0.0; vz_ = 1.0;
141 r_ = 1.0;
142 }
143
144 RCCLogicalVolume(double ix0, double iy0, double iz0,
145 double ivx, double ivy, double ivz,
146 double ir) : LogicalVolume(LogicalVolumeType::LVRCC)
147 {
148 x0_ = ix0; y0_ = iy0; z0_ = iz0;
149 vx_ = ivx; vy_ = ivy; vz_ = ivz;
150 r_ = ir;
151 }
152
153 bool Inside(const chi_mesh::Vector3& point) const override
154 {
155 typedef chi_mesh::Vector3 Vec3;
156
157 const auto& pr = point; //reference point
158 const Vec3 p0(x0_, y0_, z0_); //cylinder root
159 const Vec3 cyl_dir_vec(vx_, vy_, vz_); //cylinder direction vector
160 const Vec3 k_hat(0.0, 0.0, 1.0); //k_hat
161
162 const Vec3 p0r = pr - p0;
163 const Vec3 cyl_unit_dir = cyl_dir_vec.Normalized(); //aka cud
164 const double cyl_length = cyl_dir_vec.Norm();
165
166 //====================================== Check if point is within
167 // normal extents
168 const double p0r_dot_cud = p0r.Dot(cyl_unit_dir);
169 if (p0r_dot_cud < 0.0 or p0r_dot_cud > cyl_length)
170 return false;
171
172 //====================================== Building rotation matrix
173 //This rotation matrix must be such that,
174 //when a coordinate system is rotated with it,
175 //the new normal vector points along the
176 Vec3 binorm;
177 Vec3 tangent;
178 if (std::abs(cyl_dir_vec.Dot(k_hat) / cyl_dir_vec.Norm()) > (1.0 - 1.0e-12))
179 {
180 binorm = Vec3(0.0, 1.0, 0.0);
181 tangent = Vec3(1.0, 0.0, 0.0);
182 }
183 else
184 {
185 binorm = k_hat.Cross(cyl_dir_vec);
186 binorm = binorm/binorm.Norm();
187 tangent = binorm.Cross(cyl_dir_vec);
188 tangent = tangent/tangent.Norm();
189 }
190
191 //====================================== Project p0r onto the binorm and
192 // tangent
193 const Vec3 p0r_projected(p0r.Dot(tangent), p0r.Dot(binorm), 0.0);
194
195 //====================================== Determine if point is within cylinder
196 if (p0r_projected.NormSquare() <= r_ * r_)
197 return true;
198 else
199 return false;
200 }
201};
