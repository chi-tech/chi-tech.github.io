123 class chi_mesh::RCCLogicalVolume : public LogicalVolume

129

131  {
132  type_index = RCC;
133  x0 = 0.0; y0 = 0.0; z0 = 0.0;

135  r = 1.0;
136  }
137 
138  RCCLogicalVolume(double ix0, double iy0, double iz0,
139  double ivx, double ivy, double ivz,
140  double ir)
141  {
142  type_index = RCC;
143  x0 = ix0; y0 = iy0; z0 = iz0;
144  vx = ivx; vy = ivy; vz = ivz;
145  r = ir;
146  }
147 
148  bool Inside(chi_mesh::Vector p1)
149  {
150  chi_mesh::Vector p0(x0,y0,z0);
151  chi_mesh::Vector vd(vx,vy,vz);
152  chi_mesh::Vector k(0.0,0.0,1.0);
153 
154  chi_mesh::Vector p01 = p1-p0;
155 
156  //====================================== Building rotation matrix
157  chi_mesh::Vector binorm;
158  chi_mesh::Vector tangent;
159  if (abs(vd.Dot(k)/vd.Norm())>(1.0-1.0e-12))
160  {
161  binorm = chi_mesh::Vector(0.0,1.0,0.0);
162  tangent = chi_mesh::Vector(1.0,0.0,0.0);
163  }
164  else
165  {
166  binorm = k.Cross(vd);
167  binorm = binorm/binorm.Norm();
168  tangent = binorm.Cross(vd);
169  tangent = tangent/tangent.Norm();
170  }
171 
172  chi_mesh::Matrix3x3 R;
173 
174  R.SetColJVec(0,tangent);
175  R.SetColJVec(1,binorm);
176  R.SetColJVec(2,vd);
177 
178  //====================================== Rotate point to ref coords
179  chi_mesh::Matrix3x3 Rinv = R.Inverse();
180  chi_mesh::Vector p01T = Rinv*p01;
181 
182  chi_log.Log(LOG_0) << "Inverted p: \n" << p01T.PrintS();
183 
184  //====================================== Determine if point is within cylinder
185  bool in_cylinder = false;
186  double r2 = p01T.x*p01T.x + p01T.y*p01T.y;
187 
188  if (r2 <(this->r*this->r))
189  in_cylinder = true;
190  else
191  return false;
192 
193  //====================================== Determine if point is within extents
194  double dotP = p01T.Dot(vd);
195  if ((dotP>=0.0) and (dotP<=1.0))
196  return true;
197 
198  return false;
199  }
