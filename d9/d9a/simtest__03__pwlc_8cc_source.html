17/**This is a simple test of the Finite Volume spatial discretization applied
18 * to Laplace's problem. */
19int chiSimTest03_PWLC(lua_State*)
20{
21 Chi::log.Log() << "Coding Tutorial 3";
22
23 //============================================= Get grid
24 auto grid_ptr = chi_mesh::GetCurrentHandler().GetGrid();
25 const auto& grid = *grid_ptr;
26
27 Chi::log.Log() << "Global num cells: " << grid.GetGlobalNumberOfCells();
28
29 //============================================= Make SDM
30 typedef std::shared_ptr<chi_math::SpatialDiscretization> SDMPtr;
31 SDMPtr sdm_ptr = chi_math::SpatialDiscretization_PWLC::New(grid);
32 const auto& sdm = *sdm_ptr;
33
34 const auto& OneDofPerNode = sdm.UNITARY_UNKNOWN_MANAGER;
35
36 const size_t num_local_dofs = sdm.GetNumLocalDOFs(OneDofPerNode);
37 const size_t num_globl_dofs = sdm.GetNumGlobalDOFs(OneDofPerNode);
38
39 Chi::log.Log() << "Num local DOFs: " << num_local_dofs;
40 Chi::log.Log() << "Num globl DOFs: " << num_globl_dofs;
41
42 //============================================= Initializes Mats and Vecs
43 const auto n = static_cast<int64_t>(num_local_dofs);
44 const auto N = static_cast<int64_t>(num_globl_dofs);
45 Mat A;
46 Vec x,b;
47
48 A = chi_math::PETScUtils::CreateSquareMatrix(n,N);
49 x = chi_math::PETScUtils::CreateVector(n,N);
50 b = chi_math::PETScUtils::CreateVector(n,N);
51
52 std::vector<int64_t> nodal_nnz_in_diag;
53 std::vector<int64_t> nodal_nnz_off_diag;
54 sdm.BuildSparsityPattern(nodal_nnz_in_diag,nodal_nnz_off_diag, OneDofPerNode);
55
56 chi_math::PETScUtils::InitMatrixSparsity(A,
57 nodal_nnz_in_diag,
58 nodal_nnz_off_diag);
59
60 //============================================= Assemble the system
61 Chi::log.Log() << "Assembling system: ";
62 for (const auto& cell : grid.local_cells)
63 {
64 const auto& cell_mapping = sdm.GetCellMapping(cell);
65 const auto qp_data = cell_mapping.MakeVolumeQuadraturePointData();
66
67 const size_t num_nodes = cell_mapping.NumNodes();
68 MatDbl Acell(num_nodes, VecDbl(num_nodes, 0.0));
69 VecDbl cell_rhs(num_nodes, 0.0);
70
71 for (size_t i=0; i<num_nodes; ++i)
72 {
73 for (size_t j=0; j<num_nodes; ++j)
74 {
75 double entry_aij = 0.0;
76 for (size_t qp : qp_data.QuadraturePointIndices())
77 {
78 entry_aij +=
79 qp_data.ShapeGrad(i, qp).Dot(qp_data.ShapeGrad(j, qp)) *
80 qp_data.JxW(qp);
81 }//for qp
82 Acell[i][j] = entry_aij;
83 }//for j
84 for (size_t qp : qp_data.QuadraturePointIndices())
85 cell_rhs[i] += 1.0 * qp_data.ShapeValue(i, qp) * qp_data.JxW(qp);
86 }//for i
87
88 //======================= Flag nodes for being on dirichlet boundary
89 std::vector<bool> node_boundary_flag(num_nodes, false);
90 const size_t num_faces = cell.faces_.size();
91 for (size_t f=0; f<num_faces; ++f)
92 {
93 const auto& face = cell.faces_[f];
94 if (face.has_neighbor_) continue;
95
96 const size_t num_face_nodes = face.vertex_ids_.size();
97 for (size_t fi=0; fi<num_face_nodes; ++fi)
98 {
99 const uint i = cell_mapping.MapFaceNode(f,fi);
100 node_boundary_flag[i] = true;
101 }//for fi
102 }//for face f
103
104 //======================= Develop node mapping
105 std::vector<int64_t> imap(num_nodes, 0); //node-mapping
106 for (size_t i=0; i<num_nodes; ++i)
107 imap[i] = sdm.MapDOF(cell, i);
108
109 //======================= Assembly into system
110 for (size_t i=0; i<num_nodes; ++i)
111 {
112 if (node_boundary_flag[i]) //if dirichlet node
113 {
114 MatSetValue(A, imap[i], imap[i], 1.0, ADD_VALUES);
115 VecSetValue(b, imap[i], 0.0, ADD_VALUES);
116 }
117 else
118 {
119 for (size_t j=0; j<num_nodes; ++j)
120 {
121 if (not node_boundary_flag[j])
122 MatSetValue(A, imap[i], imap[j], Acell[i][j], ADD_VALUES);
123 }//for j
124 VecSetValue(b, imap[i], cell_rhs[i], ADD_VALUES);
125 }
126 }//for i
127 }//for cell
128
129 Chi::log.Log() << "Global assembly";
130
131 MatAssemblyBegin(A, MAT_FINAL_ASSEMBLY);
132 MatAssemblyEnd(A, MAT_FINAL_ASSEMBLY);
133 VecAssemblyBegin(b);
134 VecAssemblyEnd(b);
135
136 Chi::log.Log() << "Done global assembly";
137
138 //============================================= Create Krylov Solver
139 Chi::log.Log() << "Solving: ";
140 auto petsc_solver =
141 chi_math::PETScUtils::CreateCommonKrylovSolverSetup(
142 A, //Matrix
143 "PWLCDiffSolver", //Solver name
144 KSPCG, //Solver type
145 PCGAMG, //Preconditioner type
146 1.0e-6, //Relative residual tolerance
147 1000); //Max iterations
148
149 //============================================= Solve
150 KSPSolve(petsc_solver.ksp,b,x);
151
152 Chi::log.Log() << "Done solving";
153
154 //============================================= Extract PETSc vector
155 std::vector<double> field;
156 sdm.LocalizePETScVector(x,field,OneDofPerNode);
157
158 //============================================= Clean up
159 KSPDestroy(&petsc_solver.ksp);
160
161 VecDestroy(&x);
162 VecDestroy(&b);
163 MatDestroy(&A);
164
165 Chi::log.Log() << "Done cleanup";
166
167 //============================================= Create Field Function
168 auto ff = std::make_shared<chi_physics::FieldFunctionGridBased>(
169 "Phi", //Text name
170 sdm_ptr, //Spatial Discr.
171 chi_math::Unknown(chi_math::UnknownType::SCALAR) //Unknown
172 );
173
174 ff->UpdateFieldVector(field);
175
176
177 chi_physics::FieldFunctionGridBased::ExportMultipleToVTK("CodeTut3_PWLC", {ff});
178
179 return 0;
180}
