10 /** This functions checks the intersection of a plane with a tetrahedron.
11  * The equation of a plane is
12  * nx(x-x0) + ny(y-y0) + nz(z-z0) = 0
13  * Where the plane normal is (nx,ny,nz) and the plane point is (x0,y0,z0).
14  * If we form a dot product between the normal and a vector
15  * (x-x0,y-y0,z-z0) then sign of the result gives the sense to the surface.
16  * Therefore, if we encounter differing senses then the plane is indeed
17  * intersecting.*/
18 bool chi_mesh::FieldFunctionInterpolation::
19 CheckPlaneTetIntersect(chi_mesh::Normal plane_normal,
20  chi_mesh::Vector plane_point,
21  std::vector<chi_mesh::Vector *>* tet_points)
22 {
23  bool current_sense = false;
24 
25  size_t num_points = tet_points->size();
26  for (int i=0; i<num_points; i++)
27  {
28  chi_mesh::Vector v = (*(*tet_points)[i]) - plane_point;
29  double dotp = plane_normal.Dot(v);
30 
31  bool new_sense = (dotp >= 0.0);
32 
33  if (i==0)
34  current_sense = new_sense;
35  else if (new_sense != current_sense)
36  return true;
37  }
38  return false;
39 }
