4 * \section devman01_02 So you want to use Chi-Tech as a library
5 *
6 * \subsection devman01_02_01 Step 1 - Make a application directory and empty source file
7 *
8 * Wherever you like, create a folder that will contain your application.
9 *
10 * \code
11 * mkdir TestApp
12 * cd TestApp
13 * \endcode
14 *
15 * Now create an empty source file
16 *
17 * \code
18 * >test.cc
19 * \endcode
20 *
21 * ## _
22 * \subsection devman01_02_02 Step 2 - Create CMakeLists.txt file
23 *
24 * In the folder you just created create a text file called `CMakeLists.txt`
25 *
26 * \code
27 * >CMakeLists.txt
28 * \endcode
29 *
30 * ## _
31 * \subsection devman01_02_03 Step 3 - Edit the cmake file to connect to Chi-Tech
32 *
33 * We need to specify a few values to be included in the `CMakeLists.txt` file.
34 * - The desired application/executable name, for now we will call that <B>test</B>.
35 * - The Chi-Tech "downstream" cmake include, `Downstream.cmake`.
36 *
37\code
38cmake_minimum_required(VERSION 3.2)
39
40set(TARGET test_app)
41project(test CXX)
42
43#------------------------------------------------ DEPENDENCIES
44if (NOT DEFINED CHI_TECH_DIR)
45 if (NOT (DEFINED ENV{CHI_TECH_DIR}))
46 message(FATAL_ERROR "***** CHI_TECH_DIR is not set *****")
47 else()
48 set(CHI_TECH_DIR "$ENV{CHI_TECH_DIR}")
49 endif()
50endif()
51message(STATUS "CHI_TECH_DIR set to ${CHI_TECH_DIR}")
52
53include("${CHI_TECH_DIR}/ChiResources/Macros/Downstream.cmake")
54
55set(CMAKE_ARCHIVE_OUTPUT_DIRECTORY "${PROJECT_SOURCE_DIR}/lib")
56set(CMAKE_LIBRARY_OUTPUT_DIRECTORY "${PROJECT_SOURCE_DIR}/lib")
57set(CMAKE_RUNTIME_OUTPUT_DIRECTORY "${PROJECT_SOURCE_DIR}/bin")
58
59file (GLOB_RECURSE SOURCES "*.cc")
60add_executable(${TARGET} "${SOURCES}")
61target_link_libraries(${TARGET} ${CHI_LIBS})
62
63file(WRITE ${PROJECT_SOURCE_DIR}/Makefile "subsystem:\n" "\t$(MAKE) -C chi_build \n\n"
64 "clean:\n\t$(MAKE) -C chi_build clean\n")
65\endcode
66
67The `set(TARGET test_app)` line sets the name of the eventual executable
68to be used. You can use any name other than `test`.
69The `include` statement allows you to connect to all the resources connected
70to Chi-Tech, including lua, PETSc, etc.
71Make sure that this environment variable is set.
72You will have to find the location where you compiled Chi-Tech in order to
73properly specify the location of `Downstream.cmake`. Once this is properly
74specified, the cmake-variable `CHI_LIBS` will be defined and the necessary
75include-files will be usable.
76 The line `file (GLOB_RECURSE SOURCES "*.cc")` adds all `.cc` files,
77 in the current directory, to
78 the list of sources to compile. To specify specific
79 source-file names use `set(SOURCES "test.cc")`. Alternatively cmake
80 functionality can be googled to determine how to add sub-directories.
81

84\subsection devman01_02_04 Step 4 - Add the basic code to test.cc
85
86
87\code
88#include "chi_runtime.h"
89#include "chi_log.h"
90
91int main(int argc, char* argv[])
92{
93 chi::Initialize(argc,argv);
94 chi::RunBatch(argc, argv);
95
96 chi::log.Log() << "Hello World!";
97
98 //We will add code here
99
100 chi::Finalize();
101 return 0;
102}
103\endcode
104
105This code is the minimum needed to have everything available in Chi-Tech.
106The basic namespace access is provided via `#include "chi_runtime.h"`
107
108MPI initialization and PETSc initialization is handled via the call to
109`chi::Initialize()`.
110
111Finally all MPI and PETSc related items are destroyed via the call to
112`chi::Finalize()`.

141 * \section devman01_03 Adding lua-input to library-using apps
142 * For this section we refer to \ref devman01_02 "the section above" where we
143 * created a test application.
144 *
145 * \subsection devman01_03_01 Step 1 - Create folder to contain lua sources.
146 * Purely for categorization, create a ```lua``` folder in the ```TestApp```
147 * folder
148 *
149 * \code
150 * cd ../TestApp
151 * mkdir lua
152 * cd lua
153 * \endcode
154 *
155 * \subsection devman01_03_02 Step 2 - Create a source file
156 * Let us pretend that we will have a status update message that we want to be
157 * printed from a lua call. Create a file called ```testapp_statusmess.cc```
158 * with the following contents:
159 *
160 * \code
161 * #include"ChiLua/chi_lua.h"
162 * #include "chi::log.h"
163 *
164 * int chiPrintStatus(lua_State *L)
165 * {
166 * ChiLog& log = ChiLog::GetInstance();
167 *
168 * log.Log() << "Hello from here";
169 *
170 * return 0;
171 * }
172 * \endcode
173 *
174 * ## _
175 *
176 * \subsection devman01_03_03a Step 3 - Modify the CMakeLists.txt file to compile this code
177 * In preparation for adding many more sources to the ```lua``` folder, edit the
178 * CMakeLists.txt file as follows:
179 *
180 *
181 *
182\code{cmake}
183cmake_minimum_required(VERSION 3.2)
184
185SET(TARGET test)
186project(${TARGET} C CXX)
187
188include(~/Desktop/ChiTech/chi-tech/CHI_RESOURCES/Macros/Downstream.cmake)
189include(~/Desktop/ChiTech/chi-tech/CHI_RESOURCES/Macros/Filter.cmake)
190
191set(SOURCES "test.cc" )
192file (GLOB_RECURSE MORE_SOURCES "${CMAKE_CURRENT_SOURCE_DIR}/lua/*.cc")
193
194set(SOURCES ${SOURCES} ${MORE_SOURCES})
195
196add_executable(${TARGET} "${SOURCES}")
197target_link_libraries(${TARGET} ${CHI_LIBS})
198\endcode
199This will allow any ```.cc``` file in the ```lua``` folder to be compiled with
200the project. The line
201
202
203 \code{cmake}
204 file (GLOB_RECURSE MORE_SOURCES "${CMAKE_CURRENT_SOURCE_DIR}/lua/*.cc")
205 \endcode
206
207
208
209 recursively searches through the specified folder and pumps any filenames
210 ending with .cc into the variable ```MORE_SOURCES```. This variable gets added
211 to the manually specified ```SOURCES``` variable using the line
212
213 \code
214 set(SOURCES ${SOURCES} ${MORE_SOURCES})
215 \endcode
216
217
218 ## _
219 * \subsection devman01_03_04a Step 4 - Modify test.cc to register the lua function
220 * Modify your ```test.cc``` source file to be
221\code
222#include "chi_runtime.h"
223#include "ChiConsole/chi_console.h"
224
225int main(int argc, char* argv[])
226{
227 ChiTech::Initialize(argc,argv);
228
229 ChiConsole& console = ChiConsole::GetInstance();
230
231 auto L = console.consoleState;
232 #include "ChiMacros/lua_register_macro.h"
233
234 RegisterFunction(chiPrintStatus);
235
236 ChiTech::Finalize();
237 return 0;
238}
239\endcode
240
241Notice here that we firstly included the header file for the ```ChiConsole```
242object. This is the object that allows us to connect to the lua state.
243
244\code
245#include "ChiConsole/chi_console.h"
246\endcode
247
248
249
250Next we obtained a reference to the console through
251
252\code
253ChiConsole& console = ChiConsole::GetInstance();
254\endcode
255
256
257 We then setup the variable ```L```, which is needed by the macro file
258 ``` ChiMacros/lua_register_macro.h ```.
259
260
261\code
262auto L = console.consoleState;
263#include "ChiMacros/lua_register_macro.h"
264\endcode
265
266
267
268 This macro file exposes numerous lua macros, one of which is the
269 ```RegisterFunction(x)``` macro where we will supply the ```x``` as the
270 function call we defined earlier:
271
272 \code
273 RegisterFunction(chiPrintStatus);
274 \endcode
