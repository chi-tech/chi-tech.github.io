26 //======================================== Get material source
27 // This is done using the SetSource routine
28 // because it allows a lot of flexibility.
29 auto mat_src = phi_old_local_;

31 for (auto& groupset : groupsets_)

35 PhiOldLocal(),
36 APPLY_FIXED_SOURCES | APPLY_AGS_FISSION_SOURCES |
37 APPLY_WGS_FISSION_SOURCES);

39 }
40
41 //======================================== Compute absorption, material-source
42 // and in-flow
43 double local_out_flow = 0.0;
44 double local_in_flow = 0.0;
45 double local_absorption = 0.0;
46 double local_production = 0.0;
47 for (const auto& cell : grid_ptr_->local_cells)
48 {
49 const auto& cell_mapping = discretization_->GetCellMapping(cell);
50 const auto& transport_view = cell_transport_views_[cell.local_id_];
51 const auto& fe_intgrl_values = unit_cell_matrices_[cell.local_id_];
52 const size_t num_nodes = transport_view.NumNodes();
53 const auto& IntV_shapeI = fe_intgrl_values.Vi_vectors;
54 const auto& IntS_shapeI = fe_intgrl_values.face_Si_vectors;
55
56 //====================================== Inflow
57 // This is essentially an integration over
58 // all faces, all angles, and all groups.
59 // Only the cosines that are negative are
60 // added to the integral.
61 for (int f=0; f<cell.faces_.size(); ++f)
62 {
63 const auto& face = cell.faces_[f];
64
65 for (const auto& groupset : groupsets_)
66 {
67 for (int n = 0; n < groupset.quadrature_->omegas_.size(); ++n)
68 {
69 const auto &omega = groupset.quadrature_->omegas_[n];
70 const double wt = groupset.quadrature_->weights_[n];
71 const double mu = omega.Dot(face.normal_);
72
73 if (mu < 0.0 and (not face.has_neighbor_)) //mu<0 and bndry
74 {
75 const auto &bndry = sweep_boundaries_[face.neighbor_id_];
76 for (int fi = 0; fi < face.vertex_ids_.size(); ++fi)
77 {
78 const int i = cell_mapping.MapFaceNode(f, fi);
79 const auto &IntFi_shapeI = IntS_shapeI[f][i];
80
81 for (const auto& group : groupset.groups_)
82 {
83 const int g = group.id_;
84 const double psi = *bndry->HeterogeneousPsiIncoming(
85 cell.local_id_, f, fi, n, g, 0);

87 }//for g
88 }//for fi
89 }//if bndry
90 }//for n
91 }//for groupset
92 }//for f
93
94
95 //====================================== Outflow
96 //The group-wise outflow was determined
97 //during a solve so here we just
98 //consolidate it.
99 for (int g=0; g < num_groups_; ++g)
100 local_out_flow += transport_view.GetOutflow(g);
101
102 //====================================== Absorption and Src
103 //Isotropic flux based absorption and source
104 const auto& xs = transport_view.XS();
105 const auto& sigma_a = xs.SigmaAbsorption();
106 for (int i=0; i<num_nodes; ++i)
107 for (int g=0; g < num_groups_; ++g)
108 {
109 size_t imap = transport_view.MapDOF(i,0,g);
110 double phi_0g = phi_old_local_[imap];
111 double q_0g = mat_src[imap];
112
113 local_absorption += sigma_a[g] * phi_0g * IntV_shapeI[i];
114 local_production += q_0g * IntV_shapeI[i];
115 }//for g
116 }//for cell
117
118 //======================================== Consolidate local balances
119 double local_balance = local_production + local_in_flow
120 - local_absorption - local_out_flow;
121 double local_gain = local_production + local_in_flow;
122
123 std::vector<double> local_balance_table = {local_absorption,
124 local_production,
125 local_in_flow,
126 local_out_flow,
127 local_balance,
128 local_gain};
129 size_t table_size = local_balance_table.size();

131 std::vector<double> globl_balance_table(table_size,0.0);
132
133 MPI_Allreduce(local_balance_table.data(), //sendbuf
134 globl_balance_table.data(), //recvbuf
135 table_size,MPI_DOUBLE, //count + datatype
136 MPI_SUM, //operation
137 MPI_COMM_WORLD); //communicator
138
139 double globl_absorption = globl_balance_table.at(0);
140 double globl_production = globl_balance_table.at(1);
141 double globl_in_flow = globl_balance_table.at(2);
142 double globl_out_flow = globl_balance_table.at(3);
143 double globl_balance = globl_balance_table.at(4);
144 double globl_gain = globl_balance_table.at(5);
145
146 chi::log.Log() << "Balance table:\n"
147 << std::setprecision(5) << std::scientific
148 << " Absorption rate = " << globl_absorption << "\n"
149 << " Production rate = " << globl_production << "\n"
150 << " In-flow rate = " << globl_in_flow << "\n"
151 << " Out-flow rate = " << globl_out_flow << "\n"
152 << " Net Gain (In-flow + sources) = " << globl_gain << "\n"
153 << " Net Balance = " << globl_balance << "\n"
154 << " (Net Balance)/(Net Gain) = " << globl_balance/globl_gain << "\n";
155
156 chi::log.Log() << "\n********** Done computing balance\n";
157
158 MPI_Barrier(MPI_COMM_WORLD);
159}
