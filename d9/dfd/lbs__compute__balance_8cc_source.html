24/**Compute balance.*/
25void lbs::SteadySolver::ComputeBalance()
26{
27 MPI_Barrier(MPI_COMM_WORLD);
28 chi::log.Log() << "\n********** Computing balance\n";
29
30 auto pwld =
31 std::dynamic_pointer_cast<chi_math::SpatialDiscretization_PWLD>(discretization);
32 if (not pwld) throw std::logic_error("Trouble getting PWLD-SDM in " +
33 std::string(__FUNCTION__));
34 chi_math::SpatialDiscretization_PWLD& grid_fe_view = *pwld;
35
36 //======================================== Get material source
37 // This is done using the SetSource routine
38 // because it allows a lot of flexibility.
39 auto mat_src = phi_old_local;
40 mat_src.assign(mat_src.size(),0.0);
41 for (auto& groupset : groupsets)
42 {
43 q_moments_local.assign(q_moments_local.size(), 0.0);
44 SetSource(groupset, q_moments_local,
45 APPLY_FIXED_SOURCES | APPLY_AGS_FISSION_SOURCES |
46 APPLY_WGS_FISSION_SOURCES);
47 ScopedCopySTLvectors(groupset, q_moments_local, mat_src);
48 }
49
50 //======================================== Initialize diffusion params
51 // for xs
52 // This populates sigma_a
53 for (const auto& mat_id_xs : matid_to_xs_map)
54 {
55 const auto& xs = mat_id_xs.second;
56 if (not xs->diffusion_initialized)
57 xs->ComputeDiffusionParameters();
58 }
59
60 //======================================== Compute absorption, material-source
61 // and in-flow
62 size_t num_groups=groups.size();
63 double local_out_flow = 0.0;
64 double local_in_flow = 0.0;
65 double local_absorption = 0.0;
66 double local_production = 0.0;
67 for (const auto& cell : grid->local_cells)
68 {
69 const auto& transport_view = cell_transport_views[cell.local_id];
70 const auto& fe_intgrl_values = grid_fe_view.GetUnitIntegrals(cell);
71 const size_t num_nodes = transport_view.NumNodes();
72 const auto& IntV_shapeI = fe_intgrl_values.GetIntV_shapeI();
73 const auto& IntS_shapeI = fe_intgrl_values.GetIntS_shapeI();
74
75 //====================================== Inflow
76 // This is essentially an integration over
77 // all faces, all angles, and all groups.
78 // Only the cosines that are negative are
79 // added to the integral.
80 for (int f=0; f<cell.faces.size(); ++f)
81 {
82 const auto& face = cell.faces[f];
83
84 for (const auto& groupset : groupsets)
85 {
86 for (int n = 0; n < groupset.quadrature->omegas.size(); ++n)
87 {
88 const auto &omega = groupset.quadrature->omegas[n];
89 const double wt = groupset.quadrature->weights[n];
90 const double mu = omega.Dot(face.normal);
91
92 if (mu < 0.0 and (not face.has_neighbor)) //mu<0 and bndry
93 {
94 const auto &bndry = sweep_boundaries[face.neighbor_id];
95 for (int fi = 0; fi < face.vertex_ids.size(); ++fi)
96 {
97 const int i = fe_intgrl_values.FaceDofMapping(f, fi);
98 const auto &IntFi_shapeI = IntS_shapeI[f][i];
99
100 for (const auto &group : groupset.groups)
101 {
102 const int g = group.id;
103 const double psi = bndry->boundary_flux[g];
104 local_in_flow -= mu * wt * psi * IntFi_shapeI;
105 }//for g
106 }//for fi
107 }//if bndry
108 }//for n
109 }//for groupset
110 }//for f
111
112
113 //====================================== Outflow
114 //The group-wise outflow was determined
115 //during a solve so here we just
116 //consolidate it.
117 for (int g=0; g<num_groups; ++g)
118 local_out_flow += transport_view.GetOutflow(g);
119
120 //====================================== Absorption and Src
121 //Isotropic flux based absorption and source
122 auto& xs = transport_view.XS();
123 for (int i=0; i<num_nodes; ++i)
124 for (int g=0; g<num_groups; ++g)
125 {
126 size_t imap = transport_view.MapDOF(i,0,g);
127 double phi_0g = phi_old_local[imap];
128 double q_0g = mat_src[imap];
129
130 local_absorption += xs.sigma_a[g] * phi_0g * IntV_shapeI[i];
131 local_production += q_0g * IntV_shapeI[i];
132 }//for g
133 }//for cell
134
135 //======================================== Consolidate local balances
136 double local_balance = local_production + local_in_flow
137 - local_absorption - local_out_flow;
138 double local_gain = local_production + local_in_flow;
139
140 std::vector<double> local_balance_table = {local_absorption,
141 local_production,
142 local_in_flow,
143 local_out_flow,
144 local_balance,
145 local_gain};
146 size_t table_size = local_balance_table.size();
147
148 std::vector<double> globl_balance_table(table_size,0.0);
149
150 MPI_Allreduce(local_balance_table.data(), //sendbuf
151 globl_balance_table.data(), //recvbuf
152 table_size,MPI_DOUBLE, //count + datatype
153 MPI_SUM, //operation
154 MPI_COMM_WORLD); //communicator
155
156 double globl_absorption = globl_balance_table.at(0);
157 double globl_production = globl_balance_table.at(1);
158 double globl_in_flow = globl_balance_table.at(2);
159 double globl_out_flow = globl_balance_table.at(3);
160 double globl_balance = globl_balance_table.at(4);
161 double globl_gain = globl_balance_table.at(5);
162
163 chi::log.Log() << "Balance table:\n"
164 << std::setprecision(5) << std::scientific
165 << " Absorption rate = " << globl_absorption << "\n"
166 << " Production rate = " << globl_production << "\n"
167 << " In-flow rate = " << globl_in_flow << "\n"
168 << " Out-flow rate = " << globl_out_flow << "\n"
169 << " Integrated scalar flux = " << globl_gain << "\n"
170 << " Net Gain/Loss = " << globl_balance << "\n"
171 << " Net Gain/Loss normalized = " << globl_balance/globl_gain << "\n";
172
173 chi::log.Log() << "\n********** Done computing balance\n";
174
175 MPI_Barrier(MPI_COMM_WORLD);
176}
