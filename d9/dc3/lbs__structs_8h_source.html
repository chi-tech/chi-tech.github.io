44inline
45chi_math::CoordinateSystemType
46 MapGeometryTypeToCoordSys(const GeometryType gtype)
47{
48 using namespace chi_math;
49 switch (gtype)
50 {
51 case GeometryType::ONED_SLAB:
52 case GeometryType::TWOD_CARTESIAN:
53 case GeometryType::THREED_CARTESIAN:
54 return CoordinateSystemType::CARTESIAN;
55 case GeometryType::ONED_SPHERICAL:
56 return CoordinateSystemType::SPHERICAL;
57 case GeometryType::ONED_CYLINDRICAL:
58 case GeometryType::TWOD_CYLINDRICAL:
59 return CoordinateSystemType::CYLINDRICAL;
60 default:
61 return CoordinateSystemType::CARTESIAN;
62 }
