37inline
38chi_math::CoordinateSystemType
39 MapGeometryTypeToCoordSys(const GeometryType gtype)
40{
41 using namespace chi_math;
42 switch (gtype)
43 {
44 case GeometryType::ONED_SLAB:
45 case GeometryType::TWOD_CARTESIAN:
46 case GeometryType::THREED_CARTESIAN:
47 return CoordinateSystemType::CARTESIAN;
48 case GeometryType::ONED_SPHERICAL:
49 return CoordinateSystemType::SPHERICAL;
50 case GeometryType::ONED_CYLINDRICAL:
51 case GeometryType::TWOD_CYLINDRICAL:
52 return CoordinateSystemType::CYLINDRICAL;
53 default:
54 return CoordinateSystemType::CARTESIAN;
55 }
56}
