372 inline float Vector3::angle(const Vector3& vec) const {
373  // return angle between [0, 180]
374  float l1 = this->length();
375  float l2 = vec.length();
376  float d = this->dot(vec);
377  float angle = acosf(d / (l1 * l2)) / 3.141592f * 180.0f;
378  return angle;
379 }
