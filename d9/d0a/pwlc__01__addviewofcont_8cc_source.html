115void SpatialDiscretization_PWLC::PreComputeCellSDValues()
116{
117 MPI_Barrier(MPI_COMM_WORLD);
118 chi_log.Log(LOG_0VERBOSE_1) << chi_program_timer.GetTimeString()
119 << " Add cell SD-values.";
120
121 size_t num_local_cells = ref_grid->local_cells.size();
122
123 //================================================== Create empty view
124 // for each cell
125 {
126 using namespace chi_math::finite_element;
127 if (setup_flags & SetupFlags::COMPUTE_CELL_MAPPINGS)
128 {
129 chi_log.Log(LOG_0VERBOSE_1) << chi_program_timer.GetTimeString()
130 << " Computing unit integrals.";
131 if (!mapping_initialized)
132 {
133 for (const auto& cell : ref_grid->local_cells)
134 cell_mappings.push_back(MakeCellMappingFE(cell));
135
136 mapping_initialized = true;
137 }
138 }
139 }
140 MPI_Barrier(MPI_COMM_WORLD);
141
142 //============================================= Unit integrals
143 {
144 using namespace chi_math::finite_element;
145 if (setup_flags & SetupFlags::COMPUTE_UNIT_INTEGRALS)
146 {
147 chi_log.Log(LOG_0VERBOSE_1) << chi_program_timer.GetTimeString()
148 << " Computing unit integrals.";
149 if (not integral_data_initialized)
150 {
151 fe_unit_integrals.reserve(num_local_cells);
152 for (size_t lc=0; lc<num_local_cells; ++lc)
153 {
154 UIData ui_data;
155
156 auto cell_fe_view = GetCellMappingFE(lc);
157 cell_fe_view->ComputeUnitIntegrals(ui_data);
158
159 fe_unit_integrals.push_back(std::move(ui_data));
160 }
161
162 integral_data_initialized = true;
163 }
164 }//if compute unit intgrls
165 }
166 MPI_Barrier(MPI_COMM_WORLD);
167
168 //============================================= Quadrature data
169 {
170 using namespace chi_math::finite_element;
171 if (setup_flags & SetupFlags::COMPUTE_QP_DATA)
172 {
173 chi_log.Log(LOG_0VERBOSE_1) << chi_program_timer.GetTimeString()
174 << " Computing quadrature data.";
175 if (not qp_data_initialized)
176 {
177 fe_vol_qp_data.reserve(num_local_cells);
178 fe_srf_qp_data.reserve(num_local_cells);
179 for (size_t lc=0; lc<num_local_cells; ++lc)
180 {
181 fe_vol_qp_data.emplace_back();
182 fe_srf_qp_data.emplace_back();
183
184 auto cell_fe_view = GetCellMappingFE(lc);
185 cell_fe_view->InitializeAllQuadraturePointData(fe_vol_qp_data.back(),
186 fe_srf_qp_data.back());
187 }
188
189 qp_data_initialized = true;
190 }
191 }//if init qp data
192 }
193 MPI_Barrier(MPI_COMM_WORLD);
194 chi_log.Log(LOG_0VERBOSE_1) << chi_program_timer.GetTimeString()
195 << " Done adding cell SD-values.";
196
