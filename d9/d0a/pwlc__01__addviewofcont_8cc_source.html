116/**Adds a PWL Finite Element for each cell of the local problem.*/
117void chi_math::SpatialDiscretization_PWLC::PreComputeCellSDValues()
118{
119 MPI_Barrier(MPI_COMM_WORLD);
120 chi::log.Log0Verbose1() << chi::program_timer.GetTimeString()
121 << " Add cell SD-values.";
122
123 size_t num_local_cells = ref_grid->local_cells.size();
124
125 //================================================== Create empty view
126 // for each cell
127 {
128 using namespace chi_math::finite_element;
129 if (setup_flags & SetupFlags::COMPUTE_CELL_MAPPINGS)
130 {
131 chi::log.Log0Verbose1() << chi::program_timer.GetTimeString()
132 << " Computing unit integrals.";
133 if (!mapping_initialized)
134 {
135 for (const auto& cell : ref_grid->local_cells)
136 cell_mappings.push_back(MakeCellMappingFE(cell));
137
138 mapping_initialized = true;
139 }
140 }
141 }
142 MPI_Barrier(MPI_COMM_WORLD);
143
144 //============================================= Unit integrals
145 {
146 using namespace chi_math::finite_element;
147 if (setup_flags & SetupFlags::COMPUTE_UNIT_INTEGRALS)
148 {
149 chi::log.Log0Verbose1() << chi::program_timer.GetTimeString()
150 << " Computing unit integrals.";
151 if (not integral_data_initialized)
152 {
153 fe_unit_integrals.reserve(num_local_cells);
154 for (size_t lc=0; lc<num_local_cells; ++lc)
155 {
156 UIData ui_data;
157
158 auto cell_fe_view = GetCellMappingFE(lc);
159 cell_fe_view->ComputeUnitIntegrals(ui_data);
160
161 fe_unit_integrals.push_back(std::move(ui_data));
162 }
163
164 integral_data_initialized = true;
165 }
166 }//if compute unit intgrls
167 }
168 MPI_Barrier(MPI_COMM_WORLD);
169
170 //============================================= Quadrature data
171 {
172 using namespace chi_math::finite_element;
173 if (setup_flags & SetupFlags::COMPUTE_QP_DATA)
174 {
175 chi::log.Log0Verbose1() << chi::program_timer.GetTimeString()
176 << " Computing quadrature data.";
177 if (not qp_data_initialized)
178 {
179 fe_vol_qp_data.reserve(num_local_cells);
180 fe_srf_qp_data.reserve(num_local_cells);
181 for (size_t lc=0; lc<num_local_cells; ++lc)
182 {
183 fe_vol_qp_data.emplace_back();
184 fe_srf_qp_data.emplace_back();
185
186 auto cell_fe_view = GetCellMappingFE(lc);
187 cell_fe_view->InitializeAllQuadraturePointData(fe_vol_qp_data.back(),
188 fe_srf_qp_data.back());
189 }
190
191 qp_data_initialized = true;
192 }
193 }//if init qp data
194 }
195 MPI_Barrier(MPI_COMM_WORLD);
196 chi::log.Log0Verbose1() << chi::program_timer.GetTimeString()
197 << " Done adding cell SD-values.";
198
199}//AddViewOfLocalContinuum
