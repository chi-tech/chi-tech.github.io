556 void chi_mesh::SurfaceMesh::UpdateInternalConnectivity()
557 {
558  std::vector<chi_mesh::Face>::iterator curFace;
559  for (curFace = this->faces.begin(); curFace!=this->faces.end(); curFace++)
560  {
561  int outer_index = std::distance(this->faces.begin(),curFace);
562 
563  //Reset face connections
564  for (int e=0;e<3;e++) //Loop over current face's edges
565  {
566  curFace->e_index[e][2] = -1; //tri index
567  curFace->e_index[e][3] = -1;
568  }
569  //printf("Outer index %d\n",outer_index);
570  //=========================================== Determine face connectivity
571  //If the head v_index of the current triangle is equal to the tail
572  //of the other triangle and vice versa on the tail of the current
573  //triangle, then they are connected.
574  std::vector<chi_mesh::Face>::iterator other_face;
575  for (other_face = faces.begin(); other_face!=faces.end(); other_face++)
576  {
577  int inner_index = std::distance(this->faces.begin(),other_face);
578  //if (outer_index!=inner_index)
579  {
580  for (int e=0;e<3;e++) //Loop over current face's edges
581  {
582 
583  for (int e2=0;e2<3;e2++) //Loop over other face's edges
584  {
585  if ( (curFace->e_index[e][0]==other_face->e_index[e2][1]) &&
586  (curFace->e_index[e][1]==other_face->e_index[e2][0]) )
587  {
588  curFace->e_index[e][2] = inner_index; //tri index
589  curFace->e_index[e][3] = e2; //edge index
590  }
591  }
592  }
593  }
594 
595  }
596  }
597 
598  for (int f=0; f<this->poly_faces.size(); f++)
599  {
600  chi_mesh::PolyFace* curFace = this->poly_faces[f];
601 
602  for (int f2=0; f2<this->poly_faces.size(); f2++)
603  {
604  if (f!=f2)

608  {
609

612  if ( (curFace->edges[e][0]==other_face->edges[e2][1]) &&
613  (curFace->edges[e][1]==other_face->edges[e2][0]) )
614  {
615  curFace->edges[e][2] = f2; //tri index
