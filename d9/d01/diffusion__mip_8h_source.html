36/**Boundary condition type. We essentially only support two
37 * types: Dirichlet and Reflecting, the latter is covered under
38 * the ROBIN-type boundary condition.*/
39enum class BCType
40{
41 DIRICHLET = 1,
42 ROBIN = 2
43};
44
45/**Simple data structure to specify boundary conditions. Its stores the
46 * BC-type in `type` and an array of 3 values in `values`. For a
47 * Dirichlet-BC only `values[0]` is used to specify the value of the BC.
48 * For a robin boundary condition we use all 3 values in the form
49\f[
50a\phi + b \mathbf{n} \frac{\partial \phi}{\partial \mathbf{x}} = f
51\f]
52where \f$ a \f$, \f$ b \f$ and \f$ f \f$ map to `values[0]`, `values[1]` and
53`values[2]`, respectively.
54*/
55struct BoundaryCondition
56{
57 BCType type = BCType::DIRICHLET;
58 std::array<double,3> values = {0,0,0};
59};
