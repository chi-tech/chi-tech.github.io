160  AssembleVector(groupset,phi_old,phi_old_local.data());
161

163  sweep_chunk->SetDestinationPhi(&phi_new_local);
164  sweep_chunk->suppress_surface_src = true; //Action of Ax specific
165 
166  VecCopy(phi_old,phi_new);
