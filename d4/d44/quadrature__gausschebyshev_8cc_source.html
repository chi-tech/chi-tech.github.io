72{
73 if (verbose_)

77
78 const double pi_N = M_PI / N;
79 for (unsigned int n = 0; n < N; ++n)
80 {
81 const double xn = -std::cos((2 * n + 1) * pi_N / 2.0);
82 const double wn = pi_N;

84 qpoints_.emplace_back(xn);
