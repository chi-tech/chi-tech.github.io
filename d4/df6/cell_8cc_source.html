179 /**Computes the face area.*/
180 double chi_mesh::CellFace::ComputeFaceArea(chi_mesh::MeshContinuum& grid) const
181 {
182  if (vertex_ids.size() <= 1)
183  return 1.0;
184  else if (vertex_ids.size() == 2)
185  {
186  const auto& v0 = grid.vertices[vertex_ids[0]];
187  const auto& v1 = grid.vertices[vertex_ids[1]];
188 
189  return (v1 - v0).Norm();
190  }
191  else
192  {
193  double area = 0.0;
194  auto& v2 = centroid;
195  const auto num_verts = vertex_ids.size();
196  for (uint64_t v=0; v<num_verts; ++v)
197  {
198  uint64_t vid0 = vertex_ids[v];
199  uint64_t vid1 = (v < (num_verts-1))? vertex_ids[v+1] : vertex_ids[0];
200 
201  const auto& v0 = grid.vertices[vid0];
202  const auto& v1 = grid.vertices[vid1];
203 
204  auto v01 = v1-v0;
205  auto v02 = v2-v0;
206 
207  chi_mesh::Matrix3x3 J;
208  J.SetColJVec(0,v01);
209  J.SetColJVec(1,v02);
210  J.SetColJVec(2,chi_mesh::Vector3(1.0,1.0,1.0));
211 
212  area += 0.5*std::fabs(J.Det());
213  }
214 
215  return area;
216  }

391 /**Recomputes the cell centroid and all face centroids assuming
392  * the mesh vertices have been transformed.*/
393 void chi_mesh::Cell::
394  RecomputeCentroidsAndNormals(const chi_mesh::MeshContinuum &grid)
395 {
396  const auto k_hat = Vector3(0,0,1);
397 
398  centroid = chi_mesh::Vector3(0,0,0);
399  for (uint64_t vid : vertex_ids)
400  centroid += grid.vertices[vid];
401  centroid /= static_cast<double>(vertex_ids.size());
402 
403  for (auto& face : faces)
404  {
405  face.RecomputeCentroid(grid);
406 
407  if (cell_type == CellType::POLYGON)
408  {
409  const auto v0 = grid.vertices[face.vertex_ids[0]];
410  const auto v1 = grid.vertices[face.vertex_ids[1]];
411 
412  const auto v01 = v1 - v0;
413 
414  face.normal = v01.Cross(k_hat).Normalized();
415  }
416  else if (cell_type == CellType::POLYHEDRON)
417  {
418  // A face of a polyhedron can itself be a polygon
419  // which can be multifaceted. Here we need the
420  // average normal over all the facets computed
421  // using an area-weighted average.
422  const size_t num_face_verts = face.vertex_ids.size();
423  double total_area = 0.0;
424  auto weighted_normal = Vector3(0,0,0);
425  for (size_t fv=0; fv<num_face_verts; ++fv)
426  {
427  size_t fvp1 = (fv < (num_face_verts-1))? fv+1 : 0;
428 
429  uint64_t fvid_m = face.vertex_ids[fv ];
430  uint64_t fvid_p = face.vertex_ids[fvp1];
431 
432  auto leg_m = grid.vertices[fvid_m] - face.centroid;
433  auto leg_p = grid.vertices[fvid_p] - face.centroid;
434 
435  auto vn = leg_m.Cross(leg_p);
436 
437  double area = 0.5*vn.Norm();
438  total_area += area;
439 
440  weighted_normal = weighted_normal + area*vn.Normalized();
441  }
442  weighted_normal = weighted_normal/total_area;
443 
444  face.normal = weighted_normal.Normalized();
445  }
446  }
