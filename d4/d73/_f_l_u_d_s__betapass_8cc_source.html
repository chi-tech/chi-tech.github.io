191 {

193 
194  //======================== First entry is number of face dofs
195  face_indices.push_back(num_face_dofs);
196 
197  //======================== Second entry is amount of cells
198  face_indices.push_back(num_cells);
199 
200  //======================== Third entry is negative global cell index
201  // Each time a negative entry occurs it denotes a cell face but
202  // the actual number is -cell_g_index-1. The offset is necessary
203  // for evaluating the negative. The offset is restored during the
204  // deserialization process.
205  // It is followed by a positive number which is the store location

207  for (int c=0; c<num_cells; c++)
208  {
209  int glob_index = -(*cell_views)[c].first-1;
210 
211  std::vector<CompactFaceView>* cell_face_views =

213

215  for (size_t f=0; f<num_faces; f++)
216  {
217  face_indices.push_back(glob_index);
218  face_indices.push_back((*cell_face_views)[f].first);

220
