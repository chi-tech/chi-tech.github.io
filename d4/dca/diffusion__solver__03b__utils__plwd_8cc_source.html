31 void chi_diffusion::Solver::ReorderNodesPWLD()
32 {
33  CHI_TIMER t_stage[6];
34 
35  t_stage[0].Reset();
36  //================================================== Get reference to continuum
37  auto handler = chi_mesh::GetCurrentHandler();
38  auto region = handler->region_stack.back();
39  auto vol_continuum = region->volume_mesh_continua.back();
40 
41  auto mesher = handler->volume_mesher;
42 
43  //================================================== Get local DOF count
44  pwld_local_dof_count=0;
45  size_t num_loc_cells = vol_continuum->local_cell_glob_indices.size();
46  for (int lc=0; lc<num_loc_cells; lc++)
47  {
48  int cell_glob_index = vol_continuum->local_cell_glob_indices[lc];
49  auto cell = vol_continuum->cells[cell_glob_index];
50 
51  // %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% SLAB
52  if (typeid(*cell) == typeid(chi_mesh::CellSlab))
53  {
54  pwld_local_dof_count += 2;
55  }
56 
57  // %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% POLYGON
58  if (typeid(*cell) == typeid(chi_mesh::CellPolygon))
59  {
60  auto poly_cell = (chi_mesh::CellPolygon*)cell;
61  pwld_local_dof_count += poly_cell->v_indices.size();
62  }
63 
64  // %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% POLYHEDRON
65  if (typeid(*cell) == typeid(chi_mesh::CellPolyhedron))
66  {
67  auto polyh_cell = (chi_mesh::CellPolyhedron*)cell;
68  pwld_local_dof_count += polyh_cell->v_indices.size();
69  }
70  }
71 
72  //================================================== Get global DOF count
73  pwld_global_dof_count=0;
74  MPI_Allreduce(&pwld_local_dof_count, //Send buffer
75  &pwld_global_dof_count, //Recv buffer
76  1,MPI_INT,MPI_SUM,MPI_COMM_WORLD);
77 
78  //================================================== Ring communicate DOF start
79  pwld_local_dof_start = 0;
80  if (chi_mpi.location_id != 0)
81  {
82  MPI_Recv(&pwld_local_dof_start,
83  1,MPI_INT, //Count and type
84  chi_mpi.location_id-1, //Source
85  111, //Tag
86  MPI_COMM_WORLD,MPI_STATUS_IGNORE);
87  }
88 
89  if (chi_mpi.location_id != (chi_mpi.process_count-1))
90  {
91  int next_loc_start = pwld_local_dof_start+pwld_local_dof_count;
92  MPI_Send(&next_loc_start,
93  1,MPI_INT,
94  chi_mpi.location_id+1,
95  111,
96  MPI_COMM_WORLD);
97  }
98 
99  chi_log.Log(LOG_ALLVERBOSE_2)
100  << "Local dof count, start, total "
101  << pwld_local_dof_count << " "
102  << pwld_local_dof_start << " "
103  << pwld_global_dof_count;
104 
105 }

153 double chi_diffusion::Solver::HPerpendicularPoly(int Nv, double area, double perimeter)
154 {
155  double hp = 0.0;
156 
157  if (Nv == 3)
158  hp = 2*area/perimeter;
159  else if (Nv == 4)
160  hp = area/perimeter;
161  else //Nv > 4
162  {
163  if (Nv%2 == 0)
164  hp = 4*area/perimeter;
165  else
166  {
167  hp = 2*area/perimeter;
168  hp += sqrt(2*area / Nv*sin(2*M_PI/Nv));
169  }
170  }
171 
172  return hp;
173 }

268  }

272  fmap = af;
273  }//for adj faces

380 void chi_diffusion::Solver::SpawnBorderCell(int locI, int cell_border_index)
381 {
382  //============================================= Obtain the border cell info
383  DIFFUSION_IP_BORDERCELL* cell_info =
384  ip_locI_bordercell_info[locI][cell_border_index];
385 
386  //============================================= Create IP view
387  DIFFUSION_IP_VIEW* ip_view = new DIFFUSION_IP_VIEW;
388  ip_view->cell_dof_start = cell_info->cell_dof_start;
389  ip_locI_borderipviews[locI][cell_border_index] = ip_view;
390 
391  //============================================= Create cell
392  if (cell_info->cell_type == 0)
393  {
394  chi_mesh::CellSlab* cell = new chi_mesh::CellSlab;
395  cell->partition_id = locI;
396  cell->material_id = cell_info->cell_mat_id;
397 
398  chi_mesh::Vector vc;
399  for (int v=0; v<cell_info->cell_dof_count; v++)
400  {
401  vc = vc + *grid->nodes[cell_info->v_indices[v]];
402  cell->v_indices[v] = cell_info->v_indices[v];
403  }
404  cell->centroid = vc/cell_info->cell_dof_count;
405 
406  ip_locI_bordercells[locI][cell_border_index] = cell;
407 
408  SlabFEView* fe_view = new SlabFEView(cell, grid);
409 
410  ip_locI_borderfeviews[locI][cell_border_index] = fe_view;
411  }
412  //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% POLYGON
413  else if (cell_info->cell_type == 1)
414  {
415  chi_mesh::CellPolygon* cell = new chi_mesh::CellPolygon;
416  cell->partition_id = locI;
417  cell->material_id = cell_info->cell_mat_id;
418 
419  chi_mesh::Vector vc;
420  for (int v=0; v<cell_info->cell_dof_count; v++)
421  {
422  vc = vc + *grid->nodes[cell_info->v_indices[v]];
423  cell->v_indices.push_back(cell_info->v_indices[v]);
424  }
425  cell->centroid = vc/cell_info->cell_dof_count;
426 
427  cell->edges.resize(cell_info->cell_face_count);
428  for (int f=0; f<cell_info->cell_face_count; f++)
429  {
430  cell->edges[f] = new int[2];
431  for (int fv=0; fv<cell_info->face_v_indices[f].size(); fv++)
432  cell->edges[f][fv] = cell_info->face_v_indices[f][fv];
433  }
434 
435  ip_locI_bordercells[locI][cell_border_index] = cell;
436 
437  PolygonFEView* fe_view =
438  new PolygonFEView(cell, grid, (CHI_DISCRETIZATION_PWL*)discretization);
439 
440  fe_view->PreCompute();
441 
442  ip_locI_borderfeviews[locI][cell_border_index] = fe_view;
443 
444  }
445  //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% POLYHEDRON
446  else if (cell_info->cell_type == 2)
447  {
448  chi_mesh::CellPolyhedron* cell = new chi_mesh::CellPolyhedron;
449  cell->partition_id = locI;
450  cell->material_id = cell_info->cell_mat_id;
451 
452  chi_mesh::Vector vc;
453  for (int v=0; v<cell_info->cell_dof_count; v++)
454  {
455  vc = vc + *grid->nodes[cell_info->v_indices[v]];
456  cell->v_indices.push_back(cell_info->v_indices[v]);
457  }
458  cell->centroid = vc/cell_info->cell_dof_count;
459 
460  cell->faces.resize(cell_info->cell_face_count);
461  for (int f=0; f<cell_info->cell_face_count; f++)
462  {
463  cell->faces[f] = new chi_mesh::PolyFace;
464  chi_mesh::Vertex vfc;
465  for (int fv=0; fv<cell_info->face_v_indices[f].size(); fv++)
466  {
467  cell->faces[f]->v_indices.push_back(cell_info->face_v_indices[f][fv]);
468  vfc = vfc + *grid->nodes[cell_info->face_v_indices[f][fv]];
469  }
470  vfc = vfc/cell_info->face_v_indices[f].size();
471  cell->faces[f]->face_centroid = vfc;
472 
473  chi_mesh::Vector v0fc = vfc - *grid->nodes[cell_info->face_v_indices[f][0]];
474  chi_mesh::Vector v01 = *grid->nodes[cell_info->face_v_indices[f][1]] -
475  *grid->nodes[cell_info->face_v_indices[f][0]];
476 
477  chi_mesh::Vector n = v01.Cross(v0fc);
478  cell->faces[f]->geometric_normal = n/n.Norm();
479 
480  //Populate edges
481  int edge_count = cell->faces[f]->v_indices.size();
482  cell->faces[f]->edges.resize(edge_count);
483  for (int e=0; e<edge_count; e++)
484  {
485  cell->faces[f]->edges[e] = new int[4];
486  if (e != (edge_count-1))
487  {
488  cell->faces[f]->edges[e][0] = cell->faces[f]->v_indices[e];
489  cell->faces[f]->edges[e][1] = cell->faces[f]->v_indices[e+1];
490  } else
491  {
492  cell->faces[f]->edges[e][0] = cell->faces[f]->v_indices[e];
493  cell->faces[f]->edges[e][1] = cell->faces[f]->v_indices[0];
494  }
495  }
496 
497  }//for f
498 
499 
500  ip_locI_bordercells[locI][cell_border_index] = cell;
501 
502  PolyhedronFEView* fe_view =
503  new PolyhedronFEView(cell,grid,(CHI_DISCRETIZATION_PWL*)discretization);
504 
505  fe_view->PreCompute();
506 
507  ip_locI_borderfeviews[locI][cell_border_index] = fe_view;
508 
509  }
510  else
511  {
512  chi_log.Log(LOG_ALLERROR)
513  << "In function chi_diffusion::Solver::SpawnBorderCell: "
514  "Unsupported cell type encountered.";
515  exit(EXIT_FAILURE);
516  }
517 
518 }
