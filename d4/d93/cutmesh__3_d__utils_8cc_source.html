9 /**Checks the quality of a polyhedron against the following.
10  *
11  * Take each face and form triangles using the face edges and the face
12  * centroid. Each triangle must ccw-orientation and have a normal pointing
13  * in the same general direction as the average face normal.
14  *
15  * Now take each triangle and form a tetrahedron with the cell-centroid. The
16  * quality requirement is that no inverted tetrahedron face is inverted with
17  * respect to the cell-centroid.*/
18 bool chi_mesh::mesh_cutting::
19  CheckPolyhedronQuality(const MeshContinuum &mesh, const chi_mesh::Cell& cell)
20 {
21  const auto& C = cell.centroid;
22 
23  for (auto& face : cell.faces)
24  {
25  const auto& v0 = face.centroid;
26  size_t num_edges = face.vertex_ids.size();
27 
28  for (int e=0; e<num_edges; ++e)
29  {
30  auto edge = MakeEdgeFromPolygonEdgeIndex(face.vertex_ids,e);
31 
32  const auto& v1 = mesh.vertices[edge.first];
33  const auto& v2 = mesh.vertices[edge.second];
34 
35  auto v01 = v1-v0;
36  auto v02 = v2-v0;
37 
38  auto n = v01.Cross(v02);
39  auto C_tri = (v0+v1+v2)/3.0;
40 
41  auto CC = C_tri - C;
42 
43  if (CC.Dot(n)<0.0)
44  return false;
45  }
46  }//for face
47 
48  return true;
49 }
